Advanced RAG Architectures

Advanced RAG patterns represent the cutting edge of retrieval-augmented generation, incorporating self-reflection, error correction, autonomous agents, and multi-modal capabilities. These sophisticated architectures enable more reliable, adaptive, and intelligent information retrieval and generation systems.

Self-RAG

Self-reflecting retrieval system that evaluates its own outputs and decides when to retrieve additional information.

  • Self-evaluation mechanisms
  • Adaptive retrieval decisions
  • Quality-aware generation

Corrective RAG

Error-correcting system that validates retrieved information and refines results through iterative improvement.

  • Relevance verification
  • Iterative refinement
  • Fallback strategies

Agentic RAG

Autonomous agent-driven retrieval that can plan, execute, and adapt retrieval strategies dynamically.

  • Multi-step reasoning
  • Tool integration
  • Dynamic strategy adaptation

class ReflectionToken(Enum):
    """Special tokens for self-reflection"""
    RETRIEVE = "[Retrieve]"
    NO_RETRIEVE = "[No Retrieve]"
    RELEVANT = "[Relevant]"
    IRRELEVANT = "[Irrelevant]" 
    SUPPORTED = "[Fully Supported]"
    PARTIALLY_SUPPORTED = "[Partially Supported]"
    NOT_SUPPORTED = "[No Support]"
    USEFUL = "[Utility:5]"
    NOT_USEFUL = "[Utility:1]"
    CONTINUE = "[Continue to Use Evidence]"
    REGENERATE = "[Utility:1]"

@dataclass
class SelfRAGConfig:

    relevance_threshold: float = 0.6

    """Self-Reflective Retrieval-Augmented Generation System"""

            device_map="auto"
        )
        
        # Add special tokens
        special_tokens = [token.value for token in ReflectionToken]
        self.tokenizer.add_special_tokens({"additional_special_tokens": special_tokens})
        self.model.resize_token_embeddings(len(self.tokenizer))
        
        # Retriever component
        self.retriever = retriever
        
        # Reflection classifiers (simplified - in practice, train specialized models)

            'relevance': self._build_simple_classifier(),

        }
    
    def generate_with_reflection(
        self, 
        query: str, 
        max_length: int = 512
    ) -> Dict[str, Any]:
        """
        Generate response with self-reflection mechanism
        """
        conversation_history = []
        retrieved_docs = []
        reflection_log = []
        
        # Step 1: Determine if retrieval is needed
        retrieval_decision = self._should_retrieve(query, conversation_history)
        reflection_log.append(f"Retrieval decision: {retrieval_decision}")
        
        if retrieval_decision:
            # Step 2: Retrieve documents
            retrieved_docs = self._retrieve_documents(query)
            
            # Step 3: Assess relevance of retrieved documents
            relevant_docs = self._filter_relevant_documents(query, retrieved_docs)
            reflection_log.append(f"Filtered {len(retrieved_docs)} to {len(relevant_docs)} relevant docs")
            
            retrieved_docs = relevant_docs
        
        # Step 4: Generate response
        response = self._generate_response(query, retrieved_docs)
        
        # Step 5: Self-critique and potential regeneration
        if self.config.enable_critique:
            critique_result = self._critique_response(query, retrieved_docs, response)
            reflection_log.append(f"Critique result: {critique_result}")
            
            if critique_result['needs_improvement']:
                # Iterative improvement
                for iteration in range(self.config.max_iterations - 1):
                    if critique_result['action'] == 'retrieve_more':
                        additional_docs = self._retrieve_additional_documents(
                            query, response, retrieved_docs
                        )
                        retrieved_docs.extend(additional_docs)
                    
                    response = self._generate_response(query, retrieved_docs, previous_attempt=response)
                    critique_result = self._critique_response(query, retrieved_docs, response)
                    reflection_log.append(f"Iteration {iteration + 1} critique: {critique_result}")
                    
                    if not critique_result['needs_improvement']:
                        break
        
        return {
            'response': response,
            'retrieved_documents': retrieved_docs,
            'reflection_log': reflection_log,
            'metadata': {
                'used_retrieval': retrieval_decision,
                'num_documents': len(retrieved_docs),
                'critique_iterations': len([log for log in reflection_log if 'Iteration' in log])
            }
        }
    
    def _should_retrieve(self, query: str, context: List[str]) -> bool:
        """Determine if retrieval is needed for the query"""
        
        # Build prompt for retrieval decision
        prompt = f"""
Given the query: "{query}"

Determine if external information retrieval is needed to answer this query accurately.

Consider:
- Is this a factual question requiring specific information?
- Can this be answered with general knowledge?
- Does the query require up-to-date information?

Decision: {ReflectionToken.RETRIEVE.value} or {ReflectionToken.NO_RETRIEVE.value}
"""
        
        # Get model decision (simplified - use classification in practice)
        decision_text = self._generate_text(prompt, max_length=50)
        
        # Parse decision
        if ReflectionToken.RETRIEVE.value in decision_text:
            return True
        elif ReflectionToken.NO_RETRIEVE.value in decision_text:
            return False
        else:
            # Fallback to classifier
            return self._classify_retrieval_need(query, context)
    
    def _classify_retrieval_need(self, query: str, context: List[str]) -> bool:
        """Classify whether retrieval is needed using heuristics"""
        
        # Keywords that often require retrieval
        factual_indicators = [
            'what', 'when', 'where', 'who', 'how many', 'statistics',
            'latest', 'recent', 'current', 'today', 'specific', 'details'
        ]
        
        query_lower = query.lower()
        has_factual_indicators = any(indicator in query_lower for indicator in factual_indicators)
        
        # Length-based heuristic
        is_complex_query = len(query.split()) > 5
        
        # Combine heuristics
        confidence = 0.0
        if has_factual_indicators:
            confidence += 0.4
        if is_complex_query:
            confidence += 0.3
        if not context:  # No previous context
            confidence += 0.2
        
        return confidence > self.config.retrieval_threshold
    
    def _retrieve_documents(self, query: str) -> List[Dict]:
        """Retrieve documents using the configured retriever"""
        if not self.retriever:
            return []
        
        try:
            results = self.retriever.search(query, top_k=10)
            return [
                {
                    'content': result.content if hasattr(result, 'content') else result['content'],
                    'score': result.fusion_score if hasattr(result, 'fusion_score') else result.get('score', 1.0),
                    'id': getattr(result, 'doc_id', result.get('id', str(i)))
                }
                for i, result in enumerate(results)
            ]
        except Exception as e:
            print(f"Retrieval failed: {e}")
            return []
    
    def _filter_relevant_documents(self, query: str, documents: List[Dict]) -> List[Dict]:
        """Filter documents based on relevance assessment"""
        relevant_docs = []
        
        for doc in documents:
            relevance_score = self._assess_relevance(query, doc['content'])
            
            if relevance_score > self.config.relevance_threshold:
                doc['relevance_score'] = relevance_score
                relevant_docs.append(doc)
        
        # Sort by relevance
        relevant_docs.sort(key=lambda x: x['relevance_score'], reverse=True)
        
        return relevant_docs[:5]  # Top 5 most relevant
    
    def _assess_relevance(self, query: str, document: str) -> float:
        """Assess relevance of document to query"""
        
        prompt = f"""
Query: "{query}"
Document: "{document[:500]}..."

Assess the relevance of this document to the query.

Assessment: {ReflectionToken.RELEVANT.value} or {ReflectionToken.IRRELEVANT.value}
"""
        
        assessment = self._generate_text(prompt, max_length=30)
        
        if ReflectionToken.RELEVANT.value in assessment:
            return 0.8
        elif ReflectionToken.IRRELEVANT.value in assessment:
            return 0.2
        else:
            # Fallback to simple similarity
            return self._simple_similarity(query, document)
    
    def _simple_similarity(self, query: str, document: str) -> float:
        """Simple token-based similarity score"""
        query_tokens = set(query.lower().split())
        doc_tokens = set(document.lower().split())
        
        if not query_tokens or not doc_tokens:
            return 0.0
        
        intersection = query_tokens.intersection(doc_tokens)
        union = query_tokens.union(doc_tokens)
        
        return len(intersection) / len(union)
    
    def _generate_response(
        self, 
        query: str, 
        documents: List[Dict], 
        previous_attempt: str = None
    ) -> str:
        """Generate response using query and retrieved documents"""
        
        # Build context from documents
        context = ""
        if documents:
            context = "Retrieved information:\n"
            for i, doc in enumerate(documents[:3]):  # Use top 3 documents
                context += f"{i+1}. {doc['content'][:300]}...\n\n"
        
        # Build prompt
        if previous_attempt:
            prompt = f"""
{context}
Query: "{query}"

Previous attempt: "{previous_attempt}"

The previous response needs improvement. Generate a better response using the available information.

Improved response:"""
        else:
            prompt = f"""
{context}
Query: "{query}"

Based on the retrieved information above, provide a comprehensive and accurate answer to the query.

Response:"""
        
        return self._generate_text(prompt, max_length=256)
    
    def _critique_response(
        self, 
        query: str, 
        documents: List[Dict], 
        response: str
    ) -> Dict[str, Any]:
        """Critique the generated response"""
        
        # Assess support level
        support_score = self._assess_support(response, documents)
        
        # Assess utility
        utility_score = self._assess_utility(query, response)
        
        # Determine if improvement is needed
        needs_improvement = (
            support_score < self.config.support_threshold or
            utility_score < self.config.utility_threshold
        )
        
        # Determine action
        action = 'none'
        if needs_improvement:
            if support_score < 0.5 and documents:
                action = 'retrieve_more'
            else:
                action = 'regenerate'
        
        return {
            'needs_improvement': needs_improvement,
            'support_score': support_score,
            'utility_score': utility_score,
            'action': action,
            'reasoning': self._explain_critique(support_score, utility_score)
        }
    
    def _assess_support(self, response: str, documents: List[Dict]) -> float:
        """Assess how well the response is supported by retrieved documents"""
        
        if not documents:
            return 0.3  # Low support without evidence
        
        prompt = f"""
Response: "{response}"

Supporting documents:
{chr(10).join([f"- {doc['content'][:200]}..." for doc in documents[:3]])}

Assess how well the response is supported by the provided documents.

Support level: {ReflectionToken.SUPPORTED.value}, {ReflectionToken.PARTIALLY_SUPPORTED.value}, or {ReflectionToken.NOT_SUPPORTED.value}
"""
        
        assessment = self._generate_text(prompt, max_length=50)
        
        if ReflectionToken.SUPPORTED.value in assessment:
            return 0.9
        elif ReflectionToken.PARTIALLY_SUPPORTED.value in assessment:
            return 0.6
        elif ReflectionToken.NOT_SUPPORTED.value in assessment:
            return 0.2
        else:
            return 0.5  # Default moderate support
    
    def _assess_utility(self, query: str, response: str) -> float:
        """Assess utility/helpfulness of response"""
        
        prompt = f"""
Query: "{query}"
Response: "{response}"

Rate how useful and helpful this response is for answering the query.

Utility: {ReflectionToken.USEFUL.value} or {ReflectionToken.NOT_USEFUL.value}
"""
        
        assessment = self._generate_text(prompt, max_length=30)
        
        if ReflectionToken.USEFUL.value in assessment:
            return 0.8
        elif ReflectionToken.NOT_USEFUL.value in assessment:
            return 0.2
        else:
            return 0.5  # Default moderate utility
    
    def _explain_critique(self, support_score: float, utility_score: float) -> str:
        """Generate explanation for critique scores"""
        
        explanations = []
        
        if support_score < 0.5:
            explanations.append("Response lacks sufficient evidence support")
        if utility_score < 0.5:
            explanations.append("Response is not sufficiently helpful")
        if support_score > 0.7 and utility_score > 0.7:
            explanations.append("Response is well-supported and helpful")
        
        return "; ".join(explanations) if explanations else "Average response quality"
    
    def _retrieve_additional_documents(
        self, 
        query: str, 
        response: str, 
        existing_docs: List[Dict]
    ) -> List[Dict]:
        """Retrieve additional documents to improve response"""
        
        # Extract gaps or questions from current response
        expanded_query = f"{query} {self._extract_information_gaps(response)}"
        
        # Retrieve with expanded query
        new_docs = self._retrieve_documents(expanded_query)
        
        # Filter out duplicates
        existing_ids = {doc['id'] for doc in existing_docs}
        additional_docs = [doc for doc in new_docs if doc['id'] not in existing_ids]
        
        return additional_docs[:3]  # Limit additional documents
    
    def _extract_information_gaps(self, response: str) -> str:
        """Extract potential information gaps from response (simplified)"""
        
        gap_indicators = [
            "more information needed", "unclear", "uncertain", 
            "requires further", "additional details"
        ]
        
        response_lower = response.lower()
        if any(indicator in response_lower for indicator in gap_indicators):
            return "additional context details specific information"
        
        return ""
    
    def _build_simple_classifier(self):
        """Placeholder for building specialized classifiers"""
        # In practice, you would train specific models for each reflection task
        return lambda x: 0.5  # Dummy classifier returning neutral score
    
    def _generate_text(self, prompt: str, max_length: int = 512) -> str:
        """Generate text using the language model"""
        try:
            inputs = self.tokenizer.encode(prompt, return_tensors="pt", truncation=True, max_length=1024)
            
            with torch.no_grad():
                outputs = self.model.generate(
                    inputs,
                    max_length=inputs.shape[1] + max_length,
                    num_return_sequences=1,
                    temperature=0.7,
                    pad_token_id=self.tokenizer.eos_token_id,
                    do_sample=True
                )
            
            generated_text = self.tokenizer.decode(outputs[0], skip_special_tokens=True)
            # Extract only the generated part
            response = generated_text[len(prompt):].strip()
            
            return response
            
        except Exception as e:
            print(f"Text generation failed: {e}")
            return "I apologize, but I'm unable to generate a response at this time."

class ContentValidator(ABC):
    """Abstract base class for content validators"""
    
    @abstractmethod
    def validate(self, content: str, query: str, context: Dict) -> ValidationResult:
        pass

class RelevanceValidator(ContentValidator):
    """Validates relevance of retrieved content"""
    
    def __init__(self, threshold: float = 0.7):
        self.threshold = threshold
    
    def validate(self, content: str, query: str, context: Dict) -> ValidationResult:
        """Validate relevance of content to query"""
        
        # Simplified relevance scoring (in practice, use trained models)
        relevance_score = self._compute_relevance_score(content, query)
        
        is_valid = relevance_score >= self.threshold
        issues = [] if is_valid else [f"Low relevance score: {relevance_score:.3f}"]
        
        correction_action = CorrectionAction.ACCEPT if is_valid else CorrectionAction.RETRIEVE_MORE
        
        return ValidationResult(
            is_valid=is_valid,
            confidence_score=relevance_score,
            issues=issues,
            correction_action=correction_action,
            metadata={"validator": "relevance", "threshold": self.threshold}
        )
    
    def _compute_relevance_score(self, content: str, query: str) -> float:
        """Compute relevance score between content and query"""
        # Simplified implementation - use proper semantic similarity in production
        query_tokens = set(query.lower().split())
        content_tokens = set(content.lower().split())
        
        if not query_tokens:
            return 0.0
        
        overlap = query_tokens.intersection(content_tokens)
        return len(overlap) / len(query_tokens)

class FactualityValidator(ContentValidator):
    """Validates factual accuracy of content"""
    
    def __init__(self, fact_check_api_key: str = None):
        self.fact_check_api_key = fact_check_api_key
        self.trusted_sources = [
            "wikipedia.org", "britannica.com", ".edu", ".gov",
            "nature.com", "science.org"
        ]
    
    def validate(self, content: str, query: str, context: Dict) -> ValidationResult:
        """Validate factual accuracy of content"""
        
        # Extract factual claims
        claims = self._extract_claims(content)
        
        if not claims:
            return ValidationResult(
                is_valid=True,
                confidence_score=0.8,
                metadata={"validator": "factuality", "claims_found": 0}
            )
        
        # Validate each claim
        validation_results = []
        for claim in claims:
            claim_result = self._validate_claim(claim, context)
            validation_results.append(claim_result)
        
        # Aggregate results
        avg_confidence = sum(r['confidence'] for r in validation_results) / len(validation_results)
        failed_validations = [r for r in validation_results if not r['valid']]
        
        is_valid = len(failed_validations) == 0
        issues = [f"Failed validation: {r['claim']}" for r in failed_validations]
        
        correction_action = CorrectionAction.ACCEPT
        if not is_valid:
            if avg_confidence < 0.3:
                correction_action = CorrectionAction.WEB_SEARCH
            else:
                correction_action = CorrectionAction.COMBINE_SOURCES
        
        return ValidationResult(
            is_valid=is_valid,
            confidence_score=avg_confidence,
            issues=issues,
            correction_action=correction_action,
            metadata={
                "validator": "factuality", 
                "claims_validated": len(claims),
                "failed_validations": len(failed_validations)
            }
        )
    
    def _extract_claims(self, content: str) -> List[str]:
        """Extract factual claims from content"""
        # Simplified claim extraction - use NLP tools in production
        sentences = content.split('.')
        
        # Filter for sentences that look like factual claims
        claims = []
        factual_indicators = ["is", "was", "are", "were", "has", "have", "according to"]
        
        for sentence in sentences:
            sentence = sentence.strip()
            if (len(sentence) > 10 and 
                any(indicator in sentence.lower() for indicator in factual_indicators)):
                claims.append(sentence)
        
        return claims[:5]  # Limit to first 5 claims
    
    def _validate_claim(self, claim: str, context: Dict) -> Dict[str, Any]:
        """Validate a single factual claim"""
        
        # Check source credibility
        source_credibility = self._assess_source_credibility(context)
        
        # Check for contradictory information in context
        contradiction_score = self._check_contradictions(claim, context)
        
        # Simple validation logic
        confidence = (source_credibility + (1 - contradiction_score)) / 2
        valid = confidence > 0.5
        
        return {
            'claim': claim,
            'valid': valid,
            'confidence': confidence,
            'source_credibility': source_credibility,
            'contradiction_score': contradiction_score
        }
    
    def _assess_source_credibility(self, context: Dict) -> float:
        """Assess credibility of sources in context"""
        sources = context.get('sources', [])
        
        if not sources:
            return 0.5  # Neutral when no source info
        
        credibility_scores = []
        for source in sources:
            source_url = source.get('url', '')
            
            # Check against trusted domains
            credibility = 0.3  # Default
            for trusted in self.trusted_sources:
                if trusted in source_url.lower():
                    credibility = 0.8
                    break
            
            credibility_scores.append(credibility)
        
        return sum(credibility_scores) / len(credibility_scores)
    
    def _check_contradictions(self, claim: str, context: Dict) -> float:
        """Check for contradictory information"""
        # Simplified contradiction detection
        # In practice, use more sophisticated NLI models
        
        other_docs = context.get('retrieved_documents', [])
        contradiction_signals = 0
        total_checks = 0
        
        negation_words = ["not", "no", "never", "false", "incorrect", "wrong"]
        
        for doc in other_docs:
            doc_content = doc.get('content', '').lower()
            claim_lower = claim.lower()
            
            # Simple check for negation patterns
            for word in claim.split():
                if len(word) > 3:
                    for neg_word in negation_words:
                        if f"{neg_word} {word}" in doc_content:
                            contradiction_signals += 1
                    total_checks += 1
        
        if total_checks == 0:
            return 0.0  # No contradictions found
        
        return contradiction_signals / total_checks

class CompletenessValidator(ContentValidator):
    """Validates completeness of response"""
    
    def __init__(self, min_length: int = 50, required_aspects: List[str] = None):
        self.min_length = min_length
        self.required_aspects = required_aspects or []
    
    def validate(self, content: str, query: str, context: Dict) -> ValidationResult:
        """Validate completeness of response"""
        
        issues = []
        
        # Length check
        if len(content.strip()) < self.min_length:
            issues.append(f"Response too short: {len(content)} < {self.min_length}")
        
        # Aspect coverage check
        missing_aspects = []
        for aspect in self.required_aspects:
            if aspect.lower() not in content.lower():
                missing_aspects.append(aspect)
        
        if missing_aspects:
            issues.append(f"Missing aspects: {', '.join(missing_aspects)}")
        
        # Query component coverage
        query_components = self._extract_query_components(query)
        uncovered_components = []

                correction_action = CorrectionAction.REGENERATE
        
        return ValidationResult(
            is_valid=is_valid,
            confidence_score=coverage_ratio,
            issues=issues,
            correction_action=correction_action,
            metadata={
                "validator": "completeness",
                "coverage_ratio": coverage_ratio,
                "uncovered_components": uncovered_components

            # Execute corrections
            correction_step = self._execute_corrections(
                query, current_docs, current_response, correction_actions

            correction_history.append(correction_step)
            
            # Update current state

                current_response = correction_step.output_data.get('response', current_response)
            else:

                break
        
        return {
            'final_response': current_response,
            'final_documents': current_docs,
            'correction_history': correction_history,
            'validation_results': validation_results,
            'total_time': time.time() - start_time,

    def _analyze_query(self, query: str) -> Dict:

        # Simple analysis - use NLP models in production
        query_lower = query.lower()
        
        # Determine query type
        question_words = ["what", "how", "when", "where", "why", "who"]
        is_question = any(word in query_lower for word in question_words)

        complexity_indicators = ["compare", "analyze", "explain", "relationship", "difference"]

        query_type = "factual" if is_question else "analytical"
        if complexity > 0.5:
            query_type = "complex_analytical"
        
        return {
            "type": query_type,
            "complexity": complexity,
            "is_question": is_question,
            "indicators_found": complexity_score
        }

    def _run(self, documents: List[Dict], query: str) -> Dict:
        """Analyze documents and extract insights"""

            "confidence_scores": [],
            "coverage_analysis": {},
            "recommendation": "sufficient"  # sufficient, need_more, need_different

        for i, doc in enumerate(documents):

            analysis_results["relevant_documents"], query
        )

    def _analyze_single_document(self, document: Dict, query: str) -> Dict:
        """Analyze a single document"""
        
        content = document.get("content", "")
        
        # Simple relevance scoring
        relevance_score = self._calculate_relevance(content, query)
        
        # Extract key points
        key_points = self._extract_key_points(content)
        
        return {
            "relevance_score": relevance_score,
            "key_points": key_points,
            "length": len(content),
            "quality_score": min(relevance_score + 0.2, 1.0)  # Simple quality metric
        }
    
    def _calculate_relevance(self, content: str, query: str) -> float:
        """Calculate relevance score"""
        query_tokens = set(query.lower().split())
        content_tokens = set(content.lower().split())
        
        if not query_tokens:
            return 0.0
        
        intersection = query_tokens.intersection(content_tokens)
        return len(intersection) / len(query_tokens)

        """Extract key points from content"""
        sentences = content.split('.')
        
        # Simple key point extraction - use summarization models in production
        key_points = []
        for sentence in sentences[:5]:
            sentence = sentence.strip()
            if len(sentence) > 20 and sentence:
                key_points.append(sentence)
        
        return key_points[:3]

        # Simple insight extraction
        if all_points:

    def _build_agent_graph(self) -> StateGraph:
        """Build the agent workflow graph"""
        
        workflow = StateGraph(AgentState)
        
        # Add nodes
        workflow.add_node("planner", self._planning_node)
        workflow.add_node("retriever", self._retrieval_node)
        workflow.add_node("analyzer", self._analysis_node)
        workflow.add_node("synthesizer", self._synthesis_node)
        workflow.add_node("validator", self._validation_node)
        
        # Add edges
        workflow.add_edge("planner", "retriever")
        workflow.add_edge("retriever", "analyzer")
        workflow.add_conditional_edges(
            "analyzer",
            self._should_continue_analysis,
            {
                "continue": "retriever",
                "synthesize": "synthesizer"
            }
        )
        workflow.add_edge("synthesizer", "validator")
        workflow.add_conditional_edges(
            "validator",
            self._should_finish,
            {
                "finish": END,
                "improve": "planner"
            }
        )
        
        # Set entry point
        workflow.set_entry_point("planner")

        for search_query in state["search_queries"]:

                results = self.retriever.search(search_query, top_k=5)
                
                # Convert results to standard format
                docs = [
                    {
                        "content": result.content if hasattr(result, "content") else result["content"],
                        "score": getattr(result, "fusion_score", result.get("score", 1.0)),
                        "query": search_query,
                        "id": getattr(result, "doc_id", result.get("id", f"doc_{len(all_documents)}"))
                    }
                    for result in results
                ]
                
                all_documents.extend(docs)
                
            except Exception as e:
                state["messages"].append(
                    AIMessage(content=f"Retrieval failed for query '{search_query}': {str(e)}")
                )
        
        # Remove duplicates
        unique_docs = []
        seen_ids = set()
        for doc in all_documents:
            if doc["id"] not in seen_ids:
                unique_docs.append(doc)
                seen_ids.add(doc["id"])
        
        state["retrieved_documents"] = unique_docs

            AIMessage(content=f"Retrieved {len(unique_docs)} unique documents")

class TextEncoder(ModalityEncoder):

class AudioEncoder(ModalityEncoder):
    """Audio content encoder"""
    
    def __init__(self, model_name: str = "facebook/wav2vec2-base-960h"):
        self.processor = Wav2Vec2Processor.from_pretrained(model_name)
        self.model = Wav2Vec2ForCTC.from_pretrained(model_name)
    
    def encode(self, content: Union[bytes, np.ndarray]) -> np.ndarray:
        """Encode audio content to embeddings"""
        
        # Convert audio to appropriate format
        if isinstance(content, bytes):
            # Convert bytes to numpy array (simplified)
            audio_array = np.frombuffer(content, dtype=np.float32)
        else:
            audio_array = content
        
        # Process audio
        inputs = self.processor(
            audio_array, 
            sampling_rate=16000, 
            return_tensors="pt", 
            padding=True
        )
        
        with torch.no_grad():
            features = self.model.wav2vec2.feature_extractor(inputs.input_values)
            features = self.model.wav2vec2.feature_projection(features.transpose(1, 2))
        
        # Return mean pooled features
        return features.mean(dim=1).numpy().flatten()
    
    def transcribe(self, audio: Union[bytes, np.ndarray]) -> str:
        """Transcribe audio to text"""
        
        if isinstance(audio, bytes):
            audio_array = np.frombuffer(audio, dtype=np.float32)
        else:
            audio_array = audio
        
        inputs = self.processor(
            audio_array, 
            sampling_rate=16000, 
            return_tensors="pt"
        )
        
        with torch.no_grad():
            logits = self.model(inputs.input_values).logits
        
        predicted_ids = torch.argmax(logits, dim=-1)
        transcription = self.processor.decode(predicted_ids[0])
        
        return transcription
    
    def can_process(self, content_type: str) -> bool:
        return content_type.startswith("audio/")

class VideoEncoder(ModalityEncoder):
    """Video content encoder (frame-based)"""
    
    def __init__(self):
        self.image_encoder = ImageEncoder()
        self.frame_sample_rate = 1  # Extract 1 frame per second
    
    def encode(self, content: bytes) -> np.ndarray:
        """Encode video by sampling frames"""
        
        # Extract frames from video
        frames = self._extract_frames(content)
        
        if not frames:
            return np.zeros(512)  # Default embedding size
        
        # Encode each frame
        frame_embeddings = []
        for frame in frames:
            frame_emb = self.image_encoder.encode(frame)
            frame_embeddings.append(frame_emb)
        
        # Aggregate frame embeddings (mean pooling)
        video_embedding = np.mean(frame_embeddings, axis=0)
        
        return video_embedding
    
    def _extract_frames(self, video_bytes: bytes, max_frames: int = 10) -> List[np.ndarray]:
        """Extract frames from video bytes"""
        
        try:
            # Save bytes to temporary file and process with OpenCV
            import tempfile
            import os
            
            with tempfile.NamedTemporaryFile(suffix=".mp4", delete=False) as tmp_file:
                tmp_file.write(video_bytes)
                tmp_path = tmp_file.name
            
            # Extract frames
            cap = cv2.VideoCapture(tmp_path)
            frames = []
            frame_count = 0
            
            while cap.isOpened() and len(frames) < max_frames:
                ret, frame = cap.read()
                if not ret:
                    break
                
                # Sample frames at specified rate
                if frame_count % (int(cap.get(cv2.CAP_PROP_FPS)) * self.frame_sample_rate) == 0:
                    frames.append(cv2.cvtColor(frame, cv2.COLOR_BGR2RGB))
                
                frame_count += 1
            
            cap.release()
            os.unlink(tmp_path)  # Clean up
            
            return frames
            
        except Exception as e:
            print(f"Video processing failed: {e}")
            return []
    
    def can_process(self, content_type: str) -> bool:
        return content_type.startswith("video/")

class MultiModalRetriever:
    """Multi-modal document retrieval system"""
    
    def __init__(
        self,
        text_encoder: TextEncoder = None,
        image_encoder: ImageEncoder = None,
        audio_encoder: AudioEncoder = None,
        video_encoder: VideoEncoder = None
    ):
        self.encoders = {
            "text": text_encoder or TextEncoder(),
            "image": image_encoder or ImageEncoder(), 
            "audio": audio_encoder or AudioEncoder(),
            "video": video_encoder or VideoEncoder()
        }
        
        self.document_store: List[MultiModalDocument] = []
        self.modality_indices: Dict[str, Any] = {}
        
    def index_documents(self, documents: List[MultiModalDocument]):
        """Index multi-modal documents"""
        
        print(f"Indexing {len(documents)} multi-modal documents...")
        
        for doc in documents:
            # Encode each modality present in the document
            if doc.text_content:
                doc.modality_embeddings["text"] = self.encoders["text"].encode(doc.text_content)
            
            if doc.image_content:
                doc.modality_embeddings["image"] = self.encoders["image"].encode(doc.image_content)
                # Also generate caption for searchable text
                caption = self.encoders["image"].generate_caption(doc.image_content)
                doc.metadata["image_caption"] = caption
            
            if doc.audio_content:
                doc.modality_embeddings["audio"] = self.encoders["audio"].encode(doc.audio_content)
                # Also transcribe for searchable text
                transcription = self.encoders["audio"].transcribe(doc.audio_content)
                doc.metadata["audio_transcription"] = transcription
            
            if doc.video_content:
                doc.modality_embeddings["video"] = self.encoders["video"].encode(doc.video_content)
        
        self.document_store.extend(documents)
        print(f"Indexed {len(documents)} documents successfully")
    
    def search_multimodal(
        self, 
        query: str,
        query_image: Optional[bytes] = None,
        query_audio: Optional[bytes] = None,
        modality_weights: Dict[str, float] = None,
        top_k: int = 10
    ) -> List[Dict[str, Any]]:
        """Search across multiple modalities"""
        
        if modality_weights is None:
            modality_weights = {"text": 0.4, "image": 0.3, "audio": 0.2, "video": 0.1}
        
        # Encode query in relevant modalities
        query_embeddings = {}
        
        if query:
            query_embeddings["text"] = self.encoders["text"].encode(query)
        
        if query_image:
            query_embeddings["image"] = self.encoders["image"].encode(query_image)
        
        if query_audio:
            query_embeddings["audio"] = self.encoders["audio"].encode(query_audio)
        
        # Score documents
        document_scores = []
        
        for doc in self.document_store:
            total_score = 0.0
            total_weight = 0.0
            
            # Score each modality
            for modality, query_emb in query_embeddings.items():
                if modality in doc.modality_embeddings:
                    doc_emb = doc.modality_embeddings[modality]
                    similarity = self._cosine_similarity(query_emb, doc_emb)
                    weight = modality_weights.get(modality, 0.0)
                    
                    total_score += similarity * weight
                    total_weight += weight
            
            # Also check text-based matches for captions/transcriptions
            if query:
                caption_score = 0.0
                if "image_caption" in doc.metadata:
                    caption_similarity = self._text_similarity(query, doc.metadata["image_caption"])
                    caption_score += caption_similarity * 0.1
                
                if "audio_transcription" in doc.metadata:
                    transcript_similarity = self._text_similarity(query, doc.metadata["audio_transcription"])
                    caption_score += transcript_similarity * 0.1
                
                total_score += caption_score
                total_weight += 0.1
            
            # Normalize score
            if total_weight > 0:
                final_score = total_score / total_weight
            else:
                final_score = 0.0
            
            document_scores.append({
                "document": doc,
                "score": final_score,
                "modality_breakdown": self._get_modality_breakdown(doc, query_embeddings, modality_weights)
            })
        
        # Sort by score and return top-k
        document_scores.sort(key=lambda x: x["score"], reverse=True)
        
        return document_scores[:top_k]
    
    def _cosine_similarity(self, emb1: np.ndarray, emb2: np.ndarray) -> float:
        """Calculate cosine similarity between embeddings"""
        
        dot_product = np.dot(emb1, emb2)
        norm1 = np.linalg.norm(emb1)
        norm2 = np.linalg.norm(emb2)
        
        if norm1 == 0 or norm2 == 0:
            return 0.0
        
        return dot_product / (norm1 * norm2)
    
    def _text_similarity(self, text1: str, text2: str) -> float:
        """Simple text similarity"""
        
        tokens1 = set(text1.lower().split())
        tokens2 = set(text2.lower().split())
        
        if not tokens1 or not tokens2:
            return 0.0
        
        intersection = tokens1.intersection(tokens2)
        union = tokens1.union(tokens2)
        
        return len(intersection) / len(union)
    
    def _get_modality_breakdown(
        self, 
        document: MultiModalDocument, 
        query_embeddings: Dict[str, np.ndarray],
        modality_weights: Dict[str, float]
    ) -> Dict[str, float]:
        """Get detailed modality-wise scores"""
        
        breakdown = {}
        
        for modality, query_emb in query_embeddings.items():
            if modality in document.modality_embeddings:
                doc_emb = document.modality_embeddings[modality]
                similarity = self._cosine_similarity(query_emb, doc_emb)
                breakdown[modality] = similarity
            else:
                breakdown[modality] = 0.0
        
        return breakdown

class ResultFusion:
    """Fuse results from multiple retrieval runs"""
    
    def __init__(self, fusion_method: str = "rrf", k: int = 10):
        self.fusion_method = fusion_method
        self.k = k  # Parameter for RRF
        self.semantic_model = SentenceTransformer('all-MiniLM-L6-v2')
        
    def reciprocal_rank_fusion(
        self, 
        search_results_list: List[List[SearchResult]], 
        queries: List[str]
    ) -> List[FusedResult]:
        """Apply Reciprocal Rank Fusion to combine results"""
        
        # Collect all unique results
        all_results = {}
        
        for query_idx, results in enumerate(search_results_list):
            for rank, result in enumerate(results):
                if result.content not in all_results:
                    all_results[result.content] = {
                        'result': result,
                        'scores': [],
                        'ranks': [],
                        'queries': []
                    }
                
                # RRF score: 1 / (k + rank)
                rrf_score = 1.0 / (self.k + rank + 1)
                all_results[result.content]['scores'].append(rrf_score)
                all_results[result.content]['ranks'].append(rank + 1)
                all_results[result.content]['queries'].append(queries[query_idx])
        
        # Calculate fused scores
        fused_results = []
        for content, data in all_results.items():
            fused_score = sum(data['scores'])
            
            fused_result = FusedResult(
                content=content,
                fused_score=fused_score,
                individual_scores=data['scores'],
                sources=data['queries'],
                original_ranks=data['ranks'],
                metadata=data['result'].metadata
            )
            fused_results.append(fused_result)
        
        # Sort by fused score
        fused_results.sort(key=lambda x: x.fused_score, reverse=True)
        
        print(f"Fused {len(fused_results)} unique results from {len(search_results_list)} queries")
        return fused_results
    
    def semantic_deduplication(
        self, 
        results: List[FusedResult], 
        similarity_threshold: float = 0.85
    ) -> List[FusedResult]:
        """Remove semantically similar results"""
        
        if not results:
            return results
            
        # Get embeddings for all results
        contents = [r.content for r in results]
        embeddings = self.semantic_model.encode(contents)
        
        # Calculate similarity matrix
        similarity_matrix = cosine_similarity(embeddings)
        
        # Keep track of which results to keep
        keep_indices = []
        removed_indices = set()
        
        for i, result in enumerate(results):
            if i in removed_indices:
                continue
                
            keep_indices.append(i)
            
            # Find similar results to remove
            for j in range(i + 1, len(results)):
                if j not in removed_indices and similarity_matrix[i][j] > similarity_threshold:
                    removed_indices.add(j)
                    # Merge scores from similar results
                    results[i].fused_score += results[j].fused_score * 0.5
        
        deduplicated = [results[i] for i in keep_indices]
        
        print(f"Deduplicated from {len(results)} to {len(deduplicated)} results "
              f"(threshold: {similarity_threshold})")
        
        return deduplicated

            results[f'improvement@{k}'] = avg_fusion_recall - avg_baseline_recall

HyDE: Hypothetical Document Embeddings

HyDE (Hypothetical Document Embeddings) generates hypothetical answers to queries before retrieval, then uses these generated documents to find semantically similar real documents. This approach bridges the gap between query language and document language.

HyDE Process

  • Generate hypothetical document
  • Embed hypothetical content
  • Search using hypothetical embeddings
  • Retrieve semantically similar documents
  • Generate final answer from retrieved docs
  • Optional: hybrid scoring

Advantages

• Language Bridge: Connects query style to document style
• Semantic Matching: Better handling of conceptual similarities
• Zero-shot: Works without training data
• Domain Adaptation: Adapts to specific domains naturally

class HypotheticalDocumentGenerator:
    """Generate hypothetical documents for queries"""
    
    def __init__(
        self, 
        model_name: str = "gpt-3.5-turbo",
        temperature: float = 0.7,
        max_tokens: int = 500
    ):
        self.model_name = model_name
        self.temperature = temperature
        self.max_tokens = max_tokens
        
    async def generate_hypothetical_document(
        self, 
        query: str, 
        domain_context: str = "",
        document_type: str = "informative"
    ) -> str:
        """Generate a hypothetical document that would answer the query"""
        
        # Customize prompt based on document type
        if document_type == "academic":
            style_instruction = "Write in an academic style with technical terminology and formal language."
        elif document_type == "technical":
            style_instruction = "Write in a technical documentation style with precise details and examples."
        elif document_type == "conversational":
            style_instruction = "Write in a conversational, easy-to-understand style."
        else:
            style_instruction = "Write in a clear, informative style suitable for general audiences."
        
        domain_prompt = f"\nContext: {domain_context}" if domain_context else ""
        
        prompt = f"""
You are an expert content generator. Your task is to write a hypothetical document that would 
perfectly answer the following query. The document should contain the type of information 
someone would be looking for when asking this question.

Query: "{query}"{domain_prompt}

Instructions:
1. {style_instruction}
2. Include specific details, examples, and explanations
3. Structure the content logically
4. Make it comprehensive but focused on the query
5. Write as if this document exists in a knowledge base

Generate the hypothetical document:
"""
        
        try:
            response = await openai.ChatCompletion.acreate(
                model=self.model_name,
                messages=[{"role": "user", "content": prompt}],
                temperature=self.temperature,
                max_tokens=self.max_tokens
            )
            
            hypothetical_doc = response.choices[0].message.content.strip()
            
            print(f"Generated hypothetical document ({len(hypothetical_doc)} chars)")
            print(f"Preview: {hypothetical_doc[:150]}...")
            
            return hypothetical_doc
            
        except Exception as e:
            print(f"Hypothetical document generation failed: {e}")
            # Fallback to query expansion
            return f"Document about {query}: This document provides comprehensive information about {query}, including detailed explanations, examples, and practical applications."

class HyDERetriever:
    """HyDE-based retrieval system"""
    
    def __init__(
        self,
        base_retriever,  # Your existing vector store retriever
        doc_generator: HypotheticalDocumentGenerator,
        embedding_model: str = "all-MiniLM-L6-v2",
        hybrid_weight: float = 0.7  # Weight for HyDE vs direct retrieval
    ):
        self.base_retriever = base_retriever
        self.doc_generator = doc_generator
        self.embedding_model = SentenceTransformer(embedding_model)
        self.hybrid_weight = hybrid_weight
        
    async def retrieve_with_hyde(
        self,
        query: str,
        top_k: int = 10,
        domain_context: str = "",
        document_type: str = "informative",
        use_hybrid: bool = True
    ) -> List[HyDEResult]:
        """Perform HyDE retrieval"""
        
        print(f"Starting HyDE retrieval for: '{query}'")
        
        # Step 1: Generate hypothetical document
        hypothetical_doc = await self.doc_generator.generate_hypothetical_document(
            query=query,
            domain_context=domain_context,
            document_type=document_type
        )
        
        # Step 2: Retrieve using hypothetical document
        hyde_results = await self._retrieve_with_hypothetical(
            hypothetical_doc, top_k * 2  # Get more results for hybrid scoring
        )
        
        # Step 3: Optional hybrid scoring
        if use_hybrid:
            direct_results = await self._retrieve_direct(query, top_k * 2)
            final_results = self._combine_hybrid_scores(
                query, hypothetical_doc, hyde_results, direct_results
            )
        else:
            final_results = hyde_results
        
        # Step 4: Return top-k results
        final_results = sorted(final_results, key=lambda x: x.hybrid_score, reverse=True)
        return final_results[:top_k]
    
    async def _retrieve_with_hypothetical(
        self, 
        hypothetical_doc: str, 
        limit: int
    ) -> List[HyDEResult]:
        """Retrieve documents using hypothetical document as query"""
        
        try:
            # Use the hypothetical document as the search query
            raw_results = await self.base_retriever.search(
                hypothetical_doc, 
                limit=limit
            )
            
            results = []
            for result in raw_results:
                hyde_result = HyDEResult(
                    content=result.get('content', ''),
                    score=result.get('score', 0.0),
                    source=result.get('source', ''),
                    hyde_score=result.get('score', 0.0),
                    direct_score=0.0,  # Will be calculated if hybrid mode is used
                    hybrid_score=result.get('score', 0.0),
                    metadata=result.get('metadata', {})
                )
                results.append(hyde_result)
            
            return results
            
        except Exception as e:
            print(f"Hypothetical document retrieval failed: {e}")
            return []
    
    async def _retrieve_direct(self, query: str, limit: int) -> List[Dict[str, Any]]:
        """Direct retrieval using original query"""
        
        try:
            return await self.base_retriever.search(query, limit=limit)
        except Exception as e:
            print(f"Direct retrieval failed: {e}")
            return []
    
    def _combine_hybrid_scores(
        self,
        original_query: str,
        hypothetical_doc: str,
        hyde_results: List[HyDEResult],
        direct_results: List[Dict[str, Any]]
    ) -> List[HyDEResult]:
        """Combine HyDE and direct retrieval scores"""
        
        # Create mapping of content to direct scores
        direct_scores = {}
        for result in direct_results:
            content = result.get('content', '')
            direct_scores[content] = result.get('score', 0.0)
        
        # Update HyDE results with hybrid scores
        hybrid_results = []
        for hyde_result in hyde_results:
            content = hyde_result.content
            
            # Get direct score (0 if not found in direct results)
            direct_score = direct_scores.get(content, 0.0)
            
            # Calculate hybrid score
            hybrid_score = (
                self.hybrid_weight * hyde_result.hyde_score +
                (1 - self.hybrid_weight) * direct_score
            )
            
            # Update result
            hyde_result.direct_score = direct_score
            hyde_result.hybrid_score = hybrid_score
            
            hybrid_results.append(hyde_result)
        
        # Add results that were only found in direct retrieval
        hyde_contents = {r.content for r in hyde_results}
        for result in direct_results:
            content = result.get('content', '')
            if content not in hyde_contents:
                hybrid_result = HyDEResult(
                    content=content,
                    score=result.get('score', 0.0),
                    source=result.get('source', ''),
                    hyde_score=0.0,
                    direct_score=result.get('score', 0.0),
                    hybrid_score=(1 - self.hybrid_weight) * result.get('score', 0.0),
                    metadata=result.get('metadata', {})
                )
                hybrid_results.append(hybrid_result)
        
        print(f"Combined {len(hyde_results)} HyDE + {len(direct_results)} direct = {len(hybrid_results)} hybrid results")
        
        return hybrid_results

class AdaptiveHyDE:
    """Adaptive HyDE that learns from feedback"""
    
    def __init__(self, base_hyde_retriever: HyDERetriever):
        self.base_retriever = base_hyde_retriever
        self.performance_history = {}
        self.domain_templates = {
            "technical": "This technical document explains {topic} with detailed implementation examples and best practices.",
            "academic": "This research paper investigates {topic} through systematic analysis and empirical evaluation.",
            "business": "This business analysis examines {topic} and its impact on organizational strategy and operations.",
            "medical": "This medical reference describes {topic} including symptoms, diagnosis, and treatment approaches."
        }
    
    async def adaptive_retrieve(
        self,
        query: str,
        top_k: int = 10,
        auto_detect_domain: bool = True
    ) -> List[HyDEResult]:
        """Perform adaptive HyDE retrieval with domain detection"""
        
        # Detect domain if requested
        if auto_detect_domain:
            domain = await self._detect_query_domain(query)
            document_type = domain
        else:
            domain = "general"
            document_type = "informative"
        
        print(f"Detected domain: {domain}")
        
        # Generate domain-specific hypothetical document
        if domain in self.domain_templates:
            template = self.domain_templates[domain]
            domain_context = template.format(topic=query)
        else:
            domain_context = ""
        
        # Perform HyDE retrieval
        results = await self.base_retriever.retrieve_with_hyde(
            query=query,
            top_k=top_k,
            domain_context=domain_context,
            document_type=document_type
        )
        
        # Store performance metrics
        self._update_performance_history(query, domain, len(results))
        
        return results
    
    async def _detect_query_domain(self, query: str) -> str:
        """Detect the domain of a query using LLM"""
        
        prompt = f"""
Analyze the following query and determine its primary domain category.

Query: "{query}"

Choose the most appropriate domain from:
- technical: Programming, software, engineering, IT
- academic: Research, scientific, theoretical concepts
- business: Strategy, management, finance, marketing
- medical: Health, medicine, treatments, symptoms
- legal: Law, regulations, compliance, contracts
- general: General knowledge, everyday topics

Return only the domain name (one word).
"""
        
        try:
            response = await openai.ChatCompletion.acreate(
                model="gpt-3.5-turbo",
                messages=[{"role": "user", "content": prompt}],
                temperature=0.1,
                max_tokens=10
            )
            
            domain = response.choices[0].message.content.strip().lower()
            return domain if domain in self.domain_templates else "general"
            
        except Exception as e:
            print(f"Domain detection failed: {e}")
            return "general"
    
    def _update_performance_history(self, query: str, domain: str, num_results: int):
        """Update performance tracking"""
        
        if domain not in self.performance_history:
            self.performance_history[domain] = {
                'queries': 0,
                'avg_results': 0.0,
                'success_rate': 0.0
            }
        
        history = self.performance_history[domain]
        history['queries'] += 1
        
        # Update rolling average
        alpha = 0.1  # Learning rate
        history['avg_results'] = (
            (1 - alpha) * history['avg_results'] + 
            alpha * num_results
        )
        
        # Success if we got results
        success = 1.0 if num_results > 0 else 0.0
        history['success_rate'] = (
            (1 - alpha) * history['success_rate'] + 
            alpha * success
        )

            raw_results = await self.base_retriever.search(doc, limit=limit)
            
            results = []
            for result in raw_results:
                hyde_result = HyDEResult(
                    content=result.get('content', ''),
                    score=result.get('score', 0.0),
                    source=result.get('source', ''),
                    hyde_score=result.get('score', 0.0),
                    direct_score=0.0,
                    hybrid_score=result.get('score', 0.0),
                    metadata=result.get('metadata', {})
                )
                results.append(hyde_result)
            
            return results
            
        except Exception as e:
            print(f"Document retrieval failed: {e}")
            return []
    
    def _deduplicate_and_rank(self, all_results: List[HyDEResult]) -> List[HyDEResult]:
        """Remove duplicates and rank by aggregated scores"""
        
        # Group by content
        content_groups = {}
        for result in all_results:
            if result.content not in content_groups:
                content_groups[result.content] = []
            content_groups[result.content].append(result)
        
        # Aggregate scores for each unique content
        deduplicated = []
        for content, results in content_groups.items():
            # Take the highest scoring result as base
            best_result = max(results, key=lambda x: x.hyde_score)
            
            # Aggregate scores from all perspectives
            total_score = sum(r.hyde_score for r in results)
            avg_score = total_score / len(results)
            
            # Update with aggregated score
            best_result.hybrid_score = avg_score
            best_result.metadata['perspectives_count'] = len(results)
            best_result.metadata['total_score'] = total_score
            
            deduplicated.append(best_result)
        
        # Sort by aggregated score
        deduplicated.sort(key=lambda x: x.hybrid_score, reverse=True)
        
        print(f"Deduplicated {len(all_results)} -> {len(deduplicated)} results")
        
        return deduplicated

class TermExpander:
    """Expand queries with synonyms and related terms"""
    
    def __init__(self):
        self.stop_words = set(stopwords.words('english'))
        try:
            # Load spaCy model for better linguistic processing
            self.nlp = spacy.load('en_core_web_sm')
        except OSError:
            print("spaCy model not found. Install with: python -m spacy download en_core_web_sm")
            self.nlp = None
    
    def expand_with_wordnet(self, query: str, max_synonyms: int = 3) -> QueryExpansion:
        """Expand query using WordNet synonyms"""
        
        tokens = word_tokenize(query.lower())
        tokens = [token for token in tokens if token.isalnum() and token not in self.stop_words]
        
        expansion_terms = []
        expansion_sources = []
        confidence_scores = []
        
        for token in tokens:
            synonyms = set()
            
            # Get synonyms from WordNet
            for syn in wordnet.synsets(token):
                for lemma in syn.lemmas():
                    synonym = lemma.name().replace('_', ' ')
                    if synonym != token and len(synonym) > 2:
                        synonyms.add(synonym)
            
            # Add top synonyms
            for synonym in list(synonyms)[:max_synonyms]:
                expansion_terms.append(synonym)
                expansion_sources.append('wordnet')
                confidence_scores.append(0.7)  # Base confidence for WordNet
        
        # Build expanded query
        expanded_query = query
        if expansion_terms:
            expanded_query += " " + " ".join(expansion_terms)
        
        return QueryExpansion(
            original_query=query,
            expanded_query=expanded_query,
            expansion_terms=expansion_terms,
            expansion_sources=expansion_sources,
            confidence_scores=confidence_scores
        )
    
    def expand_with_embeddings(
        self, 
        query: str, 
        corpus_terms: List[str],
        similarity_threshold: float = 0.7,
        max_expansions: int = 5
    ) -> QueryExpansion:
        """Expand query using semantic similarity with corpus terms"""
        
        if not hasattr(self, 'embedding_model'):
            self.embedding_model = SentenceTransformer('all-MiniLM-L6-v2')
        
        # Get query embedding
        query_embedding = self.embedding_model.encode([query])
        
        # Get embeddings for corpus terms
        term_embeddings = self.embedding_model.encode(corpus_terms)
        
        # Calculate similarities
        from sklearn.metrics.pairwise import cosine_similarity
        similarities = cosine_similarity(query_embedding, term_embeddings)[0]
        
        # Find similar terms
        expansion_terms = []
        expansion_sources = []
        confidence_scores = []
        
        for i, similarity in enumerate(similarities):
            if similarity > similarity_threshold:
                term = corpus_terms[i]
                if term.lower() not in query.lower():  # Don't add terms already in query
                    expansion_terms.append(term)
                    expansion_sources.append('embeddings')
                    confidence_scores.append(float(similarity))
        
        # Sort by similarity and take top expansions
        sorted_expansions = sorted(
            zip(expansion_terms, expansion_sources, confidence_scores),
            key=lambda x: x[2], reverse=True
        )[:max_expansions]
        
        if sorted_expansions:
            expansion_terms, expansion_sources, confidence_scores = zip(*sorted_expansions)
            expansion_terms = list(expansion_terms)
            expansion_sources = list(expansion_sources)
            confidence_scores = list(confidence_scores)
        else:
            expansion_terms, expansion_sources, confidence_scores = [], [], []
        
        # Build expanded query
        expanded_query = query
        if expansion_terms:
            expanded_query += " " + " ".join(expansion_terms)
        
        return QueryExpansion(
            original_query=query,
            expanded_query=expanded_query,
            expansion_terms=expansion_terms,
            expansion_sources=expansion_sources,
            confidence_scores=confidence_scores
        )

class IntentClassifier:
    """Classify query intent to guide rewriting strategy"""
    
    def __init__(self, model_name: str = "gpt-3.5-turbo"):
        self.model_name = model_name
        self.intent_patterns = {
            'definition': ['what is', 'define', 'meaning of', 'definition'],
            'howto': ['how to', 'how do', 'steps to', 'guide to', 'tutorial'],
            'comparison': ['vs', 'versus', 'compare', 'difference between', 'better than'],
            'factual': ['when', 'where', 'who', 'which', 'list of'],
            'explanation': ['why', 'explain', 'reason for', 'cause of'],
            'troubleshooting': ['error', 'problem', 'issue', 'fix', 'solve', 'troubleshoot']
        }
    
    def classify_intent(self, query: str) -> Tuple[str, float]:
        """Classify query intent using pattern matching and LLM"""
        
        query_lower = query.lower()
        
        # Pattern-based classification
        intent_scores = {}
        for intent, patterns in self.intent_patterns.items():
            score = sum(1 for pattern in patterns if pattern in query_lower)
            if score > 0:
                intent_scores[intent] = score
        
        if intent_scores:
            best_intent = max(intent_scores, key=intent_scores.get)
            confidence = min(intent_scores[best_intent] / len(self.intent_patterns[best_intent]), 1.0)
            return best_intent, confidence
        
        # Fallback to general intent
        return 'general', 0.5
    
    async def classify_with_llm(self, query: str) -> Tuple[str, float]:
        """Use LLM for more sophisticated intent classification"""
        
        prompt = f"""
Analyze the following query and classify its intent. Choose the most appropriate category:

Query: "{query}"

Intent categories:
- definition: Asking for definitions or meanings
- howto: Seeking instructions or procedures
- comparison: Comparing items or concepts
- factual: Requesting factual information
- explanation: Seeking explanations or reasoning
- troubleshooting: Solving problems or errors
- general: General information seeking

Return only the intent category name and confidence (0.0-1.0) in format: "intent_name confidence_score"
"""
        
        try:
            response = await openai.ChatCompletion.acreate(
                model=self.model_name,
                messages=[{"role": "user", "content": prompt}],
                temperature=0.1,
                max_tokens=20
            )
            
            result = response.choices[0].message.content.strip().split()
            intent = result[0] if result else 'general'
            confidence = float(result[1]) if len(result) > 1 else 0.5
            
            return intent, confidence
            
        except Exception as e:
            print(f"LLM intent classification failed: {e}")
            return self.classify_intent(query)

class ContextualRewriter:
    """Rewrite queries based on context and intent"""
    
    def __init__(self, model_name: str = "gpt-3.5-turbo"):
        self.model_name = model_name
        self.intent_classifier = IntentClassifier()
        
        # Rewriting templates for different intents
        self.rewrite_templates = {
            'definition': "What is the definition and meaning of {query}?",
            'howto': "Step-by-step guide on {query}",
            'comparison': "Compare and contrast {query}",
            'factual': "Provide factual information about {query}",
            'explanation': "Explain the reasoning and causes behind {query}",
            'troubleshooting': "Solutions and fixes for {query} problems"
        }
    
    async def rewrite_for_clarity(
        self, 
        query: str, 
        context: str = "",
        domain: str = ""
    ) -> RewrittenQuery:
        """Rewrite query for better clarity and specificity"""
        
        # Classify intent
        intent, confidence = await self.intent_classifier.classify_with_llm(query)
        
        # Build context for rewriting
        context_prompt = ""
        if context:
            context_prompt = f"Context: {context}\n"
        if domain:
            context_prompt += f"Domain: {domain}\n"
        
        prompt = f"""
Rewrite the following query to make it clearer and more specific for information retrieval.
{context_prompt}
Original Query: "{query}"
Query Intent: {intent}

Guidelines:
1. Make the query more specific and targeted
2. Add relevant keywords that would help find documents
3. Maintain the original intent
4. Use domain-appropriate terminology if applicable
5. Avoid making the query too long or complex

Return only the rewritten query:
"""
        
        try:
            response = await openai.ChatCompletion.acreate(
                model=self.model_name,
                messages=[{"role": "user", "content": prompt}],
                temperature=0.3,
                max_tokens=100
            )
            
            rewritten = response.choices[0].message.content.strip()
            
            # Clean up the response
            rewritten = rewritten.strip('"').strip()
            
            return RewrittenQuery(
                original_query=query,
                rewritten_query=rewritten,
                rewrite_type="clarity",
                confidence=confidence,
                metadata={
                    'intent': intent,
                    'context': context,
                    'domain': domain
                }
            )
            
        except Exception as e:
            print(f"Query rewriting failed: {e}")
            return RewrittenQuery(
                original_query=query,
                rewritten_query=query,  # Fallback to original
                rewrite_type="none",
                confidence=0.0
            )
    
    def rewrite_with_template(self, query: str, intent: str) -> RewrittenQuery:
        """Rewrite query using intent-specific templates"""
        
        if intent in self.rewrite_templates:
            template = self.rewrite_templates[intent]
            rewritten = template.format(query=query)
            
            return RewrittenQuery(
                original_query=query,
                rewritten_query=rewritten,
                rewrite_type="template",
                confidence=0.8,
                metadata={'intent': intent, 'template': template}
            )
        
        return RewrittenQuery(
            original_query=query,
            rewritten_query=query,
            rewrite_type="none",
            confidence=0.0
        )

class ChainOfThoughtRewriter:
    """Multi-step query rewriting using chain-of-thought"""
    
    def __init__(self, model_name: str = "gpt-3.5-turbo"):
        self.model_name = model_name
    
    async def rewrite_with_cot(
        self, 
        query: str, 
        max_steps: int = 3,
        domain_knowledge: str = ""
    ) -> List[RewrittenQuery]:
        """Perform multi-step query rewriting with reasoning"""
        
        rewritten_queries = []
        current_query = query
        
        for step in range(max_steps):
            prompt = f"""
You are helping to improve a search query through iterative refinement.

Current query: "{current_query}"
Step: {step + 1}/{max_steps}
{f"Domain context: {domain_knowledge}" if domain_knowledge else ""}

Think step by step:
1. What information is the user really seeking?
2. What specific terms would help find relevant documents?
3. How can we make this query more precise?

Provide your reasoning and then the improved query:

Reasoning: [Your step-by-step analysis]
Improved Query: [Your improved version]
"""
            
            try:
                response = await openai.ChatCompletion.acreate(
                    model=self.model_name,
                    messages=[{"role": "user", "content": prompt}],
                    temperature=0.3,
                    max_tokens=200
                )
                
                content = response.choices[0].message.content.strip()
                
                # Extract the improved query
                lines = content.split('\n')
                improved_query = current_query  # Default fallback
                reasoning = ""
                
                for line in lines:
                    if line.startswith('Reasoning:'):
                        reasoning = line.replace('Reasoning:', '').strip()
                    elif line.startswith('Improved Query:'):
                        improved_query = line.replace('Improved Query:', '').strip()
                        improved_query = improved_query.strip('"').strip()
                
                if improved_query != current_query:
                    rewritten_query = RewrittenQuery(
                        original_query=query,
                        rewritten_query=improved_query,
                        rewrite_type=f"cot_step_{step + 1}",
                        confidence=0.7,
                        metadata={
                            'step': step + 1,
                            'reasoning': reasoning,
                            'previous_query': current_query
                        }
                    )
                    rewritten_queries.append(rewritten_query)
                    current_query = improved_query
                else:
                    # No improvement, stop the chain
                    break
                    
            except Exception as e:
                print(f"CoT rewriting step {step + 1} failed: {e}")
                break
        
        return rewritten_queries

class DomainAdaptiveRewriter:
    """Domain-specific query rewriting"""
    
    def __init__(self):
        self.domain_vocabularies = {
            'medical': {
                'symptoms': ['pain', 'ache', 'discomfort', 'swelling'],
                'treatments': ['therapy', 'medication', 'procedure', 'intervention'],
                'conditions': ['disease', 'disorder', 'syndrome', 'condition']
            },
            'technical': {
                'programming': ['code', 'function', 'algorithm', 'implementation'],
                'systems': ['architecture', 'framework', 'platform', 'infrastructure'],
                'tools': ['software', 'library', 'package', 'utility']
            },
            'legal': {
                'documents': ['contract', 'agreement', 'statute', 'regulation'],
                'processes': ['litigation', 'compliance', 'filing', 'proceeding'],
                'entities': ['corporation', 'plaintiff', 'defendant', 'party']
            }
        }
    
    def adapt_to_domain(
        self, 
        query: str, 
        domain: str,
        corpus_terms: Optional[List[str]] = None
    ) -> RewrittenQuery:
        """Adapt query to domain-specific terminology"""
        
        if domain not in self.domain_vocabularies:
            return RewrittenQuery(
                original_query=query,
                rewritten_query=query,
                rewrite_type="none",
                confidence=0.0
            )
        
        domain_vocab = self.domain_vocabularies[domain]
        query_lower = query.lower()
        
        # Find applicable domain terms
        applicable_terms = []
        for category, terms in domain_vocab.items():
            for term in terms:
                if any(word in query_lower for word in term.split()):
                    applicable_terms.extend([t for t in terms if t != term])
        
        # Add domain-specific terms
        if applicable_terms:
            expanded_query = query + " " + " ".join(set(applicable_terms[:5]))
            
            return RewrittenQuery(
                original_query=query,
                rewritten_query=expanded_query,
                rewrite_type="domain_adaptation",
                confidence=0.8,
                expanded_terms=list(set(applicable_terms[:5])),
                metadata={'domain': domain, 'category_matches': len(applicable_terms)}
            )
        
        return RewrittenQuery(
            original_query=query,
            rewritten_query=query,
            rewrite_type="none",
            confidence=0.0
        )

class ComprehensiveQueryRewriter:
    """Main query rewriting system combining all techniques"""
    
    def __init__(
        self,
        model_name: str = "gpt-3.5-turbo",
        corpus_terms: Optional[List[str]] = None
    ):
        self.term_expander = TermExpander()
        self.contextual_rewriter = ContextualRewriter(model_name)
        self.cot_rewriter = ChainOfThoughtRewriter(model_name)
        self.domain_rewriter = DomainAdaptiveRewriter()
        self.corpus_terms = corpus_terms or []
    
    async def comprehensive_rewrite(
        self,
        query: str,
        strategies: List[str] = None,
        context: str = "",
        domain: str = "",
        max_variants: int = 5
    ) -> List[RewrittenQuery]:
        """Apply multiple rewriting strategies"""
        
        if strategies is None:
            strategies = ['clarity', 'expansion', 'domain', 'cot']
        
        rewritten_queries = []
        
        # Strategy 1: Clarity rewriting
        if 'clarity' in strategies:
            clarity_rewrite = await self.contextual_rewriter.rewrite_for_clarity(
                query, context, domain
            )
            rewritten_queries.append(clarity_rewrite)
        
        # Strategy 2: Term expansion
        if 'expansion' in strategies:
            if self.corpus_terms:
                expansion = self.term_expander.expand_with_embeddings(query, self.corpus_terms)
            else:
                expansion = self.term_expander.expand_with_wordnet(query)
            
            expansion_rewrite = RewrittenQuery(
                original_query=query,
                rewritten_query=expansion.expanded_query,
                rewrite_type="expansion",
                confidence=0.6,
                expanded_terms=expansion.expansion_terms,
                metadata={'sources': expansion.expansion_sources}
            )
            rewritten_queries.append(expansion_rewrite)
        
        # Strategy 3: Domain adaptation
        if 'domain' in strategies and domain:
            domain_rewrite = self.domain_rewriter.adapt_to_domain(query, domain, self.corpus_terms)
            if domain_rewrite.rewrite_type != "none":
                rewritten_queries.append(domain_rewrite)
        
        # Strategy 4: Chain-of-thought rewriting
        if 'cot' in strategies:
            cot_rewrites = await self.cot_rewriter.rewrite_with_cot(
                query, max_steps=2, domain_knowledge=f"{context} {domain}".strip()
            )
            rewritten_queries.extend(cot_rewrites)
        
        # Rank and return top variants
        rewritten_queries = sorted(rewritten_queries, key=lambda x: x.confidence, reverse=True)
        return rewritten_queries[:max_variants]
    
    def analyze_rewrite_effectiveness(
        self, 
        original_results: List[Dict], 
        rewritten_results: Dict[str, List[Dict]]
    ) -> Dict[str, float]:
        """Analyze effectiveness of different rewriting strategies"""
        
        baseline_score = len(original_results)
        
        effectiveness = {}
        for rewrite_type, results in rewritten_results.items():
            if baseline_score > 0:
                improvement = (len(results) - baseline_score) / baseline_score
            else:
                improvement = 1.0 if results else 0.0
            
            effectiveness[rewrite_type] = improvement
        
        return effectiveness

                result['source_query'] = query
            all_results.extend(original_results)
        except Exception as e:
            print(f"Original query retrieval failed: {e}")
        
        # Use rewritten queries if enabled
        if self.use_multiple_queries:
            for rewrite in rewrites:
                try:
                    rewrite_results = await self.base_retriever.search(
                        rewrite.rewritten_query, limit=top_k
                    )
                    for result in rewrite_results:
                        result['query_type'] = rewrite.rewrite_type
                        result['source_query'] = rewrite.rewritten_query
                        result['rewrite_confidence'] = rewrite.confidence
                    all_results.extend(rewrite_results)
                except Exception as e:
                    print(f"Rewritten query retrieval failed: {e}")
        
        # Deduplicate and rank results
        final_results = self._deduplicate_results(all_results)
        
        return final_results[:top_k]
    
    def _deduplicate_results(self, results: List[Dict]) -> List[Dict]:
        """Deduplicate results while preserving best scores"""
        
        seen_content = {}
        deduplicated = []
        
        for result in results:
            content = result.get('content', '')
            score = result.get('score', 0.0)
            
            if content not in seen_content or score > seen_content[content]['score']:
                seen_content[content] = result
        
        deduplicated = list(seen_content.values())
        deduplicated.sort(key=lambda x: x.get('score', 0.0), reverse=True)
        
        return deduplicated

class QuestionDecomposer:
    """Decompose complex questions into sub-questions"""
    
    def __init__(self, model_name: str = "gpt-3.5-turbo"):
        self.model_name = model_name
        
    async def decompose_question(
        self, 
        question: str, 
        max_subquestions: int = 5
    ) -> List[Dict[str, Any]]:
        """Decompose complex question into simpler sub-questions"""
        
        prompt = f"""
You are an expert at breaking down complex questions into simpler sub-questions that can be answered step by step.

Complex Question: "{question}"

Decompose this into {max_subquestions} or fewer sub-questions that:
1. Can be answered individually using document retrieval
2. Build upon each other logically
3. Together provide the information needed for the main question
4. Are ordered from basic facts to more complex reasoning

For each sub-question, also specify:
- The type of information needed (factual, definitional, comparative, etc.)
- Dependencies on other sub-questions (if any)

Return a JSON list where each item has:
{{
    "sub_question": "the sub-question text",
    "question_type": "factual|definitional|comparative|analytical",
    "depends_on": ["list of sub-question indices this depends on"],
    "priority": 1-5 (1=highest priority)
}}

JSON:
"""
        
        try:
            response = await openai.ChatCompletion.acreate(
                model=self.model_name,
                messages=[{"role": "user", "content": prompt}],
                temperature=0.3,
                max_tokens=500
            )
            
            content = response.choices[0].message.content.strip()
            
            # Extract JSON from response
            json_start = content.find('[')
            json_end = content.rfind(']') + 1
            
            if json_start != -1 and json_end != 0:
                json_content = content[json_start:json_end]
                sub_questions = json.loads(json_content)
                
                # Add step IDs
                for i, sq in enumerate(sub_questions):
                    sq['step_id'] = f"step_{i+1}"
                
                print(f"Decomposed question into {len(sub_questions)} sub-questions")
                return sub_questions
            
        except Exception as e:
            print(f"Question decomposition failed: {e}")
        
        # Fallback: create single sub-question
        return [{
            "step_id": "step_1",
            "sub_question": question,
            "question_type": "factual",
            "depends_on": [],
            "priority": 1
        }]
    
    def create_dependency_graph(self, sub_questions: List[Dict]) -> nx.DiGraph:
        """Create dependency graph for sub-questions"""
        
        graph = nx.DiGraph()
        
        # Add nodes
        for sq in sub_questions:
            graph.add_node(
                sq['step_id'], 
                question=sq['sub_question'],
                question_type=sq['question_type'],
                priority=sq['priority']
            )
        
        # Add edges for dependencies
        for sq in sub_questions:
            for dep_idx in sq['depends_on']:
                if isinstance(dep_idx, int) and 0 <= dep_idx < len(sub_questions):
                    dep_step_id = sub_questions[dep_idx]['step_id']
                    graph.add_edge(dep_step_id, sq['step_id'])
        
        return graph

class EvidenceRetriever:
    """Retrieve evidence for reasoning steps"""
    
    def __init__(self, base_retriever, embedding_model: str = "all-MiniLM-L6-v2"):
        self.base_retriever = base_retriever
        self.embedding_model = SentenceTransformer(embedding_model)
        self.evidence_cache = {}
    
    async def retrieve_evidence(
        self,
        question: str,
        previous_evidence: List[str] = None,
        context_docs: List[Dict] = None,
        top_k: int = 5
    ) -> List[Dict[str, Any]]:
        """Retrieve evidence for a specific question"""
        
        # Check cache first
        cache_key = f"{question}_{top_k}"
        if cache_key in self.evidence_cache:
            return self.evidence_cache[cache_key]
        
        try:
            # Build enhanced query with context
            enhanced_query = question
            
            if previous_evidence:
                # Add context from previous evidence
                context_terms = self._extract_key_terms(previous_evidence)
                enhanced_query += " " + " ".join(context_terms[:5])
            
            # Retrieve documents
            raw_results = await self.base_retriever.search(enhanced_query, limit=top_k * 2)
            
            # Filter and rank results
            filtered_results = self._filter_relevant_evidence(
                question, raw_results, context_docs
            )
            
            # Cache results
            self.evidence_cache[cache_key] = filtered_results[:top_k]
            
            return filtered_results[:top_k]
            
        except Exception as e:
            print(f"Evidence retrieval failed for '{question}': {e}")
            return []
    
    def _extract_key_terms(self, text_list: List[str]) -> List[str]:
        """Extract key terms from previous evidence"""
        
        import nltk
        from nltk.corpus import stopwords
        from nltk.tokenize import word_tokenize
        from collections import Counter
        
        try:
            stop_words = set(stopwords.words('english'))
        except LookupError:
            nltk.download('stopwords')
            stop_words = set(stopwords.words('english'))
        
        all_tokens = []
        for text in text_list:
            tokens = word_tokenize(text.lower())
            filtered_tokens = [
                token for token in tokens 
                if token.isalnum() and len(token) > 2 and token not in stop_words
            ]
            all_tokens.extend(filtered_tokens)
        
        # Get most common terms
        term_counts = Counter(all_tokens)
        key_terms = [term for term, count in term_counts.most_common(10)]
        
        return key_terms
    
    def _filter_relevant_evidence(
        self,
        question: str,
        raw_results: List[Dict],
        context_docs: List[Dict] = None
    ) -> List[Dict]:
        """Filter and rank evidence by relevance"""
        
        if not raw_results:
            return []
        
        # Get question embedding
        question_embedding = self.embedding_model.encode([question])
        
        # Score documents
        scored_results = []
        
        for result in raw_results:
            content = result.get('content', '')
            
            if content:
                # Calculate semantic similarity
                content_embedding = self.embedding_model.encode([content])
                from sklearn.metrics.pairwise import cosine_similarity
                similarity = cosine_similarity(question_embedding, content_embedding)[0][0]
                
                # Boost score if document contains key question terms
                question_terms = set(question.lower().split())
                content_terms = set(content.lower().split())
                term_overlap = len(question_terms.intersection(content_terms))
                term_boost = min(term_overlap * 0.1, 0.3)
                
                final_score = float(similarity) + term_boost
                
                result['relevance_score'] = final_score
                scored_results.append(result)
        
        # Sort by relevance score
        scored_results.sort(key=lambda x: x['relevance_score'], reverse=True)
        
        return scored_results

class ReasoningEngine:
    """Core reasoning engine for multi-hop questions"""
    
    def __init__(self, model_name: str = "gpt-4"):
        self.model_name = model_name
        
    async def reason_step(
        self,
        question: str,
        evidence_docs: List[Dict],
        previous_steps: List[ReasoningStep] = None,
        reasoning_type: str = "factual"
    ) -> ReasoningStep:
        """Perform reasoning for a single step"""
        
        # Build context from previous steps
        previous_context = ""
        if previous_steps:
            context_parts = []
            for step in previous_steps:
                context_parts.append(f"Q: {step.question}\nA: {step.answer}")
            previous_context = "\n\n".join(context_parts)
        
        # Build evidence context
        evidence_context = ""
        if evidence_docs:
            evidence_parts = []
            for i, doc in enumerate(evidence_docs[:5]):  # Limit to top 5
                content = doc.get('content', '')[:500]  # Truncate long content
                source = doc.get('source', f'Document {i+1}')
                evidence_parts.append(f"[{source}]: {content}")
            evidence_context = "\n\n".join(evidence_parts)
        
        # Create reasoning prompt based on type
        if reasoning_type == "comparative":
            reasoning_instruction = """
Compare and contrast the information from the evidence. Identify similarities, 
differences, and relationships between the concepts or entities mentioned.
"""
        elif reasoning_type == "analytical":
            reasoning_instruction = """
Analyze the evidence to identify patterns, causes, effects, or underlying principles.
Provide logical analysis based on the information provided.
"""
        elif reasoning_type == "definitional":
            reasoning_instruction = """
Provide a clear, comprehensive definition based on the evidence. Include key 
characteristics, properties, or features mentioned in the sources.
"""
        else:  # factual
            reasoning_instruction = """
Extract and synthesize the factual information from the evidence to answer 
the question directly and accurately.
"""
        
        prompt = f"""
You are an expert reasoning assistant. Answer the question using only the provided evidence.

Question: {question}

{f"Previous reasoning steps:\n{previous_context}\n" if previous_context else ""}

Evidence:
{evidence_context}

Instructions:
{reasoning_instruction}

Requirements:
1. Base your answer strictly on the provided evidence
2. If the evidence is insufficient, state what information is missing
3. Cite specific evidence sources when making claims
4. Provide a confidence level (0.0-1.0) for your answer
5. Identify the key evidence that supports your reasoning

Format your response as:
Answer: [Your answer]
Confidence: [0.0-1.0]
Key Evidence: [List of supporting evidence quotes]
Reasoning: [Your step-by-step reasoning process]
"""
        
        try:
            response = await openai.ChatCompletion.acreate(
                model=self.model_name,
                messages=[{"role": "user", "content": prompt}],
                temperature=0.2,
                max_tokens=600
            )
            
            content = response.choices[0].message.content.strip()
            
            # Parse response
            answer = self._extract_section(content, "Answer:")
            confidence_str = self._extract_section(content, "Confidence:")
            key_evidence = self._extract_section(content, "Key Evidence:")
            reasoning = self._extract_section(content, "Reasoning:")
            
            try:
                confidence = float(confidence_str.split()[0]) if confidence_str else 0.5
            except (ValueError, IndexError):
                confidence = 0.5
            
            # Extract evidence quotes
            evidence_quotes = []
            if key_evidence:
                # Simple extraction of quoted text
                import re
                quotes = re.findall(r'"([^"]*)"', key_evidence)
                evidence_quotes.extend(quotes)
            
            step_id = f"step_{len(previous_steps) + 1}" if previous_steps else "step_1"
            
            return ReasoningStep(
                step_id=step_id,
                question=question,
                retrieved_docs=evidence_docs,
                answer=answer or "Unable to determine from available evidence",
                confidence=confidence,
                evidence=evidence_quotes,
                reasoning_type=reasoning_type,
                metadata={
                    'raw_response': content,
                    'reasoning': reasoning,
                    'evidence_count': len(evidence_docs)
                }
            )
            
        except Exception as e:
            print(f"Reasoning step failed: {e}")
            
            # Return fallback reasoning step
            return ReasoningStep(
                step_id=f"step_{len(previous_steps) + 1}" if previous_steps else "step_1",
                question=question,
                retrieved_docs=evidence_docs,
                answer="Unable to process reasoning step due to error",
                confidence=0.0,
                reasoning_type=reasoning_type,
                metadata={'error': str(e)}
            )
    
    def _extract_section(self, content: str, section_name: str) -> str:
        """Extract a section from formatted response"""
        
        lines = content.split('\n')
        section_content = []
        in_section = False
        
        for line in lines:
            if line.startswith(section_name):
                in_section = True
                section_content.append(line.replace(section_name, '').strip())
            elif in_section:
                if line.strip() and any(line.startswith(s) for s in ['Answer:', 'Confidence:', 'Key Evidence:', 'Reasoning:']):
                    break
                section_content.append(line.strip())
        
        return '\n'.join(section_content).strip()

class MultiHopRAGSystem:
    """Complete multi-hop reasoning RAG system"""
    
    def __init__(
        self,
        base_retriever,
        model_name: str = "gpt-4",
        max_hops: int = 5
    ):
        self.decomposer = QuestionDecomposer(model_name)
        self.evidence_retriever = EvidenceRetriever(base_retriever)
        self.reasoning_engine = ReasoningEngine(model_name)
        self.max_hops = max_hops
    
    async def answer_complex_question(
        self,
        question: str,
        max_subquestions: int = 5
    ) -> ReasoningChain:
        """Answer complex question using multi-hop reasoning"""
        
        print(f"Starting multi-hop reasoning for: '{question}'")
        
        # Step 1: Decompose question
        sub_questions = await self.decomposer.decompose_question(
            question, max_subquestions
        )
        
        # Step 2: Create dependency graph
        dep_graph = self.decomposer.create_dependency_graph(sub_questions)
        
        # Step 3: Execute reasoning steps in topological order
        reasoning_steps = []
        completed_steps = {}
        
        # Get execution order (topological sort)
        try:
            execution_order = list(nx.topological_sort(dep_graph))
        except nx.NetworkXError:
            # Fallback to original order if graph has cycles
            execution_order = [sq['step_id'] for sq in sub_questions]
        
        print(f"Executing {len(execution_order)} reasoning steps")
        
        for step_id in execution_order:
            # Find corresponding sub-question
            sub_q = next(sq for sq in sub_questions if sq['step_id'] == step_id)
            
            print(f"Processing {step_id}: {sub_q['sub_question']}")
            
            # Get evidence for this step
            previous_evidence = []
            context_docs = []
            
            # Collect evidence from dependent steps
            for dep_step_id in sub_q.get('depends_on', []):
                if isinstance(dep_step_id, int) and dep_step_id < len(sub_questions):
                    dep_step_actual_id = sub_questions[dep_step_id]['step_id']
                    if dep_step_actual_id in completed_steps:
                        prev_step = completed_steps[dep_step_actual_id]
                        previous_evidence.extend(prev_step.evidence)
                        context_docs.extend(prev_step.retrieved_docs)
            
            # Retrieve evidence
            evidence_docs = await self.evidence_retriever.retrieve_evidence(
                sub_q['sub_question'],
                previous_evidence=previous_evidence,
                context_docs=context_docs
            )
            
            # Perform reasoning
            reasoning_step = await self.reasoning_engine.reason_step(
                question=sub_q['sub_question'],
                evidence_docs=evidence_docs,
                previous_steps=[completed_steps[sid] for sid in completed_steps],
                reasoning_type=sub_q['question_type']
            )
            
            reasoning_steps.append(reasoning_step)
            completed_steps[step_id] = reasoning_step
        
        # Step 4: Synthesize final answer
        final_answer = await self._synthesize_final_answer(
            question, reasoning_steps
        )
        
        # Step 5: Calculate overall confidence
        step_confidences = [step.confidence for step in reasoning_steps]
        overall_confidence = sum(step_confidences) / len(step_confidences) if step_confidences else 0.0
        
        # Step 6: Create reasoning chain
        reasoning_chain = ReasoningChain(
            original_question=question,
            steps=reasoning_steps,
            final_answer=final_answer,
            confidence=overall_confidence,
            reasoning_path=[step.step_id for step in reasoning_steps]
        )
        
        print(f"Multi-hop reasoning completed with {len(reasoning_steps)} steps")
        
        return reasoning_chain
    
    async def _synthesize_final_answer(
        self,
        original_question: str,
        reasoning_steps: List[ReasoningStep]
    ) -> str:
        """Synthesize final answer from reasoning steps"""
        
        # Build context from all reasoning steps
        step_summaries = []
        for step in reasoning_steps:
            step_summaries.append(f"Q: {step.question}\nA: {step.answer}\nConfidence: {step.confidence:.2f}")
        
        steps_context = "\n\n".join(step_summaries)
        
        prompt = f"""
You are synthesizing a final answer from multiple reasoning steps.

Original Question: {original_question}

Reasoning Steps:
{steps_context}

Instructions:
1. Synthesize a comprehensive answer that addresses the original question
2. Use information from all relevant reasoning steps
3. Ensure the answer is coherent and well-structured
4. Highlight any uncertainties or limitations
5. Make the answer self-contained and understandable

Provide a clear, comprehensive final answer:
"""
        
        try:
            response = await openai.ChatCompletion.acreate(
                model=self.reasoning_engine.model_name,
                messages=[{"role": "user", "content": prompt}],
                temperature=0.3,
                max_tokens=400
            )
            
            return response.choices[0].message.content.strip()
            
        except Exception as e:
            print(f"Final answer synthesis failed: {e}")
            
            # Fallback: combine step answers
            step_answers = [step.answer for step in reasoning_steps if step.answer]
            return f"Based on the reasoning steps: {' '.join(step_answers)}"
    
    def visualize_reasoning_chain(self, reasoning_chain: ReasoningChain) -> str:
        """Create a text visualization of the reasoning chain"""
        
        viz = f"Multi-hop Reasoning Chain for: '{reasoning_chain.original_question}'\n"
        viz += "=" * 60 + "\n\n"
        
        for i, step in enumerate(reasoning_chain.steps):
            viz += f"Step {i+1}: {step.step_id}\n"
            viz += f"Question: {step.question}\n"
            viz += f"Answer: {step.answer}\n"
            viz += f"Confidence: {step.confidence:.2f}\n"
            
            if step.evidence:
                viz += "Key Evidence:\n"
                for evidence in step.evidence[:3]:  # Show top 3
                    viz += f"  • {evidence}\n"
            
            viz += f"Documents Used: {len(step.retrieved_docs)}\n"
            viz += "-" * 40 + "\n\n"
        
        viz += f"FINAL ANSWER (Confidence: {reasoning_chain.confidence:.2f}):\n"
        viz += reasoning_chain.final_answer + "\n"
        
        return viz

        # Implementation would synthesize final answer
        return state

    from langgraph.graph import StateGraph

    workflow.add_node("reason", reasoning_node)

    workflow.add_edge("reason", "synthesize")
    
    workflow.set_entry_point("decompose")
    workflow.set_finish_point("synthesize")
    
    return workflow.compile()

            print("Please install spaCy model: python -m spacy download en_core_web_sm")
            self.nlp = None
        
        self.graph = nx.MultiDiGraph()
        self.entities = {}
        self.relationships = []
    
    async def build_from_documents(self, documents: List[Dict]) -> nx.MultiDiGraph:
        """Build knowledge graph from document collection"""
        
        for doc in documents:
            content = doc.get('content', '')
            source = doc.get('source', 'unknown')
            
            # Extract entities and relationships
            entities = self._extract_entities(content)
            relationships = await self._extract_relationships(content, entities, source)
            
            # Add to graph
            self._add_to_graph(entities, relationships)
        
        return self.graph
    
    def _extract_entities(self, text: str) -> List[Entity]:
        """Extract named entities from text"""
        
        if not self.nlp:
            return []
        
        doc = self.nlp(text)
        entities = []
        
        for ent in doc.ents:
            entity = Entity(
                id=f"ent_{len(self.entities)}",
                name=ent.text.strip(),
                type=ent.label_,
                properties={
                    'start_char': ent.start_char,
                    'end_char': ent.end_char,
                    'confidence': 1.0
                }
            )
            
            # Avoid duplicates
            existing_id = self._find_existing_entity(entity.name, entity.type)
            if existing_id:
                entity.id = existing_id
            else:
                self.entities[entity.id] = entity
            
            entities.append(entity)
        
        return entities
    
    def _find_existing_entity(self, name: str, entity_type: str) -> Optional[str]:
        """Find existing entity by name and type"""
        
        for ent_id, entity in self.entities.items():
            if entity.name.lower() == name.lower() and entity.type == entity_type:
                return ent_id
        return None
    
    async def _extract_relationships(
        self, 
        text: str, 
        entities: List[Entity], 
        source: str
    ) -> List[Relationship]:
        """Extract relationships using LLM"""
        
        if len(entities) < 2:
            return []
        
        entity_names = [ent.name for ent in entities]
        
        prompt = f"""
Extract relationships between entities in the following text.

Text: {text[:1000]}

Entities: {', '.join(entity_names)}

For each relationship, specify:
- Subject entity
- Relationship type (e.g., "is_part_of", "located_in", "works_for", etc.)
- Object entity

Format: Subject|Relationship|Object
Only include relationships explicitly stated in the text.

Relationships:
"""
        
        try:
            response = await openai.ChatCompletion.acreate(
                model="gpt-3.5-turbo",
                messages=[{"role": "user", "content": prompt}],
                temperature=0.1,
                max_tokens=300
            )
            
            content = response.choices[0].message.content.strip()
            relationships = []
            
            for line in content.split('\n'):
                if '|' in line:
                    parts = line.split('|')
                    if len(parts) == 3:
                        subj, pred, obj = [p.strip() for p in parts]
                        
                        # Map entity names to IDs
                        subj_id = self._find_entity_id_by_name(subj, entities)
                        obj_id = self._find_entity_id_by_name(obj, entities)
                        
                        if subj_id and obj_id:
                            relationships.append(Relationship(
                                subject=subj_id,
                                predicate=pred.lower().replace(' ', '_'),
                                object=obj_id,
                                source=source
                            ))
            
            return relationships
            
        except Exception as e:
            print(f"Relationship extraction failed: {e}")
            return []
    
    def _find_entity_id_by_name(self, name: str, entities: List[Entity]) -> Optional[str]:
        """Find entity ID by name"""
        
        name_lower = name.lower()
        for entity in entities:
            if entity.name.lower() == name_lower:
                return entity.id
        return None
    
    def _add_to_graph(self, entities: List[Entity], relationships: List[Relationship]):
        """Add entities and relationships to the graph"""
        
        # Add entity nodes
        for entity in entities:
            self.graph.add_node(
                entity.id,
                name=entity.name,
                type=entity.type,
                **entity.properties
            )
        
        # Add relationship edges
        for rel in relationships:
            self.graph.add_edge(
                rel.subject,
                rel.object,
                predicate=rel.predicate,
                confidence=rel.confidence,
                source=rel.source
            )
            
            self.relationships.append(rel)

class GraphRetriever:
    """Retrieve information using graph structure"""
    
    def __init__(self, knowledge_graph: nx.MultiDiGraph, embedding_model: str = "all-MiniLM-L6-v2"):
        self.graph = knowledge_graph
        self.embedding_model = SentenceTransformer(embedding_model)
        
        # Create embeddings for entities
        self.entity_embeddings = {}
        self._create_entity_embeddings()
    
    def _create_entity_embeddings(self):
        """Create embeddings for all entities in the graph"""
        
        for node_id in self.graph.nodes():
            node_data = self.graph.nodes[node_id]
            entity_text = f"{node_data.get('name', '')} {node_data.get('type', '')}"
            
            if entity_text.strip():
                embedding = self.embedding_model.encode([entity_text])[0]
                self.entity_embeddings[node_id] = embedding
    
    async def retrieve_by_entities(self, query_entities: List[str], max_hops: int = 2) -> List[Dict]:
        """Retrieve information by finding paths between entities"""
        
        # Find entity nodes matching the query
        matched_entities = []
        
        for entity_name in query_entities:
            for node_id in self.graph.nodes():
                node_data = self.graph.nodes[node_id]
                if entity_name.lower() in node_data.get('name', '').lower():
                    matched_entities.append(node_id)
        
        if not matched_entities:
            return []
        
        # Find paths between matched entities
        relevant_paths = []
        
        for i, start_entity in enumerate(matched_entities):
            for end_entity in matched_entities[i+1:]:
                try:
                    # Find shortest path
                    if nx.has_path(self.graph, start_entity, end_entity):
                        path = nx.shortest_path(self.graph, start_entity, end_entity)
                        if len(path) <= max_hops + 1:
                            relevant_paths.append(path)
                except nx.NetworkXNoPath:
                    continue
        
        # Convert paths to readable information
        results = []
        
        for path in relevant_paths:
            path_info = self._path_to_info(path)
            results.append({
                'content': path_info['description'],
                'entities': path_info['entities'],
                'relationships': path_info['relationships'],
                'path_length': len(path) - 1,
                'confidence': path_info['confidence']
            })
        
        return results
    
    def _path_to_info(self, path: List[str]) -> Dict:
        """Convert graph path to readable information"""
        
        entities = []
        relationships = []
        descriptions = []
        confidences = []
        
        for i, node_id in enumerate(path):
            node_data = self.graph.nodes[node_id]
            entity_name = node_data.get('name', node_id)
            entity_type = node_data.get('type', 'Unknown')
            
            entities.append(f"{entity_name} ({entity_type})")
            
            if i < len(path) - 1:
                # Get edge information
                next_node = path[i + 1]
                edge_data = self.graph.get_edge_data(node_id, next_node)
                
                if edge_data:
                    # Handle multiple edges
                    edge_info = list(edge_data.values())[0]
                    predicate = edge_info.get('predicate', 'related_to')
                    confidence = edge_info.get('confidence', 1.0)
                    
                    relationships.append(predicate)
                    confidences.append(confidence)
                    
                    descriptions.append(f"{entity_name} {predicate.replace('_', ' ')} {self.graph.nodes[next_node].get('name', next_node)}")
        
        avg_confidence = sum(confidences) / len(confidences) if confidences else 1.0
        
        return {
            'description': ' -> '.join(descriptions),
            'entities': entities,
            'relationships': relationships,
            'confidence': avg_confidence
        }
    
    async def semantic_graph_search(self, query: str, top_k: int = 5) -> List[Dict]:
        """Combine semantic search with graph structure"""
        
        # Get query embedding
        query_embedding = self.embedding_model.encode([query])[0]
        
        # Calculate similarities with entities
        entity_scores = []
        
        for entity_id, entity_embedding in self.entity_embeddings.items():
            from sklearn.metrics.pairwise import cosine_similarity
            import numpy as np
            
            similarity = cosine_similarity([query_embedding], [entity_embedding])[0][0]
            
            if similarity > 0.3:  # Threshold for relevance
                entity_scores.append((entity_id, similarity))
        
        # Sort by similarity
        entity_scores.sort(key=lambda x: x[1], reverse=True)
        top_entities = entity_scores[:top_k]
        
        # Expand results using graph structure
        expanded_results = []
        
        for entity_id, score in top_entities:
            # Get entity info
            node_data = self.graph.nodes[entity_id]
            entity_name = node_data.get('name', entity_id)
            entity_type = node_data.get('type', 'Unknown')
            
            # Get connected entities
            neighbors = list(self.graph.neighbors(entity_id))
            neighbor_info = []
            
            for neighbor_id in neighbors[:3]:  # Limit to 3 neighbors
                neighbor_data = self.graph.nodes[neighbor_id]
                neighbor_name = neighbor_data.get('name', neighbor_id)
                
                # Get edge relationship
                edge_data = self.graph.get_edge_data(entity_id, neighbor_id)
                if edge_data:
                    predicate = list(edge_data.values())[0].get('predicate', 'related_to')
                    neighbor_info.append(f"{predicate.replace('_', ' ')} {neighbor_name}")
            
            result = {
                'content': f"{entity_name} ({entity_type}): {'; '.join(neighbor_info)}",
                'entity_id': entity_id,
                'entity_name': entity_name,
                'entity_type': entity_type,
                'neighbors': neighbor_info,
                'similarity_score': float(score),
                'graph_context': True
            }
            
            expanded_results.append(result)
        
        return expanded_results

class GraphEnhancedRAG:
    """Complete graph-enhanced RAG system"""
    
    def __init__(
        self,
        base_retriever,
        knowledge_graph: nx.MultiDiGraph = None,
        model_name: str = "gpt-4"
    ):
        self.base_retriever = base_retriever
        self.knowledge_graph = knowledge_graph
        self.graph_retriever = GraphRetriever(knowledge_graph) if knowledge_graph else None
        self.model_name = model_name
        
        if not knowledge_graph:
            self.kg_builder = KnowledgeGraphBuilder()
    
    async def build_knowledge_graph(self, documents: List[Dict]) -> nx.MultiDiGraph:
        """Build knowledge graph from documents"""
        
        print(f"Building knowledge graph from {len(documents)} documents...")
        
        self.knowledge_graph = await self.kg_builder.build_from_documents(documents)
        self.graph_retriever = GraphRetriever(self.knowledge_graph)
        
        print(f"Knowledge graph built with {self.knowledge_graph.number_of_nodes()} entities "
              f"and {self.knowledge_graph.number_of_edges()} relationships")
        
        return self.knowledge_graph
    
    async def retrieve_with_graph(
        self,
        query: str,
        use_semantic_search: bool = True,
        use_entity_search: bool = True,
        top_k: int = 5
    ) -> List[Dict]:
        """Retrieve using both traditional and graph-based methods"""
        
        all_results = []
        
        # Traditional retrieval
        try:
            traditional_results = await self.base_retriever.search(query, limit=top_k)
            for result in traditional_results:
                result['retrieval_method'] = 'traditional'
            all_results.extend(traditional_results)
        except Exception as e:
            print(f"Traditional retrieval failed: {e}")
        
        # Graph-based retrieval
        if self.graph_retriever:
            try:
                # Semantic graph search
                if use_semantic_search:
                    semantic_results = await self.graph_retriever.semantic_graph_search(query, top_k)
                    for result in semantic_results:
                        result['retrieval_method'] = 'graph_semantic'
                    all_results.extend(semantic_results)
                
                # Entity-based search (if entities detected in query)
                if use_entity_search:
                    query_entities = self._extract_query_entities(query)
                    if query_entities:
                        entity_results = await self.graph_retriever.retrieve_by_entities(query_entities)
                        for result in entity_results:
                            result['retrieval_method'] = 'graph_entity'
                        all_results.extend(entity_results)
            
            except Exception as e:
                print(f"Graph retrieval failed: {e}")
        
        # Deduplicate and rank
        final_results = self._merge_and_rank_results(all_results)
        
        return final_results[:top_k]
    
    def _extract_query_entities(self, query: str) -> List[str]:
        """Extract entity mentions from query"""
        
        if not hasattr(self, 'kg_builder') or not self.kg_builder.nlp:
            return []
        
        doc = self.kg_builder.nlp(query)
        entities = [ent.text for ent in doc.ents if ent.label_ in ['PERSON', 'ORG', 'GPE', 'PRODUCT']]
        
        return entities
    
    def _merge_and_rank_results(self, results: List[Dict]) -> List[Dict]:
        """Merge and rank results from different retrieval methods"""
        
        # Group by content similarity
        merged_results = []
        seen_content = set()
        
        for result in results:
            content = result.get('content', '')
            
            # Simple deduplication based on content similarity
            if content and len(content) > 20:
                content_key = content[:100].lower().strip()
                if content_key not in seen_content:
                    seen_content.add(content_key)
                    
                    # Calculate combined score
                    traditional_score = result.get('score', 0.0) if result.get('retrieval_method') == 'traditional' else 0.0
                    graph_score = result.get('similarity_score', 0.0) if 'graph' in result.get('retrieval_method', '') else 0.0
                    
                    # Weighted combination
                    combined_score = 0.6 * traditional_score + 0.4 * graph_score
                    result['combined_score'] = combined_score
                    
                    merged_results.append(result)
        
        # Sort by combined score
        merged_results.sort(key=lambda x: x.get('combined_score', 0.0), reverse=True)
        
        return merged_results
    
    async def answer_with_graph_context(self, query: str, top_k: int = 5) -> str:
        """Generate answer using graph-enhanced retrieval"""
        
        # Retrieve with graph enhancement
        results = await self.retrieve_with_graph(query, top_k=top_k)
        
        # Build context
        context_parts = []
        
        for i, result in enumerate(results):
            content = result.get('content', '')
            method = result.get('retrieval_method', 'unknown')
            
            context_parts.append(f"[Source {i+1} - {method}]: {content}")
        
        context = "\n\n".join(context_parts)
        
        # Generate answer
        prompt = f"""
Answer the following question using the provided context. The context includes information 
from both traditional document retrieval and knowledge graph-based retrieval.

Question: {query}

Context:
{context}

Instructions:
1. Provide a comprehensive answer based on the available information
2. When possible, mention specific entities and their relationships
3. Indicate if information comes from structured knowledge (graph) vs. unstructured text
4. If the information is incomplete, state what additional details would be helpful

Answer:
"""
        
        try:
            response = await openai.ChatCompletion.acreate(
                model=self.model_name,
                messages=[{"role": "user", "content": prompt}],
                temperature=0.3,
                max_tokens=400
            )
            
            return response.choices[0].message.content.strip()
            
        except Exception as e:
            print(f"Answer generation failed: {e}")
            return f"Based on the retrieved information: {context[:500]}..."

Graph RAG Considerations

• Graph Quality: Accuracy depends on entity extraction and relationship identification
• Scalability: Large graphs require efficient storage and querying
• Maintenance: Graphs need updates as new information becomes available
• Integration: Balance between graph-based and traditional retrieval methods

Iterative RAG

Iterative RAG performs multiple rounds of retrieval and generation, refining the search and improving the answer quality through iterative feedback loops.

Iterative Process

  • Initial query processing
  • Retrieval and generation
  • Answer quality assessment
  • Query refinement
  • Additional retrieval rounds
  • Final answer synthesis

Benefits

• Progressive Improvement: Each iteration refines the answer
• Gap Filling: Identify and fill information gaps
• Quality Control: Self-assessment and correction
• Comprehensive Coverage: Explore multiple information angles

@dataclass
class IterationResult:

    query: str
    retrieved_docs: List[Dict]

class AnswerAssessor:
    """Assess answer quality and identify gaps"""
    
    def __init__(self, model_name: str = "gpt-3.5-turbo"):
        self.model_name = model_name
    
    async def assess_answer_quality(
        self,
        question: str,
        answer: str,
        retrieved_docs: List[Dict]
    ) -> Tuple[float, List[str]]:
        """Assess answer quality and identify information gaps"""
        
        doc_summaries = []
        for i, doc in enumerate(retrieved_docs[:3]):  # Limit for prompt size
            content = doc.get('content', '')[:200]
            doc_summaries.append(f"Doc {i+1}: {content}")
        
        docs_context = "\n".join(doc_summaries)
        
        prompt = f"""
Assess the quality of this answer and identify any information gaps.

Question: {question}
Answer: {answer}

Retrieved Documents:
{docs_context}

Evaluate the answer on:
1. Completeness: Does it fully address the question?
2. Accuracy: Is the information correct based on the documents?
3. Relevance: Does it stay focused on the question?
4. Clarity: Is it well-structured and understandable?

Provide:
1. A confidence score (0.0-1.0)
2. List of missing information or gaps (if any)

Format:
Confidence: [score]
Gaps: [list specific information that's missing or unclear]
Assessment: [brief explanation of the scoring]
"""
        
        try:
            response = await openai.ChatCompletion.acreate(
                model=self.model_name,
                messages=[{"role": "user", "content": prompt}],
                temperature=0.1,
                max_tokens=300
            )
            
            content = response.choices[0].message.content.strip()
            
            # Parse response
            confidence_score = 0.5
            gaps = []
            
            lines = content.split('\n')
            for line in lines:
                if line.startswith('Confidence:'):
                    try:
                        confidence_score = float(line.split(':')[1].strip())
                    except ValueError:
                        pass
                elif line.startswith('Gaps:'):
                    gap_text = line.split(':', 1)[1].strip()
                    if gap_text.lower() not in ['none', 'no gaps', 'n/a']:
                        gaps = [gap.strip() for gap in gap_text.split(',') if gap.strip()]
            
            return confidence_score, gaps
            
        except Exception as e:
            print(f"Answer assessment failed: {e}")
            return 0.5, []

class QueryRefiner:
    """Refine queries based on identified gaps"""
    
    def __init__(self, model_name: str = "gpt-3.5-turbo"):
        self.model_name = model_name
    
    async def refine_query(
        self,
        original_query: str,
        current_answer: str,
        identified_gaps: List[str],
        previous_queries: List[str] = None
    ) -> str:
        """Generate refined query to address information gaps"""
        
        if not identified_gaps:
            return original_query
        
        previous_context = ""
        if previous_queries:
            previous_context = f"\nPrevious queries tried: {', '.join(previous_queries[-3:])}"
        
        gaps_text = "; ".join(identified_gaps)
        
        prompt = f"""
Refine the search query to address the identified information gaps.

Original Question: {original_query}
Current Answer: {current_answer[:300]}...
Information Gaps: {gaps_text}
{previous_context}

Create a new, more specific search query that would help find the missing information.
Focus on the most important gaps while keeping the query focused and searchable.

Refined Query:
"""
        
        try:
            response = await openai.ChatCompletion.acreate(
                model=self.model_name,
                messages=[{"role": "user", "content": prompt}],
                temperature=0.3,
                max_tokens=100
            )
            
            refined_query = response.choices[0].message.content.strip()
            return refined_query
            
        except Exception as e:
            print(f"Query refinement failed: {e}")
            return original_query

class IterativeRAGSystem:
    """Complete iterative RAG system"""
    
    def __init__(
        self,
        base_retriever,
        model_name: str = "gpt-4",
        max_iterations: int = 3,
        confidence_threshold: float = 0.8
    ):
        self.base_retriever = base_retriever
        self.model_name = model_name
        self.max_iterations = max_iterations
        self.confidence_threshold = confidence_threshold
        
        self.assessor = AnswerAssessor(model_name)
        self.refiner = QueryRefiner(model_name)
    
    async def iterative_answer(
        self,
        question: str,
        initial_context: str = ""
    ) -> List[IterationResult]:
        """Perform iterative RAG to generate comprehensive answer"""
        
        print(f"Starting iterative RAG for: '{question}'")
        
        iterations = []
        current_query = question
        previous_queries = []
        accumulated_context = initial_context
        
        for iteration in range(1, self.max_iterations + 1):
            print(f"\nIteration {iteration}: {current_query}")
            
            # Retrieve documents
            try:
                retrieved_docs = await self.base_retriever.search(current_query, limit=5)
            except Exception as e:
                print(f"Retrieval failed: {e}")
                retrieved_docs = []
            
            # Generate answer
            answer = await self._generate_answer(
                question, current_query, retrieved_docs, accumulated_context
            )
            
            # Assess answer quality
            confidence, gaps = await self.assessor.assess_answer_quality(
                question, answer, retrieved_docs
            )
            
            # Create iteration result
            iteration_result = IterationResult(
                iteration=iteration,
                query=current_query,
                retrieved_docs=retrieved_docs,
                generated_answer=answer,
                confidence_score=confidence,
                gaps_identified=gaps,
                refinement_needed=confidence < self.confidence_threshold and len(gaps) > 0,
                metadata={
                    'doc_count': len(retrieved_docs),
                    'context_length': len(accumulated_context)
                }
            )
            
            iterations.append(iteration_result)
            
            # Update accumulated context
            accumulated_context += f"\n\nIteration {iteration} findings: {answer}"
            
            print(f"Confidence: {confidence:.2f}, Gaps: {len(gaps)}")
            
            # Check if we should continue
            if not iteration_result.refinement_needed or iteration == self.max_iterations:
                print(f"Stopping after {iteration} iterations")
                break
            
            # Refine query for next iteration
            previous_queries.append(current_query)
            current_query = await self.refiner.refine_query(
                question, answer, gaps, previous_queries
            )
            
            # Avoid infinite loops with identical queries
            if current_query in previous_queries:
                print("Query refinement converged, stopping")
                break
        
        return iterations
    
    async def _generate_answer(
        self,
        original_question: str,
        current_query: str,
        retrieved_docs: List[Dict],
        accumulated_context: str
    ) -> str:
        """Generate answer for current iteration"""
        
        # Build context from retrieved documents
        doc_context = ""
        if retrieved_docs:
            doc_parts = []
            for i, doc in enumerate(retrieved_docs):
                content = doc.get('content', '')[:400]  # Limit length
                source = doc.get('source', f'Document {i+1}')
                doc_parts.append(f"[{source}]: {content}")
            doc_context = "\n\n".join(doc_parts)
        
        # Include accumulated context from previous iterations
        full_context = f"{accumulated_context}\n\n{doc_context}".strip()
        
        prompt = f"""
Answer the question using the provided information. This is part of an iterative process 
to build a comprehensive answer.

Original Question: {original_question}
Current Search Focus: {current_query}

Information Available:
{full_context}

Instructions:
1. Focus on directly answering the original question
2. Use information from both current retrieval and previous iterations
3. Be specific and cite sources when possible
4. If information is incomplete, acknowledge gaps clearly
5. Build upon previous findings rather than repeating them

Answer:
"""
        
        try:
            response = await openai.ChatCompletion.acreate(
                model=self.model_name,
                messages=[{"role": "user", "content": prompt}],
                temperature=0.3,
                max_tokens=400
            )
            
            return response.choices[0].message.content.strip()
            
        except Exception as e:
            print(f"Answer generation failed: {e}")
            return f"Unable to generate answer: {str(e)}"
    
    def synthesize_final_answer(self, iterations: List[IterationResult]) -> str:
        """Synthesize final answer from all iterations"""
        
        if not iterations:
            return "No iterations completed."
        
        # Use the last iteration as the base, but incorporate insights from all
        final_iteration = iterations[-1]
        
        # Combine key insights from all iterations
        all_insights = []
        for iter_result in iterations:
            if iter_result.generated_answer:
                all_insights.append(f"Iteration {iter_result.iteration}: {iter_result.generated_answer}")
        
        # Return the most comprehensive answer (usually the last one)
        return final_iteration.generated_answer
    
    def get_iteration_summary(self, iterations: List[IterationResult]) -> Dict:
        """Get summary of iterative process"""
        
        if not iterations:
            return {}
        
        total_docs = sum(len(iter_result.retrieved_docs) for iter_result in iterations)
        final_confidence = iterations[-1].confidence_score if iterations else 0.0
        queries_used = [iter_result.query for iter_result in iterations]
        
        return {
            'total_iterations': len(iterations),
            'final_confidence': final_confidence,
            'total_documents_retrieved': total_docs,
            'queries_explored': queries_used,
            'improvement_trend': [iter_result.confidence_score for iter_result in iterations],
            'final_gaps': iterations[-1].gaps_identified if iterations else []
        }

class DocumentRanker(RAGModule):
    """Re-rank retrieved documents"""
    
    def __init__(self):
        super().__init__("DocumentRanker", ModuleType.CONTEXT_PROCESSOR)
    
    async def process(self, context: RAGContext) -> RAGContext:
        if not context.retrieved_docs:
            return context
        
        # Simulate re-ranking logic
        query = context.processed_query or context.query
        query_words = set(query.lower().split())
        
        for doc in context.retrieved_docs:
            content_words = set(doc['content'].lower().split())
            word_overlap = len(query_words.intersection(content_words))
            
            # Boost score based on word overlap
            doc['rerank_score'] = doc.get('score', 0.0) + (word_overlap * 0.1)

        context.processed_docs = sorted(
            context.retrieved_docs,

            reverse=True
        )

        return context

class ContextCompressor(RAGModule):
    """Compress context to fit model limits"""
    
    def __init__(self, max_tokens: int = 2000):
        super().__init__("ContextCompressor", ModuleType.CONTEXT_PROCESSOR)
        self.max_tokens = max_tokens
    
    async def process(self, context: RAGContext) -> RAGContext:
        docs_to_process = context.processed_docs or context.retrieved_docs
        if not docs_to_process:
            return context
        
        compressed_docs = []
        current_tokens = 0
        
        for doc in docs_to_process:
            content = doc['content']
            # Rough token estimation (1 token ≈ 4 characters)
            doc_tokens = len(content) // 4
            
            if current_tokens + doc_tokens <= self.max_tokens:
                compressed_docs.append(doc)
                current_tokens += doc_tokens
            else:
                # Truncate the document to fit
                remaining_tokens = self.max_tokens - current_tokens
                truncated_content = content[:remaining_tokens * 4]
                
                truncated_doc = doc.copy()
                truncated_doc['content'] = truncated_content + "..."
                truncated_doc['truncated'] = True
                compressed_docs.append(truncated_doc)
                break
        
        context.processed_docs = compressed_docs
        context.metadata['compression_applied'] = True
        context.metadata['compressed_doc_count'] = len(compressed_docs)
        return context

class RAGPipeline:
    """Manages the execution of RAG modules"""
    
    def __init__(self, name: str):
        self.name = name
        self.modules = []
        self.config = {}
    
    def add_module(self, module: RAGModule) -> 'RAGPipeline':
        """Add a module to the pipeline"""
        self.modules.append(module)
        return self
    
    def configure(self, **kwargs) -> 'RAGPipeline':
        """Configure pipeline parameters"""
        self.config.update(kwargs)
        return self
    
    async def execute(self, query: str, initial_context: Dict = None) -> RAGContext:
        """Execute the pipeline"""
        
        # Initialize context
        context = RAGContext(
            query=query,
            metadata=initial_context or {}
        )
        
        context.metadata['pipeline_name'] = self.name
        context.metadata['module_count'] = len(self.modules)
        
        # Execute modules in sequence
        for i, module in enumerate(self.modules):
            try:
                print(f"Executing module {i+1}/{len(self.modules)}: {module.name}")
                context = await module.process(context)
                context.metadata[f'module_{i+1}_executed'] = module.name
                
            except Exception as e:
                print(f"Module {module.name} failed: {e}")
                context.metadata[f'module_{i+1}_error'] = str(e)
                # Continue with next module or implement error handling strategy
        
        return context
    
    def get_module_by_name(self, name: str) -> Optional[RAGModule]:
        """Get a module by name"""
        for module in self.modules:
            if module.name == name:
                return module
        return None
    
    def get_modules_by_type(self, module_type: ModuleType) -> List[RAGModule]:
        """Get all modules of a specific type"""
        return [m for m in self.modules if m.module_type == module_type]

@dataclass
class DocumentChunk:
    """Represents a chunk of a long document"""
    content: str
    chunk_id: str
    start_pos: int
    end_pos: int
    level: int = 0  # Hierarchical level
    parent_chunk: Optional[str] = None
    child_chunks: List[str] = field(default_factory=list)

class HierarchicalChunker:
    """Create hierarchical chunks from long documents"""
    
    def __init__(
        self, 
        base_chunk_size: int = 1000,
        overlap: int = 200,
        max_levels: int = 3
    ):
        self.base_chunk_size = base_chunk_size
        self.overlap = overlap
        self.max_levels = max_levels
    
    def chunk_document(self, text: str, doc_id: str) -> List[DocumentChunk]:
        """Create hierarchical chunks from document"""
        
        chunks = []
        
        # Level 0: Base chunks
        base_chunks = self._create_base_chunks(text, doc_id)
        chunks.extend(base_chunks)
        
        # Higher levels: Aggregate chunks
        current_level_chunks = base_chunks
        
        for level in range(1, self.max_levels):
            if len(current_level_chunks) <= 2:
                break  # No need for more levels
            
            parent_chunks = self._create_parent_chunks(
                current_level_chunks, doc_id, level
            )
            chunks.extend(parent_chunks)
            current_level_chunks = parent_chunks
        
        return chunks
    
    def _create_base_chunks(self, text: str, doc_id: str) -> List[DocumentChunk]:
        """Create base-level chunks"""
        
        chunks = []
        words = text.split()
        
        for i in range(0, len(words), self.base_chunk_size - self.overlap):
            chunk_words = words[i:i + self.base_chunk_size]
            chunk_text = ' '.join(chunk_words)
            
            if not chunk_text.strip():
                continue
            
            start_pos = len(' '.join(words[:i])) if i > 0 else 0
            end_pos = start_pos + len(chunk_text)
            
            chunk = DocumentChunk(
                content=chunk_text,
                chunk_id=f"{doc_id}_chunk_{len(chunks)}",
                start_pos=start_pos,
                end_pos=end_pos,
                level=0,
                metadata={
                    'word_count': len(chunk_words),
                    'doc_id': doc_id
                }
            )
            
            chunks.append(chunk)
        
        return chunks
    
    def _create_parent_chunks(
        self, 
        child_chunks: List[DocumentChunk], 
        doc_id: str, 
        level: int
    ) -> List[DocumentChunk]:
        """Create parent chunks by combining child chunks"""
        
        parent_chunks = []
        group_size = max(2, len(child_chunks) // 3)  # Group child chunks
        
        for i in range(0, len(child_chunks), group_size):
            group = child_chunks[i:i + group_size]
            
            # Combine content
            combined_content = ' '.join([chunk.content for chunk in group])
            
            # Truncate if too long
            if len(combined_content) > self.base_chunk_size * 2:
                combined_content = combined_content[:self.base_chunk_size * 2] + "..."
            
            parent_chunk = DocumentChunk(
                content=combined_content,
                chunk_id=f"{doc_id}_level{level}_chunk_{len(parent_chunks)}",
                start_pos=group[0].start_pos,
                end_pos=group[-1].end_pos,
                level=level,
                child_chunks=[chunk.chunk_id for chunk in group],
                metadata={
                    'child_count': len(group),
                    'doc_id': doc_id
                }
            )
            
            # Update child chunks with parent reference
            for chunk in group:
                chunk.parent_chunk = parent_chunk.chunk_id
            
            parent_chunks.append(parent_chunk)
        
        return parent_chunks

class ContextCompressor:
    """Compress long contexts to fit model limits"""
    
    def __init__(self, model_name: str = "gpt-3.5-turbo", max_tokens: int = 3000):
        self.model_name = model_name
        self.max_tokens = max_tokens
        self.embedding_model = SentenceTransformer('all-MiniLM-L6-v2')
    
    async def compress_context(
        self, 
        chunks: List[DocumentChunk], 
        query: str,
        compression_ratio: float = 0.5
    ) -> str:
        """Compress context using summarization and selection"""
        
        # Step 1: Rank chunks by relevance to query
        ranked_chunks = await self._rank_chunks_by_relevance(chunks, query)
        
        # Step 2: Select chunks within token budget
        selected_chunks = self._select_chunks_within_budget(
            ranked_chunks, int(self.max_tokens * compression_ratio)
        )
        
        # Step 3: Generate summaries for remaining chunks
        compressed_content = await self._generate_compressed_summaries(
            selected_chunks, query
        )
        
        return compressed_content
    
    async def _rank_chunks_by_relevance(
        self, 
        chunks: List[DocumentChunk], 
        query: str
    ) -> List[Tuple[DocumentChunk, float]]:
        """Rank chunks by semantic similarity to query"""
        
        # Get query embedding
        query_embedding = self.embedding_model.encode([query])[0]
        
        # Calculate similarities
        chunk_scores = []
        
        for chunk in chunks:
            chunk_embedding = self.embedding_model.encode([chunk.content])[0]
            
            from sklearn.metrics.pairwise import cosine_similarity
            similarity = cosine_similarity([query_embedding], [chunk_embedding])[0][0]
            
            chunk_scores.append((chunk, float(similarity)))
        
        # Sort by similarity score
        return sorted(chunk_scores, key=lambda x: x[1], reverse=True)
    
    def _select_chunks_within_budget(
        self, 
        ranked_chunks: List[Tuple[DocumentChunk, float]], 
        token_budget: int
    ) -> List[DocumentChunk]:
        """Select chunks that fit within token budget"""
        
        selected_chunks = []
        current_tokens = 0
        
        for chunk, score in ranked_chunks:
            # Rough token estimation (1 token ≈ 4 characters)
            chunk_tokens = len(chunk.content) // 4
            
            if current_tokens + chunk_tokens <= token_budget:
                selected_chunks.append(chunk)
                current_tokens += chunk_tokens
            else:
                break
        
        return selected_chunks
    
    async def _generate_compressed_summaries(
        self, 
        chunks: List[DocumentChunk], 
        query: str
    ) -> str:
        """Generate compressed summaries of selected chunks"""
        
        if not chunks:
            return ""
        
        # Group chunks by level for better organization
        chunks_by_level = {}
        for chunk in chunks:
            level = chunk.level
            if level not in chunks_by_level:
                chunks_by_level[level] = []
            chunks_by_level[level].append(chunk)
        
        compressed_parts = []
        
        # Process each level
        for level in sorted(chunks_by_level.keys()):
            level_chunks = chunks_by_level[level]
            
            # Combine chunks for this level
            level_content = '\n\n'.join([
                f"[Chunk {chunk.chunk_id}]: {chunk.content[:500]}..."
                for chunk in level_chunks
            ])
            
            # Generate summary for this level
            summary = await self._summarize_level_content(level_content, query, level)
            compressed_parts.append(f"Level {level} Summary: {summary}")
        
        return '\n\n'.join(compressed_parts)
    
    async def _summarize_level_content(
        self, 
        content: str, 
        query: str, 
        level: int
    ) -> str:
        """Generate summary for content at specific level"""
        
        prompt = f"""
Summarize the following content focusing on information relevant to this query: "{query}"

Content (Level {level}):
{content[:1500]}

Instructions:
1. Focus on information that helps answer the query
2. Maintain key facts and important details
3. Keep the summary concise but comprehensive
4. Preserve any quantitative data or specific examples

Summary:
"""
        
        try:
            response = await openai.ChatCompletion.acreate(
                model=self.model_name,
                messages=[{"role": "user", "content": prompt}],
                temperature=0.2,
                max_tokens=200
            )
            
            return response.choices[0].message.content.strip()
            
        except Exception as e:
            print(f"Summarization failed: {e}")
            return content[:200] + "..."

class AdaptiveContextSelector:
    """Adaptively select context based on query complexity"""
    
    def __init__(self):
        self.query_complexity_indicators = [
            'compare', 'analyze', 'explain', 'relationship', 'difference',
            'multiple', 'various', 'different', 'several', 'many'
        ]
    
    async def select_optimal_context(
        self, 
        chunks: List[DocumentChunk], 
        query: str,
        base_token_limit: int = 2000
    ) -> Tuple[List[DocumentChunk], int]:
        """Select optimal context based on query complexity"""
        
        # Assess query complexity
        complexity_score = self._assess_query_complexity(query)
        
        # Adjust token limit based on complexity
        adjusted_limit = int(base_token_limit * (1 + complexity_score))
        
        # Select chunks using different strategies based on complexity
        if complexity_score > 0.7:
            # High complexity: use hierarchical selection
            selected_chunks = await self._hierarchical_selection(
                chunks, query, adjusted_limit
            )
        elif complexity_score > 0.3:
            # Medium complexity: use relevance-based selection
            selected_chunks = await self._relevance_selection(
                chunks, query, adjusted_limit
            )
        else:
            # Low complexity: use simple top-k selection
            selected_chunks = await self._simple_selection(
                chunks, query, adjusted_limit
            )
        
        return selected_chunks, adjusted_limit
    
    def _assess_query_complexity(self, query: str) -> float:
        """Assess query complexity (0.0 - 1.0)"""
        
        query_lower = query.lower()
        
        # Count complexity indicators
        complexity_indicators = sum(
            1 for indicator in self.query_complexity_indicators
            if indicator in query_lower
        )
        
        # Factor in query length
        length_factor = min(len(query.split()) / 20.0, 1.0)
        
        # Factor in question words
        question_words = ['what', 'how', 'why', 'when', 'where', 'which']
        question_factor = sum(1 for word in question_words if word in query_lower) * 0.1
        
        complexity = (
            (complexity_indicators * 0.3) +
            (length_factor * 0.4) +
            (question_factor * 0.3)
        )
        
        return min(complexity, 1.0)
    
    async def _hierarchical_selection(
        self, 
        chunks: List[DocumentChunk], 
        query: str, 
        token_limit: int
    ) -> List[DocumentChunk]:
        """Select chunks using hierarchical strategy"""
        
        # Prefer higher-level chunks for overview, lower-level for details
        chunks_by_level = {}
        for chunk in chunks:
            level = chunk.level
            if level not in chunks_by_level:
                chunks_by_level[level] = []
            chunks_by_level[level].append(chunk)
        
        selected_chunks = []
        remaining_tokens = token_limit
        
        # Start with highest level for overview
        for level in sorted(chunks_by_level.keys(), reverse=True):
            level_chunks = chunks_by_level[level][:2]  # Limit high-level chunks
            
            for chunk in level_chunks:
                chunk_tokens = len(chunk.content) // 4
                if chunk_tokens <= remaining_tokens:
                    selected_chunks.append(chunk)
                    remaining_tokens -= chunk_tokens
        
        # Add detailed chunks from lower levels
        for level in sorted(chunks_by_level.keys()):
            if remaining_tokens <= 0:
                break
                
            level_chunks = chunks_by_level[level]
            
            for chunk in level_chunks:
                if chunk in selected_chunks:
                    continue
                    
                chunk_tokens = len(chunk.content) // 4
                if chunk_tokens <= remaining_tokens:
                    selected_chunks.append(chunk)
                    remaining_tokens -= chunk_tokens
        
        return selected_chunks
    
    async def _relevance_selection(
        self, 
        chunks: List[DocumentChunk], 
        query: str, 
        token_limit: int
    ) -> List[DocumentChunk]:
        """Select chunks based on relevance scores"""
        
        # This would use the same logic as ContextCompressor._rank_chunks_by_relevance
        # Simplified for brevity
        selected_chunks = []
        remaining_tokens = token_limit
        
        for chunk in chunks:
            chunk_tokens = len(chunk.content) // 4
            if chunk_tokens <= remaining_tokens:
                selected_chunks.append(chunk)
                remaining_tokens -= chunk_tokens
            
            if remaining_tokens <= 0:
                break
        
        return selected_chunks[:10]  # Limit to top 10 chunks
    
    async def _simple_selection(
        self, 
        chunks: List[DocumentChunk], 
        query: str, 
        token_limit: int
    ) -> List[DocumentChunk]:
        """Simple selection of first chunks that fit"""
        
        selected_chunks = []
        remaining_tokens = token_limit
        
        for chunk in chunks[:5]:  # Limit to first 5 chunks
            chunk_tokens = len(chunk.content) // 4
            if chunk_tokens <= remaining_tokens:
                selected_chunks.append(chunk)
                remaining_tokens -= chunk_tokens
        
        return selected_chunks

class LongContextRAG:
    """Complete long-context RAG system"""
    
    def __init__(
        self,
        model_name: str = "gpt-4",
        max_context_tokens: int = 8000,
        chunk_size: int = 1000
    ):
        self.model_name = model_name
        self.max_context_tokens = max_context_tokens
        
        self.chunker = HierarchicalChunker(
            base_chunk_size=chunk_size,
            overlap=200,
            max_levels=3
        )
        self.compressor = ContextCompressor(model_name, max_context_tokens)
        self.selector = AdaptiveContextSelector()
    
    async def process_long_document(
        self, 
        document_text: str, 
        query: str,
        doc_id: str = "long_doc"
    ) -> LongContextResult:
        """Process a long document with the given query"""
        
        print(f"Processing long document ({len(document_text)} chars) for query: '{query}'")
        
        # Step 1: Create hierarchical chunks
        chunks = self.chunker.chunk_document(document_text, doc_id)
        print(f"Created {len(chunks)} chunks across multiple levels")
        
        # Step 2: Select optimal context
        selected_chunks, adjusted_limit = await self.selector.select_optimal_context(
            chunks, query, self.max_context_tokens // 2
        )
        print(f"Selected {len(selected_chunks)} chunks (limit: {adjusted_limit} tokens)")
        
        # Step 3: Compress context if needed
        compressed_context = await self.compressor.compress_context(
            selected_chunks, query, compression_ratio=0.7
        )
        
        # Step 4: Generate answer
        answer = await self._generate_answer(query, compressed_context)
        
        # Step 5: Compile results
        result = LongContextResult(
            original_query=query,
            selected_chunks=selected_chunks,
            compressed_context=compressed_context,
            generated_answer=answer,
            processing_stats={
                'total_chunks': len(chunks),
                'selected_chunks': len(selected_chunks),
                'original_length': len(document_text),
                'compressed_length': len(compressed_context),
                'compression_ratio': len(compressed_context) / len(document_text) if document_text else 0,
                'token_limit_used': adjusted_limit
            }
        )
        
        return result
    
    async def _generate_answer(self, query: str, context: str) -> str:
        """Generate answer from compressed context"""
        
        prompt = f"""
Answer the following question using the provided context. The context has been processed 
and compressed from a much larger document to focus on relevant information.

Question: {query}

Context:
{context}

Instructions:
1. Provide a comprehensive answer based on the context
2. Reference specific information from different sections when relevant
3. If the context seems incomplete, acknowledge any limitations
4. Maintain accuracy and avoid hallucination

Answer:
"""
        
        try:
            response = await openai.ChatCompletion.acreate(
                model=self.model_name,
                messages=[{"role": "user", "content": prompt}],
                temperature=0.3,
                max_tokens=500
            )
            
            return response.choices[0].message.content.strip()
            
        except Exception as e:
            print(f"Answer generation failed: {e}")
            return f"Unable to generate answer due to error: {str(e)}"
