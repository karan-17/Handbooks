What is Kubeflow?

Kubeflow is an open-source machine learning toolkit for Kubernetes that makes deployments of machine learning workflows on Kubernetes simple, portable, and scalable. It provides a complete ML platform that enables data scientists and engineers to build and deploy ML models at scale.

Key Benefits

  • Cloud Native: Built specifically for Kubernetes environments
  • Scalable: Leverage Kubernetes for horizontal scaling
  • Portable: Run anywhere Kubernetes runs (cloud, on-premises, edge)
  • Composable: Use individual components or the full platform
  • Multi-Framework: Support for TensorFlow, PyTorch, XGBoost, and more
  • End-to-End: Complete ML lifecycle from development to production

Why Kubeflow?
Traditional ML development often involves fragmented tools and manual processes. Kubeflow provides a unified platform that standardizes ML workflows while leveraging Kubernetes for infrastructure management.

import kfp
from kfp import dsl

def preprocess_op():
    return dsl.ContainerOp(
        name='Preprocess Data',
        image='python:3.8',
        command=['python', '-c'],
        arguments=['''
import pandas as pd
import pickle

# Load and preprocess data
df = pd.read_csv('/data/raw/dataset.csv')
df_processed = df.dropna().reset_index(drop=True)

# Save processed data
with open('/data/processed/data.pkl', 'wb') as f:
    pickle.dump(df_processed, f)
print("Data preprocessing completed")
'''],
        file_outputs={'processed_data': '/data/processed/data.pkl'}
    )

def train_op(processed_data):
    return dsl.ContainerOp(
        name='Train Model',
        image='tensorflow/tensorflow:2.8.0',
        command=['python', '-c'],
        arguments=[f'''
import pickle
import tensorflow as tf
from tensorflow import keras

# Load processed data
with open("{processed_data}", "rb") as f:
    data = pickle.load(f)

# Create and train model
model = keras.Sequential([
    keras.layers.Dense(64, activation='relu'),
    keras.layers.Dense(32, activation='relu'),
    keras.layers.Dense(1, activation='sigmoid')
])

model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])
# Training logic here...

# Save model
model.save('/model/trained_model')
print("Model training completed")
'''],
        file_outputs={'model': '/model/trained_model'}
    )

def deploy_op(model):
    return dsl.ContainerOp(
        name='Deploy Model',
        image='kfserving/sklearnserver:latest',
        command=['python', '-c'],
        arguments=[f'''
# Deploy model to KServe
print("Model deployed successfully")
''']
    )

@dsl.pipeline(
    name='ML Training Pipeline',
    description='A simple ML training pipeline'
)
def ml_pipeline():
    preprocess_task = preprocess_op()
    train_task = train_op(preprocess_task.outputs['processed_data'])
    deploy_task = deploy_op(train_task.outputs['model'])

# Compile and submit pipeline
if __name__ == '__main__':
    kfp.compiler.Compiler().compile(ml_pipeline, 'ml_pipeline.yaml')
    
    # Submit to Kubeflow Pipelines
    client = kfp.Client(host='http://localhost:8080')
    run = client.run_pipeline(
        experiment_id='default',
        job_name='ml-training-run',
        pipeline_package_path='ml_pipeline.yaml'
    )
    print(f"Pipeline run created: {run.id}")

from kfp import components

@components.create_component_from_func
def data_validation_op(
    data_path: str,
    validation_rules: str
) -> str:
    """Validates input data against predefined rules"""
    import pandas as pd
    import json
    
    df = pd.read_csv(data_path)
    rules = json.loads(validation_rules)
    
    # Validation logic
    for rule in rules:
        if rule['type'] == 'not_null':
            assert not df[rule['column']].isnull().any(), f"Null values found in {rule['column']}"
        elif rule['type'] == 'range':
            assert df[rule['column']].between(rule['min'], rule['max']).all(), f"Values out of range in {rule['column']}"
    
    return "validation_passed"

Model Serving with KServe

KServe (formerly KFServing) provides serverless model serving with features like autoscaling, canary deployments, and multi-framework support.

Deploy TensorFlow Model

# tensorflow-serving.yaml
apiVersion: serving.kserve.io/v1beta1
kind: InferenceService
metadata:
  name: tensorflow-model
  namespace: kubeflow-user-example-com
spec:
  predictor:
    tensorflow:
      storageUri: gs://my-bucket/tensorflow-model/
      resources:
        limits:
          cpu: 1000m
          memory: 2Gi
        requests:
          cpu: 100m
          memory: 512Mi
      env:
        - name: TF_CPP_MIN_LOG_LEVEL
          value: "2"

# Deploy the model
kubectl apply -f tensorflow-serving.yaml

# Check status
kubectl get inferenceservice tensorflow-model -n kubeflow-user-example-com

# Custom predictor with Python runtime
import kserve
import joblib
import numpy as np
from typing import Dict, List

class CustomPredictor(kserve.Model):
    def __init__(self, name: str):
        super().__init__(name)
        self.name = name
        self.ready = False
        
    def load(self) -> bool:
        # Load your custom model
        self.model = joblib.load('/mnt/models/model.pkl')
        self.ready = True
        return self.ready
    
    def predict(self, payload: Dict, headers: Dict[str, str] = None) -> Dict:
        inputs = np.array(payload["instances"])
        predictions = self.model.predict(inputs)
        return {"predictions": predictions.tolist()}

if __name__ == "__main__":
    model = CustomPredictor("custom-model")
    model.load()
    kserve.ModelServer().start(models=[model])

End-to-End Image Classification Pipeline

import kfp
from kfp import dsl
from kfp.components import InputPath, OutputPath, create_component_from_func

@create_component_from_func
def download_data_op(
    dataset_url: str,
    output_path: OutputPath(str)
):
    """Download and prepare image dataset"""
    import os
    import urllib.request
    import zipfile
    
    # Download dataset
    urllib.request.urlretrieve(dataset_url, '/tmp/dataset.zip')
    
    # Extract dataset
    with zipfile.ZipFile('/tmp/dataset.zip', 'r') as zip_ref:
        zip_ref.extractall(output_path)
    
    print(f"Dataset downloaded and extracted to {output_path}")

@create_component_from_func
def preprocess_images_op(
    input_path: InputPath(str),
    output_path: OutputPath(str),
    image_size: int = 224
):
    """Preprocess images for training"""
    import os
    import cv2
    import numpy as np
    from tensorflow.keras.utils import to_categorical
    
    # Image preprocessing logic
    X, y = [], []
    class_names = sorted(os.listdir(input_path))
    
    for idx, class_name in enumerate(class_names):
        class_dir = os.path.join(input_path, class_name)
        for img_file in os.listdir(class_dir):
            img_path = os.path.join(class_dir, img_file)
            img = cv2.imread(img_path)
            img = cv2.resize(img, (image_size, image_size))
            img = img.astype(np.float32) / 255.0
            X.append(img)
            y.append(idx)
    
    X = np.array(X)
    y = to_categorical(y, len(class_names))
    
    # Save preprocessed data
    np.save(os.path.join(output_path, 'X.npy'), X)
    np.save(os.path.join(output_path, 'y.npy'), y)
    
    with open(os.path.join(output_path, 'classes.txt'), 'w') as f:
        f.write('\n'.join(class_names))
    
    print(f"Preprocessed {len(X)} images with {len(class_names)} classes")

@create_component_from_func
def train_model_op(
    input_path: InputPath(str),
    model_path: OutputPath(str),
    epochs: int = 50,
    batch_size: int = 32
):
    """Train CNN model"""
    import os
    import numpy as np
    import tensorflow as tf
    from tensorflow.keras import layers, models
    
    # Load preprocessed data
    X = np.load(os.path.join(input_path, 'X.npy'))
    y = np.load(os.path.join(input_path, 'y.npy'))
    
    # Split train/validation
    split_idx = int(0.8 * len(X))
    X_train, X_val = X[:split_idx], X[split_idx:]
    y_train, y_val = y[:split_idx], y[split_idx:]
    
    # Build model
    model = models.Sequential([
        layers.Conv2D(32, (3, 3), activation='relu', input_shape=X.shape[1:]),
        layers.MaxPooling2D((2, 2)),
        layers.Conv2D(64, (3, 3), activation='relu'),
        layers.MaxPooling2D((2, 2)),
        layers.Conv2D(64, (3, 3), activation='relu'),
        layers.Flatten(),
        layers.Dense(64, activation='relu'),
        layers.Dropout(0.5),
        layers.Dense(y.shape[1], activation='softmax')
    ])
    
    model.compile(
        optimizer='adam',
        loss='categorical_crossentropy',
        metrics=['accuracy']
    )
    
    # Train model
    history = model.fit(
        X_train, y_train,
        epochs=epochs,
        batch_size=batch_size,
        validation_data=(X_val, y_val),
        verbose=1
    )
    
    # Save model
    model.save(model_path)
    print(f"Model training completed. Final accuracy: {history.history['val_accuracy'][-1]:.4f}")

@create_component_from_func
def evaluate_model_op(
    model_path: InputPath(str),
    test_data_path: InputPath(str)
) -> float:
    """Evaluate model performance"""
    import os
    import numpy as np
    import tensorflow as tf
    
    # Load model and test data
    model = tf.keras.models.load_model(model_path)
    X_test = np.load(os.path.join(test_data_path, 'X.npy'))
    y_test = np.load(os.path.join(test_data_path, 'y.npy'))
    
    # Evaluate
    test_loss, test_accuracy = model.evaluate(X_test, y_test, verbose=0)
    print(f"Test accuracy: {test_accuracy:.4f}")
    
    return test_accuracy

@dsl.pipeline(
    name='Image Classification Pipeline',
    description='End-to-end image classification with CNN'
)
def image_classification_pipeline(
    dataset_url: str = "https://example.com/dataset.zip",
    epochs: int = 50,
    batch_size: int = 32,
    accuracy_threshold: float = 0.85
):
    # Download and preprocess data
    download_task = download_data_op(dataset_url)
    preprocess_task = preprocess_images_op(download_task.output)
    
    # Train model
    train_task = train_model_op(
        preprocess_task.output,
        epochs=epochs,
        batch_size=batch_size
    )
    
    # Evaluate model
    evaluate_task = evaluate_model_op(
        train_task.output,
        preprocess_task.output
    )
    
    # Conditional deployment based on accuracy
    with dsl.Condition(evaluate_task.output >= accuracy_threshold):
        deploy_task = deploy_model_to_kserve_op(train_task.output)

# Compile and run pipeline
if __name__ == '__main__':
    kfp.compiler.Compiler().compile(
        image_classification_pipeline, 
        'image_classification_pipeline.yaml'
    )

Hyperparameter Tuning with Katib

# katib-experiment.yaml
apiVersion: kubeflow.org/v1beta1
kind: Experiment
metadata:
  namespace: kubeflow
  name: hyperparameter-tuning
spec:
  algorithm:
    algorithmName: random
  objective:
    type: maximize
    goal: 0.99
    objectiveMetricName: Validation-accuracy
    additionalMetricNames:
      - Train-accuracy
  parameters:
    - name: lr
      parameterType: double
      feasibleSpace:
        min: "0.001"
        max: "0.1"
    - name: batch-size
      parameterType: int
      feasibleSpace:
        min: "16"
        max: "128"
    - name: epochs
      parameterType: int
      feasibleSpace:
        min: "10"
        max: "100"
  trialTemplate:
    primaryContainerName: training-container
    trialParameters:
      - name: learningRate
        description: Learning rate for the training model
        reference: lr
      - name: batchSize
        description: Batch size for training
        reference: batch-size
      - name: epochs
        description: Number of training epochs
        reference: epochs
    trialSpec:
      apiVersion: batch/v1
      kind: Job
      spec:
        template:
          spec:
            containers:
              - name: training-container
                image: my-registry/ml-trainer:latest
                command:
                  - "python3"
                  - "/opt/model.py"
                  - "--lr=${trialParameters.learningRate}"
                  - "--batch-size=${trialParameters.batchSize}"
                  - "--epochs=${trialParameters.epochs}"
            restartPolicy: Never

# Apply the experiment
kubectl apply -f katib-experiment.yaml

# Monitor experiment progress
kubectl get experiment hyperparameter-tuning -n kubeflow -w

# Get best trial results
kubectl get trials -n kubeflow -l experiment=hyperparameter-tuning --sort-by=.status.observation.metrics[0].max
