1. GitHub Actions Fundamentals

Architecture Overview

GitHub Actions Architecture

Event → Workflow → Job → Step → Action

📅 Events (push, PR, schedule)
↓
🔄 Workflows (.github/workflows/*.yml)
↓
🏗️ Jobs (parallel execution units)
↓
📝 Steps (sequential commands)
↓
⚙️ Actions (reusable components)

Core Components Explained

🔄 Workflows

Automated processes defined in YAML files that respond to repository events.

  • Stored in .github/workflows/
  • Triggered by events or schedules
  • Can contain multiple jobs
  • Support for matrix builds

🏗️ Jobs

Sets of steps that execute on the same runner environment.

  • Run in parallel by default
  • Can depend on other jobs
  • Execute on GitHub-hosted or self-hosted runners
  • Isolated environments

📝 Steps

Individual tasks within a job that run sequentially.

  • Shell commands or actions
  • Share the same runner environment
  • Can pass data between steps
  • Support conditional execution

⚙️ Actions

Reusable units of code that perform specific tasks.

  • Public actions from GitHub Marketplace
  • Custom actions for your organization
  • JavaScript, Docker, or composite actions
  • Versioned and shareable

Basic Workflow Structure

Starter Workflow Template

# .github/workflows/ci.yml
name: Continuous Integration

# Trigger events
on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main ]
  schedule:
    - cron: '0 2 * * 1'  # Weekly on Monday at 2 AM

# Environment variables
env:
  NODE_VERSION: '18'
  PYTHON_VERSION: '3.11'

# Workflow jobs
jobs:
  # Job 1: Code Quality
  lint-and-test:
    runs-on: ubuntu-latest
    
    steps:
    - name: Checkout repository
      uses: actions/checkout@v4
      
    - name: Setup Node.js
      uses: actions/setup-node@v4
      with:
        node-version: ${{ env.NODE_VERSION }}
        cache: 'npm'
    
    - name: Install dependencies
      run: npm ci
    
    - name: Run linter
      run: npm run lint
    
    - name: Run tests
      run: npm test -- --coverage
    
    - name: Upload coverage reports
      uses: codecov/codecov-action@v3

  # Job 2: Security Scanning
  security-scan:
    runs-on: ubuntu-latest
    
    steps:
    - uses: actions/checkout@v4
    
    - name: Run security audit
      run: npm audit --audit-level high
    
    - name: Dependency Review
      uses: actions/dependency-review-action@v3

  # Job 3: Build and Package
  build:
    needs: [lint-and-test, security-scan]
    runs-on: ubuntu-latest
    
    steps:
    - uses: actions/checkout@v4
    
    - name: Setup Node.js
      uses: actions/setup-node@v4
      with:
        node-version: ${{ env.NODE_VERSION }}
        cache: 'npm'
    
    - name: Install dependencies
      run: npm ci
    
    - name: Build application
      run: npm run build
    
    - name: Upload build artifacts
      uses: actions/upload-artifact@v3
      with:
        name: build-files
        path: dist/

Event Type Trigger Use Case Example

2. Advanced Workflow Design & Configuration

Matrix Strategy for Multi-Environment Testing

Cross-Platform Testing Matrix
# .github/workflows/matrix-testing.yml
name: Cross-Platform Testing

on: [push, pull_request]

jobs:
  test:
    runs-on: ${{ matrix.os }}
    strategy:
      fail-fast: false
      matrix:
        os: [ubuntu-latest, windows-latest, macos-latest]
        node-version: [16, 18, 20]
        include:
          # Additional configurations
          - os: ubuntu-latest
            node-version: 20
            experimental: true
            coverage: true
        exclude:
          # Remove problematic combinations
          - os: windows-latest
            node-version: 16
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v4
    
    - name: Setup Node.js ${{ matrix.node-version }}
      uses: actions/setup-node@v4
      with:
        node-version: ${{ matrix.node-version }}
    
    - name: Install dependencies
      run: npm ci
    
    - name: Run tests
      run: npm test
      continue-on-error: ${{ matrix.experimental == true }}
    
    - name: Generate coverage report
      if: matrix.coverage == true
      run: npm run test:coverage
    
    - name: Upload coverage
      if: matrix.coverage == true
      uses: codecov/codecov-action@v3

Reusable Deployment Workflow
# .github/workflows/reusable-deploy.yml
name: Reusable Deployment

on:
  workflow_call:
    inputs:
      environment:
        description: 'Target environment'
        required: true
        type: string
      app-name:
        description: 'Application name'
        required: true
        type: string
      docker-image:
        description: 'Docker image to deploy'
        required: true
        type: string
      health-check-url:
        description: 'Health check endpoint'
        required: false
        type: string
        default: '/health'
    secrets:
      KUBE_CONFIG:
        description: 'Kubernetes config'
        required: true
      SLACK_WEBHOOK:
        description: 'Slack notification webhook'
        required: false
    outputs:
      deployment-url:
        description: 'Deployed application URL'
        value: ${{ jobs.deploy.outputs.app-url }}

jobs:
  deploy:
    runs-on: ubuntu-latest
    environment: ${{ inputs.environment }}
    outputs:
      app-url: ${{ steps.deploy.outputs.app-url }}
    
    steps:
    - name: Checkout repository
      uses: actions/checkout@v4
    
    - name: Configure kubectl
      uses: azure/k8s-set-context@v3
      with:
        method: kubeconfig
        kubeconfig: ${{ secrets.KUBE_CONFIG }}
    
    - name: Deploy to Kubernetes
      id: deploy
      run: |
        # Replace image in deployment manifest
        sed -i 's|IMAGE_PLACEHOLDER|${{ inputs.docker-image }}|g' k8s/deployment.yaml
        
        # Apply deployment
        kubectl apply -f k8s/
        
        # Wait for rollout
        kubectl rollout status deployment/${{ inputs.app-name }} -n ${{ inputs.environment }}
        
        # Get service URL
        APP_URL=$(kubectl get service ${{ inputs.app-name }} -n ${{ inputs.environment }} -o jsonpath='{.status.loadBalancer.ingress[0].hostname}')
        echo "app-url=${APP_URL}" >> $GITHUB_OUTPUT
    
    - name: Health check
      run: |
        for i in {1..30}; do
          if curl -f "http://${{ steps.deploy.outputs.app-url }}${{ inputs.health-check-url }}"; then
            echo "Health check passed"
            exit 0
          fi
          echo "Health check attempt $i failed, retrying..."
          sleep 10
        done
        echo "Health check failed after 30 attempts"
        exit 1
    
    - name: Notify Slack
      if: always() && secrets.SLACK_WEBHOOK
      uses: 8398a7/action-slack@v3
      with:
        status: ${{ job.status }}
        text: |
          Deployment to ${{ inputs.environment }}: ${{ job.status }}
          App: ${{ inputs.app-name }}
          URL: http://${{ steps.deploy.outputs.app-url }}
      env:
        SLACK_WEBHOOK_URL: ${{ secrets.SLACK_WEBHOOK }}

Security Best Practices

  • Pin actions to specific commit SHAs for security
  • Use OpenID Connect (OIDC) for cloud provider authentication
  • Enable secret scanning and push protection
  • Use environment protection rules for sensitive deployments
  • Regularly audit and rotate secrets

3. Jenkins Integration with GitHub Actions

Hybrid CI/CD Strategy

Combine GitHub Actions for lightweight tasks with Jenkins for complex, enterprise workflows.

When to Use Each Platform

GitHub Actions - Best For:

  • Simple CI/CD workflows
  • Open source projects
  • GitHub-native integrations
  • Matrix builds and parallel testing
  • Container-based workflows

Jenkins - Best For:

  • Complex enterprise workflows
  • Legacy system integration
  • Custom plugins and extensions
  • On-premises environments
  • Advanced pipeline orchestration

Jenkins Pipeline Integration

GitHub Actions → Jenkins Trigger
# .github/workflows/trigger-jenkins.yml
name: Trigger Jenkins Deployment

on:
  workflow_run:
    workflows: ["CI"]
    types: [completed]
    branches: [main]

jobs:
  trigger-jenkins:
    if: ${{ github.event.workflow_run.conclusion == 'success' }}
    runs-on: ubuntu-latest
    
    steps:
    - name: Trigger Jenkins job
      env:
        JENKINS_URL: ${{ secrets.JENKINS_URL }}
        JENKINS_USER: ${{ secrets.JENKINS_USER }}
        JENKINS_TOKEN: ${{ secrets.JENKINS_TOKEN }}
      run: |
        # Trigger Jenkins job with parameters
        curl -X POST \
          "${JENKINS_URL}/job/production-deployment/buildWithParameters" \
          --user "${JENKINS_USER}:${JENKINS_TOKEN}" \
          --data-urlencode "GIT_COMMIT=${GITHUB_SHA}" \
          --data-urlencode "GIT_BRANCH=${GITHUB_REF_NAME}" \
          --data-urlencode "BUILD_NUMBER=${GITHUB_RUN_NUMBER}" \
          --data-urlencode "TRIGGERED_BY=github-actions"
    
    - name: Wait for Jenkins job completion
      env:
        JENKINS_URL: ${{ secrets.JENKINS_URL }}
        JENKINS_USER: ${{ secrets.JENKINS_USER }}
        JENKINS_TOKEN: ${{ secrets.JENKINS_TOKEN }}
      run: |
        # Get latest build number
        BUILD_NUMBER=$(curl -s "${JENKINS_URL}/job/production-deployment/api/json" \
          --user "${JENKINS_USER}:${JENKINS_TOKEN}" | \
          jq '.nextBuildNumber - 1')
        
        # Poll for completion
        while true; do
          STATUS=$(curl -s "${JENKINS_URL}/job/production-deployment/${BUILD_NUMBER}/api/json" \
            --user "${JENKINS_USER}:${JENKINS_TOKEN}" | \
            jq -r '.result // "BUILDING"')
          
          if [ "$STATUS" = "SUCCESS" ]; then
            echo "Jenkins job completed successfully"
            exit 0
          elif [ "$STATUS" = "FAILURE" ]; then
            echo "Jenkins job failed"
            exit 1
          fi
          
          echo "Job status: $STATUS, waiting..."
          sleep 30
        done

Enterprise Jenkins Pipeline

Multi-Stage Production Pipeline

// Jenkinsfile
@Library('enterprise-shared-library@v2.1.0') _

pipeline {
    agent {
        kubernetes {
            yaml """
                apiVersion: v1
                kind: Pod
                spec:
                  containers:
                  - name: terraform
                    image: hashicorp/terraform:1.7
                    command: ['cat']
                    tty: true
                  - name: docker
                    image: docker:24-dind
                    securityContext:
                      privileged: true
                  - name: kubectl
                    image: bitnami/kubectl:latest
                    command: ['cat']
                    tty: true
                  - name: sonar
                    image: sonarsource/sonar-scanner-cli:latest
                    command: ['cat']
                    tty: true
            """
        }
    }
    
    parameters {
        choice(
            name: 'ENVIRONMENT',
            choices: ['staging', 'production'],
            description: 'Target deployment environment'
        )
        string(
            name: 'GIT_COMMIT',
            defaultValue: '',
            description: 'Git commit hash from GitHub Actions'
        )
        booleanParam(
            name: 'SKIP_TESTS',
            defaultValue: false,
            description: 'Skip test execution'
        )
    }
    
    environment {
        DOCKER_REGISTRY = credentials('docker-registry')
        AWS_DEFAULT_REGION = 'us-west-2'
        SONAR_TOKEN = credentials('sonar-token')
        SLACK_CHANNEL = '#deployments'
    }
    
    stages {
        stage('Checkout') {
            steps {
                script {
                    if (params.GIT_COMMIT) {
                        checkout([
                            $class: 'GitSCM',
                            branches: [[name: params.GIT_COMMIT]],
                            userRemoteConfigs: [[url: env.GIT_URL]]
                        ])
                    } else {
                        checkout scm
                    }
                }
            }
        }
        
        stage('Infrastructure Validation') {
            parallel {
                stage('Terraform Plan') {
                    steps {
                        container('terraform') {
                            script {
                                dir('terraform') {
                                    sh '''
                                        terraform init -backend-config="bucket=${S3_BACKEND_BUCKET}"
                                        terraform plan -out=tfplan -var="environment=${ENVIRONMENT}"
                                        terraform show -json tfplan > plan.json
                                    '''
                                    
                                    // Archive terraform plan
                                    archiveArtifacts artifacts: 'plan.json', allowEmptyArchive: false
                                }
                            }
                        }
                    }
                }
                
                stage('Security Scanning') {
                    steps {
                        container('docker') {
                            script {
                                sh '''
                                    # Build image for scanning
                                    docker build -t security-scan:${BUILD_NUMBER} .
                                    
                                    # Run Trivy scan
                                    docker run --rm -v /var/run/docker.sock:/var/run/docker.sock \
                                        aquasec/trivy image --format json \
                                        --output scan-results.json \
                                        security-scan:${BUILD_NUMBER}
                                '''
                                
                                publishHTML([
                                    allowMissing: false,
                                    alwaysLinkToLastBuild: true,
                                    keepAll: true,
                                    reportDir: '.',
                                    reportFiles: 'scan-results.json',
                                    reportName: 'Security Scan Report'
                                ])
                            }
                        }
                    }
                }
            }
        }
        
        stage('Quality Gate') {
            when {
                not { params.SKIP_TESTS }
            }
            steps {
                container('sonar') {
                    script {
                        sh '''
                            sonar-scanner \
                                -Dsonar.projectKey=${JOB_NAME} \
                                -Dsonar.sources=. \
                                -Dsonar.host.url=${SONAR_HOST_URL} \
                                -Dsonar.login=${SONAR_TOKEN}
                        '''
                        
                        // Wait for quality gate result
                        timeout(time: 10, unit: 'MINUTES') {
                            script {
                                def qualityGate = waitForQualityGate()
                                if (qualityGate.status != 'OK') {
                                    error "Pipeline aborted due to quality gate failure: ${qualityGate.status}"
                                }
                            }
                        }
                    }
                }
            }
        }
        
        stage('Build and Push') {
            steps {
                container('docker') {
                    script {
                        def image = docker.build("${DOCKER_REGISTRY}/${JOB_NAME}:${BUILD_NUMBER}")
                        
                        docker.withRegistry("https://${DOCKER_REGISTRY}", 'registry-credentials') {
                            image.push()
                            image.push("latest")
                        }
                        
                        // Set image tag for deployment
                        env.IMAGE_TAG = "${DOCKER_REGISTRY}/${JOB_NAME}:${BUILD_NUMBER}"
                    }
                }
            }
        }
        
        stage('Deploy Infrastructure') {
            when {
                anyOf {
                    branch 'main'
                    params.ENVIRONMENT == 'production'
                }
            }
            steps {
                container('terraform') {
                    script {
                        dir('terraform') {
                            sh '''
                                terraform apply -auto-approve tfplan
                                terraform output -json > outputs.json
                            '''
                            
                            // Store terraform outputs for deployment
                            archiveArtifacts artifacts: 'outputs.json', allowEmptyArchive: false
                        }
                    }
                }
            }
        }
        
        stage('Application Deployment') {
            steps {
                container('kubectl') {
                    script {
                        sh '''
                            # Update deployment manifest with new image
                            sed -i "s|IMAGE_PLACEHOLDER|${IMAGE_TAG}|g" k8s/deployment.yaml
                            
                            # Apply Kubernetes manifests
                            kubectl apply -f k8s/ -n ${ENVIRONMENT}
                            
                            # Wait for rollout to complete
                            kubectl rollout status deployment/${JOB_NAME} -n ${ENVIRONMENT} --timeout=300s
                        '''
                        
                        // Health check
                        retry(5) {
                            script {
                                sh '''
                                    HEALTH_URL=$(kubectl get service ${JOB_NAME} -n ${ENVIRONMENT} -o jsonpath='{.status.loadBalancer.ingress[0].hostname}')
                                    curl -f "http://${HEALTH_URL}/health" || exit 1
                                '''
                            }
                        }
                    }
                }
            }
        }
        
        stage('Post-Deployment Tests') {
            parallel {
                stage('Smoke Tests') {
                    steps {
                        script {
                            sh '''
                                # Run smoke tests against deployed application
                                npm run test:smoke -- --environment=${ENVIRONMENT}
                            '''
                        }
                    }
                }
                
                stage('Performance Tests') {
                    when {
                        params.ENVIRONMENT == 'staging'
                    }
                    steps {
                        script {
                            sh '''
                                # Run performance tests
                                npm run test:performance
                            '''
                            
                            publishHTML([
                                allowMissing: false,
                                alwaysLinkToLastBuild: true,
                                keepAll: true,
                                reportDir: 'performance-results',
                                reportFiles: 'report.html',
                                reportName: 'Performance Test Report'
                            ])
                        }
                    }
                }
            }
        }
    }
    
    post {
        always {
            // Clean up
            sh 'docker system prune -f || true'
        }
        
        success {
            script {
                slackSend(
                    channel: env.SLACK_CHANNEL,
                    color: 'good',
                    message: """
                        ✅ Deployment Successful!
                        Job: ${env.JOB_NAME}
                        Build: ${env.BUILD_NUMBER}
                        Environment: ${params.ENVIRONMENT}
                        Commit: ${params.GIT_COMMIT ?: env.GIT_COMMIT}
                    """
                )
            }
        }
        
        failure {
            script {
                slackSend(
                    channel: env.SLACK_CHANNEL,
                    color: 'danger',
                    message: """
                        ❌ Deployment Failed!
                        Job: ${env.JOB_NAME}
                        Build: ${env.BUILD_NUMBER}
                        Environment: ${params.ENVIRONMENT}
                        Build URL: ${env.BUILD_URL}
                    """
                )
            }
        }
        
        unstable {
            script {
                slackSend(
                    channel: env.SLACK_CHANNEL,
                    color: 'warning',
                    message: """
                        ⚠️ Deployment Unstable!
                        Job: ${env.JOB_NAME}
                        Build: ${env.BUILD_NUMBER}
                        Please check the build logs.
                    """
                )
            }
        }
    }
}

Automated Terraform Deployment
# .github/workflows/terraform.yml
name: Terraform Infrastructure

on:
  push:
    branches: [main]
    paths: ['terraform/**']
  pull_request:
    paths: ['terraform/**']
  workflow_dispatch:
    inputs:
      action:
        description: 'Terraform action to perform'
        required: true
        type: choice
        options:
          - plan
          - apply
          - destroy
      environment:
        description: 'Target environment'
        required: true
        type: choice
        options:
          - staging
          - production

env:
  TF_VERSION: '1.7.0'
  AWS_REGION: 'us-west-2'

jobs:
  terraform-validate:
    runs-on: ubuntu-latest
    
    steps:
    - name: Checkout repository
      uses: actions/checkout@v4
    
    - name: Setup Terraform
      uses: hashicorp/setup-terraform@v3
      with:
        terraform_version: ${{ env.TF_VERSION }}
    
    - name: Terraform Format Check
      run: terraform fmt -check -recursive terraform/
    
    - name: Terraform Validate
      run: |
        cd terraform/environments/staging
        terraform init -backend=false
        terraform validate

  terraform-plan:

      continue-on-error: true
    
    - name: Generate Plan Summary
      working-directory: terraform/environments/${{ matrix.environment }}
      run: |
        terraform show -no-color tfplan > plan_output.txt

        cat << EOF > plan_summary.md

        \`\`\`

# .github/workflows/production-pipeline.yml
name: Production Deployment Pipeline

on:
  push:
    branches: [main]
  release:
    types: [published]

env:
  REGISTRY: ghcr.io
  IMAGE_NAME: ${{ github.repository }}
  AWS_REGION: us-west-2

jobs:
  # Stage 1: Code Quality and Security
  quality-gate:
    runs-on: ubuntu-latest
    outputs:
      version: ${{ steps.version.outputs.version }}
      should-deploy: ${{ steps.changes.outputs.should-deploy }}
    
    steps:
    - name: Checkout repository
      uses: actions/checkout@v4
      with:
        fetch-depth: 0
    
    - name: Detect changes
      id: changes
      uses: dorny/paths-filter@v2
      with:
        filters: |
          app:
            - 'src/**'
            - 'Dockerfile'
            - 'go.mod'
            - 'go.sum'
          infrastructure:
            - 'terraform/**'
          k8s:
            - 'k8s/**'
    
    - name: Generate version
      id: version
      run: |
        if [[ "${{ github.event_name }}" == "release" ]]; then
          echo "version=${GITHUB_REF#refs/tags/}" >> $GITHUB_OUTPUT
        else
          echo "version=${GITHUB_SHA:0:8}" >> $GITHUB_OUTPUT
        fi
    
    - name: Setup Go
      uses: actions/setup-go@v4
      with:
        go-version: '1.21'
    
    - name: Cache Go modules
      uses: actions/cache@v3
      with:
        path: ~/go/pkg/mod
        key: ${{ runner.os }}-go-${{ hashFiles('**/go.sum') }}
    
    - name: Run tests
      run: |
        go test -v -race -coverprofile=coverage.out ./...
        go tool cover -html=coverage.out -o coverage.html
    
    - name: Run security scan
      uses: securecodewarrior/github-action-gosec@master
      with:
        args: './...'
    
    - name: Upload coverage
      uses: codecov/codecov-action@v3
    
    - name: Set deployment flag
      id: deploy-flag
      run: |
        if [[ "${{ steps.changes.outputs.app }}" == "true" || "${{ github.event_name }}" == "release" ]]; then
          echo "should-deploy=true" >> $GITHUB_OUTPUT
        else
          echo "should-deploy=false" >> $GITHUB_OUTPUT
        fi

  # Stage 2: Infrastructure Validation
  infrastructure-plan:
    if: needs.quality-gate.outputs.should-deploy == 'true'
    needs: quality-gate
    runs-on: ubuntu-latest
    permissions:
      id-token: write
      contents: read
    
    strategy:
      matrix:
        environment: [staging, production]
    
    steps:
    - name: Checkout repository
      uses: actions/checkout@v4
    
    - name: Setup Terraform
      uses: hashicorp/setup-terraform@v3
      with:
        terraform_version: '1.7.0'
    
    - name: Configure AWS credentials
      uses: aws-actions/configure-aws-credentials@v4
      with:
        role-to-assume: ${{ secrets[format('AWS_ROLE_ARN_{0}', matrix.environment)] }}
        aws-region: ${{ env.AWS_REGION }}
    
    - name: Terraform Plan
      working-directory: terraform/environments/${{ matrix.environment }}
      run: |
        terraform init
        terraform plan -out=tfplan

  # Stage 3: Build and Security Scan
  build-and-scan:
    if: needs.quality-gate.outputs.should-deploy == 'true'
    needs: quality-gate
    runs-on: ubuntu-latest
    permissions:
      contents: read
      packages: write
      security-events: write
    
    outputs:
      image-digest: ${{ steps.build.outputs.digest }}
      image-tags: ${{ steps.meta.outputs.tags }}
    
    steps:
    - name: Checkout repository
      uses: actions/checkout@v4
    
    - name: Set up Docker Buildx
      uses: docker/setup-buildx-action@v3
    
    - name: Log in to Container Registry
      uses: docker/login-action@v3
      with:
        registry: ${{ env.REGISTRY }}
        username: ${{ github.actor }}
        password: ${{ secrets.GITHUB_TOKEN }}
    
    - name: Extract metadata
      id: meta
      uses: docker/metadata-action@v5
      with:
        images: ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}
        tags: |
          type=ref,event=branch
          type=ref,event=pr
          type=semver,pattern={{version}}
          type=raw,value=${{ needs.quality-gate.outputs.version }}
          type=raw,value=latest,enable={{is_default_branch}}
    
    - name: Build and push Docker image
      id: build
      uses: docker/build-push-action@v5
      with:
        context: .
        platforms: linux/amd64,linux/arm64
        push: true
        target: production
        tags: ${{ steps.meta.outputs.tags }}
        labels: ${{ steps.meta.outputs.labels }}
        cache-from: type=gha
        cache-to: type=gha,mode=max
        provenance: true
        sbom: true
    
    - name: Run Trivy vulnerability scanner
      uses: aquasecurity/trivy-action@master
      with:
        image-ref: ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:${{ needs.quality-gate.outputs.version }}
        format: 'sarif'
        output: 'trivy-results.sarif'
    
    - name: Upload Trivy scan results
      uses: github/codeql-action/upload-sarif@v3
      with:
        sarif_file: 'trivy-results.sarif'

  # Stage 4: Deploy to Staging
  deploy-staging:
    needs: [quality-gate, infrastructure-plan, build-and-scan]
    runs-on: ubuntu-latest
    environment: staging
    permissions:
      id-token: write
      contents: read
    
    steps:
    - name: Checkout k8s manifests
      uses: actions/checkout@v4
      with:
        repository: company/k8s-manifests
        token: ${{ secrets.K8S_REPO_TOKEN }}
        path: k8s-manifests
    
    - name: Configure AWS credentials
      uses: aws-actions/configure-aws-credentials@v4
      with:
        role-to-assume: ${{ secrets.AWS_ROLE_ARN_STAGING }}
        aws-region: ${{ env.AWS_REGION }}
    
    - name: Update image in GitOps repo
      working-directory: k8s-manifests
      run: |
        # Update image tag in staging overlay
        cd applications/web-app/overlays/staging
        kustomize edit set image ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:${{ needs.quality-gate.outputs.version }}
        
        # Commit and push changes
        git config user.name "github-actions[bot]"
        git config user.email "github-actions[bot]@users.noreply.github.com"
        git add .
        git commit -m "Update staging image to ${{ needs.quality-gate.outputs.version }}"
        git push origin main
    
    - name: Wait for ArgoCD sync
      run: |
        # Install ArgoCD CLI
        curl -sSL -o /usr/local/bin/argocd https://github.com/argoproj/argo-cd/releases/latest/download/argocd-linux-amd64
        chmod +x /usr/local/bin/argocd
        
        # Login to ArgoCD
        argocd login ${{ secrets.ARGOCD_SERVER }} --username ${{ secrets.ARGOCD_USERNAME }} --password ${{ secrets.ARGOCD_PASSWORD }}
        
        # Sync and wait
        argocd app sync web-app-staging
        argocd app wait web-app-staging --timeout 600

  # Stage 5: Integration Tests
  integration-tests:
    needs: deploy-staging
    runs-on: ubuntu-latest
    
    steps:
    - name: Checkout repository
      uses: actions/checkout@v4
    
    - name: Run integration tests
      run: |
        # Run integration tests against staging
        npm run test:integration -- --base-url=${{ secrets.STAGING_BASE_URL }}
    
    - name: Run performance tests
      run: |
        # Run k6 performance tests
        docker run --rm -v $(pwd)/tests:/tests grafana/k6 run /tests/performance.js
    
    - name: Security penetration test
      run: |
        # Run OWASP ZAP scan
        docker run -v $(pwd):/zap/wrk/:rw -t owasp/zap2docker-stable zap-baseline.py \
          -t ${{ secrets.STAGING_BASE_URL }} -J zap-report.json

  # Stage 6: Production Deployment
  deploy-production:
    if: github.event_name == 'release'
    needs: [quality-gate, build-and-scan, integration-tests]
    runs-on: ubuntu-latest
    environment: production
    permissions:
      id-token: write
      contents: read
    
    steps:
    - name: Checkout k8s manifests
      uses: actions/checkout@v4
