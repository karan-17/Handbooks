Data Cleaning Overview

Effective data cleaning and preprocessing are critical foundations for successful RAG systems. Poor quality input data leads to reduced retrieval accuracy, irrelevant responses, and degraded user experience. This chapter covers comprehensive techniques for transforming raw documents into clean, structured, and retrieval-optimized content.

Why Data Cleaning Matters

  • Improves semantic similarity matching
  • Reduces noise in vector embeddings
  • Enhances retrieval precision
  • Standardizes content format
  • Removes irrelevant metadata

Common Data Issues

  • OCR errors and garbled text
  • Inconsistent formatting
  • HTML/XML artifacts
  • Duplicate content
  • Encoding problems

Advanced Text Cleaning Techniques

1. HTML and Markup Removal

Web-scraped content often contains HTML tags, CSS, and JavaScript that can interfere with semantic understanding. Here's a comprehensive cleaning approach:

HTML Cleaning Implementation
import re
import html
from bs4 import BeautifulSoup
from markdownify import markdownify
import unicodedata

class AdvancedTextCleaner:
    def __init__(self):
        # Compile regex patterns for efficiency
        self.url_pattern = re.compile(r'http[s]?://(?:[a-zA-Z]|[0-9]|[$-_@.&+]|[!*\\(\\),]|(?:%[0-9a-fA-F][0-9a-fA-F]))+')
        self.email_pattern = re.compile(r'\S+@\S+\.\S+')
        self.phone_pattern = re.compile(r'\b\d{3}[-.]?\d{3}[-.]?\d{4}\b')
        self.whitespace_pattern = re.compile(r'\s+')
        self.special_chars = re.compile(r'[^\w\s\.\,\!\?\;\:\(\)\-\'\"]')
        
    def clean_html_content(self, html_content: str, preserve_structure: bool = True) -> str:
        """
        Clean HTML content while preserving semantic structure
        """
        try:
            # Parse HTML
            soup = BeautifulSoup(html_content, 'html.parser')
            
            # Remove script and style elements
            for script in soup(["script", "style", "meta", "link", "noscript"]):
                script.decompose()
            
            # Remove comments
            for comment in soup.find_all(string=lambda text: isinstance(text, Comment)):
                comment.extract()
            
            # Handle specific tags
            self._process_semantic_tags(soup)
            
            if preserve_structure:
                # Convert to markdown to preserve structure
                text = markdownify(str(soup), heading_style="ATX")
                # Clean up markdown artifacts
                text = re.sub(r'\n\s*\n\s*\n', '\n\n', text)  # Remove excess newlines
            else:
                # Extract plain text
                text = soup.get_text()
            
            return self._post_process_text(text)
            
        except Exception as e:
            # Fallback to regex cleaning
            return self._fallback_html_clean(html_content)
    
    def _process_semantic_tags(self, soup):
        """Process semantic HTML tags to preserve meaning"""
        # Convert headers to consistent format
        for header in soup.find_all(['h1', 'h2', 'h3', 'h4', 'h5', 'h6']):
            level = int(header.name[1])
            header.string = f"{'#' * level} {header.get_text().strip()}"
        
        # Handle lists
        for ul in soup.find_all('ul'):
            items = ul.find_all('li')
            list_text = '\n'.join([f"• {item.get_text().strip()}" for item in items])
            ul.replace_with(list_text + '\n')
        
        for ol in soup.find_all('ol'):
            items = ol.find_all('li')
            list_text = '\n'.join([f"{i+1}. {item.get_text().strip()}" 
                                  for i, item in enumerate(items)])
            ol.replace_with(list_text + '\n')
    
    def _fallback_html_clean(self, text: str) -> str:
        """Regex-based HTML cleaning as fallback"""
        # Remove HTML tags
        text = re.sub(r'<[^>]+>', ' ', text)
        # Decode HTML entities
        text = html.unescape(text)
        return self._post_process_text(text)
    
    def _post_process_text(self, text: str) -> str:
        """Final text cleaning steps"""
        # Normalize whitespace
        text = self.whitespace_pattern.sub(' ', text)
        # Remove excessive punctuation
        text = re.sub(r'[\.]{3,}', '...', text)
        text = re.sub(r'[!]{2,}', '!', text)
        text = re.sub(r'[\?]{2,}', '?', text)
        # Clean up spacing
        text = text.strip()
        return text

2. OCR Error Correction

OCR-processed documents often contain character recognition errors that can significantly impact retrieval quality. Here's an advanced correction system:

OCR Error Correction
import nltk
from spellchecker import SpellChecker
from difflib import SequenceMatcher
import language_tool_python

class OCRCorrector:
    def __init__(self, custom_vocabulary: list = None):
        self.spell_checker = SpellChecker()
        self.grammar_tool = language_tool_python.LanguageTool('en-US')
        
        # Add custom vocabulary for domain-specific terms
        if custom_vocabulary:
            self.spell_checker.word_frequency.load_words(custom_vocabulary)
        
        # Common OCR character mappings
        self.ocr_corrections = {
            'rn': 'm', 'vv': 'w', 'cl': 'd', 'ii': 'll', 'Ii': 'Il',
            '0': 'O', '5': 'S', '1': 'l', '6': 'G', '8': 'B',
            'ﬁ': 'fi', 'ﬂ': 'fl', ''': "'", '"': '"', '"': '"'
        }
        
    def correct_ocr_errors(self, text: str, confidence_threshold: float = 0.7) -> str:
        """
        Correct OCR errors using multiple strategies
        """
        # Stage 1: Character-level corrections
        text = self._fix_character_errors(text)
        
        # Stage 2: Word-level corrections
        text = self._fix_word_errors(text, confidence_threshold)
        
        # Stage 3: Context-aware corrections
        text = self._fix_contextual_errors(text)
        
        return text
    
    def _fix_character_errors(self, text: str) -> str:
        """Fix common OCR character substitution errors"""
        for error, correction in self.ocr_corrections.items():
            text = text.replace(error, correction)
        
        # Fix common spacing issues
        text = re.sub(r'\b(\w)\s+(\w)\b', r'\1\2', text)  # Remove spaces in words
        text = re.sub(r'(\w)([\.,:;!?])', r'\1 \2', text)  # Add space before punctuation
        
        return text
    
    def _fix_word_errors(self, text: str, threshold: float) -> str:
        """Fix word-level spelling errors"""
        words = text.split()
        corrected_words = []
        
        for word in words:
            # Skip words with numbers or special characters
            if not word.isalpha():
                corrected_words.append(word)
                continue
            
            # Check if word needs correction
            if word.lower() not in self.spell_checker:
                candidates = self.spell_checker.candidates(word)
                if candidates:
                    # Find best match based on similarity
                    best_match = max(candidates, 
                                   key=lambda x: SequenceMatcher(None, word.lower(), x).ratio())
                    similarity = SequenceMatcher(None, word.lower(), best_match).ratio()
                    
                    if similarity >= threshold:
                        # Preserve original case
                        corrected_word = self._preserve_case(word, best_match)
                        corrected_words.append(corrected_word)
                    else:
                        corrected_words.append(word)
                else:
                    corrected_words.append(word)
            else:
                corrected_words.append(word)
        
        return ' '.join(corrected_words)
    
    def _fix_contextual_errors(self, text: str) -> str:
        """Fix errors using grammatical context"""
        try:
            matches = self.grammar_tool.check(text)
            corrected_text = language_tool_python.utils.correct(text, matches)
            return corrected_text
        except Exception:
            return text
    
    def _preserve_case(self, original: str, correction: str) -> str:
        """Preserve the case pattern of the original word"""
        if original.isupper():
            return correction.upper()
        elif original.istitle():
            return correction.capitalize()
        else:
            return correction.lower()

OCR Correction Best Practices

• Use domain-specific dictionaries for technical content
• Validate corrections manually for critical documents
• Consider confidence scores when applying corrections
• Preserve original formatting and structure when possible

Text Normalization and Standardization

1. Unicode Normalization

Unicode normalization ensures consistent character representation across different sources and encoding schemes:

Unicode and Encoding Normalization
import unicodedata
import ftfy
from unidecode import unidecode

class UnicodeNormalizer:
    def __init__(self, target_encoding='utf-8'):
        self.target_encoding = target_encoding
        
    def normalize_unicode(self, text: str, form: str = 'NFC') -> str:
        """
        Normalize Unicode characters to a consistent form
        
        Forms:
        - NFC: Canonical Decomposition, followed by Canonical Composition
        - NFD: Canonical Decomposition
        - NFKC: Compatibility Decomposition, followed by Canonical Composition
        - NFKD: Compatibility Decomposition
        """
        # Fix encoding issues first
        text = ftfy.fix_text(text)
        
        # Normalize Unicode
        text = unicodedata.normalize(form, text)
        
        return text
    
    def handle_special_characters(self, text: str, strategy: str = 'preserve') -> str:
        """
        Handle special characters and accents
        
        Strategies:
        - preserve: Keep original characters
        - transliterate: Convert to ASCII equivalents
        - remove: Remove non-ASCII characters
        """
        if strategy == 'transliterate':
            # Convert accented characters to ASCII
            text = unidecode(text)
        elif strategy == 'remove':
            # Remove non-ASCII characters
            text = ''.join(char for char in text if ord(char) < 128)
        elif strategy == 'preserve':
            # Keep original but ensure proper encoding
            text = text.encode(self.target_encoding, 'ignore').decode(self.target_encoding)
        
        return text
    
    def standardize_quotation_marks(self, text: str) -> str:
        """Standardize different types of quotation marks"""
        quote_mappings = {
            '"': '"', '"': '"',  # Smart quotes to straight quotes
            ''': "'", ''': "'",  # Smart apostrophes
            '‚': "'", '„': '"',  # Various quote styles
            '«': '"', '»': '"',  # French quotes
            '‹': "'", '›': "'"   # Single French quotes
        }
        
        for old_quote, new_quote in quote_mappings.items():
            text = text.replace(old_quote, new_quote)
        
        return text
    
    def normalize_whitespace(self, text: str) -> str:
        """Normalize different types of whitespace characters"""
        # Replace various whitespace characters with regular space
        whitespace_chars = [
            '\u00A0',  # Non-breaking space
            '\u2007',  # Figure space
            '\u200B',  # Zero-width space
            '\u2060',  # Word joiner
            '\uFEFF',  # Zero-width no-break space
        ]
        
        for ws_char in whitespace_chars:
            text = text.replace(ws_char, ' ')
        
        # Normalize line endings
        text = text.replace('\r\n', '\n').replace('\r', '\n')
        
        # Remove excessive whitespace
        text = re.sub(r'\n\s*\n\s*\n', '\n\n', text)  # Multiple newlines to double
        text = re.sub(r' +', ' ', text)  # Multiple spaces to single
        
        return text.strip()

2. Content Structure Normalization

Consistent content structure improves chunking and retrieval performance:

Structure Normalization
import re
from typing import Dict, List, Tuple

class ContentStructureNormalizer:
    def __init__(self):
        self.section_patterns = {
            'heading': re.compile(r'^#{1,6}\s+.+$', re.MULTILINE),
            'list_item': re.compile(r'^\s*[•\-\*]\s+.+$', re.MULTILINE),
            'numbered_item': re.compile(r'^\s*\d+\.\s+.+$', re.MULTILINE),
            'table_row': re.compile(r'\|.+\|', re.MULTILINE)
        }
    
    def normalize_headings(self, text: str) -> str:
        """Standardize heading formats"""
        lines = text.split('\n')
        normalized_lines = []
        
        for line in lines:
            stripped = line.strip()
            
            # Convert underlined headings to markdown style
            if len(stripped) > 0 and all(c in '=-~^' for c in stripped):
                if normalized_lines and normalized_lines[-1].strip():
                    # Previous line is the heading text
                    heading_text = normalized_lines[-1].strip()
                    level = 1 if '=' in stripped else 2
                    normalized_lines[-1] = f"{'#' * level} {heading_text}"
                    continue
            
            # Standardize markdown headings spacing
            heading_match = re.match(r'^(#{1,6})\s*(.+)$', stripped)
            if heading_match:
                level = len(heading_match.group(1))
                title = heading_match.group(2).strip()
                normalized_lines.append(f"{'#' * level} {title}")
            else:
                normalized_lines.append(line)
        
        return '\n'.join(normalized_lines)
    
    def normalize_lists(self, text: str) -> str:
        """Standardize list formatting"""
        lines = text.split('\n')
        normalized_lines = []
        in_list = False
        list_level = 0
        
        for line in lines:
            stripped = line.strip()
            leading_spaces = len(line) - len(line.lstrip())
            
            # Detect list items
            bullet_match = re.match(r'^[•\-\*\+]\s+(.+)$', stripped)
            numbered_match = re.match(r'^(\d+)\.\s+(.+)$', stripped)
            
            if bullet_match:
                content = bullet_match.group(1)
                current_level = leading_spaces // 2  # Estimate indentation level
                normalized_lines.append('  ' * current_level + f'• {content}')
                in_list = True
            elif numbered_match:
                content = numbered_match.group(2)
                current_level = leading_spaces // 2
                normalized_lines.append('  ' * current_level + f'1. {content}')
                in_list = True
            else:
                if in_list and not stripped:
                    # Empty line in list
                    normalized_lines.append('')
                elif in_list and stripped:
                    # End of list
                    in_list = False
                    normalized_lines.append('')
                    normalized_lines.append(line)
                else:
                    normalized_lines.append(line)
        
        return '\n'.join(normalized_lines)
    
    def normalize_tables(self, text: str) -> str:
        """Normalize table formatting"""
        lines = text.split('\n')
        normalized_lines = []
        in_table = False
        
        for line in lines:
            # Detect table rows (containing pipes)
            if '|' in line and line.strip():
                # Clean up table formatting
                parts = [part.strip() for part in line.split('|')]
                # Remove empty first/last parts if they exist
                if parts and not parts[0]:
                    parts = parts[1:]
                if parts and not parts[-1]:
                    parts = parts[:-1]
                
                if parts:  # Valid table row
                    formatted_line = '| ' + ' | '.join(parts) + ' |'
                    normalized_lines.append(formatted_line)
                    in_table = True
                else:
                    normalized_lines.append(line)
            else:
                if in_table and not line.strip():
                    # End of table
                    in_table = False
                normalized_lines.append(line)
        
        return '\n'.join(normalized_lines)
    
    def extract_metadata(self, text: str) -> Tuple[str, Dict]:
        """Extract and normalize metadata from document"""
        metadata = {}
        content_lines = []
        
        lines = text.split('\n')
        in_frontmatter = False
        frontmatter_start = False
        
        for line in lines:
            stripped = line.strip()
            
            # YAML frontmatter detection
            if stripped == '---' and not frontmatter_start:
                in_frontmatter = True
                frontmatter_start = True
                continue
            elif stripped == '---' and in_frontmatter:
                in_frontmatter = False
                continue
            
            if in_frontmatter:
                # Parse YAML-like metadata
                if ':' in stripped:
                    key, value = stripped.split(':', 1)
                    metadata[key.strip()] = value.strip()
            else:
                content_lines.append(line)
        
        # Clean content
        clean_content = '\n'.join(content_lines).strip()
        
        return clean_content, metadata

Complete Cleaning Pipeline Implementation

Here's a comprehensive implementation that combines all cleaning techniques into a production-ready pipeline:

Complete Data Cleaning Pipeline
from typing import Dict, List, Optional, Callable
import logging
from dataclasses import dataclass
from pathlib import Path
import json

@dataclass
class CleaningConfig:
    """Configuration for the cleaning pipeline"""
    preserve_structure: bool = True
    correct_ocr: bool = True
    normalize_unicode: bool = True
    unicode_form: str = 'NFC'
    character_strategy: str = 'preserve'  # preserve, transliterate, remove
    confidence_threshold: float = 0.7
    custom_vocabulary: Optional[List[str]] = None
    remove_urls: bool = False
    remove_emails: bool = False
    remove_phone_numbers: bool = False
    min_sentence_length: int = 10
    max_sentence_length: int = 1000

class DocumentCleaningPipeline:
    def __init__(self, config: CleaningConfig = None):
        self.config = config or CleaningConfig()
        self.logger = logging.getLogger(__name__)
        
        # Initialize components
        self.text_cleaner = AdvancedTextCleaner()
        self.ocr_corrector = OCRCorrector(self.config.custom_vocabulary) if self.config.correct_ocr else None
        self.unicode_normalizer = UnicodeNormalizer()
        self.structure_normalizer = ContentStructureNormalizer()
        
        # Statistics tracking
        self.stats = {
            'documents_processed': 0,
            'total_characters_before': 0,
            'total_characters_after': 0,
            'corrections_applied': 0
        }
    
    def clean_document(self, content: str, metadata: Dict = None) -> Dict:
        """
        Clean a single document through the complete pipeline
        """
        if metadata is None:
            metadata = {}
        
        original_length = len(content)
        self.stats['total_characters_before'] += original_length
        
        try:
            # Stage 1: Initial cleaning
            cleaned_content = self._stage_1_initial_cleaning(content)
            
            # Stage 2: Structure normalization
            cleaned_content = self._stage_2_structure_normalization(cleaned_content)
            
            # Stage 3: Content normalization
            cleaned_content = self._stage_3_content_normalization(cleaned_content)
            
            # Stage 4: OCR correction (if enabled)
            if self.config.correct_ocr and self.ocr_corrector:
                cleaned_content = self._stage_4_ocr_correction(cleaned_content)
            
            # Stage 5: Final validation and filtering
            cleaned_content = self._stage_5_final_validation(cleaned_content)
            
            # Extract metadata
            cleaned_content, extracted_metadata = self.structure_normalizer.extract_metadata(cleaned_content)
            metadata.update(extracted_metadata)
            
            # Update statistics
            self.stats['documents_processed'] += 1
            self.stats['total_characters_after'] += len(cleaned_content)
            
            return {
                'content': cleaned_content,
                'metadata': metadata,
                'cleaning_stats': {
                    'original_length': original_length,
                    'cleaned_length': len(cleaned_content),
                    'compression_ratio': len(cleaned_content) / original_length if original_length > 0 else 0
                }
            }
            
        except Exception as e:
            self.logger.error(f"Error cleaning document: {str(e)}")
            return {
                'content': content,  # Return original on error
                'metadata': metadata,
                'cleaning_stats': {
                    'error': str(e),
                    'original_length': original_length,
                    'cleaned_length': original_length,
                    'compression_ratio': 1.0
                }
            }
    
    def _stage_1_initial_cleaning(self, content: str) -> str:
        """Stage 1: Remove HTML, basic cleanup"""
        # HTML cleaning
        content = self.text_cleaner.clean_html_content(
            content, 
            preserve_structure=self.config.preserve_structure
        )
        
        # URL/Email/Phone removal if configured
        if self.config.remove_urls:
            content = self.text_cleaner.url_pattern.sub('', content)
        if self.config.remove_emails:
            content = self.text_cleaner.email_pattern.sub('', content)
        if self.config.remove_phone_numbers:
            content = self.text_cleaner.phone_pattern.sub('', content)
        
        return content
    
    def _stage_2_structure_normalization(self, content: str) -> str:
        """Stage 2: Normalize document structure"""
        # Normalize headings
        content = self.structure_normalizer.normalize_headings(content)
        
        # Normalize lists
        content = self.structure_normalizer.normalize_lists(content)
        
        # Normalize tables
        content = self.structure_normalizer.normalize_tables(content)
        
        return content
    
    def _stage_3_content_normalization(self, content: str) -> str:
        """Stage 3: Unicode and character normalization"""
        if self.config.normalize_unicode:
            # Unicode normalization
            content = self.unicode_normalizer.normalize_unicode(
                content, 
                form=self.config.unicode_form
            )
            
            # Handle special characters
            content = self.unicode_normalizer.handle_special_characters(
                content, 
                strategy=self.config.character_strategy
            )
            
            # Standardize quotes and whitespace
            content = self.unicode_normalizer.standardize_quotation_marks(content)
            content = self.unicode_normalizer.normalize_whitespace(content)
        
        return content
    
    def _stage_4_ocr_correction(self, content: str) -> str:
        """Stage 4: OCR error correction"""
        try:
            corrected = self.ocr_corrector.correct_ocr_errors(
                content, 
                confidence_threshold=self.config.confidence_threshold
            )
            return corrected
        except Exception as e:
            self.logger.warning(f"OCR correction failed: {str(e)}")
            return content
    
    def _stage_5_final_validation(self, content: str) -> str:
        """Stage 5: Final validation and filtering"""
        lines = content.split('\n')
        valid_lines = []
        
        for line in lines:
            line = line.strip()
            
            # Skip empty lines (but preserve paragraph breaks)
            if not line:
                if valid_lines and valid_lines[-1] != '':
                    valid_lines.append('')
                continue
            
            # Filter by length
            if (len(line) >= self.config.min_sentence_length and 
                len(line) <= self.config.max_sentence_length):
                valid_lines.append(line)
        
        # Remove trailing empty lines
        while valid_lines and valid_lines[-1] == '':
            valid_lines.pop()
        
        return '\n'.join(valid_lines)
    
    def get_statistics(self) -> Dict:
        """Get pipeline statistics"""
        if self.stats['documents_processed'] > 0:
            avg_compression = (self.stats['total_characters_after'] / 
                             self.stats['total_characters_before'])
        else:
            avg_compression = 0
        
        return {
            **self.stats,
            'average_compression_ratio': avg_compression,
            'total_size_reduction': (self.stats['total_characters_before'] - 
                                   self.stats['total_characters_after'])
        }

# Usage example
def clean_document_collection(documents: List[Dict], config: CleaningConfig = None) -> List[Dict]:
    """
    Clean a collection of documents
    """
    pipeline = DocumentCleaningPipeline(config)
    cleaned_documents = []
    
    for doc in documents:
        content = doc.get('content', '')
        metadata = doc.get('metadata', {})
        
        cleaned_doc = pipeline.clean_document(content, metadata)
        cleaned_documents.append(cleaned_doc)
    
    # Log statistics
    stats = pipeline.get_statistics()
    logging.info(f"Cleaned {stats['documents_processed']} documents")
    logging.info(f"Average compression ratio: {stats['average_compression_ratio']:.2f}")
    
    return cleaned_documents

Quality Assessment and Validation

Implement quality assessment to ensure cleaning effectiveness:

Quality Assessment Framework
class CleaningQualityAssessor:
    def __init__(self):
        self.metrics = {}
    
    def assess_cleaning_quality(self, original: str, cleaned: str) -> Dict:
        """Assess the quality of cleaning operations"""
        return {
            'content_preservation': self._assess_content_preservation(original, cleaned),
            'readability_improvement': self._assess_readability(original, cleaned),
            'structure_quality': self._assess_structure_quality(cleaned),
            'encoding_quality': self._assess_encoding_quality(cleaned),
            'overall_score': self._calculate_overall_score(original, cleaned)
        }
    
    def _assess_content_preservation(self, original: str, cleaned: str) -> float:
        """Measure how well important content is preserved"""
        # Compare word overlap
        original_words = set(original.lower().split())
        cleaned_words = set(cleaned.lower().split())
        
        if len(original_words) == 0:
            return 1.0
        
        overlap = len(original_words.intersection(cleaned_words))
        return overlap / len(original_words)
    
    def _assess_readability(self, original: str, cleaned: str) -> float:
        """Assess readability improvement"""
        # Simple readability metrics
        original_sentences = len([s for s in original.split('.') if s.strip()])
        cleaned_sentences = len([s for s in cleaned.split('.') if s.strip()])
        
        original_avg_length = len(original) / max(original_sentences, 1)
        cleaned_avg_length = len(cleaned) / max(cleaned_sentences, 1)
        
        # Prefer moderate sentence lengths (50-150 characters)
        optimal_range = (50, 150)
        
        def score_length(avg_len):
            if optimal_range[0] <= avg_len <= optimal_range[1]:
                return 1.0
            elif avg_len < optimal_range[0]:
                return avg_len / optimal_range[0]
            else:
                return optimal_range[1] / avg_len
        
        return score_length(cleaned_avg_length) - score_length(original_avg_length) + 1.0
    
    def _assess_structure_quality(self, text: str) -> float:
        """Assess structural quality of cleaned text"""
        score = 0.0
        
        # Check for proper heading structure
        lines = text.split('\n')
        has_headings = any(line.strip().startswith('#') for line in lines)
        if has_headings:
            score += 0.3
        
        # Check for proper list formatting
        has_lists = any(re.match(r'^\s*[•\-\*]\s+', line) for line in lines)
        if has_lists:
            score += 0.2
        
        # Check for consistent paragraph breaks
        paragraph_breaks = text.count('\n\n')
        total_lines = len(lines)
        if total_lines > 0 and paragraph_breaks / total_lines > 0.1:
            score += 0.3
        
        # Check for balanced text distribution
        avg_line_length = sum(len(line) for line in lines) / max(len(lines), 1)
        if 30 <= avg_line_length <= 200:
            score += 0.2
        
        return min(score, 1.0)
    
    def _assess_encoding_quality(self, text: str) -> float:
        """Assess encoding and character quality"""
        # Count problematic characters
        total_chars = len(text)
        if total_chars == 0:
            return 1.0
        
        problematic_chars = 0
        
        # Count encoding artifacts
        problematic_chars += text.count('�')  # Replacement character
        problematic_chars += text.count('?')  # Question marks (potential encoding issues)
        problematic_chars += len(re.findall(r'&[a-zA-Z]+;', text))  # HTML entities
        
        # Count excessive whitespace
        excessive_spaces = len(re.findall(r'  +', text))  # Multiple spaces
        excessive_newlines = len(re.findall(r'\n\n\n+', text))  # Multiple newlines
        
        problematic_chars += excessive_spaces + excessive_newlines
        
        return max(0.0, 1.0 - (problematic_chars / total_chars))
    
    def _calculate_overall_score(self, original: str, cleaned: str) -> float:
        """Calculate overall cleaning quality score"""
        preservation = self._assess_content_preservation(original, cleaned)
        readability = self._assess_readability(original, cleaned)
        structure = self._assess_structure_quality(cleaned)
        encoding = self._assess_encoding_quality(cleaned)
        
        # Weighted average
        weights = {
            'preservation': 0.4,
            'readability': 0.2,
            'structure': 0.2,
            'encoding': 0.2
        }
        
        return (preservation * weights['preservation'] +
                readability * weights['readability'] +
                structure * weights['structure'] +
                encoding * weights['encoding'])
