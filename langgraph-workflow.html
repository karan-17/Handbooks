1. LangGraph Overview & Architecture

LangGraph 1.0 is now generally available as the first stable major release in the durable agent framework space. It's a low-level orchestration framework used by companies like Klarna, Uber, LinkedIn, and J.P. Morgan for production-ready AI systems.

Core Architecture Principles

LangGraph is built on the principle of structured agents with minimal abstraction. The framework focuses on control and durability rather than hiding complexity.

Durable Execution

Build agents that persist through failures and can run for extended periods, resuming from exactly where they left off.

Human-in-the-Loop

Incorporate human oversight by inspecting and modifying agent state at any point during execution.

Comprehensive Memory

Create stateful agents with both short-term working memory and long-term persistent memory across sessions.

Streaming Support

Native token-by-token streaming showing agent reasoning and actions in real time.

2025 Key Features

  • Node/Task Level Caching: Cache results of individual nodes to reduce redundant computation
  • Deferred Nodes: Postpone node execution until all upstream paths complete
  • Command Support: Special return type that specifies both state updates and next node routing
  • Multi-Agent Coordination: Built-in support for supervisor, collaborative, and sequential patterns

3. Workflow Orchestration Patterns

Basic Linear Workflow

from langgraph.graph import StateGraph, START, END # Create workflow workflow = StateGraph(AgentState) # Define nodes def input_processor(state: AgentState): """Process user input""" processed = process_input(state["user_input"]) return {"current_task": processed} def task_executor(state: AgentState): """Execute the current task""" result = execute_task(state["current_task"]) return {"agent_output": result} def output_formatter(state: AgentState): """Format final output""" formatted = format_output(state["agent_output"]) return {"final_output": formatted} # Add nodes to workflow workflow.add_node("process_input", input_processor) workflow.add_node("execute_task", task_executor) workflow.add_node("format_output", output_formatter) # Add edges workflow.add_edge(START, "process_input") workflow.add_edge("process_input", "execute_task") workflow.add_edge("execute_task", "format_output") workflow.add_edge("format_output", END) # Compile workflow app = workflow.compile()

Conditional Workflow with Branching

def routing_function(state: AgentState) -> str: """Determine next node based on state""" if "error" in state.get("current_task", "").lower(): return "error_handler" elif "analysis" in state.get("current_task", "").lower(): return "data_analyzer" else: return "general_processor" # Add conditional edges workflow.add_conditional_edges( "task_router", routing_function, { "error_handler": "error_handler", "data_analyzer": "data_analyzer", "general_processor": "general_processor" } )

Cyclic Workflows with Loop Control

def should_continue(state: AgentState) -> str: """Control loop continuation""" if state.get("retry_count", 0) >= 3: return "end" if state.get("task_completed", False): return "end" return "continue" def retry_processor(state: AgentState): """Handle retry logic""" retry_count = state.get("retry_count", 0) + 1 return { "retry_count": retry_count, "current_task": f"Retry {retry_count}: {state['current_task']}" } # Add loop with conditional exit workflow.add_conditional_edges( "task_processor", should_continue, { "continue": "retry_processor", "end": END } ) workflow.add_edge("retry_processor", "task_processor") # Close the loop

4. Agent Coordination & Communication

Supervisor Pattern

The supervisor pattern involves a coordinator agent managing specialized worker agents:

from typing import Literal class SupervisorState(TypedDict): messages: list current_agent: str task_queue: list results: dict next_action: Literal["research", "analyze", "write", "review", "end"] def supervisor_node(state: SupervisorState): """Supervisor decides which agent to route to""" last_message = state["messages"][-1] if state["messages"] else "" if "research" in last_message.lower(): return { "current_agent": "researcher", "next_action": "research" } elif "analyze" in last_message.lower(): return { "current_agent": "analyst", "next_action": "analyze" } elif "write" in last_message.lower(): return { "current_agent": "writer", "next_action": "write" } else: return {"next_action": "end"} def research_agent(state: SupervisorState): """Specialized research agent""" research_result = perform_research(state["messages"][-1]) return { "messages": state["messages"] + [f"Research: {research_result}"], "results": {**state["results"], "research": research_result} } def analysis_agent(state: SupervisorState): """Specialized analysis agent""" analysis = analyze_data(state["results"].get("research", "")) return { "messages": state["messages"] + [f"Analysis: {analysis}"], "results": {**state["results"], "analysis": analysis} } # Build supervisor workflow supervisor_workflow = StateGraph(SupervisorState) supervisor_workflow.add_node("supervisor", supervisor_node) supervisor_workflow.add_node("researcher", research_agent) supervisor_workflow.add_node("analyst", analysis_agent) supervisor_workflow.add_conditional_edges( "supervisor", lambda x: x["next_action"], { "research": "researcher", "analyze": "analyst", "end": END } )

Collaborative Multi-Agent Pattern

class CollaborativeState(TypedDict): shared_context: dict agent_outputs: dict collaboration_log: list current_phase: str def collaborative_agent_a(state: CollaborativeState): """Agent A contributes to shared context""" contribution = process_with_agent_a(state["shared_context"]) return { "agent_outputs": { **state["agent_outputs"], "agent_a": contribution }, "collaboration_log": state["collaboration_log"] + [f"Agent A: {contribution}"], "shared_context": { **state["shared_context"], "agent_a_input": contribution } } def collaborative_agent_b(state: CollaborativeState): """Agent B builds on Agent A's work""" # Access Agent A's contribution agent_a_work = state["agent_outputs"].get("agent_a", "") contribution = process_with_agent_b(agent_a_work, state["shared_context"]) return { "agent_outputs": { **state["agent_outputs"], "agent_b": contribution }, "collaboration_log": state["collaboration_log"] + [f"Agent B: {contribution}"] } def synthesizer_agent(state: CollaborativeState): """Combine all agent contributions""" all_outputs = state["agent_outputs"] synthesis = synthesize_results(all_outputs) return { "final_result": synthesis, "collaboration_log": state["collaboration_log"] + [f"Synthesis: {synthesis}"] }

6. Error Handling & Recovery

Multi-Level Error Management

LangGraph promotes a comprehensive approach to error handling at node, graph, and application levels:

from langgraph.errors import GraphRecursionError import traceback from typing import Any, Dict class ErrorState(TypedDict): error_log: list error_count: int last_error: dict recovery_attempts: int fallback_active: bool def error_aware_node(state: ErrorState) -> Dict[str, Any]: """Node with comprehensive error handling""" try: # Main processing logic result = risky_operation(state) # Reset error tracking on success return { "result": result, "error_count": 0, "recovery_attempts": 0, "fallback_active": False } except ValueError as e: # Handle specific error types error_info = { "type": "ValueError", "message": str(e), "node": "error_aware_node", "timestamp": datetime.now().isoformat(), "recoverable": True } return { "error_log": state.get("error_log", []) + [error_info], "error_count": state.get("error_count", 0) + 1, "last_error": error_info } except Exception as e: # Handle unexpected errors error_info = { "type": type(e).__name__, "message": str(e), "traceback": traceback.format_exc(), "node": "error_aware_node", "timestamp": datetime.now().isoformat(), "recoverable": False } return { "error_log": state.get("error_log", []) + [error_info], "error_count": state.get("error_count", 0) + 1, "last_error": error_info, "requires_intervention": True }

Retry Mechanisms with Exponential Backoff

import time import random from langgraph.constants import Send def retry_node(state: ErrorState) -> Dict[str, Any]: """Implement retry with exponential backoff""" retry_count = state.get("recovery_attempts", 0) max_retries = 3 if retry_count >= max_retries: return { "fallback_active": True, "error_message": "Max retries exceeded, activating fallback" } # Exponential backoff calculation base_delay = 1 # seconds delay = base_delay * (2 ** retry_count) + random.uniform(0, 1) try: time.sleep(delay) result = attempt_operation(state) return { "result": result, "recovery_attempts": 0, "error_count": 0, "recovery_successful": True } except Exception as e: return { "recovery_attempts": retry_count + 1, "last_retry_error": str(e), "next_retry_delay": delay * 2 } def retry_controller(state: ErrorState) -> str: """Control retry flow""" if state.get("recovery_successful", False): return "success_path" elif state.get("fallback_active", False): return "fallback_handler" elif state.get("recovery_attempts", 0) < 3: return "retry_node" else: return "escalation_handler"

Fallback Strategies

def fallback_handler(state: ErrorState) -> Dict[str, Any]: """Implement fallback mechanisms""" error_type = state.get("last_error", {}).get("type", "Unknown") # Model fallback strategy if error_type == "ModelUnavailableError": fallback_model = get_fallback_model() try: result = fallback_model.invoke(state.get("input", "")) return { "result": result, "fallback_model_used": True, "resolution": "model_fallback" } except Exception: pass # Data fallback strategy elif error_type == "DataNotFoundError": cached_data = get_cached_data(state.get("query", "")) if cached_data: return { "result": cached_data, "data_source": "cache", "resolution": "cache_fallback" } # Service fallback strategy elif error_type == "ServiceUnavailableError": backup_service = get_backup_service() try: result = backup_service.process(state.get("request", {})) return { "result": result, "service_used": "backup", "resolution": "service_fallback" } except Exception: pass # Ultimate fallback - graceful degradation return { "result": generate_graceful_response(state), "resolution": "graceful_degradation", "requires_manual_review": True } def circuit_breaker_node(state: ErrorState) -> Dict[str, Any]: """Implement circuit breaker pattern""" failure_threshold = 5 recovery_timeout = 300 # 5 minutes current_time = time.time() last_failure = state.get("last_failure_time", 0) consecutive_failures = state.get("consecutive_failures", 0) # Check if circuit should be open if consecutive_failures >= failure_threshold: if current_time - last_failure < recovery_timeout: return { "circuit_open": True, "error_message": "Circuit breaker open - service unavailable" } else: # Try to close circuit (half-open state) return { "circuit_half_open": True, "test_request": True } # Circuit is closed, normal operation try: result = normal_operation(state) return { "result": result, "consecutive_failures": 0, "circuit_open": False } except Exception as e: return { "consecutive_failures": consecutive_failures + 1, "last_failure_time": current_time, "error": str(e) }

Human-in-the-Loop Error Recovery

from langgraph.checkpoint.sqlite import SqliteSaver def human_intervention_required(state: ErrorState) -> bool: """Determine if human intervention is needed""" critical_errors = ["SecurityError", "DataCorruptionError", "AuthenticationError"] error_type = state.get("last_error", {}).get("type", "") return ( error_type in critical_errors or state.get("error_count", 0) > 5 or state.get("requires_intervention", False) ) def create_human_review_checkpoint(state: ErrorState): """Create checkpoint for human review""" checkpointer = SqliteSaver.from_conn_string("checkpoint.db") thread_config = { "configurable": { "thread_id": f"error_review_{datetime.now().strftime('%Y%m%d_%H%M%S')}" } } review_data = { "state": state, "error_summary": generate_error_summary(state), "suggested_actions": suggest_recovery_actions(state), "escalation_level": determine_escalation_level(state) } # Pause execution for human review return { "human_review_required": True, "review_checkpoint": thread_config["configurable"]["thread_id"], "review_data": review_data, "execution_paused": True } def resume_after_human_review(state: ErrorState) -> Dict[str, Any]: """Resume execution after human intervention""" human_decision = state.get("human_decision", {}) if human_decision.get("action") == "retry": return { "recovery_attempts": 0, "human_guidance": human_decision.get("instructions", ""), "execution_paused": False } elif human_decision.get("action") == "fallback": return { "fallback_active": True, "human_approved_fallback": True, "execution_paused": False } elif human_decision.get("action") == "escalate": return { "escalated": True, "escalation_ticket": human_decision.get("ticket_id", ""), "execution_paused": True } return {"execution_terminated": True}
Best Practice: Always implement error categorization to handle recoverable vs. non-recoverable errors differently. Use structured error objects that can be easily logged and analyzed.

from enum import Enum class ModelTier(Enum): FAST = "gpt-3.5-turbo" # Low cost, fast BALANCED = "gpt-4o-mini" # Balanced cost/performance PREMIUM = "gpt-4o" # High cost, best quality class ModelRouter: def __init__(self): self.model_costs = { ModelTier.FAST: {"input": 0.0005, "output": 0.0015}, ModelTier.BALANCED: {"input": 0.00015, "output": 0.0006}, ModelTier.PREMIUM: {"input": 0.005, "output": 0.015} } self.performance_history = {} def select_model(self, task_complexity: str, quality_requirement: float, budget_constraint: float) -> ModelTier: """Intelligent model selection""" if task_complexity == "simple" and quality_requirement < 0.7: return ModelTier.FAST elif task_complexity == "complex" or quality_requirement > 0.9: return ModelTier.PREMIUM elif budget_constraint < 0.001: # Low budget return ModelTier.FAST else: return ModelTier.BALANCED def estimate_cost(self, model: ModelTier, input_tokens: int, output_tokens: int) -> float: """Estimate request cost""" costs = self.model_costs[model] return (input_tokens * costs["input"] + output_tokens * costs["output"]) / 1000 def cost_optimized_node(state: AgentState): """Node with intelligent model routing""" router = ModelRouter() task_complexity = analyze_task_complexity(state["user_input"]) quality_requirement = state.get("quality_threshold", 0.8) budget_constraint = state.get("budget_per_request", 0.01) selected_model = router.select_model( task_complexity, quality_requirement, budget_constraint ) # Initialize appropriate model llm = ChatOpenAI( model=selected_model.value, temperature=0.1 if selected_model == ModelTier.PREMIUM else 0.3 ) # Process request start_time = time.time() response = llm.invoke(state["user_input"]) processing_time = time.time() - import pickle import gzip from typing import Any, Dict class StateOptimizer: def __init__(self, max_history_size: int = 20, compression_threshold: int = 1000): self.max_history_size = max_history_size self.compression_threshold = compression_threshold def optimize_state(self, state: Dict[str, Any]) -> Dict[str, Any]: """Optimize state size and structure""" optimized_state = state.copy() # Compress large text fields for key, value in optimized_state.items(): if isinstance(value, str) and len(value) > self.compression_threshold: compressed = gzip.compress(value.encode('utf-8')) if len(compressed) < len(value.encode('utf-8')) * 0.8: # Remove "_compressed" decompressed_data = gzip.decompress(state[compressed_key]).decode('utf-8') decompressed_state[original_key] = decompressed_data del decompressed_state[compressed_key] del decompressed_state[f"{original_key}_is_compressed"] return decompressed_state def optimized_state_node(state: AgentState): """Node that optimizes state before processing""" optimizer = StateOptimizer() #
