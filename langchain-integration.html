1. LangChain Overview & Core Architecture (2026)

LangChain provides the engineering platform and open source frameworks developers use to build, test, and deploy reliable AI agents. The platform emphasizes modularity, production readiness, and seamless integration across the AI development lifecycle.

Core Components

Language models and chat models with unified interfaces

Prompts

Chains

Composable sequences connecting multiple components

Retrieval

Vector stores, embeddings, and document processing

Memory

Context retention and conversation management

Agents

Intelligent reasoning and dynamic tool usage

2026 Update: LangChain has evolved to emphasize LCEL (LangChain Expression Language) as the primary method for building chains, deprecating legacy chain classes in favor of more flexible, composable patterns.

# Persistent Memory with LangGraph from langgraph.graph import StateGraph from langgraph.checkpoint.sqlite import SqliteSaver from typing import TypedDict, List class ConversationState(TypedDict): messages: List[str] user_context: dict session_summary: str def conversation_node(state: ConversationState): # Process conversation with access to full state return state # Create graph with persistent checkpointing graph = StateGraph(ConversationState) graph.add_node("conversation", conversation_node) graph.set_entry_point("conversation") # SQLite-based persistence memory = SqliteSaver.from_conn_string("conversations.db") app = graph.compile(checkpointer=memory) # Conversations persist across sessions result = app.invoke( {"messages": ["Hello"]}, config={"configurable": {"thread_id": "user123"}} )

5. LangChain Expression Language (LCEL) Patterns

Core LCEL Concepts

LCEL is a declarative language for composing chains using the pipe operator (|). It provides automatic support for streaming, async execution, parallelism, and more.

# Basic LCEL Chain from langchain_core.prompts import ChatPromptTemplate from langchain_openai import ChatOpenAI from langchain_core.output_parsers import StrOutputParser prompt = ChatPromptTemplate.from_template("Tell me a joke about {topic}") model = ChatOpenAI(model="gpt-3.5-turbo") output_parser = StrOutputParser() # Chain composition with pipe operator chain = prompt | model | output_parser # Automatic async, streaming, and batch support result = chain.invoke({"topic": "programming"})

Advanced LCEL Patterns

Parallel Execution

from langchain_core.runnables import RunnableParallel # Parallel chain execution analysis_chain = RunnableParallel( sentiment=prompt_sentiment | model | StrOutputParser(), summary=prompt_summary | model | StrOutputParser(), keywords=prompt_keywords | model | StrOutputParser() ) # All branches execute simultaneously results = analysis_chain.invoke({"text": "Long document content..."})

Conditional Branching

from langchain_core.runnables import RunnableBranch # Conditional routing based on input branch = RunnableBranch( (lambda x: "code" in x["query"].lower(), code_analysis_chain), (lambda x: "data" in x["query"].lower(), data_analysis_chain), general_chain # Default branch ) result = branch.invoke({"query": "How to debug Python code?"})

from langchain_core.runnables import Runnable from typing import Any, Dict, Optional class CustomDataProcessor(Runnable): """Custom tool implementing Runnable protocol.""" def __init__(self, api_key: str): self.api_key = api_key def invoke( self, input: Dict[str, Any], config: Optional[Dict] = None ) -> Dict[str, Any]: # Synchronous processing processed_data = self._process_data(input["data"]) return {"processed": processed_data} async def ainvoke( self, input: Dict[str, Any], config: Optional[Dict] = None ) -> Dict[str, Any]: # Asynchronous processing processed_data = await self._aprocess_data(input["data"]) return {"processed": processed_data} def _process_data(self, data: str) -> str: # Sync implementation return data.upper() async def _aprocess_data(self, data: str) -> str: # Async implementation import asyncio await asyncio.sleep(0.1) # Simulate async work return data.upper() # Integration with LCEL processor = CustomDataProcessor("api_key") chain = prompt | processor | model | output_parser

LCEL chains automatically support streaming through the .stream() and .astream() methods.

from langchain_core.runnables import Runnable import asyncio import aiohttp class AsyncWebRetriever(Runnable): """Async web content retriever.""" async def ainvoke(self, input: dict, config=None) -> dict: urls = input.get("urls", []) async with aiohttp.ClientSession() as session: tasks = [self._fetch_url(session, url) for url in urls] results = await asyncio.gather(*tasks) return {"retrieved_content": results} async def _fetch_url(self, session: aiohttp.ClientSession, url: str) -> str: try: async with session.get(url) as response: return await response.text() except Exception as e: return f"Error fetching {url}: {str(e)}" # Integration in async chain async_retriever = AsyncWebRetriever() async_chain = async_retriever | processing_chain # Concurrent execution results = await asyncio.gather(*[ async_chain.ainvoke({"urls": urls_batch_1}), async_chain.ainvoke({"urls": urls_batch_2}), ])

from langchain_core.runnables import RunnablePassthrough from langchain_core.runnables.utils import Input, Output # Fallback chains for error recovery primary_chain = prompt | model | output_parser fallback_chain = simple_prompt | backup_model | output_parser robust_chain = primary_chain.with_fallbacks([fallback_chain]) # Custom error handling def safe_invoke(chain, input_data): try: return chain.invoke(input_data) except Exception as e: print(f"Chain execution failed: {e}") return {"error": str(e), "fallback": True} # Retry logic with exponential backoff from langchain_core.runnables import RunnableRetry retry_chain = RunnableRetry( bound=chain, max_attempt_number=3, wait_exponential_jitter=True )

Common Debugging Patterns

# Add debugging middleware to chains class DebugRunnable(Runnable): def __init__(self, runnable: Runnable, name: str): self.runnable = runnable self.name = name def invoke(self, input, config=None): print(f"[{self.name}] Input: {input}") try: output = self.runnable.invoke(input, config) print(f"[{self.name}] Output: {output}") return output except Exception as e: print(f"[{self.name}] Error: {e}") raise # Wrap chain components for debugging debug_chain = ( DebugRunnable(prompt, "prompt") | DebugRunnable(model, "model") | DebugRunnable(output_parser, "parser") ) # Token usage tracking def track_token_usage(chain_result, input_data): """Track and log token usage for cost monitoring.""" if hasattr(chain_result, 'response_metadata'): usage = chain_result.response_metadata.get('token_usage', {}) print(f"Token usage: {usage}") # Log to monitoring system logger.info("chain_execution", { "input_tokens": usage.get('prompt_tokens', 0), "output_tokens": usage.get('completion_tokens', 0), "total_tokens": usage.get('total_tokens', 0) })
