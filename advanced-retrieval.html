1. Dense-Sparse Fusion

Hybrid search combines the semantic understanding of dense embeddings with the exact term matching capabilities of sparse methods like BM25. This approach provides comprehensive coverage across different query types and document characteristics.

@dataclass
class SearchResult:
    """Represents a search result with relevance scores"""
    doc_id: str
    content: str
    dense_score: float
    sparse_score: float
    fusion_score: float
    metadata: Dict = None

class FusionStrategy(ABC):
    """Abstract base class for score fusion strategies"""
    
    @abstractmethod
    def fuse_scores(self, dense_scores: List[float], sparse_scores: List[float]) -> List[float]:
        pass

class ReciprocalRankFusion(FusionStrategy):
    """Reciprocal Rank Fusion (RRF) for combining rankings"""
    
    def __init__(self, k: int = 60):
        self.k = k
    
    def fuse_scores(self, dense_results: List[Tuple], sparse_results: List[Tuple]) -> List[SearchResult]:
        """
        Fuse results using Reciprocal Rank Fusion
        
        Args:
            dense_results: List of (doc_id, score) tuples from dense search
            sparse_results: List of (doc_id, score) tuples from sparse search
        """
        # Create rank mappings
        dense_ranks = {doc_id: rank + 1 for rank, (doc_id, _) in enumerate(dense_results)}
        sparse_ranks = {doc_id: rank + 1 for rank, (doc_id, _) in enumerate(sparse_results)}
        
        # Get all unique document IDs
        all_doc_ids = set(dense_ranks.keys()) | set(sparse_ranks.keys())
        
        fused_scores = {}
        for doc_id in all_doc_ids:
            rrf_score = 0
            
            if doc_id in dense_ranks:
                rrf_score += 1 / (self.k + dense_ranks[doc_id])
            
            if doc_id in sparse_ranks:
                rrf_score += 1 / (self.k + sparse_ranks[doc_id])
            
            fused_scores[doc_id] = rrf_score
        
        # Sort by fused score
        return sorted(fused_scores.items(), key=lambda x: x[1], reverse=True)

class LinearCombination(FusionStrategy):
    """Linear combination of normalized scores"""
    
    def __init__(self, dense_weight: float = 0.7, sparse_weight: float = 0.3):
        self.dense_weight = dense_weight
        self.sparse_weight = sparse_weight
        assert abs(dense_weight + sparse_weight - 1.0) < 1e-6, "Weights must sum to 1"
    
    def fuse_scores(self, dense_scores: List[float], sparse_scores: List[float]) -> List[float]:
        # Normalize scores to [0, 1] range
        dense_normalized = self._min_max_normalize(dense_scores)
        sparse_normalized = self._min_max_normalize(sparse_scores)
        
        # Combine linearly
        return [self.dense_weight * d + self.sparse_weight * s 
                for d, s in zip(dense_normalized, sparse_normalized)]
    
    def _min_max_normalize(self, scores: List[float]) -> List[float]:
        if not scores or max(scores) == min(scores):
            return [0.5] * len(scores)
        
        min_score, max_score = min(scores), max(scores)
        return [(score - min_score) / (max_score - min_score) for score in scores]

class HybridRetriever:
    """Advanced hybrid retrieval system"""
    
    def __init__(
        self,
        embedding_model_name: str = "all-MiniLM-L6-v2",
        fusion_strategy: FusionStrategy = None,
        dense_top_k: int = 100,
        sparse_top_k: int = 100,
        final_top_k: int = 20
    ):
        # Initialize dense retrieval
        self.embedding_model = SentenceTransformer(embedding_model_name)
        self.dense_index = None
        self.dense_top_k = dense_top_k
        
        # Initialize sparse retrieval
        self.bm25 = None
        self.sparse_top_k = sparse_top_k
        
        # Fusion configuration
        self.fusion_strategy = fusion_strategy or ReciprocalRankFusion()
        self.final_top_k = final_top_k
        
        # Document storage
        self.documents = []
        self.doc_embeddings = None
        self.tokenized_docs = None
        
    def index_documents(self, documents: List[Dict]):
        """
        Index documents for both dense and sparse retrieval
        
        Args:
            documents: List of document dictionaries with 'content' and 'metadata'
        """
        self.documents = documents
        
        # Extract content for indexing
        doc_contents = [doc['content'] for doc in documents]
        
        # Build dense index
        print("Building dense embeddings...")
        self.doc_embeddings = self.embedding_model.encode(doc_contents, show_progress_bar=True)
        
        # Build FAISS index for efficient similarity search
        self.dense_index = faiss.IndexFlatIP(self.doc_embeddings.shape[1])
        # Normalize embeddings for cosine similarity
        faiss.normalize_L2(self.doc_embeddings)
        self.dense_index.add(self.doc_embeddings.astype(np.float32))
        
        # Build sparse index (BM25)
        print("Building sparse index...")
        self.tokenized_docs = [doc.lower().split() for doc in doc_contents]
        self.bm25 = BM25Okapi(self.tokenized_docs)
        
        print(f"Indexed {len(documents)} documents")
    
    def search(self, query: str, top_k: int = None) -> List[SearchResult]:
        """
        Perform hybrid search combining dense and sparse retrieval
        
        Args:
            query: Search query
            top_k: Number of results to return (defaults to self.final_top_k)
        
        Returns:
            List of SearchResult objects ranked by fusion score
        """
        if top_k is None:
            top_k = self.final_top_k
        
        # Dense search
        dense_results = self._dense_search(query, self.dense_top_k)
        
        # Sparse search
        sparse_results = self._sparse_search(query, self.sparse_top_k)
        
        # Fuse results
        if isinstance(self.fusion_strategy, ReciprocalRankFusion):
            fused_rankings = self.fusion_strategy.fuse_scores(dense_results, sparse_results)
            
            # Create SearchResult objects
            results = []
            for doc_id, fusion_score in fused_rankings[:top_k]:
                # Get original scores
                dense_score = next((score for did, score in dense_results if did == doc_id), 0.0)
                sparse_score = next((score for did, score in sparse_results if did == doc_id), 0.0)
                
                doc_idx = int(doc_id)
                result = SearchResult(
                    doc_id=doc_id,
                    content=self.documents[doc_idx]['content'],
                    dense_score=dense_score,
                    sparse_score=sparse_score,
                    fusion_score=fusion_score,
                    metadata=self.documents[doc_idx].get('metadata', {})
                )
                results.append(result)
            
            return results
        
        else:
            # For other fusion strategies, implement score-based fusion
            # This is a simplified implementation - extend as needed
            return self._score_based_fusion(dense_results, sparse_results, top_k)
    
    def _dense_search(self, query: str, top_k: int) -> List[Tuple[str, float]]:
        """Perform dense vector search"""
        # Encode query
        query_embedding = self.embedding_model.encode([query])
        faiss.normalize_L2(query_embedding)
        
        # Search
        scores, indices = self.dense_index.search(query_embedding.astype(np.float32), top_k)
        
        # Return (doc_id, score) tuples
        return [(str(idx), float(score)) for score, idx in zip(scores[0], indices[0]) if idx != -1]
    
    def _sparse_search(self, query: str, top_k: int) -> List[Tuple[str, float]]:
        """Perform sparse BM25 search"""
        query_tokens = query.lower().split()
        scores = self.bm25.get_scores(query_tokens)
        
        # Get top-k indices
        top_indices = np.argsort(scores)[-top_k:][::-1]
        
        # Return (doc_id, score) tuples
        return [(str(idx), float(scores[idx])) for idx in top_indices]
    
    def _score_based_fusion(self, dense_results: List[Tuple], sparse_results: List[Tuple], top_k: int) -> List[SearchResult]:
        """Alternative fusion method based on score combination"""
        # Create score dictionaries
        dense_scores = {doc_id: score for doc_id, score in dense_results}
        sparse_scores = {doc_id: score for doc_id, score in sparse_results}
        
        # Get all document IDs
        all_doc_ids = set(dense_scores.keys()) | set(sparse_scores.keys())
        
        # Calculate fusion scores
        fusion_scores = []
        for doc_id in all_doc_ids:
            dense_score = dense_scores.get(doc_id, 0.0)
            sparse_score = sparse_scores.get(doc_id, 0.0)
            
            # Use fusion strategy
            fusion_score = self.fusion_strategy.fuse_scores([dense_score], [sparse_score])[0]
            
            fusion_scores.append((doc_id, dense_score, sparse_score, fusion_score))
        
        # Sort by fusion score
        fusion_scores.sort(key=lambda x: x[3], reverse=True)
        
        # Create SearchResult objects
        results = []
        for doc_id, dense_score, sparse_score, fusion_score in fusion_scores[:top_k]:
            doc_idx = int(doc_id)
            result = SearchResult(
                doc_id=doc_id,
                content=self.documents[doc_idx]['content'],
                dense_score=dense_score,
                sparse_score=sparse_score,
                fusion_score=fusion_score,
                metadata=self.documents[doc_idx].get('metadata', {})
            )
            results.append(result)
        
        return results

2. ColBERT-Style Late Interaction

ColBERT introduces a novel approach to dense retrieval through late interaction, where query and document tokens interact at the similarity computation stage rather than during encoding.

ColBERT Late Interaction Implementation
import torch
import torch.nn as nn
from transformers import AutoTokenizer, AutoModel
from typing import List, Tuple
import numpy as np

class ColBERTRetriever:
    """Implementation of ColBERT-style late interaction retrieval"""
    
    def __init__(
        self,
        model_name: str = "colbert-ir/colbertv2.0",
        max_query_length: int = 32,
        max_doc_length: int = 180,
        dim: int = 128
    ):
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        self.max_query_length = max_query_length
        self.max_doc_length = max_doc_length
        self.dim = dim
        
        # Load tokenizer and model
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModel.from_pretrained(model_name)
        self.model.to(self.device)
        self.model.eval()
        
        # Add special tokens
        if "[Q]" not in self.tokenizer.get_vocab():
            self.tokenizer.add_special_tokens({"additional_special_tokens": ["[Q]", "[D]"]})
            self.model.resize_token_embeddings(len(self.tokenizer))
        
        # Linear layer for dimension reduction
        self.linear = nn.Linear(self.model.config.hidden_size, self.dim)
        self.linear.to(self.device)
        
    def encode_queries(self, queries: List[str]) -> torch.Tensor:
        """Encode queries with [Q] prefix"""
        encoded_queries = []
        
        for query in queries:
            # Add query prefix
            query = f"[Q] {query}"
            
            # Tokenize
            tokens = self.tokenizer(
                query,
                max_length=self.max_query_length,
                padding='max_length',
                truncation=True,
                return_tensors='pt'
            )
            
            # Move to device
            input_ids = tokens['input_ids'].to(self.device)
            attention_mask = tokens['attention_mask'].to(self.device)
            
            with torch.no_grad():
                # Get embeddings
                outputs = self.model(input_ids=input_ids, attention_mask=attention_mask)
                embeddings = outputs.last_hidden_state
                
                # Apply linear layer
                embeddings = self.linear(embeddings)
                
                # Normalize embeddings
                embeddings = nn.functional.normalize(embeddings, p=2, dim=-1)
                
                # Mask padded tokens
                embeddings = embeddings * attention_mask.unsqueeze(-1)
                
            encoded_queries.append(embeddings.squeeze(0))
        
        return torch.stack(encoded_queries)
    
    def encode_documents(self, documents: List[str]) -> torch.Tensor:
        """Encode documents with [D] prefix"""
        encoded_docs = []
        
        for doc in documents:
            # Add document prefix
            doc = f"[D] {doc}"
            
            # Tokenize
            tokens = self.tokenizer(
                doc,
                max_length=self.max_doc_length,
                padding='max_length',
                truncation=True,
                return_tensors='pt'
            )
            
            # Move to device
            input_ids = tokens['input_ids'].to(self.device)
            attention_mask = tokens['attention_mask'].to(self.device)
            
            with torch.no_grad():
                # Get embeddings
                outputs = self.model(input_ids=input_ids, attention_mask=attention_mask)
                embeddings = outputs.last_hidden_state
                
                # Apply linear layer
                embeddings = self.linear(embeddings)
                
                # Normalize embeddings
                embeddings = nn.functional.normalize(embeddings, p=2, dim=-1)
                
                # Mask padded tokens
                embeddings = embeddings * attention_mask.unsqueeze(-1)
                
            encoded_docs.append(embeddings.squeeze(0))
        
        return torch.stack(encoded_docs)
    
    def late_interaction_score(
        self, 
        query_embeddings: torch.Tensor, 
        doc_embeddings: torch.Tensor
    ) -> float:
        """
        Compute late interaction score between query and document
        
        Args:
            query_embeddings: [query_len, dim]
            doc_embeddings: [doc_len, dim]
        
        Returns:
            Interaction score
        """
        # Compute all pairwise similarities
        # query_embeddings: [query_len, dim]
        # doc_embeddings: [doc_len, dim]
        # similarities: [query_len, doc_len]
        similarities = torch.matmul(query_embeddings, doc_embeddings.T)
        
        # For each query token, find the maximum similarity with any document token
        max_similarities = torch.max(similarities, dim=1)[0]
        
        # Sum over all query tokens (excluding padding)
        # This requires checking for non-zero embeddings (non-padded tokens)
        query_mask = torch.norm(query_embeddings, dim=1) > 0
        score = torch.sum(max_similarities * query_mask)
        
        return score.item()
    
    def search(
        self, 
        query: str, 
        document_embeddings: torch.Tensor, 
        top_k: int = 10
    ) -> List[Tuple[int, float]]:
        """
        Search for most relevant documents using late interaction
        
        Args:
            query: Search query
            document_embeddings: Pre-computed document embeddings [num_docs, max_doc_len, dim]
            top_k: Number of results to return
        
        Returns:
            List of (doc_index, score) tuples
        """
        # Encode query
        query_embedding = self.encode_queries([query])[0]  # [query_len, dim]
        
        # Compute scores for all documents
        scores = []
        for doc_idx, doc_embedding in enumerate(document_embeddings):
            score = self.late_interaction_score(query_embedding, doc_embedding)
            scores.append((doc_idx, score))
        
        # Sort by score and return top-k
        scores.sort(key=lambda x: x[1], reverse=True)
        return scores[:top_k]
    
    def batch_search(
        self, 
        queries: List[str], 
        document_embeddings: torch.Tensor, 
        top_k: int = 10
    ) -> List[List[Tuple[int, float]]]:
        """Batch search for multiple queries"""
        results = []
        query_embeddings = self.encode_queries(queries)
        
        for query_embedding in query_embeddings:
            doc_scores = []
            for doc_idx, doc_embedding in enumerate(document_embeddings):
                score = self.late_interaction_score(query_embedding, doc_embedding)
                doc_scores.append((doc_idx, score))
            
            doc_scores.sort(key=lambda x: x[1], reverse=True)
            results.append(doc_scores[:top_k])
        
        return results

class QueryExpansionSystem:
    """Advanced query expansion using multiple strategies"""
    
    def __init__(
        self,
        embedding_model: str = "all-MiniLM-L6-v2",
        llm_model: str = "gpt-3.5-turbo",
        use_wordnet: bool = True,
        use_llm: bool = True
    ):
        # Initialize models
        self.embedding_model = SentenceTransformer(embedding_model)
        self.nlp = spacy.load("en_core_web_sm")
        self.llm_model = llm_model
        self.use_wordnet = use_wordnet
        self.use_llm = use_llm
        
        # Initialize NLI pipeline for query relevance
        self.nli_pipeline = pipeline(
            "text-classification",
            model="microsoft/DialoGPT-medium"
        )
        
        # Download required NLTK data
        if use_wordnet:
            try:
                nltk.data.find('corpora/wordnet')
            except LookupError:
                nltk.download('wordnet')
        
        # Knowledge base for domain-specific expansions
        self.domain_expansions = defaultdict(list)
        
    def expand_query(
        self, 
        query: str, 
        corpus: List[str] = None,
        max_expansions: int = 5,
        expansion_strategies: List[str] = None
    ) -> Dict[str, List[str]]:
        """
        Expand query using multiple strategies
        
        Args:
            query: Original query
            corpus: Optional corpus for corpus-based expansion
            max_expansions: Maximum number of expansion terms per strategy
            expansion_strategies: List of strategies to use
        
        Returns:
            Dictionary with expansion terms for each strategy
        """
        if expansion_strategies is None:
            expansion_strategies = ["wordnet", "embedding", "llm", "corpus"]
        
        expansions = {}
        
        # Preprocess query
        processed_query = self._preprocess_query(query)
        
        # WordNet-based expansion
        if "wordnet" in expansion_strategies and self.use_wordnet:
            expansions["wordnet"] = self._wordnet_expansion(
                processed_query, max_expansions
            )
        
        # Embedding-based expansion
        if "embedding" in expansion_strategies:
            expansions["embedding"] = self._embedding_expansion(
                processed_query, corpus, max_expansions
            )
        
        # LLM-based expansion
        if "llm" in expansion_strategies and self.use_llm:
            expansions["llm"] = self._llm_expansion(
                processed_query, max_expansions
            )
        
        # Corpus-based expansion
        if "corpus" in expansion_strategies and corpus:
            expansions["corpus"] = self._corpus_expansion(
                processed_query, corpus, max_expansions
            )
        
        # Combine and rank expansions
        combined_expansions = self._combine_expansions(
            expansions, processed_query, max_expansions
        )
        
        return {
            "original_query": query,
            "processed_query": processed_query,
            "expansions_by_strategy": expansions,
            "final_expansions": combined_expansions,
            "expanded_query": self._create_expanded_query(processed_query, combined_expansions)
        }
    
    def _preprocess_query(self, query: str) -> str:
        """Preprocess query for better expansion"""
        # Parse with spaCy
        doc = self.nlp(query)
        
        # Extract important terms (remove stop words, keep nouns and adjectives)
        important_terms = []
        for token in doc:
            if (not token.is_stop and 
                not token.is_punct and 
                token.pos_ in ['NOUN', 'ADJ', 'VERB'] and
                len(token.text) > 2):
                important_terms.append(token.lemma_.lower())
        
        return " ".join(important_terms) if important_terms else query.lower()
    
    def _wordnet_expansion(self, query: str, max_terms: int) -> List[str]:
        """Expand query using WordNet synonyms"""
        if not self.use_wordnet:
            return []
        
        expansion_terms = set()
        words = query.split()
        
        for word in words:
            # Get synonyms from WordNet
            synsets = wordnet.synsets(word)
            for synset in synsets[:3]:  # Limit to top 3 synsets
                for lemma in synset.lemmas()[:2]:  # Top 2 lemmas per synset
                    synonym = lemma.name().replace('_', ' ').lower()
                    if synonym != word and len(synonym) > 2:
                        expansion_terms.add(synonym)
        
        return list(expansion_terms)[:max_terms]
    
    def _embedding_expansion(
        self, 
        query: str, 
        corpus: List[str], 
        max_terms: int
    ) -> List[str]:
        """Expand query using embedding similarity"""
        if not corpus:
            return []
        
        # Get query embedding
        query_embedding = self.embedding_model.encode([query])
        
        # Get corpus embeddings (sample for efficiency)
        sample_size = min(1000, len(corpus))
        sample_corpus = corpus[:sample_size]
        corpus_embeddings = self.embedding_model.encode(sample_corpus)
        
        # Find most similar documents
        similarities = cosine_similarity(query_embedding, corpus_embeddings)[0]
        top_doc_indices = np.argsort(similarities)[-10:][::-1]
        
        # Extract terms from similar documents
        expansion_terms = set()
        for idx in top_doc_indices:
            doc = sample_corpus[idx]
            doc_processed = self.nlp(doc)
            
            for token in doc_processed:
                if (not token.is_stop and 
                    not token.is_punct and 
                    token.pos_ in ['NOUN', 'ADJ'] and
                    len(token.text) > 2 and
                    token.lemma_.lower() not in query.lower()):
                    expansion_terms.add(token.lemma_.lower())
        
        # Rank terms by frequency in similar documents
        term_freq = defaultdict(int)
        for idx in top_doc_indices:
            doc_terms = set(token.lemma_.lower() for token in self.nlp(sample_corpus[idx])
                           if not token.is_stop and not token.is_punct)
            for term in expansion_terms:
                if term in doc_terms:
                    term_freq[term] += 1
        
        # Return top terms by frequency
        ranked_terms = sorted(term_freq.items(), key=lambda x: x[1], reverse=True)
        return [term for term, freq in ranked_terms[:max_terms]]
    
    def _llm_expansion(self, query: str, max_terms: int) -> List[str]:
        """Expand query using Large Language Model"""
        if not self.use_llm:
            return []
        
        try:
            prompt = f"""
Given the search query: "{query}"

Generate {max_terms} semantically related terms or phrases that would help improve search results. 
Focus on synonyms, related concepts, and domain-specific terminology.

Return only the terms, separated by commas, without explanations.

Related terms:"""

            response = openai.ChatCompletion.create(
                model=self.llm_model,
                messages=[{"role": "user", "content": prompt}],
                temperature=0.3,
                max_tokens=100
            )
            
            # Parse response
            content = response.choices[0].message.content.strip()
            terms = [term.strip().lower() for term in content.split(',')]
            
            # Filter and validate terms
            valid_terms = []
            for term in terms:
                if (len(term) > 2 and 
                    term not in query.lower() and
                    not any(char.isdigit() for char in term)):
                    valid_terms.append(term)
            
            return valid_terms[:max_terms]
            
        except Exception as e:
            print(f"LLM expansion failed: {e}")
            return []
    
    def _corpus_expansion(
        self, 
        query: str, 
        corpus: List[str], 
        max_terms: int
    ) -> List[str]:
        """Expand query using corpus statistics (co-occurrence)"""
        # Build TF-IDF vectorizer
        vectorizer = TfidfVectorizer(
            max_features=5000,
            ngram_range=(1, 2),
            stop_words='english'
        )
        
        try:
            tfidf_matrix = vectorizer.fit_transform(corpus)
            feature_names = vectorizer.get_feature_names_out()
            
            # Get query terms
            query_terms = set(query.lower().split())
            
            # Find terms that co-occur with query terms
            expansion_candidates = defaultdict(float)
            
            for doc_idx, doc in enumerate(corpus[:500]):  # Sample for efficiency
                doc_lower = doc.lower()
                if any(term in doc_lower for term in query_terms):
                    # Get TF-IDF scores for this document
                    doc_tfidf = tfidf_matrix[doc_idx].toarray()[0]
                    
                    # Add high-scoring terms as candidates
                    for term_idx, score in enumerate(doc_tfidf):
                        if score > 0.1:  # Threshold for relevance
                            term = feature_names[term_idx]
                            if term not in query_terms and len(term) > 2:
                                expansion_candidates[term] += score
            
            # Return top candidates
            ranked_candidates = sorted(
                expansion_candidates.items(), 
                key=lambda x: x[1], 
                reverse=True
            )
            
            return [term for term, score in ranked_candidates[:max_terms]]
            
        except Exception as e:
            print(f"Corpus expansion failed: {e}")
            return []
    
    def _combine_expansions(
        self, 
        expansions: Dict[str, List[str]], 
        query: str, 
        max_terms: int
    ) -> List[str]:
        """Combine and rank expansion terms from different strategies"""
        # Count term frequencies across strategies
        term_counts = defaultdict(int)
        term_strategies = defaultdict(set)
        
        for strategy, terms in expansions.items():
            for term in terms:
                term_counts[term] += 1
                term_strategies[term].add(strategy)
        
        # Score terms based on frequency and strategy diversity
        term_scores = {}
        for term, count in term_counts.items():
            diversity_bonus = len(term_strategies[term]) * 0.2
            term_scores[term] = count + diversity_bonus
        
        # Filter out terms too similar to original query
        filtered_terms = []
        for term in term_scores:
            if not self._is_too_similar(term, query):
                filtered_terms.append(term)
        
        # Sort by score and return top terms
        ranked_terms = sorted(
            [(term, term_scores[term]) for term in filtered_terms],
            key=lambda x: x[1],
            reverse=True
        )
        
        return [term for term, score in ranked_terms[:max_terms]]
    
    def _is_too_similar(self, term: str, query: str) -> bool:
        """Check if expansion term is too similar to original query"""
        query_words = set(query.lower().split())
        term_words = set(term.lower().split())
        
        # Check for exact matches or high overlap
        if term.lower() in query.lower() or query.lower() in term.lower():
            return True
        
        # Check word-level overlap
        if len(term_words.intersection(query_words)) / max(len(term_words), 1) > 0.7:
            return True
        
        return False
    
    def _create_expanded_query(self, query: str, expansions: List[str]) -> str:
        """Create final expanded query string"""
        if not expansions:
            return query
        
        # Add expansions with boosting syntax (for Elasticsearch-style queries)
        boosted_expansions = [f"{term}^0.5" for term in expansions[:3]]
        
        return f"{query} {' '.join(boosted_expansions)}"

1. Cross-Encoder Re-ranking

Re-ranking refines initial retrieval results using more sophisticated models that can capture complex query-document interactions. Cross-encoders provide the highest accuracy by jointly encoding query and document pairs.

class RerankingResult:

class NeuralReranker:
    """Neural re-ranking using cross-encoders and ensemble methods"""
    
    def __init__(
        self,
        cross_encoder_model: str = "cross-encoder/ms-marco-MiniLM-L-12-v2",
        device: str = None,
        batch_size: int = 32
    ):
        self.device = device or ("cuda" if torch.cuda.is_available() else "cpu")
        self.batch_size = batch_size
        
        # Initialize cross-encoder
        self.cross_encoder = CrossEncoder(cross_encoder_model, device=self.device)
        
        # Model weights for ensemble (if using multiple models)
        self.model_weights = {"cross_encoder": 1.0}
    
    def rerank(
        self,
        query: str,
        documents: List[Dict],
        initial_scores: List[float] = None,
        top_k: int = None,
        score_combination: str = "weighted_sum"
    ) -> List[RerankingResult]:
        """
        Re-rank documents using cross-encoder
        
        Args:
            query: Search query
            documents: List of document dictionaries
            initial_scores: Original retrieval scores
            top_k: Number of results to return
            score_combination: Method to combine scores
        
        Returns:
            List of RerankingResult objects
        """
        if not documents:
            return []
        
        if initial_scores is None:
            initial_scores = [1.0] * len(documents)
        
        # Prepare query-document pairs
        query_doc_pairs = [(query, doc['content']) for doc in documents]
        
        # Get cross-encoder scores
        rerank_scores = self._get_cross_encoder_scores(query_doc_pairs)
        
        # Combine scores
        final_scores = self._combine_scores(
            initial_scores, rerank_scores, method=score_combination
        )
        
        # Create results
        results = []
        for i, doc in enumerate(documents):
            result = RerankingResult(
                doc_id=doc.get('id', str(i)),
                content=doc['content'],
                original_score=initial_scores[i],
                rerank_score=rerank_scores[i],
                final_score=final_scores[i],
                metadata=doc.get('metadata', {})
            )
            results.append(result)
        
        # Sort by final score
        results.sort(key=lambda x: x.final_score, reverse=True)
        
        # Return top-k
        if top_k:
            return results[:top_k]
        return results
    
    def _get_cross_encoder_scores(self, query_doc_pairs: List[Tuple[str, str]]) -> List[float]:
        """Get scores from cross-encoder model"""
        # Process in batches for efficiency
        all_scores = []
        
        for i in range(0, len(query_doc_pairs), self.batch_size):
            batch = query_doc_pairs[i:i + self.batch_size]
            
            # Get scores for batch
            batch_scores = self.cross_encoder.predict(batch, show_progress_bar=False)
            
            # Convert to list if numpy array
            if isinstance(batch_scores, np.ndarray):
                batch_scores = batch_scores.tolist()
            
            all_scores.extend(batch_scores)
        
        return all_scores
    
    def _combine_scores(
        self,
        initial_scores: List[float],
        rerank_scores: List[float],
        method: str = "weighted_sum"
    ) -> List[float]:
        """Combine initial and re-ranking scores"""
        
        if method == "weighted_sum":
            # Normalize scores to [0, 1] range
            initial_norm = self._normalize_scores(initial_scores)
            rerank_norm = self._normalize_scores(rerank_scores)
            
            # Weighted combination (favor re-ranking)
            alpha = 0.3  # Weight for initial scores
            beta = 0.7   # Weight for re-ranking scores
            
            return [alpha * init + beta * rerank 
                    for init, rerank in zip(initial_norm, rerank_norm)]
        
        elif method == "rerank_only":
            return rerank_scores
        
        elif method == "rank_fusion":
            return self._rank_fusion(initial_scores, rerank_scores)
        
        else:
            raise ValueError(f"Unknown score combination method: {method}")
    
    def _normalize_scores(self, scores: List[float]) -> List[float]:
        """Normalize scores to [0, 1] range"""
        if not scores or max(scores) == min(scores):
            return [0.5] * len(scores)
        
        min_score, max_score = min(scores), max(scores)
        return [(score - min_score) / (max_score - min_score) for score in scores]
    
    def _rank_fusion(self, scores1: List[float], scores2: List[float]) -> List[float]:
        """Combine scores using rank fusion"""
        # Convert scores to ranks
        ranks1 = self._scores_to_ranks(scores1)
        ranks2 = self._scores_to_ranks(scores2)
        
        # Reciprocal rank fusion
        k = 60
        fusion_scores = []
        for r1, r2 in zip(ranks1, ranks2):
            rrf_score = 1/(k + r1) + 1/(k + r2)
            fusion_scores.append(rrf_score)
        
        return fusion_scores
    
    def _scores_to_ranks(self, scores: List[float]) -> List[int]:
        """Convert scores to ranks (1-indexed)"""
        indexed_scores = list(enumerate(scores))
        indexed_scores.sort(key=lambda x: x[1], reverse=True)
        
        ranks = [0] * len(scores)
        for rank, (original_idx, _) in enumerate(indexed_scores, 1):
            ranks[original_idx] = rank
        
        return ranks

class EnsembleReranker(NeuralReranker):
    """Ensemble re-ranking using multiple models"""
    
    def __init__(
        self,
        models: List[str] = None,
        model_weights: Dict[str, float] = None,
        device: str = None,
        batch_size: int = 16
    ):
        super().__init__(device=device, batch_size=batch_size)
        
        # Initialize multiple models
        if models is None:
            models = [
                "cross-encoder/ms-marco-MiniLM-L-12-v2",
                "cross-encoder/ms-marco-TinyBERT-L-6-v2"
            ]
        
        self.models = {f"model_{i}": CrossEncoder(model, device=self.device) 
                      for i, model in enumerate(models)}
        
        # Set model weights
        if model_weights is None:
            self.model_weights = {name: 1.0/len(self.models) for name in self.models.keys()}
        else:
            self.model_weights = model_weights
    
    def rerank(
        self,
        query: str,
        documents: List[Dict],
        initial_scores: List[float] = None,
        top_k: int = None,
        score_combination: str = "ensemble_weighted"
    ) -> List[RerankingResult]:
        """Re-rank using ensemble of models"""
        
        if not documents:
            return []
        
        if initial_scores is None:
            initial_scores = [1.0] * len(documents)
        
        # Prepare query-document pairs
        query_doc_pairs = [(query, doc['content']) for doc in documents]
        
        # Get scores from all models
        all_model_scores = {}
        for model_name, model in self.models.items():
            scores = self._get_model_scores(model, query_doc_pairs)
            all_model_scores[model_name] = scores
        
        # Combine ensemble scores
        ensemble_scores = self._combine_ensemble_scores(all_model_scores)
        
        # Combine with initial scores
        final_scores = self._combine_scores(
            initial_scores, ensemble_scores, method="weighted_sum"
        )
        
        # Create results
        results = []
        for i, doc in enumerate(documents):
            result = RerankingResult(
                doc_id=doc.get('id', str(i)),
                content=doc['content'],
                original_score=initial_scores[i],
                rerank_score=ensemble_scores[i],
                final_score=final_scores[i],
                metadata={
                    **doc.get('metadata', {}),
                    'model_scores': {name: scores[i] for name, scores in all_model_scores.items()}
                }
            )
            results.append(result)
        
        # Sort and return
        results.sort(key=lambda x: x.final_score, reverse=True)
        return results[:top_k] if top_k else results
    
    def _get_model_scores(self, model: CrossEncoder, query_doc_pairs: List[Tuple[str, str]]) -> List[float]:
        """Get scores from a specific model"""
        all_scores = []
        
        for i in range(0, len(query_doc_pairs), self.batch_size):
            batch = query_doc_pairs[i:i + self.batch_size]
            batch_scores = model.predict(batch, show_progress_bar=False)
            
            if isinstance(batch_scores, np.ndarray):
                batch_scores = batch_scores.tolist()
            
            all_scores.extend(batch_scores)
        
        return all_scores
    
    def _combine_ensemble_scores(self, model_scores: Dict[str, List[float]]) -> List[float]:
        """Combine scores from multiple models"""
        num_docs = len(next(iter(model_scores.values())))
        ensemble_scores = []
        
        for i in range(num_docs):
            weighted_score = 0.0
            total_weight = 0.0
            
            for model_name, scores in model_scores.items():
                weight = self.model_weights.get(model_name, 1.0)
                weighted_score += weight * scores[i]
                total_weight += weight
            
            ensemble_scores.append(weighted_score / total_weight if total_weight > 0 else 0.0)
        
        return ensemble_scores

            context_boost = " ".join(f"{term}^0.3" for term in context_terms[:5])

        self,

        original_query: str
    ) -> List[RerankingResult]:

        # Preference-based boosting
        for result in results:
            boost_factor = 1.0

                if content_type in self.user_preferences['preferred_content_type']:
                    boost_factor *= 1.1
            
            # Apply boost
            result.final_score *= boost_factor
        
        # Re-sort after adjustments
        results.sort(key=lambda x: x.final_score, reverse=True)
        
        return results

        valid_strategies = ["cascade", "parallel_fusion", "adaptive"]
        if self.fusion_strategy not in valid_strategies:
            raise ValueError(f"Fusion strategy must be one of {valid_strategies}")
    
    def retrieve(
        self, 
        query: str, 
        final_top_k: int = 20,
        stage_config: Dict[str, Any] = None
    ) -> PipelineResult:
        """
        Execute multi-stage retrieval pipeline
        
        Args:
            query: Search query
            final_top_k: Number of final results to return
            stage_config: Per-stage configuration overrides
        
        Returns:
            PipelineResult with documents and metadata
        """
        start_time = time.time()
        
        # Check cache
        cache_key = f"{query}:{final_top_k}:{self.fusion_strategy}"
        if self.cache and cache_key in self.cache:
            self.logger.info(f"Returning cached result for query: {query}")
            return self.cache[cache_key]
        
        stage_results = []
        current_candidates = None
        
        try:
            if self.fusion_strategy == "cascade":
                stage_results = self._execute_cascade_retrieval(
                    query, final_top_k, stage_config
                )
            elif self.fusion_strategy == "parallel_fusion":
                stage_results = self._execute_parallel_retrieval(
                    query, final_top_k, stage_config
                )
            elif self.fusion_strategy == "adaptive":
                stage_results = self._execute_adaptive_retrieval(
                    query, final_top_k, stage_config
                )
            
            # Combine results from all stages
            final_documents, final_scores = self._combine_stage_results(
                stage_results, final_top_k
            )
            
            # Create result object
            result = PipelineResult(
                final_documents=final_documents,
                final_scores=final_scores,
                stage_results=stage_results,
                total_execution_time=time.time() - start_time,
                pipeline_metadata={
                    "fusion_strategy": self.fusion_strategy,
                    "num_stages": len([s for s in self.stages if s.enabled]),
                    "query_length": len(query)
                }
            )
            
            # Cache result
            if self.cache:
                self.cache[cache_key] = result
            
            return result
            
        except Exception as e:
            self.logger.error(f"Pipeline execution failed: {str(e)}")
            raise
    
    def _execute_cascade_retrieval(
        self, 
        query: str, 
        final_top_k: int, 
        stage_config: Dict[str, Any]
    ) -> List[StageResult]:
        """Execute stages in cascade (each stage refines the previous)"""
        stage_results = []
        current_candidates = None
        
        for stage in self.stages:
            if not stage.enabled:
                continue
            
            stage_start = time.time()
            
            try:
                if current_candidates is None:
                    # First stage - retrieve from full corpus
                    documents, scores = self._execute_stage(
                        stage, query, None, stage_config
                    )
                else:
                    # Subsequent stages - refine previous results
                    documents, scores = self._execute_stage(
                        stage, query, current_candidates, stage_config
                    )
                
                # Create stage result
                stage_result = StageResult(
                    stage_name=stage.name,
                    documents=documents,
                    scores=scores,
                    execution_time=time.time() - stage_start,
                    metadata={"input_size": len(current_candidates) if current_candidates else "full_corpus"}
                )
                
                stage_results.append(stage_result)
                
                # Update candidates for next stage
                current_candidates = list(zip(documents, scores))
                
                self.logger.info(
                    f"Stage {stage.name} completed: {len(documents)} results in "
                    f"{stage_result.execution_time:.3f}s"
                )
                
            except Exception as e:
                self.logger.error(f"Stage {stage.name} failed: {str(e)}")
                # Continue with previous results if available
                if current_candidates is None:
                    raise  # Can't continue without any results
        
        return stage_results
    
    def _execute_parallel_retrieval(
        self, 
        query: str, 
        final_top_k: int, 
        stage_config: Dict[str, Any]
    ) -> List[StageResult]:
        """Execute all stages in parallel and fuse results"""
        stage_results = []
        
        # Prepare stages for parallel execution
        enabled_stages = [stage for stage in self.stages if stage.enabled]
        
        # Execute stages in parallel
        with ThreadPoolExecutor(max_workers=self.max_workers) as executor:
            # Submit all stages
            future_to_stage = {}
            for stage in enabled_stages:
                future = executor.submit(
                    self._execute_stage_with_timing, 
                    stage, query, None, stage_config
                )
                future_to_stage[future] = stage
            
            # Collect results
            for future in as_completed(future_to_stage, timeout=30):
                stage = future_to_stage[future]
                try:
                    stage_result = future.result()
                    stage_results.append(stage_result)
                    
                    self.logger.info(
                        f"Stage {stage.name} completed: {len(stage_result.documents)} results"
                    )
                    
                except Exception as e:
                    self.logger.error(f"Stage {stage.name} failed: {str(e)}")
        
        return stage_results
    
    def _execute_adaptive_retrieval(
        self, 
        query: str, 
        final_top_k: int, 
        stage_config: Dict[str, Any]
    ) -> List[StageResult]:
        """Execute stages adaptively based on intermediate results"""
        stage_results = []
        current_quality_score = 0.0
        quality_threshold = 0.8
        
        for stage in self.stages:
            if not stage.enabled:
                continue
            
            # Decide whether to execute this stage
            if current_quality_score >= quality_threshold and len(stage_results) > 1:
                self.logger.info(
                    f"Skipping stage {stage.name} due to sufficient quality: {current_quality_score:.3f}"
                )
                continue
            
            stage_start = time.time()
            
            try:
                # Execute stage
                if not stage_results:
                    documents, scores = self._execute_stage(stage, query, None, stage_config)
                else:
                    # Use best results from previous stages
                    prev_candidates = list(zip(
                        stage_results[-1].documents, 
                        stage_results[-1].scores
                    ))
                    documents, scores = self._execute_stage(stage, query, prev_candidates, stage_config)
                
                # Create stage result
                stage_result = StageResult(
                    stage_name=stage.name,
                    documents=documents,
                    scores=scores,
                    execution_time=time.time() - stage_start,
                    metadata={"quality_threshold": quality_threshold}
                )
                
                stage_results.append(stage_result)
                
                # Update quality score
                current_quality_score = self._assess_result_quality(documents, scores, query)
                stage_result.metadata["quality_score"] = current_quality_score
                
                self.logger.info(
                    f"Stage {stage.name} completed: quality={current_quality_score:.3f}"
                )
                
            except Exception as e:
                self.logger.error(f"Adaptive stage {stage.name} failed: {str(e)}")
        
        return stage_results
    
    def _execute_stage(
        self, 
        stage: RetrievalStage, 
        query: str, 
        candidates: Optional[List[Tuple]], 
        stage_config: Dict[str, Any]
    ) -> Tuple[List[Dict], List[float]]:
        """Execute a single retrieval stage"""
        
        # Get stage-specific configuration
        config = stage_config.get(stage.name, {}) if stage_config else {}
        top_k = config.get('top_k', stage.top_k)
        
        # Execute retriever
        if candidates is None:
            # First stage - search full corpus
            if hasattr(stage.retriever, 'search'):
                results = stage.retriever.search(query, top_k=top_k)
            elif hasattr(stage.retriever, 'retrieve'):
                results = stage.retriever.retrieve(query, top_k=top_k)
            else:
                raise ValueError(f"Stage {stage.name} retriever has no search/retrieve method")
        else:
            # Refinement stage - operate on candidates
            if hasattr(stage.retriever, 'rerank'):
                candidate_docs = [{'content': doc['content'] if isinstance(doc, dict) else doc[0]['content'], 
                                  'id': str(i)} for i, doc in enumerate(candidates)]
                candidate_scores = [score if isinstance(score, float) else score[1] 
                                  for score in candidates]
                
                results = stage.retriever.rerank(
                    query=query,
                    documents=candidate_docs,
                    initial_scores=candidate_scores,
                    top_k=top_k
                )
            else:
                # Fallback to search with limited candidates
                results = stage.retriever.search(query, top_k=top_k)
        
        # Extract documents and scores
        if isinstance(results, list) and results:
            if hasattr(results[0], 'content'):  # SearchResult or RerankingResult objects
                documents = [{'content': r.content, 'id': r.doc_id, 
                            'metadata': getattr(r, 'metadata', {})} for r in results]
                scores = [getattr(r, 'final_score', getattr(r, 'fusion_score', 1.0)) for r in results]
            elif isinstance(results[0], dict):
                documents = results
                scores = [doc.get('score', 1.0) for doc in documents]
            else:
                raise ValueError(f"Unexpected result format from stage {stage.name}")
        else:
            documents, scores = [], []
        
        return documents, scores
    
    def _execute_stage_with_timing(
        self, 
        stage: RetrievalStage, 
        query: str, 
        candidates: Optional[List[Tuple]], 
        stage_config: Dict[str, Any]
    ) -> StageResult:
        """Execute stage and return StageResult with timing"""
        start_time = time.time()
        
        documents, scores = self._execute_stage(stage, query, candidates, stage_config)
        
        return StageResult(
            stage_name=stage.name,
            documents=documents,
            scores=scores,
            execution_time=time.time() - start_time,
            metadata={"parallel_execution": True}
        )
    
    def _combine_stage_results(
        self, 
        stage_results: List[StageResult], 
        final_top_k: int
    ) -> Tuple[List[Dict], List[float]]:
        """Combine results from multiple stages"""
        
        if not stage_results:
            return [], []
        
        if self.fusion_strategy == "cascade":
            # Return results from last stage
            last_result = stage_results[-1]
            return last_result.documents[:final_top_k], last_result.scores[:final_top_k]
        
        elif self.fusion_strategy in ["parallel_fusion", "adaptive"]:
            # Combine results using weighted fusion
            doc_scores = {}  # doc_id -> (document, total_score, count)
            
            for stage_result in stage_results:
                stage = next(s for s in self.stages if s.name == stage_result.stage_name)
                weight = stage.weight
                
                for doc, score in zip(stage_result.documents, stage_result.scores):
                    doc_id = doc.get('id', doc['content'][:50])  # Use content prefix as fallback ID
                    
                    if doc_id in doc_scores:
                        current_doc, current_score, count = doc_scores[doc_id]
                        doc_scores[doc_id] = (current_doc, current_score + weight * score, count + 1)
                    else:
                        doc_scores[doc_id] = (doc, weight * score, 1)
            
            # Calculate final scores and sort
            final_items = []
            for doc_id, (doc, total_score, count) in doc_scores.items():
                # Average weighted score
                final_score = total_score / count
                final_items.append((doc, final_score))
            
            # Sort by final score
            final_items.sort(key=lambda x: x[1], reverse=True)
            
            # Extract top-k
            final_documents = [doc for doc, score in final_items[:final_top_k]]
            final_scores = [score for doc, score in final_items[:final_top_k]]
            
            return final_documents, final_scores
    
    def _assess_result_quality(self, documents: List[Dict], scores: List[float], query: str) -> float:
        """Assess quality of results for adaptive execution"""
        if not documents or not scores:
            return 0.0
        
        # Simple quality metrics
        avg_score = sum(scores) / len(scores)
        score_variance = sum((s - avg_score) ** 2 for s in scores) / len(scores)
        
        # Higher average score and lower variance indicate better quality
        quality_score = avg_score * (1.0 - min(score_variance, 1.0))
        
        return quality_score

class RetrievalMonitor:
    """Monitor and track retrieval system performance"""
    
    def __init__(self, window_size: int = 1000):
        self.window_size = window_size
        self.query_times = deque(maxlen=window_size)
        self.query_scores = deque(maxlen=window_size)
        self.cache_hits = deque(maxlen=window_size)
        self.errors = deque(maxlen=window_size)
        self.stage_times = defaultdict(lambda: deque(maxlen=window_size))
        
        self.logger = logging.getLogger(__name__)
    
    def record_query(
        self,
        query: str,
        response_time: float,
        avg_score: float,
        cache_hit: bool = False,
        error: bool = False,
        stage_times: Dict[str, float] = None
    ):
        """Record metrics for a single query"""
        
        self.query_times.append(response_time)
        self.query_scores.append(avg_score)
        self.cache_hits.append(1 if cache_hit else 0)
        self.errors.append(1 if error else 0)
        
        if stage_times:
            for stage_name, stage_time in stage_times.items():
                self.stage_times[stage_name].append(stage_time)
    
    def get_metrics(self) -> RetrievalMetrics:
        """Get current performance metrics"""
        
        if not self.query_times:
            return RetrievalMetrics()
        
        return RetrievalMetrics(
            query_count=len(self.query_times),
            avg_response_time=statistics.mean(self.query_times),
            avg_retrieval_score=statistics.mean(self.query_scores),
            cache_hit_rate=statistics.mean(self.cache_hits),
            error_rate=statistics.mean(self.errors),
            stage_performance={
                stage: statistics.mean(times) 
                for stage, times in self.stage_times.items()
            }
        )
    
    def log_performance_summary(self):
        """Log current performance summary"""
        metrics = self.get_metrics()
        
        self.logger.info(f"Retrieval Performance Summary:")
        self.logger.info(f"  Queries processed: {metrics.query_count}")
        self.logger.info(f"  Avg response time: {metrics.avg_response_time:.3f}s")
        self.logger.info(f"  Avg retrieval score: {metrics.avg_retrieval_score:.3f}")
        self.logger.info(f"  Cache hit rate: {metrics.cache_hit_rate:.3f}")
        self.logger.info(f"  Error rate: {metrics.error_rate:.3f}")
        
        for stage, avg_time in metrics.stage_performance.items():
            self.logger.info(f"  {stage}: {avg_time:.3f}s")

        """Search with monitoring"""
        start_time = time.time()
        error = False
        cache_hit = False
        
        try:
            # Execute search
            result = self.pipeline.retrieve(query, final_top_k=top_k)
            
            # Calculate metrics
            response_time = time.time() - start_time
            avg_score = statistics.mean(result.final_scores) if result.final_scores else 0.0
            
            # Check if result was cached
            cache_hit = response_time < 0.01  # Simplified cache detection
            
            # Extract stage timing
            stage_times = {
                stage_result.stage_name: stage_result.execution_time
                for stage_result in result.stage_results
            }
            
            # Record metrics
            self.monitor.record_query(
                query=query,
                response_time=response_time,
                avg_score=avg_score,
                cache_hit=cache_hit,
                error=error,
                stage_times=stage_times
            )
            
            return {
                'documents': result.final_documents,
                'scores': result.final_scores,
                'metadata': {
                    'response_time': response_time,
                    'stage_count': len(result.stage_results),
                    'total_pipeline_time': result.total_execution_time
                }
            }
            
        except Exception as e:
            error = True
            response_time = time.time() - start_time
            
            self.monitor.record_query(
                query=query,
                response_time=response_time,
                avg_score=0.0,
                cache_hit=False,
                error=True
            )
            
            raise e
