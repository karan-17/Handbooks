🚀 CI/CD Practices

Security-First CI in 2025

Modern CI/CD pipelines integrate security checks from the first commit, implementing shift-left security practices.

# GitHub Actions Security-First Pipeline
name: Secure CI/CD Pipeline

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main ]

jobs:
  security-scan:
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v3
    
    - name: Run Trivy vulnerability scanner
      uses: aquasecurity/trivy-action@master
      with:
        scan-type: 'fs'
        scan-ref: '.'
        format: 'sarif'
        output: 'trivy-results.sarif'
    
    - name: SAST Scan
      uses: github/codeql-action/init@v2
      with:
        languages: javascript, python
    
    - name: Secrets Detection
      uses: gitguardian/ggshield-action@v1
      env:
        GITGUARDIAN_API_KEY: ${{ secrets.GITGUARDIAN_API_KEY }}

  build-and-test:
    needs: security-scan
    runs-on: ubuntu-latest
    steps:
    - name: Build Application
      run: |
        docker build -t app:${{ github.sha }} .
        
    - name: Run Tests
      run: |
        npm test
        npm run test:integration
        
    - name: Container Security Scan
      run: |
        trivy image app:${{ github.sha }}

    npm audit --audit-level high

FROM node:18-alpine AS runtime
# Create non-root user
RUN addgroup -g 1001 -S nodejs && \
    adduser -S nextjs -u 1001

# Install security updates
RUN apk update && apk upgrade && \
    rm -rf /var/cache/apk/*

WORKDIR /app
COPY --from=builder --chown=nextjs:nodejs /app .

# Security configurations
USER nextjs
EXPOSE 3000
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD curl -f http://localhost:3000/health || exit 1

# Read-only root filesystem
CMD ["node", "server.js"]
