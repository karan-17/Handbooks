5. Embedding Optimization Techniques

Compression Strategies

Technique Compression Ratio Performance Impact Best Use Case
Float8 Quantization 4x <0.3% degradation Production systems
Int8 Quantization 4x Moderate degradation Resource-constrained
PCA (50% dimensions) 2x Minimal impact High-dimensional data
PCA + Float8 8x <1% degradation Optimal balance
Binary Quantization 32x Significant loss Approximate search

Dimensionality Reduction

# PCA Dimensionality Reduction from sklearn.decomposition import PCA import numpy as np # Original embeddings (e.g., 768-dimensional BERT) embeddings = np.random.random((10000, 768)).astype('float32') # Reduce to 256 dimensions (retains ~95% variance) pca = PCA(n_components=256) reduced_embeddings = pca.fit_transform(embeddings) print(f"Original shape: {embeddings.shape}") print(f"Reduced shape: {reduced_embeddings.shape}") print(f"Variance retained: {pca.explained_variance_ratio_.sum():.3f}") # Quantization after PCA def quantize_float8(embeddings): """Simulate float8 quantization""" return np.round(embeddings * 127) / 127 quantized = quantize_float8(reduced_embeddings) compression_ratio = embeddings.nbytes / quantized.nbytes print(f"Total compression ratio: {compression_ratio:.1f}x")

6. Multi-Modal Embeddings (Text + Images)

CLIP Architecture and Capabilities

CLIP Key Features:
  • Dual Encoders: Text transformer + Vision transformer
  • Shared Space: 512-dimensional embeddings for both modalities
  • Training Data: 400 million image-text pairs
  • Zero-shot: Classification without explicit training

CLIP Implementation Examples

# CLIP for Multi-modal Embeddings import torch import clip from PIL import Image # Load CLIP model device = "cuda" if torch.cuda.is_available() else "cpu" model, preprocess = clip.load("ViT-B/32", device=device) # Process image image = preprocess(Image.open("image.jpg")).unsqueeze(0).to(device) text = clip.tokenize(["a photo of a cat", "a photo of a dog"]).to(device) # Generate embeddings with torch.no_grad(): image_features = model.encode_image(image) text_features = model.encode_text(text) # Calculate similarities logits_per_image, logits_per_text = model(image, text) probs = logits_per_image.softmax(dim=-1).cpu().numpy() print("Image-text similarity probabilities:", probs) # OpenAI API approach (if available) # Using OpenAI's CLIP via API import openai client = openai.OpenAI() # Text embedding text_embedding = client.embeddings.create( model="clip-text-encoder", input="a photo of a cat" ) # Note: Image embedding would require separate endpoint # image_embedding = client.embeddings.create( # model="clip-image-encoder", # input=base64_encoded_image # )

Multi-modal RAG Implementation

# Multi-modal RAG with CLIP embeddings from sentence_transformers import SentenceTransformer import numpy as np from PIL import Image class MultiModalRAG: def __init__(self): self.clip_model = SentenceTransformer('clip-ViT-B-32') self.text_embeddings = [] self.image_embeddings = [] self.documents = [] self.images = [] def add_text_document(self, text): embedding = self.clip_model.encode([text]) self.text_embeddings.append(embedding[0]) self.documents.append(text) def add_image_document(self, image_path): image = Image.open(image_path) embedding = self.clip_model.encode([image]) self.image_embeddings.append(embedding[0]) self.images.append(image_path) def search(self, query, modality="text", top_k=5): if modality == "text": query_embedding = self.clip_model.encode([query])[0] # Search both text and image embeddings all_embeddings = self.text_embeddings + self.image_embeddings similarities = [np.dot(query_embedding, emb) for emb in all_embeddings] # Return top results top_indices = np.argsort(similarities)[-top_k:][::-1] results = [] for idx in top_indices: if idx < len(self.documents): results.append(("text", self.documents[idx], similarities[idx])) else: img_idx = idx - len(self.documents) results.append(("image", self.images[img_idx], similarities[idx])) return results # Usage example rag = MultiModalRAG() rag.add_text_document("A technical specification document") rag.add_image_document("diagram.png") results = rag.search("technical diagram", top_k=3) for modality, content, score in results: print(f"{modality}: {content} (score: {score:.3f})")
