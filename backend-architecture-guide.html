Core System Components

  • Controllers: Handle incoming requests and route to appropriate services
  • Services: Contain business logic and orchestrate operations
  • Repositories: Abstract data access and persistence operations
  • Models/Entities: Represent business objects and data structures
  • Middleware: Cross-cutting concerns like logging, authentication
  • Validators: Input validation and sanitization
  • Error Handlers: Centralized error processing and responses

Infrastructure Components

  • Load Balancers: Distribute incoming requests across multiple servers
  • Databases: Primary data storage (SQL/NoSQL)
  • Caches: Redis, Memcached for performance optimization
  • Message Queues: Asynchronous processing (RabbitMQ, Kafka)
  • CDN: Content delivery networks for static assets
  • Monitoring: Logging, metrics, and alerting systems
  • Security: Firewalls, encryption, rate limiting

MVC (Model-View-Controller) Architecture

MVC is one of the most fundamental architectural patterns, providing clear separation of concerns by dividing applications into three interconnected components.

🏗️

Core Components

  • Model: Manages data, business logic, and rules independent of UI
  • View: Handles presentation layer and user interface rendering
  • Controller: Routes commands, processes user input, coordinates between Model and View

const bcrypt = require('bcrypt');
const db = require('./database');

class User {
    constructor(userData) {
        this.id = userData.id;
        this.email = userData.email;
        this.username = userData.username;
        this.password = userData.password;
        this.createdAt = userData.createdAt;
        this.updatedAt = userData.updatedAt;
    }

    // Business Logic Methods
    static async create(userData) {
        const hashedPassword = await bcrypt.hash(userData.password, 10);
        
        const query = `
            INSERT INTO users (email, username, password, created_at)
            VALUES ($1, $2, $3, NOW())
            RETURNING *
        `;
        
        const values = [userData.email, userData.username, hashedPassword];
        const result = await db.query(query, values);
        
        return new User(result.rows[0]);
    }

    static async findById(id) {
        const query = 'SELECT * FROM users WHERE id = $1';
        const result = await db.query(query, [id]);
        
        if (result.rows.length === 0) {
            throw new Error('User not found');
        }
        
        return new User(result.rows[0]);
    }

    static async findByEmail(email) {
        const query = 'SELECT * FROM users WHERE email = $1';
        const result = await db.query(query, [email]);
        
        if (result.rows.length === 0) {
            return null;
        }
        
        return new User(result.rows[0]);
    }

    async validatePassword(password) {
        return await bcrypt.compare(password, this.password);
    }

    async update(userData) {
        const fields = [];
        const values = [];
        let paramCount = 1;

        Object.keys(userData).forEach(key => {
            if (userData[key] !== undefined && key !== 'id') {
                fields.push(`${key} = $${paramCount}`);
                values.push(userData[key]);
                paramCount++;
            }
        });

        if (fields.length === 0) {
            throw new Error('No fields to update');
        }

        const query = `
            UPDATE users 
            SET ${fields.join(', ')}, updated_at = NOW()
            WHERE id = $${paramCount}
            RETURNING *
        `;
        
        values.push(this.id);
        const result = await db.query(query, values);
        
        return new User(result.rows[0]);
    }

    async delete() {
        const query = 'DELETE FROM users WHERE id = $1';
        await db.query(query, [this.id]);
    }

    // Return safe user data (without password)
    toJSON() {
        return {
            id: this.id,
            email: this.email,
            username: this.username,
            createdAt: this.createdAt,
            updatedAt: this.updatedAt
        };
    }
}

module.exports = User;

controllers/userController.js 

const User = require('../models/user');
const jwt = require('jsonwebtoken');
const { validationResult } = require('express-validator');

class UserController {
    // Display user registration form
    static showRegister(req, res) {
        res.render('user/register', {
            title: 'Create Account',
            errors: [],
            formData: {}
        });
    }

    // Handle user registration
    static async register(req, res) {
        try {
            // Check validation errors
            const errors = validationResult(req);
            if (!errors.isEmpty()) {
                return res.status(400).render('user/register', {
                    title: 'Create Account',
                    errors: errors.array(),
                    formData: req.body
                });
            }

            // Check if user already exists
            const existingUser = await User.findByEmail(req.body.email);
            if (existingUser) {
                return res.status(400).render('user/register', {
                    title: 'Create Account',
                    errors: [{ msg: 'User with this email already exists' }],
                    formData: req.body
                });
            }

            // Create new user
            const userData = {
                email: req.body.email,
                username: req.body.username,
                password: req.body.password
            };

            const newUser = await User.create(userData);

            // Generate JWT token
            const token = jwt.sign(
                { userId: newUser.id, email: newUser.email },
                process.env.JWT_SECRET,
                { expiresIn: '24h' }
            );

            // Set secure cookie
            res.cookie('auth_token', token, {
                httpOnly: true,
                secure: process.env.NODE_ENV === 'production',
                maxAge: 24 * 60 * 60 * 1000 // 24 hours
            });

            res.redirect('/dashboard');

        } catch (error) {
            console.error('Registration error:', error);
            res.status(500).render('user/register', {
                title: 'Create Account',
                errors: [{ msg: 'An error occurred during registration' }],
                formData: req.body
            });
        }
    }

    // Display login form
    static showLogin(req, res) {
        res.render('user/login', {
            title: 'Login',
            errors: []
        });
    }

    // Handle user login
    static async login(req, res) {
        try {
            const errors = validationResult(req);
            if (!errors.isEmpty()) {
                return res.status(400).render('user/login', {
                    title: 'Login',
                    errors: errors.array()
                });
            }

            const { email, password } = req.body;

            // Find user by email
            const user = await User.findByEmail(email);
            if (!user) {
                return res.status(400).render('user/login', {
                    title: 'Login',
                    errors: [{ msg: 'Invalid email or password' }]
                });
            }

            // Validate password
            const isValidPassword = await user.validatePassword(password);
            if (!isValidPassword) {
                return res.status(400).render('user/login', {
                    title: 'Login',
                    errors: [{ msg: 'Invalid email or password' }]
                });
            }

            // Generate JWT token
            const token = jwt.sign(
                { userId: user.id, email: user.email },
                process.env.JWT_SECRET,
                { expiresIn: '24h' }
            );

            res.cookie('auth_token', token, {
                httpOnly: true,
                secure: process.env.NODE_ENV === 'production',
                maxAge: 24 * 60 * 60 * 1000
            });

            res.redirect('/dashboard');

        } catch (error) {
            console.error('Login error:', error);
            res.status(500).render('user/login', {
                title: 'Login',
                errors: [{ msg: 'An error occurred during login' }]
            });
        }
    }

    // Display user profile
    static async showProfile(req, res) {
        try {
            const user = await User.findById(req.user.userId);
            res.render('user/profile', {
                title: 'User Profile',
                user: user.toJSON()
            });
        } catch (error) {
            console.error('Profile error:', error);
            res.status(500).send('Error loading profile');
        }
    }

    // Update user profile
    static async updateProfile(req, res) {
        try {
            const errors = validationResult(req);
            if (!errors.isEmpty()) {
                return res.status(400).json({
                    success: false,
                    errors: errors.array()
                });
            }

            const user = await User.findById(req.user.userId);
            const updatedUser = await user.update(req.body);

            res.json({
                success: true,
                message: 'Profile updated successfully',
                user: updatedUser.toJSON()
            });

        } catch (error) {
            console.error('Update profile error:', error);
            res.status(500).json({
                success: false,
                message: 'Error updating profile'
            });
        }
    }

    // Get user by ID (API endpoint)
    static async getUser(req, res) {
        try {
            const user = await User.findById(req.params.id);
            res.json({
                success: true,
                user: user.toJSON()
            });
        } catch (error) {
            console.error('Get user error:', error);
            res.status(404).json({
                success: false,
                message: 'User not found'
            });
        }
    }

    // Delete user account
    static async deleteUser(req, res) {
        try {
            const user = await User.findById(req.user.userId);
            await user.delete();

            // Clear auth cookie
            res.clearCookie('auth_token');
            
            res.json({
                success: true,
                message: 'Account deleted successfully'
            });

        } catch (error) {
            console.error('Delete user error:', error);
            res.status(500).json({
                success: false,
                message: 'Error deleting account'
            });
        }
    }

    // Logout user
    static logout(req, res) {
        res.clearCookie('auth_token');
        res.redirect('/login');
    }
}

Benefits of MVC Architecture

  • Enhanced Organization: Clear separation improves codebase structure and maintainability
  • Parallel Development: Teams can work on different components simultaneously
  • Code Reusability: Multiple views can share the same model components
  • Improved Testability: Decoupled nature facilitates unit testing
  • Separation of Concerns: UI, business logic, and data are clearly separated

Drawbacks and Challenges

  • Increased Complexity: Can be overkill for simple applications
  • Learning Curve: Requires understanding of separation principles
  • Initial Setup Overhead: More planning and structure required
  • Potential Over-Engineering: May add unnecessary abstraction for small projects

When to Use MVC

  • Complex Applications: Multiple user interfaces and business logic components
  • Team Development: Multiple developers working on different aspects
  • Reusable Components: Business logic needs to be shared across views
  • Testing Requirements: Need comprehensive unit testing capabilities
  • Long-term Maintenance: Projects requiring long-term maintainability

MVVM (Model-View-ViewModel) Architecture

MVVM extends the separation of concerns concept by introducing a ViewModel layer that acts as a binding layer between the View and Model, enabling powerful data binding capabilities. The ViewModel serves as an abstraction of the View, exposing public properties and commands that the View can bind to.

🔗

Core Components

  • Model: Business logic, data access, and domain entities
  • View: User interface and presentation logic
  • ViewModel: Abstraction layer exposing data and commands for View binding

class UserService {
    constructor(apiClient) {
        this.apiClient = apiClient;
        this.cache = new Map();
    }

    // Get user by ID with caching
    async getUser(userId) {
        if (this.cache.has(userId)) {
            return this.cache.get(userId);
        }

        try {
            const response = await this.apiClient.get(`/users/${userId}`);
            const user = new User(response.data);
            this.cache.set(userId, user);
            return user;
        } catch (error) {
            throw new Error(`Failed to fetch user: ${error.message}`);
        }
    }

    // Update user profile
    async updateUser(userId, userData) {
        try {
            const response = await this.apiClient.put(`/users/${userId}`, userData);
            const updatedUser = new User(response.data);
            
            // Update cache
            this.cache.set(userId, updatedUser);
            
            return updatedUser;
        } catch (error) {
            throw new Error(`Failed to update user: ${error.message}`);
        }
    }

    // Search users with pagination
    async searchUsers(query, page = 1, limit = 10) {
        try {
            const response = await this.apiClient.get('/users/search', {
                params: { q: query, page, limit }
            });
            
            return {
                users: response.data.users.map(userData => new User(userData)),
                totalCount: response.data.totalCount,
                currentPage: page,
                hasNextPage: response.data.hasNextPage
            };
        } catch (error) {
            throw new Error(`Failed to search users: ${error.message}`);
        }
    }

    // Validate user data before operations
    validateUserData(userData) {
        const errors = [];
        
        if (!userData.email || !/\S+@\S+\.\S+/.test(userData.email)) {
            errors.push('Valid email is required');
        }
        
        if (!userData.username || userData.username.length < 3) {
            errors.push('Username must be at least 3 characters');
        }
        
        if (userData.password && userData.password.length < 8) {
            errors.push('Password must be at least 8 characters');
        }
        
        return errors;
    }

    // Clear cache
    clearCache() {
        this.cache.clear();
    }
}

class User {
    constructor(data) {
        this.id = data.id;
        this.email = data.email;
        this.username = data.username;
        this.firstName = data.firstName || '';
        this.lastName = data.lastName || '';
        this.avatar = data.avatar || null;
        this.isActive = data.isActive !== false;
        this.createdAt = data.createdAt ? new Date(data.createdAt) : new Date();
        this.updatedAt = data.updatedAt ? new Date(data.updatedAt) : new Date();
    }

    // Computed properties
    get fullName() {
        return `${this.firstName} ${this.lastName}`.trim() || this.username;
    }

    get displayAvatar() {
        return this.avatar || '/images/default-avatar.png';
    }

    get isNewUser() {
        const daysSinceCreation = (new Date() - this.createdAt) / (1000 * 60 * 60 * 24);
        return daysSinceCreation <= 7;
    }

    // Methods
    toJSON() {
        return {
            id: this.id,
            email: this.email,
            username: this.username,
            firstName: this.firstName,
            lastName: this.lastName,
            fullName: this.fullName,
            avatar: this.avatar,
            displayAvatar: this.displayAvatar,
            isActive: this.isActive,
            isNewUser: this.isNewUser,
            createdAt: this.createdAt,
            updatedAt: this.updatedAt
        };
    }

    update(newData) {
        Object.assign(this, newData);
        this.updatedAt = new Date();
        return this;
    }
}

import { useState, useEffect, useCallback, useMemo } from 'react';
import { UserService } from '../models/userService';
import { useNotification } from '../hooks/useNotification';

// Custom hook that serves as ViewModel
export const useUser = (userId, userService) => {
    // State management
    const [user, setUser] = useState(null);
    const [isLoading, setIsLoading] = useState(false);
    const [isUpdating, setIsUpdating] = useState(false);
    const [error, setError] = useState(null);
    const [validationErrors, setValidationErrors] = useState([]);
    
    const { showNotification } = useNotification();

    // Computed properties (memoized)
    const userDisplayData = useMemo(() => {
        if (!user) return null;
        
        return {
            ...user.toJSON(),
            initials: user.fullName.split(' ').map(n => n[0]).join('').toUpperCase(),
            membershipDuration: Math.floor((new Date() - user.createdAt) / (1000 * 60 * 60 * 24)),
            statusBadge: user.isActive ? 'Active' : 'Inactive',
            statusColor: user.isActive ? 'success' : 'warning'
        };
    }, [user]);

    // Load user data
    const loadUser = useCallback(async () => {
        if (!userId) return;
        
        setIsLoading(true);
        setError(null);
        
        try {
            const userData = await userService.getUser(userId);
            setUser(userData);
        } catch (err) {
            setError(err.message);
            showNotification(err.message, 'error');
        } finally {
            setIsLoading(false);
        }
    }, [userId, userService, showNotification]);

    // Update user profile
    const updateProfile = useCallback(async (updates) => {
        if (!user) return false;
        
        // Validate input
        const errors = userService.validateUserData(updates);
        setValidationErrors(errors);
        
        if (errors.length > 0) {
            showNotification('Please fix validation errors', 'warning');
            return false;
        }
        
        setIsUpdating(true);
        setError(null);
        
        try {
            const updatedUser = await userService.updateUser(user.id, updates);
            setUser(updatedUser);
            showNotification('Profile updated successfully!', 'success');
            return true;
        } catch (err) {
            setError(err.message);
            showNotification(err.message, 'error');
            return false;
        } finally {
            setIsUpdating(false);
        }
    }, [user, userService, showNotification]);

    // Refresh user data
    const refreshUser = useCallback(() => {
        userService.clearCache();
        loadUser();
    }, [loadUser, userService]);

    // Reset validation errors
    const clearValidationErrors = useCallback(() => {
        setValidationErrors([]);
    }, []);

    // Upload avatar
    const uploadAvatar = useCallback(async (file) => {
        if (!user || !file) return false;
        
        setIsUpdating(true);
        
        try {
            // Create form data
            const formData = new FormData();
            formData.append('avatar', file);
            
            // Upload via service (assuming the service handles file uploads)
            const response = await fetch(`/api/users/${user.id}/avatar`, {
                method: 'POST',
                body: formData
            });
            
            if (!response.ok) {
                throw new Error('Failed to upload avatar');
            }
            
            const result = await response.json();
            
            // Update user with new avatar URL
            const updatedUser = user.update({ avatar: result.avatarUrl });
            setUser(updatedUser);
            
            showNotification('Avatar updated successfully!', 'success');
            return true;
        } catch (err) {
            setError(err.message);
            showNotification(err.message, 'error');
            return false;
        } finally {
            setIsUpdating(false);
        }
    }, [user, showNotification]);

    // Form helpers for two-way binding
    const createFormField = useCallback((fieldName) => {
        return {
            value: user?.[fieldName] || '',
            onChange: (value) => {
                if (user) {
                    const updatedUser = { ...user, [fieldName]: value };
                    setUser(new UserService.User(updatedUser));
                }
            },
            error: validationErrors.find(err => err.includes(fieldName.toLowerCase()))
        };
    }, [user, validationErrors]);

    // Load user on mount or when userId changes
    useEffect(() => {
        loadUser();
    }, [loadUser]);

    // Public interface (what the View binds to)
    return {
        // Data
        user: userDisplayData,
        isLoading,
        isUpdating,
        error,
        validationErrors,
        
        // Computed properties
        hasUser: !!user,
        canEdit: !!user && !isUpdating,
        
        // Commands
        updateProfile,
        refreshUser,
        uploadAvatar,
        clearValidationErrors,
        
        // Form binding helpers
        createFormField,
        
        // Utility methods
        retry: loadUser
    };
};

interface UserProfileViewModel {

  selector: 'app-user-profile',

export class UserProfileComponent implements OnInit, OnDestroy {
  // Private subjects for state management
  private userSubject = new BehaviorSubject(null);
  private loadingSubject = new BehaviorSubject(false);
  private updatingSubject = new BehaviorSubject(false);
  private errorSubject = new BehaviorSubject(null);
  private validationErrorsSubject = new BehaviorSubject([]);
  private destroy$ = new Subject();

  // Public observables (ViewModel)
  public viewModel$: Observable;
  public profileForm: FormGroup;
  public isEditing = false;

  constructor(
    private userService: UserService,
    private formBuilder: FormBuilder,
    private notificationService: NotificationService
  ) {
    this.initializeForm();
    this.initializeViewModel();
  }

  ngOnInit(): void {
    this.loadUser();

  }

    this.destroy$.next();

  private initializeForm(): void {
    this.profileForm = this.formBuilder.group({
      firstName: ['', Validators.required],
      lastName: ['', Validators.required],
      email: ['', [Validators.required, Validators.email]],
      username: ['', [Validators.required, Validators.minLength(3)]]
    });
  }

  // Initialize ViewModel as combined observables
  private initializeViewModel(): void {
    this.viewModel$ = combineLatest([
      this.userSubject.asObservable(),
      this.loadingSubject.asObservable(),
      this.updatingSubject.asObservable(),
      this.errorSubject.asObservable(),
      this.validationErrorsSubject.asObservable()
    ]).pipe(
      map(([user, isLoading, isUpdating, error, validationErrors]) => {
        const displayData = user ? {
          ...user,
          fullName: `${user.firstName} ${user.lastName}`.trim() || user.username,
          initials: user.firstName && user.lastName 
            ? `${user.firstName[0]}${user.lastName[0]}`.toUpperCase()
            : user.username.substring(0, 2).toUpperCase(),
          membershipDuration: Math.floor(
            (new Date().getTime() - new Date(user.createdAt).getTime()) / (1000 * 60 * 60 * 24)
          ),
          statusBadge: user.isActive ? 'Active' : 'Inactive',
          statusColor: user.isActive ? 'success' : 'warning',
          displayAvatar: user.avatar || '/assets/images/default-avatar.png'
        } : null;

        return {
          user,
          isLoading,
          isUpdating,
          error,
          validationErrors,
          canEdit: !!user && !isUpdating,
          displayData
        };
      })
    );
  }

  // Setup real-time form validation
  private setupFormValidation(): void {
    this.profileForm.valueChanges.pipe(
      debounceTime(300),
      distinctUntilChanged(),
      takeUntil(this.destroy$)
    ).subscribe(() => {
      this.validateForm();
    });
  }

  // Load user data
  private async loadUser(): Promise {
    this.loadingSubject.next(true);
    this.errorSubject.next(null);

    try {
      // Assuming we get userId from route params or service
      const userId = 1; // Replace with actual user ID
      const user = await this.userService.getUser(userId).toPromise();
      
      this.userSubject.next(user);
      this.populateForm(user);
    } catch (error) {
      this.errorSubject.next('Failed to load user profile');
      this.notificationService.showError('Failed to load user profile');
    } finally {
      this.loadingSubject.next(false);
    }
  }

  private populateForm(user: User): void {
    this.profileForm.patchValue({
      firstName: user.firstName,
      lastName: user.lastName,
      email: user.email,
      username: user.username
    });
  }

  // Validate form and update validation errors
  private validateForm(): void {
    const errors: string[] = [];
    const formErrors = this.profileForm.errors;
    
    if (formErrors) {
      Object.keys(formErrors).forEach(key => {
        switch (key) {
          case 'required':
            errors.push('All fields are required');
            break;
          case 'email':
            errors.push('Please enter a valid email address');
            break;
          case 'minlength':
            errors.push('Username must be at least 3 characters long');
            break;
        }
      });
    }

    this.validationErrorsSubject.next(errors);
  }

  // Public methods (Commands)
  public startEditing(): void {
    this.isEditing = true;
    const currentUser = this.userSubject.value;
    if (currentUser) {
      this.populateForm(currentUser);
    }
  }

  public cancelEditing(): void {
    this.isEditing = false;
    const currentUser = this.userSubject.value;
    if (currentUser) {
      this.populateForm(currentUser);
    }
    this.validationErrorsSubject.next([]);
  }

  public async saveProfile(): Promise {
    if (this.profileForm.invalid) {
      this.validateForm();
      return;
    }

    this.updatingSubject.next(true);
    this.errorSubject.next(null);

    try {
      const currentUser = this.userSubject.value;
      if (!currentUser) return;

      const formData = this.profileForm.value;
      const updatedUser = await this.userService
        .updateUser(currentUser.id, formData)
        .toPromise();
      
      this.userSubject.next(updatedUser);
      this.isEditing = false;
      this.notificationService.showSuccess('Profile updated successfully!');
    } catch (error) {
      this.errorSubject.next('Failed to update profile');
      this.notificationService.showError('Failed to update profile');
    } finally {
      this.updatingSubject.next(false);
    }
  }

  public async refreshProfile(): Promise {
    await this.loadUser();
  }

  public async uploadAvatar(file: File): Promise {
    this.updatingSubject.next(true);

    try {
      const currentUser = this.userSubject.value;
      if (!currentUser) return;

      const avatarUrl = await this.userService
        .uploadAvatar(currentUser.id, file)
        .toPromise();
      
      const updatedUser = { ...currentUser, avatar: avatarUrl };
      this.userSubject.next(updatedUser);
      
      this.notificationService.showSuccess('Avatar updated successfully!');
    } catch (error) {
      this.errorSubject.next('Failed to upload avatar');
      this.notificationService.showError('Failed to upload avatar');
    } finally {
      this.updatingSubject.next(false);
    }
  }

MVVM vs MVC Comparison

Aspect MVC MVVM
Data Binding Manual updates between View and Model Automatic two-way data binding
View Logic Controller handles view logic ViewModel abstracts view concerns
Testability Controller testing requires mocking views ViewModel easily testable without UI
Complexity Simpler, more straightforward More complex setup, but powerful
UI Framework Support Universal pattern Best with frameworks supporting data binding
Separation of Concerns Good separation Excellent separation with binding layer

Benefits of MVVM Architecture

  • Powerful Data Binding: Automatic synchronization between view and data
  • Enhanced Testability: ViewModels can be tested independently of UI
  • Clean Separation: Clear distinction between business logic and presentation
  • Reactive Programming: Supports observable patterns and reactive updates
  • Reusable ViewModels: Same ViewModel can serve multiple views
  • Maintainable Code: Changes in business logic don't affect view layer

Drawbacks and Challenges

  • Learning Curve: Requires understanding of reactive programming concepts
  • Framework Dependency: Works best with specific UI frameworks
  • Memory Management: Risk of memory leaks with improper observable cleanup
  • Over-Engineering: Can be excessive for simple CRUD applications
  • Debugging Complexity: Data binding can make debugging more challenging

When to Use MVVM

  • Rich UI Applications: Complex user interfaces with dynamic data
  • Real-time Updates: Applications requiring live data synchronization
  • Cross-platform Development: Sharing ViewModels across platforms
  • Large Teams: Clear separation enables parallel development
  • Testable Applications: High requirements for unit test coverage
  • Reactive Applications: Heavy use of observables and reactive patterns

Clean Architecture

Clean Architecture, introduced by Robert Martin, is a software design philosophy that emphasizes separation of concerns through dependency inversion. It creates systems that are independent of frameworks, UI, databases, and external agencies.

🎯

Core Principles

  • Dependency Inversion: High-level modules should not depend on low-level modules
  • Separation of Concerns: Each layer has a single responsibility
  • Independence: Framework, UI, database, and external agency independence
  • Testability: Business rules can be tested without external elements

const Email = require('../value-objects/Email');

class User {
    constructor(id, email, username, firstName, lastName, createdAt = new Date()) {
        this.validateConstructorParameters(id, email, username);
        
        this.id = id;
        this.email = new Email(email);
        this.username = username;
        this.firstName = firstName || '';
        this.lastName = lastName || '';
        this.isActive = true;
        this.createdAt = createdAt;
        this.updatedAt = createdAt;
        this.lastLoginAt = null;
    }

    // Business rules and validation
    validateConstructorParameters(id, email, username) {
        if (!id) {
            throw new Error('User ID is required');
        }
        
        if (!email) {
            throw new Error('Email is required');
        }
        
        if (!username || username.length < 3) {
            throw new Error('Username must be at least 3 characters long');
        }
        
        if (username.length > 50) {
            throw new Error('Username cannot exceed 50 characters');
        }
    }

    // Domain methods
    updateProfile(firstName, lastName) {
        if (firstName !== undefined) {
            this.firstName = firstName.trim();
        }
        
        if (lastName !== undefined) {
            this.lastName = lastName.trim();
        }
        
        this.updatedAt = new Date();
        return this;
    }

    changeEmail(newEmail) {
        const emailVO = new Email(newEmail);
        this.email = emailVO;
        this.updatedAt = new Date();
        return this;
    }

    deactivate() {
        if (!this.isActive) {
            throw new Error('User is already deactivated');
        }
        
        this.isActive = false;
        this.updatedAt = new Date();
        return this;
    }

    activate() {
        if (this.isActive) {
            throw new Error('User is already active');
        }
        
        this.isActive = true;
        this.updatedAt = new Date();
        return this;
    }

    recordLogin() {
        this.lastLoginAt = new Date();
        return this;
    }

    // Business logic methods
    canBeDeleted() {
        // Business rule: Users can only be deleted if inactive for more than 30 days
        if (this.isActive) {
            return false;
        }
        
        const thirtyDaysAgo = new Date();
        thirtyDaysAgo.setDate(thirtyDaysAgo.getDate() - 30);
        
        return this.updatedAt < thirtyDaysAgo;
    }

    isNewUser() {
        const sevenDaysAgo = new Date();
        sevenDaysAgo.setDate(sevenDaysAgo.getDate() - 7);
        
        return this.createdAt > sevenDaysAgo;
    }

    // Computed properties
    get fullName() {
        return `${this.firstName} ${this.lastName}`.trim() || this.username;
    }

    get emailAddress() {
        return this.email.value;
    }

    get hasLoggedIn() {
        return this.lastLoginAt !== null;
    }

    get daysSinceCreation() {
        const now = new Date();
        const diffTime = Math.abs(now - this.createdAt);
        return Math.ceil(diffTime / (1000 * 60 * 60 * 24));

    // Equality and comparison

        if (!(other instanceof User)) {

        }
        
        return this.id === other.id;
    }

        return {
            id: this.id,
            email: this.email.value,
            username: this.username,
            firstName: this.firstName,
            lastName: this.lastName,

const User = require('../../domain/entities/User');

class CreateUserUseCase {
    constructor(userRepository, emailService, passwordHashingService) {
        this.userRepository = userRepository;
        this.emailService = emailService;
        this.passwordHashingService = passwordHashingService;
    }

    async execute(request) {
        // Input validation
        await this.validateRequest(request);
        
        // Business rules validation
        await this.validateBusinessRules(request);
        
        // Create domain entity
        const user = User.create(
            request.email,
            request.username,
            request.firstName,
            request.lastName
        );
        
        // Hash password (application service)
        const hashedPassword = await this.passwordHashingService.hash(request.password);
        
        try {
            // Persist user
            const savedUser = await this.userRepository.save(user, hashedPassword);
            
            // Send welcome email (side effect)
            await this.sendWelcomeEmail(savedUser);
            
            // Return response
            return {
                success: true,
                data: {
                    userId: savedUser.id,
                    email: savedUser.emailAddress,
                    username: savedUser.username,
                    fullName: savedUser.fullName,
                    createdAt: savedUser.createdAt
                }
            };
            
        } catch (error) {
            // Handle persistence errors
            if (error.code === 'DUPLICATE_EMAIL') {
                throw new Error('User with this email already exists');
            }
            
            if (error.code === 'DUPLICATE_USERNAME') {
                throw new Error('Username is already taken');
            }
            
            throw new Error('Failed to create user: ' + error.message);
        }
    }

    async validateRequest(request) {
        const errors = [];
        
        // Required fields validation
        if (!request.email) {
            errors.push('Email is required');
        }
        
        if (!request.username) {
            errors.push('Username is required');
        }
        
        if (!request.password) {
            errors.push('Password is required');
        }
        
        // Email format validation
        if (request.email && !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(request.email)) {
            errors.push('Invalid email format');
        }
        
        // Username validation
        if (request.username && (request.username.length < 3 || request.username.length > 50)) {
            errors.push('Username must be between 3 and 50 characters');
        }
        
        // Password strength validation
        if (request.password && request.password.length < 8) {
            errors.push('Password must be at least 8 characters long');
        }
        
        if (request.password && !/(?=.*[a-z])(?=.*[A-Z])(?=.*\d)/.test(request.password)) {
            errors.push('Password must contain at least one lowercase letter, one uppercase letter, and one number');
        }
        
        if (errors.length > 0) {
            throw new ValidationError('Invalid input data', errors);
        }
    }

    async validateBusinessRules(request) {
        // Check if email is already in use
        const existingUserByEmail = await this.userRepository.findByEmail(request.email);
        if (existingUserByEmail) {
            throw new BusinessRuleError('User with this email already exists');
        }
        
        // Check if username is already taken
        const existingUserByUsername = await this.userRepository.findByUsername(request.username);
        if (existingUserByUsername) {
            throw new BusinessRuleError('Username is already taken');
        }
        
        // Check for forbidden usernames
        const forbiddenUsernames = ['admin', 'root', 'administrator', 'moderator'];
        if (forbiddenUsernames.includes(request.username.toLowerCase())) {
            throw new BusinessRuleError('Username is not allowed');
        }
        
        // Email domain validation (business rule)
        const allowedDomains = process.env.ALLOWED_EMAIL_DOMAINS?.split(',') || [];
        if (allowedDomains.length > 0) {
            const emailDomain = request.email.split('@')[1];
            if (!allowedDomains.includes(emailDomain)) {
                throw new BusinessRuleError('Email domain is not allowed');
            }
        }
    }

    async sendWelcomeEmail(user) {
        try {
            await this.emailService.sendWelcomeEmail({
                to: user.emailAddress,
                username: user.username,
                fullName: user.fullName
            });
        } catch (error) {
            // Email sending is not critical for user creation
            // Log error but don't fail the use case
            console.error('Failed to send welcome email:', error);
        }

    async save(user, hashedPassword) {
        const client = await this.db.getClient();
        
        try {
            await client.query('BEGIN');
            
            const userQuery = `
                INSERT INTO users (id, email, username, first_name, last_name, 
                                 is_active, created_at, updated_at)
                VALUES ($1, $2, $3, $4, $5, $6, $7, $8)
                RETURNING *
            `;
            
            const userValues = [
                user.id,
                user.emailAddress,
                user.username,
                user.firstName,
                user.lastName,
                user.isActive,
                user.createdAt,
                user.updatedAt
            ];
            
            const userResult = await client.query(userQuery, userValues);
            
            // Save password separately (security consideration)
            const passwordQuery = `
                INSERT INTO user_passwords (user_id, password_hash, created_at)
                VALUES ($1, $2, $3)
            `;
            
            await client.query(passwordQuery, [user.id, hashedPassword, user.createdAt]);
            
            await client.query('COMMIT');
            
            // Return domain entity
            return User.fromPersistence(userResult.rows[0]);
            
        } catch (error) {
            await client.query('ROLLBACK');
            
            // Map database errors to domain errors
            if (error.constraint === 'users_email_unique') {
                const domainError = new Error('User with this email already exists');
                domainError.code = 'DUPLICATE_EMAIL';
                throw domainError;
            }
            
            if (error.constraint === 'users_username_unique') {
                const domainError = new Error('Username is already taken');
                domainError.code = 'DUPLICATE_USERNAME';
                throw domainError;
            }
            
            throw error;
        } finally {
            client.release();
        }
    }

            offset = 0,
            sortBy = 'created_at',
            sortOrder = 'DESC',
            isActive = null,
            search = null
        } = options;
        
        try {
            let query = `
                SELECT id, email, username, first_name, last_name,
                       is_active, created_at, updated_at, last_login_at
                FROM users
            `;
            
            const conditions = [];
            const values = [];
            let paramCount = 1;
            
            if (isActive !== null) {
                conditions.push(`is_active = $${paramCount++}`);
                values.push(isActive);
            }
            
            if (search) {
                conditions.push(`(
                    LOWER(username) LIKE LOWER($${paramCount}) OR 
                    LOWER(email) LIKE LOWER($${paramCount}) OR
                    LOWER(first_name) LIKE LOWER($${paramCount}) OR
                    LOWER(last_name) LIKE LOWER($${paramCount})
                )`);
                values.push(`%${search}%`);
                paramCount++;
            }
            
            if (conditions.length > 0) {
                query += ' WHERE ' + conditions.join(' AND ');
            }
            
            query += ` ORDER BY ${sortBy} ${sortOrder} LIMIT $${paramCount} OFFSET $${paramCount + 1}`;
            values.push(limit, offset);
            
            const result = await this.db.query(query, values);
            
            return result.rows.map(row =>

            res.json(response);
            
        } catch (error) {
            this.handleError(error, res);
        }
    }

        try {
            const userId = req.params.id;
            
            const result = await this.deleteUserUseCase.execute({ userId });
            
            res.status(204).send();
            
        } catch (error) {
            this.handleError(error, res);
        }
    }

    async listUsers(req, res) {
        try {
            const request = {
                page: parseInt(req.query.page) || 1,
                limit: parseInt(req.query.limit) || 10,
                search: req.query.search,
                sortBy: req.query.sortBy || 'createdAt',
                sortOrder: req.query.sortOrder || 'desc',
                isActive: req.query.isActive ? req.query.isActive === 'true' : null
            };

            const result = await this.listUsersUseCase.execute(request);
            
            const response = this.userPresenter.presentUserList(result.data);
            res.json(response);
            
        } catch (error) {
            this.handleError(error, res);

        // Map domain errors to HTTP responses
        if (error.name === 'ValidationError') {
            return res.status(400).json({
                success: false,
                error: 'Validation failed',
                details: error.validationErrors
            });
        }

        if (error.name === 'BusinessRuleError') {
            return res.status(409).json({
                success: false,
                error: error.message
            });
        }

        if (error.message === 'User not found') {
            return res.status(404).json({
                success: false,
                error: 'User not found'
            });
        }

        if (error.message.includes('already exists')) {
            return res.status(409).json({
                success: false,
                error: error.message
            });
        }

        // Internal server error
        console.error('Unexpected error:', error);
        res.status(500).json({
            success: false,
            error: 'An unexpected error occurred'
        });
    }

Benefits of Clean Architecture

  • Framework Independence: Core logic doesn't depend on specific frameworks
  • Database Independence: Can switch databases without changing business logic
  • Highly Testable: Business logic can be tested without external dependencies
  • UI Independence: Same business logic works with different UIs
  • Maintainable: Changes in one layer don't cascade to others
  • Flexible: Easy to add new features without breaking existing code

Drawbacks and Challenges

  • Initial Complexity: Higher upfront investment in architecture setup
  • More Code: Requires more interfaces and abstraction layers
  • Learning Curve: Team needs to understand dependency inversion principles
  • Over-Engineering Risk: Can be excessive for simple applications
  • Performance Overhead: Multiple layers can add latency

When to Use Clean Architecture

  • Long-term Projects: Applications expected to evolve over years
  • Large Teams: Multiple developers working on different components
  • Complex Business Logic: Rich domain models with intricate rules
  • Multiple Interfaces: Same logic serves web, mobile, API clients
  • Changing Requirements: Frequent business rule modifications
  • Enterprise Applications: Mission-critical systems requiring stability

Hexagonal Architecture (Ports & Adapters)

Hexagonal Architecture, also known as Ports and Adapters, isolates the core business logic from external concerns by defining clear boundaries through ports (interfaces) and adapters (implementations). This creates a testable, maintainable system where the business logic is completely independent of external systems.

⬡

Core Concepts

  • Ports: Interfaces that define contracts between the core and external world
  • Adapters: Implementations that handle external communication
  • Core Domain: Business logic completely isolated from external concerns
  • Driver Adapters: Input adapters that trigger business operations
  • Driven Adapters: Output adapters called by business logic

        this.userRepository = userRepository;
        this.notificationService = notificationService;
        this.passwordHasher = passwordHasher;
    }

    async registerUser(userData) {
        // Business rules validation
        await this.validateRegistrationData(userData);
        
        // Check if user already exists
        const existingUser = await this.userRepository.findByEmail(userData.email);
        if (existingUser) {
            throw new BusinessError('User already exists with this email');
        }

        // Create user entity
        const user = new User({
            email: userData.email,
            username: userData.username,
            firstName: userData.firstName,
            lastName: userData.lastName
        });

        // Hash password
        const hashedPassword = await this.passwordHasher.hash(userData.password);
        user.setPassword(hashedPassword);

        // Apply business rules
        user.activate();

        // Persist user
        const savedUser = await this.userRepository.save(user);

        // Send welcome notification (side effect)
        await this.notificationService.sendEmail(
            user.email,
            'Welcome to our platform!',
            `Hello ${user.fullName}, welcome to our platform.`
        );

        // Log registration event
        await this.notificationService.logEvent('USER_REGISTERED', {
            userId: savedUser.id,
            email: savedUser.email,
            timestamp: new Date().toISOString()
        });

        return savedUser;
    }

    async authenticateUser(email, password) {
        // Find user by email
        const user = await this.userRepository.findByEmail(email);
        if (!user) {
            throw new AuthenticationError('Invalid credentials');
        }

        // Verify password
        const isValidPassword = await this.passwordHasher.verify(password, user.password);
        if (!isValidPassword) {
            throw new AuthenticationError('Invalid credentials');
        }

        // Business rule: Check if user is active
        if (!user.isActive) {
            throw new BusinessError('Account is deactivated');
        }

        // Update last login
        user.recordLogin();
        await this.userRepository.update(user);

        // Log authentication event
        await this.notificationService.logEvent('USER_LOGIN', {
            userId: user.id,
            email: user.email,
            timestamp: new Date().toISOString()
        });

        return user;
    }

    async updateUserProfile(userId, profileData) {
        // Find user
        const user = await this.userRepository.findById(userId);
        if (!user) {
            throw new NotFoundError('User not found');
        }

        // Validate profile data
        await this.validateProfileData(profileData);

        // Check email uniqueness if email is being changed
        if (profileData.email && profileData.email !== user.email) {
            const existingUser = await this.userRepository.findByEmail(profileData.email);
            if (existingUser) {
                throw new BusinessError('Email already in use');
            }
        }

        // Update user entity
        if (profileData.email) user.changeEmail(profileData.email);
        if (profileData.firstName) user.firstName = profileData.firstName;
        if (profileData.lastName) user.lastName = profileData.lastName;
        if (profileData.username) user.username = profileData.username;

        user.touch(); // Update timestamp

        // Persist changes
        const updatedUser = await this.userRepository.update(user);

        // Send update notification if email changed
        if (profileData.email && profileData.email !== user.email) {
            await this.notificationService.sendEmail(
                updatedUser.email,
                'Email address updated',
                'Your email address has been successfully updated.'
            );
        }

        return updatedUser;
    }

    async deactivateUser(userId) {
        const user = await this.userRepository.findById(userId);
        if (!user) {
            throw new NotFoundError('User not found');
        }

        // Business rule: Don't deactivate already inactive users
        if (!user.isActive) {
            throw new BusinessError('User is already deactivated');
        }

        user.deactivate();
        const updatedUser = await this.userRepository.update(user);

        // Send deactivation notification
        await this.notificationService.sendEmail(
            user.email,
            'Account deactivated',
            'Your account has been deactivated. Contact support if this was unexpected.'
        );

        // Log deactivation event
        await this.notificationService.logEvent('USER_DEACTIVATED', {
            userId: user.id,
            timestamp: new Date().toISOString()
        });

        return updatedUser;
    }

    async getUserProfile(userId) {
        const user = await this.userRepository.findById(userId);
        if (!user) {
            throw new NotFoundError('User not found');
        }

        return user;
    }

    async resetPassword(email) {
        const user = await this.userRepository.findByEmail(email);
        if (!user) {
            // Don't reveal that user doesn't exist for security
            return { message: 'If the email exists, a reset link has been sent.' };
        }

        // Generate reset token (simplified)
        const resetToken = require('crypto').randomBytes(32).toString('hex');
        user.setPasswordResetToken(resetToken, new Date(Date.now() + 3600000)); // 1 hour

        await this.userRepository.update(user);

        // Send reset email
        await this.notificationService.sendEmail(
            user.email,
            'Password Reset Request',
            `Click here to reset your password: /reset-password?token=${resetToken}`
        );

        return { message: 'If the email exists, a reset link has been sent.' };
    }

    // Private validation methods
    async validateRegistrationData(userData) {
        if (!userData.email || !this.isValidEmail(userData.email)) {
            throw new ValidationError('Valid email is required');
        }

        if (!userData.password || userData.password.length < 8) {
            throw new ValidationError('Password must be at least 8 characters');
        }

        if (!userData.username || userData.username.length < 3) {
            throw new ValidationError('Username must be at least 3 characters');
        }
    }

    async validateProfileData(profileData) {
        if (profileData.email && !this.isValidEmail(profileData.email)) {
            throw new ValidationError('Valid email is required');
        }

        if (profileData.username && profileData.username.length < 3) {
            throw new ValidationError('Username must be at least 3 characters');
        }
    }

    isValidEmail(email) {
        return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
    }

                return null;
            }
            
            return this.mapRowToUser(result.rows[0]);
        } catch (error) {
            throw new Error(`Failed to find user by ID: ${error.message}`);
        }
    }

    async findByEmail(email) {
        try {
            const query = 'SELECT * FROM users WHERE LOWER(email) = LOWER($1)';
            const result = await this.db.query(query, [email]);
            
            if (result.rows.length === 0) {
                return null;
            }
            
            return this.mapRowToUser(result.rows[0]);
        } catch (error) {
            throw new Error(`Failed to find user by email: ${error.message}`);
        }
    }

                UPDATE users 
                SET email = $2, username = $3, first_name = $4, 
                    last_name = $5, password_hash = $6, is_active = $7, 
                    updated_at = $8, last_login_at = $9, 
                    password_reset_token = $10, password_reset_expires = $11
                WHERE id = $1
                RETURNING *
            `;
            
            const values = [
                user.id,
                user.email,
                user.username,
                user.firstName,
                user.lastName,
                user.password,
                user.isActive,
                user.updatedAt,
                user.lastLoginAt,

    async exists(criteria) {
        try {
            let query = 'SELECT 1 FROM users WHERE ';
            const values = [];
            const conditions = [];
            let paramCount = 1;

            if (criteria.email) {
                conditions.push(`LOWER(email) = LOWER($${paramCount++})`);
                values.push(criteria.email);
            }

            if (criteria.username) {
                conditions.push(`LOWER(username) = LOWER($${paramCount++})`);
                values.push(criteria.username);
            }

            if (criteria.id) {
                conditions.push(`id = $${paramCount++}`);
                values.push(criteria.id);
            }

            query += conditions.join(' AND ') + ' LIMIT 1';
            
            const result = await this.db.query(query, values);
            return result.rows.length > 0;
        } catch (error) {
            throw new Error(`Failed to check user existence: ${error.message}`);

Benefits of Hexagonal Architecture

  • Complete Isolation: Business logic is completely independent of external systems
  • Easy Testing: Core domain can be tested with simple mock implementations
  • Flexibility: Easy to swap external systems without changing business logic
  • Clear Boundaries: Explicit contracts through ports make dependencies obvious
  • Maintainability: Changes in external systems don't affect the core
  • Technology Independence: Core logic doesn't depend on specific technologies

Drawbacks and Challenges

  • Increased Complexity: More interfaces and abstractions to manage
  • Over-Engineering: Can be overkill for simple applications
  • Learning Curve: Requires understanding of ports and adapters concepts
  • More Code: Additional boilerplate code for interfaces
  • Initial Setup Time: More upfront investment in architecture

When to Use Hexagonal Architecture

  • Complex Business Logic: Rich domain models with intricate business rules
  • Multiple Interfaces: Same logic needs to serve different input/output channels
  • High Test Coverage: Applications requiring comprehensive unit testing
  • Changing External Systems: Frequent changes in databases or third-party services
  • Long-term Maintenance: Applications expected to evolve over many years
  • Team Collaboration: Large teams working on different parts simultaneously

Layered Architecture (N-Tier Architecture)

Layered Architecture organizes code into horizontal layers where each layer only communicates with the layer directly below it. This creates a clear separation of concerns and makes the system easier to maintain and test.

📚

Common Layers

  • Presentation Layer: User interface and user interaction handling
  • Business Logic Layer: Core business rules and processing
  • Data Access Layer: Database and external service interactions
  • Cross-cutting Concerns: Logging, security, caching across all layers

const UserRepository = require('../../data/repositories/UserRepository');
const User = require('../models/User');
const userValidator = require('../validators/userValidator');
const EmailService = require('../../data/external/EmailService');

class UserService {
    constructor() {
        this.userRepository = new UserRepository();
        this.emailService = new EmailService();
    }

    async createUser(userData) {
        // Business logic layer responsibilities:
        // 1. Validation
        // 2. Business rules
        // 3. Orchestration
        
        // Validate input
        const validation = userValidator.validateCreate(userData);
        if (!validation.isValid) {
            throw new Error(`Validation failed: ${validation.errors.join(', ')}`);
        }

        // Business rule: Check for duplicate email
        const existingUser = await this.userRepository.findByEmail(userData.email);
        if (existingUser) {
            throw new Error('User with this email already exists');
        }

        // Create business entity
        const user = new User(userData);
        
        // Apply business rules
        user.hashPassword();
        user.generateId();
        user.setCreatedDate();

        // Persist through data layer
        const savedUser = await this.userRepository.create(user.toJSON());

        // Side effects
        await this.emailService.sendWelcomeEmail(user.email, user.username);

        return savedUser;
    }

    async getUsers(options) {
        // Business logic for pagination and filtering
        const { page, limit, search } = options;
        
        const offset = (page - 1) * limit;
        
        return await this.userRepository.findMany({
            offset,
            limit,
            search
        });
    }

    async updateUser(id, updateData) {
        // Validation
        const validation = userValidator.validateUpdate(updateData);
        if (!validation.isValid) {
            throw new Error(`Validation failed: ${validation.errors.join(', ')}`);
        }

        // Business rules
        const existingUser = await this.userRepository.findById(id);
        if (!existingUser) {
            throw new Error('User not found');
        }

        // Check email uniqueness if email is being updated
        if (updateData.email && updateData.email !== existingUser.email) {
            const emailExists = await this.userRepository.findByEmail(updateData.email);
            if (emailExists) {
                throw new Error('Email already in use');
            }
        }

        // Update business entity
        const user = new User(existingUser);
        user.update(updateData);
        user.setUpdatedDate();

        return await this.userRepository.update(id, user.toJSON());
    }
}

            return this.mapToBusinessObject(result.rows[0]);
        } catch (error) {
            throw new Error(`Database error: ${error.message}`);
        }
    }

    async findById(id) {
        try {
            const query = 'SELECT * FROM users WHERE id = $1';
            const result = await database.query(query, [id]);
            
            if (result.rows.length === 0) {
                return null;
            }
            
            return this.mapToBusinessObject(result.rows[0]);
        } catch (error) {
            throw new Error(`Database error: ${error.message}`);
        }
    }

    async findByEmail(email) {
        try {
            const query = 'SELECT * FROM users WHERE email = $1';
            const result = await database.query(query, [email]);
            
            if (result.rows.length === 0) {
                return null;
            }
            
            return this.mapToBusinessObject(result.rows[0]);
        } catch (error) {
            throw new Error(`Database error: ${error.message}`);
        }
    }

    async findMany(options) {
        try {
            let query = 'SELECT * FROM users';
            const values = [];
            const conditions = [];
            let paramCount = 1;

            // Add search condition if provided
            if (options.search) {
                conditions.push(`(username ILIKE $${paramCount} OR email ILIKE $${paramCount})`);
                values.push(`%${options.search}%`);
                paramCount++;
            }

            // Add WHERE clause if conditions exist
            if (conditions.length > 0) {
                query += ' WHERE ' + conditions.join(' AND ');
            }

            // Add pagination
            query += ` ORDER BY created_at DESC LIMIT $${paramCount} OFFSET $${paramCount + 1}`;
            values.push(options.limit, options.offset);

            const result = await database.query(query, values);
            return result.rows.map(row => this.mapToBusinessObject(row));
        } catch (error) {
            throw new Error(`Database error: ${error.message}`);
        }
    }

    async update(id, userData) {
        try {
            const query = `
                UPDATE users 
                SET email = $2, username = $3, updated_at = $4
                WHERE id = $1
                RETURNING *
            `;
            
            const values = [
                id,
                userData.email,
                userData.username,
                userData.updatedAt
            ];
            
            const result = await database.query(query, values);
            
            if (result.rows.length === 0) {
                throw new Error('User not found');
            }
            
            return this.mapToBusinessObject(result.rows[0]);
        } catch (error) {
            throw new Error(`Database error: ${error.message}`);
        }
    }

    // Map database row to business object format
    mapToBusinessObject(row) {
        return {
            id: row.id,
            email: row.email,
            username: row.username,
            passwordHash: row.password_hash,
            createdAt: row.created_at,
            updatedAt: row.updated_at
        };
    }
}

Rules:
• Each layer only communicates with the layer directly below
• No layer should skip levels
• Data flows down, responses flow up
• Cross-cutting concerns (logging, security) span all layers

Benefits of Layered Architecture

  • Clear Separation: Each layer has distinct responsibilities
  • Easy to Understand: Simple, intuitive structure
  • Team Organization: Different teams can work on different layers
  • Reusability: Business logic can be reused across different presentation layers
  • Testability: Each layer can be tested independently
  • Maintainability: Changes in one layer have minimal impact on others

Drawbacks and Challenges

  • Performance Overhead: Data passes through multiple layers
  • Monolithic Tendency: Can become a large, single deployable unit
  • Rigid Structure: Changes that span layers can be complex
  • Database Coupling: Often tightly coupled to database schema
  • Limited Scalability: Entire application scales as one unit

When to Use Layered Architecture

  • Traditional Applications: CRUD-heavy business applications
  • Small to Medium Teams: Teams familiar with layered approaches
  • Rapid Development: Quick prototyping and development cycles
  • Limited Complexity: Applications without complex business rules
  • Monolithic Deployment: Single deployment unit is acceptable

Microservices Architecture

Microservices architecture structures an application as a collection of small, independent services that communicate over well-defined APIs. Each service is responsible for a specific business capability and can be developed, deployed, and scaled independently.

🔗

Core Principles

  • Single Responsibility: Each service handles one business capability
  • Independence: Services can be developed and deployed independently
  • Decentralized: No central coordination or shared databases
  • Resilient: Failure in one service doesn't cascade to others
  • Technology Agnostic: Services can use different technologies

const express = require('express');
const bcrypt = require('bcrypt');
const jwt = require('jsonwebtoken');
const { Pool } = require('pg');

class UserService {
    constructor() {
        this.app = express();
        this.db = new Pool({
            connectionString: process.env.USER_DB_URL
        });
        this.setupMiddleware();
        this.setupRoutes();
    }

    setupMiddleware() {
        this.app.use(express.json());
        this.app.use(this.corsMiddleware());
        this.app.use(this.loggingMiddleware());
    }

    setupRoutes() {
        // Health check for service discovery
        this.app.get('/health', (req, res) => {
            res.json({ 
                service: 'user-service', 
                status: 'healthy',
                timestamp: new Date().toISOString()
            });
        });

        // User authentication
        this.app.post('/auth/login', this.login.bind(this));
        this.app.post('/auth/register', this.register.bind(this));
        this.app.post('/auth/refresh', this.refreshToken.bind(this));

        // User management
        this.app.get('/users/:id', this.authMiddleware, this.getUser.bind(this));
        this.app.put('/users/:id', this.authMiddleware, this.updateUser.bind(this));
        this.app.delete('/users/:id', this.authMiddleware, this.deleteUser.bind(this));

        // User preferences
        this.app.get('/users/:id/preferences', this.authMiddleware, this.getPreferences.bind(this));
        this.app.put('/users/:id/preferences', this.authMiddleware, this.updatePreferences.bind(this));
    }

    async register(req, res) {
        try {
            const { email, password, username, firstName, lastName } = req.body;
            
            // Validate input
            if (!email || !password || !username) {
                return res.status(400).json({ 
                    error: 'Email, password, and username are required' 
                });
            }

            // Check if user exists
            const existingUser = await this.db.query(
                'SELECT id FROM users WHERE email = $1 OR username = $2',
                [email, username]
            );

            if (existingUser.rows.length > 0) {
                return res.status(409).json({ 
                    error: 'User already exists' 
                });
            }

            // Hash password
            const hashedPassword = await bcrypt.hash(password, 10);

            // Create user
            const result = await this.db.query(
                `INSERT INTO users (email, username, password_hash, first_name, last_name, created_at)
                 VALUES ($1, $2, $3, $4, $5, NOW())
                 RETURNING id, email, username, first_name, last_name, created_at`,
                [email, username, hashedPassword, firstName, lastName]
            );

            const user = result.rows[0];

            // Generate JWT
            const token = jwt.sign(
                { userId: user.id, email: user.email },
                process.env.JWT_SECRET,
                { expiresIn: '24h' }
            );

            // Publish user created event
            await this.publishEvent('user.created', {
                userId: user.id,
                email: user.email,
                username: user.username,
                timestamp: new Date().toISOString()
            });

            res.status(201).json({
                success: true,
                data: {
                    user: {
                        id: user.id,
                        email: user.email,
                        username: user.username,
                        firstName: user.first_name,
                        lastName: user.last_name
                    },
                    token
                }
            });
        } catch (error) {
            console.error('Registration error:', error);
            res.status(500).json({ 
                error: 'Internal server error' 
            });
        }
    }

    async login(req, res) {
        try {
            const { email, password } = req.body;

            // Find user
            const result = await this.db.query(
                'SELECT * FROM users WHERE email = $1',
                [email]
            );

            if (result.rows.length === 0) {
                return res.status(401).json({ 
                    error: 'Invalid credentials' 
                });
            }

            const user = result.rows[0];

            // Verify password
            const isValidPassword = await bcrypt.compare(password, user.password_hash);
            if (!isValidPassword) {
                return res.status(401).json({ 
                    error: 'Invalid credentials' 
                });
            }

            // Update last login
            await this.db.query(
                'UPDATE users SET last_login_at = NOW() WHERE id = $1',
                [user.id]
            );

            // Generate JWT
            const token = jwt.sign(
                { userId: user.id, email: user.email },
                process.env.JWT_SECRET,
                { expiresIn: '24h' }
            );

            // Publish login event
            await this.publishEvent('user.logged_in', {
                userId: user.id,
                timestamp: new Date().toISOString()
            });

            res.json({
                success: true,
                data: {
                    user: {
                        id: user.id,
                        email: user.email,
                        username: user.username,
                        firstName: user.first_name,
                        lastName: user.last_name
                    },
                    token
                }
            });
        } catch (error) {
            console.error('Login error:', error);
            res.status(500).json({ 
                error: 'Internal server error' 
            });
        }
    }

    async getUser(req, res) {
        try {
            const userId = req.params.id;

            // Check authorization
            if (req.user.userId !== userId && !req.user.isAdmin) {
                return res.status(403).json({ 
                    error: 'Forbidden' 
                });
            }

            const result = await this.db.query(
                'SELECT id, email, username, first_name, last_name, created_at, last_login_at FROM users WHERE id = $1',
                [userId]
            );

            if (result.rows.length === 0) {
                return res.status(404).json({ 
                    error: 'User not found' 
                });
            }

            const user = result.rows[0];
            res.json({
                success: true,
                data: {
                    id: user.id,
                    email: user.email,
                    username: user.username,
                    firstName: user.first_name,
                    lastName: user.last_name,
                    createdAt: user.created_at,
                    lastLoginAt: user.last_login_at
                }
            });
        } catch (error) {
            console.error('Get user error:', error);
            res.status(500).json({ 
                error: 'Internal server error' 
            });
        }
    }

    // Event publishing for inter-service communication
    async publishEvent(eventType, data) {
        try {
            // This could use RabbitMQ, Apache Kafka, or cloud messaging services
            const event = {
                type: eventType,
                data,
                timestamp: new Date().toISOString(),
                service: 'user-service'
            };

            // Example with HTTP webhook (in production use proper message queue)
            if (process.env.EVENT_WEBHOOK_URL) {
                await fetch(process.env.EVENT_WEBHOOK_URL, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(event)
                });
            }

            console.log('Event published:', event);
        } catch (error) {
            console.error('Event publishing error:', error);
        }
    }

    // Middleware
    authMiddleware(req, res, next) {
        try {
            const token = req.headers.authorization?.split(' ')[1];
            if (!token) {
                return res.status(401).json({ error: 'No token provided' });
            }

            const decoded = jwt.verify(token, process.env.JWT_SECRET);
            req.user = decoded;
            next();
        } catch (error) {
            res.status(401).json({ error: 'Invalid token' });
        }
    }

    corsMiddleware() {
        return (req, res, next) => {
            res.header('Access-Control-Allow-Origin', '*');
            res.header('Access-Control-Allow-Headers', 'Origin, X-Requested-With, Content-Type, Accept, Authorization');
            res.header('Access-Control-Allow-Methods', 'GET, POST, PUT, DELETE, OPTIONS');
            next();
        };
    }

    loggingMiddleware() {
        return (req, res, next) => {
            console.log(`[${new Date().toISOString()}] ${req.method} ${req.path}`);
            next();
        };
    }

    start(port = 3001) {
        this.app.listen(port, () => {
            console.log(`User Service running on port ${port}`);
        });
    }
}

        this.app.use(this.corsMiddleware());

            for (const [name, service] of this.services.entries()) {
                serviceHealth[name] = {
                    totalInstances: service.urls.length,
                    healthyInstances: service.healthyInstances.length
                };
            }
            
            res.json({
                gateway: 'healthy',
                services: serviceHealth,
                timestamp: new Date().toISOString()

        this.app.use('/api/users',

        );

        // Product routes (public read, auth for write)
        this.app.use('/api/products',

            this.createProxy('product-service', '/products')
        );

        // Order routes (auth required)
        this.app.use('/api/orders',
            this.authMiddleware,
            this.createProxy('order-service', '/orders')
        );

        // Payment routes (auth required)
        this.app.use('/api/payments',
            this.authMiddleware,
            this.createProxy('payment-service', '/payments')
        );

        // Catch-all for undefined routes
        this.app.use('*', (req, res) => {
            res.status(404).json({
                error: 'Endpoint not found',
                path: req.originalUrl
            });
        });
    }

    createProxy(serviceName, pathRewrite = '') {
        return httpProxy({
            target: 'http://placeholder', // Will be replaced by router
            changeOrigin: true,
            pathRewrite: pathRewrite ? {
                [`^/api/${serviceName.split('-')[0]}s?`]: pathRewrite
            } : {},
            router: (req) => {
                return this.getHealthyServiceUrl(serviceName);
            },
            onError: (err, req, res) => {
                console.error(`Proxy error for ${serviceName}:`, err.message);
                res.status(503).json({
                    error: 'Service temporarily unavailable',
                    service: serviceName
                });
            },
            onProxyReq: (proxyReq, req, res) => {
                // Add headers for service identification
                proxyReq.setHeader('X-Gateway-Request-ID', req.id || Math.random().toString(36));
                proxyReq.setHeader('X-Forwarded-For', req.ip);
                proxyReq.setHeader('X-Service-Name', serviceName);
            }
        });
    }

    getHealthyServiceUrl(serviceName) {
        const service = this.services.get(serviceName);
        if (!service || service.healthyInstances.length === 0) {
            throw new Error(`No healthy instances available for ${serviceName}`);
        }

        // Round-robin load balancing
        const url = service.healthyInstances[service.currentIndex];
        service.currentIndex = (service.currentIndex + 1) % service.healthyInstances.length;
        
        return url;
    }

    async checkServiceHealth(serviceName, url) {
        try {
            const response = await fetch(`${url}/health`, {
                method: 'GET',
                timeout: 5000
            });
            
            if (response.ok) {
                return true;
            }
        } catch (error) {
            console.error(`Health check failed for ${url}:`, error.message);
        }
        return false;
    }

    async updateHealthyInstances() {
        for (const [serviceName, service] of this.services.entries()) {
            const healthyInstances = [];
            
            for (const url of service.urls) {
                if (await this.checkServiceHealth(serviceName, url)) {
                    healthyInstances.push(url);
                }
            }
            
            service.healthyInstances = healthyInstances;
            
            if (healthyInstances.length === 0) {
                console.warn(`No healthy instances for ${serviceName}`);
            }
        }
    }

    startHealthChecks() {
        // Initial health check
        this.updateHealthyInstances();
        
        // Periodic health checks every 30 seconds
        setInterval(() => {
            this.updateHealthyInstances();
        }, 30000);
    }

    // Authentication middleware
    authMiddleware(req, res, next) {
        try {
            const token = req.headers.authorization?.split(' ')[1];
            if (!token) {
                return res.status(401).json({ 
                    error: 'Authentication required' 
                });
            }

            const decoded = jwt.verify(token, process.env.JWT_SECRET);
            req.user = decoded;
            next();
        } catch (error) {
            res.status(401).json({ 
                error: 'Invalid or expired token' 
            });
        }
    }

    // Conditional authentication for certain HTTP methods
    conditionalAuth(requireAuthMethods) {
        return (req, res, next) => {
            if (requireAuthMethods.includes(req.method)) {
                return this.authMiddleware(req, res, next);
            }
            next();
        };
    }

    loggingMiddleware() {
        return (req, res, next) => {
            const start = Date.now();
            req.id = Math.random().toString(36).substring(2, 15);
            
            console.log(`[${new Date().toISOString()}] ${req.id} ${req.method} ${req.path} - Started`);
            
            res.on('finish', () => {
                const duration = Date.now() - start;
                console.log(`[${new Date().toISOString()}] ${req.id} ${req.method} ${req.path} - ${res.statusCode} (${duration}ms)`);
            });
            
            next();
        };
    }

    corsMiddleware() {
        return (req, res, next) => {
            res.header('Access-Control-Allow-Origin', '*');
            res.header('Access-Control-Allow-Headers', 'Origin, X-Requested-With, Content-Type, Accept, Authorization');
            res.header('Access-Control-Allow-Methods', 'GET, POST, PUT, DELETE, OPTIONS');
            
            if (req.method === 'OPTIONS') {
                res.sendStatus(200);
            } else {
                next();
            }
        };
    }

    start(port = 3000) {
        this.app.listen(port, () => {
            console.log(`API Gateway running on port ${port}`);
            console.log('Registered services:', Array.from(this.services.keys()));
        });
    }
}

class EventBus {
    constructor() {
        this.connection = null;
        this.channel = null;
        this.exchanges = {
            'user.events': 'user.events',
            'order.events': 'order.events',
            'payment.events': 'payment.events'
        };
    }

    async connect() {
        try {
            this.connection = await amqp.connect(process.env.RABBITMQ_URL || 'amqp://localhost');
            this.channel = await this.connection.createChannel();
            
            // Declare exchanges
            for (const exchange of Object.values(this.exchanges)) {
                await this.channel.assertExchange(exchange, 'topic', { durable: true });
            }
            
            console.log('Connected to message queue');
        } catch (error) {
            console.error('Failed to connect to message queue:', error);
            throw error;
        }
    }

    async publishEvent(exchange, routingKey, eventData) {
        try {
            const event = {
                id: Math.random().toString(36).substring(2, 15),
                timestamp: new Date().toISOString(),
                service: process.env.SERVICE_NAME,
                version: '1.0',
                data: eventData
            };

            await this.channel.publish(
                exchange,
                routingKey,
                Buffer.from(JSON.stringify(event)),
                { persistent: true }
            );
            
            console.log(`Event published: ${exchange}.${routingKey}`, event.id);
        } catch (error) {
            console.error('Failed to publish event:', error);
            throw error;
        }
    }

    async subscribe(exchange, routingKey, handler) {
        try {
            const queue = `${process.env.SERVICE_NAME}.${exchange}.${routingKey}`;
            await this.channel.assertQueue(queue, { durable: true });
            await this.channel.bindQueue(queue, exchange, routingKey);
            
            await this.channel.consume(queue, async (message) => {
                if (message) {
                    try {
                        const event = JSON.parse(message.content.toString());
                        console.log(`Event received: ${exchange}.${routingKey}`, event.id);
                        
                        await handler(event);
                        this.channel.ack(message);
                    } catch (error) {
                        console.error('Error processing event:', error);
                        // Reject and requeue for retry
                        this.channel.nack(message, false, true);
                    }
                }
            });
            
            console.log(`Subscribed to ${exchange}.${routingKey}`);
        } catch (error) {
            console.error('Failed to subscribe to event:', error);
            throw error;
        }
    }
}

// Service-to-Service HTTP Communication
class ServiceClient {
    constructor(baseURL, serviceName) {
        this.baseURL = baseURL;
        this.serviceName = serviceName;
        this.timeout = 5000;
    }

    async makeRequest(method, endpoint, data = null, headers = {}) {
        const url = `${this.baseURL}${endpoint}`;
        const requestHeaders = {
            'Content-Type': 'application/json',
            'X-Service-Name': process.env.SERVICE_NAME,
            'X-Request-ID': Math.random().toString(36).substring(2, 15),
            ...headers
        };

        try {
            const controller = new AbortController();
            const timeoutId = setTimeout(() => controller.abort(), this.timeout);

            const response = await fetch(url, {
                method,
                headers: requestHeaders,
                body: data ? JSON.stringify(data) : null,
                signal: controller.signal
            });

            clearTimeout(timeoutId);

            if (!response.ok) {
                throw new Error(`${this.serviceName} service error: ${response.status}`);
            }

            return await response.json();
        } catch (error) {
            if (error.name === 'AbortError') {
                throw new Error(`${this.serviceName} service timeout`);
            }
            throw new Error(`${this.serviceName} service unavailable: ${error.message}`);
        }
    }

    async get(endpoint, headers = {}) {
        return this.makeRequest('GET', endpoint, null, headers);
    }

    async post(endpoint, data, headers = {}) {
        return this.makeRequest('POST', endpoint, data, headers);
    }

    async put(endpoint, data, headers = {}) {
        return this.makeRequest('PUT', endpoint, data, headers);
    }

    async delete(endpoint, headers = {}) {
        return this.makeRequest('DELETE', endpoint, null, headers);
    }
}

class UserDataService {
    constructor() {
        this.localDb = new Pool({ connectionString: process.env.LOCAL_DB_URL });
        this.userServiceClient = new ServiceClient(
            process.env.USER_SERVICE_URL,
            'user-service'
        );

    async getUserData(userId, authToken) {
        try {
            // 1. Try local cache first (fastest)
            if (this.cache.has(userId)) {
                const cached = this.cache.get(userId);
                if (Date.now() - cached.timestamp < 60000) { // 1 minute cache
                    return cached.data;
                }
            }

            // 2. Try local database replica (fast)
            const localResult = await this.localDb.query(
                'SELECT * FROM user_profiles WHERE user_id = $1 AND is_deleted = false',
                [userId]
            );

            if (localResult.rows.length > 0) {
                const userData = localResult.rows[0];
                
                // Cache the result
                this.cache.set(userId, {
                    data: userData,
                    timestamp: Date.now()
                });
                
                // Optionally refresh from source asynchronously
                this.refreshUserDataAsync(userId, authToken);
                
                return userData;
            }

            // 3. Fallback to user service (slower but authoritative)
            const serviceResult = await this.userServiceClient.get(
                `/users/${userId}`,
                { Authorization: `Bearer ${authToken}` }
            );

            if (serviceResult.success) {
                const userData = serviceResult.data;
                
                // Update local database
                await this.updateLocalUserData(userData);
                
                // Cache the result
                this.cache.set(userId, {
                    data: userData,
                    timestamp: Date.now()
                });
                
                return userData;
            }

            throw new Error('User not found');
        } catch (error) {
            console.error('Error fetching user data:', error);
            
            // Last resort: return stale data if available
            const staleResult = await this.localDb.query(
                'SELECT * FROM user_profiles WHERE user_id = $1',
                [userId]
            );

            if (staleResult.rows.length > 0) {

                    ...staleResult.rows[0],

Benefits of Microservices Architecture

  • Independent Scaling: Scale services based on individual demand
  • Technology Diversity: Use the best technology for each service
  • Fault Isolation: Failures in one service don't bring down the entire system
  • Team Autonomy: Teams can develop and deploy independently
  • Faster Development: Parallel development across multiple teams
  • Easy Deployment: Deploy services independently without affecting others

Drawbacks and Challenges

  • Network Latency: Inter-service communication adds latency
  • Data Consistency: Maintaining consistency across services is complex
  • Testing Complexity: Integration testing becomes more difficult
  • Operational Overhead: More services to monitor and maintain
  • Transaction Management: Distributed transactions are challenging
  • Service Discovery: Need mechanisms to locate and communicate with services

When to Use Microservices Architecture

  • Large Applications: Complex systems with multiple business domains
  • Large Teams: Multiple development teams working on different features
  • Scalability Requirements: Different parts of the system have different scaling needs
  • Technology Diversity: Different services benefit from different technologies
  • Continuous Deployment: Need to deploy different parts independently
  • Organizational Structure: Teams organized around business capabilities

Event-Driven Architecture

Event-Driven Architecture is a software design pattern where components communicate through events - notifications that something significant has happened in the system. This pattern enables loose coupling, scalability, and real-time responsiveness.

⚡

Core Components

  • Event Producers: Components that publish events when something happens
  • Event Consumers: Components that subscribe to and process events
  • Event Bus/Broker: Infrastructure for routing events between producers and consumers
  • Event Store: Persistent storage for events (used in Event Sourcing)

class EventStore {
    constructor(database) {
        this.db = database;
    }

    async saveEvents(streamId, expectedVersion, events) {
        const client = await this.db.getClient();
        
        try {
            await client.query('BEGIN');
            
            // Check expected version to prevent concurrent modifications
            const versionResult = await client.query(
                'SELECT COALESCE(MAX(version), 0) as current_version FROM events WHERE stream_id = $1',
                [streamId]
            );
            
            const currentVersion = versionResult.rows[0].current_version;
            
            if (expectedVersion !== -1 && currentVersion !== expectedVersion) {
                throw new Error(`Concurrency conflict. Expected version ${expectedVersion}, but current is ${currentVersion}`);
            }
            
            // Save events
            for (let i = 0; i < events.length; i++) {
                const event = events[i];
                const version = currentVersion + i + 1;
                
                await client.query(
                    `INSERT INTO events (
                        id, stream_id, version, event_type, event_data, 
                        metadata, created_at
                    ) VALUES ($1, $2, $3, $4, $5, $6, NOW())`,
                    [
                        event.id,
                        streamId,
                        version,
                        event.type,
                        JSON.stringify(event.data),
                        JSON.stringify(event.metadata || {})
                    ]
                );
            }
            
            await client.query('COMMIT');
            
            // Publish events to event bus
            for (const event of events) {
                await this.publishEvent(streamId, event);
            }
            
            return currentVersion + events.length;
        } catch (error) {
            await client.query('ROLLBACK');
            throw error;
        } finally {
            client.release();
        }
    }

    async getEvents(streamId, fromVersion = 0) {
        const result = await this.db.query(
            `SELECT * FROM events 
             WHERE stream_id = $1 AND version > $2 
             ORDER BY version ASC`,
            [streamId, fromVersion]
        );
        
        return result.rows.map(row => ({
            id: row.id,
            type: row.event_type,
            data: JSON.parse(row.event_data),
            metadata: JSON.parse(row.metadata),
            version: row.version,
            createdAt: row.created_at
        }));
    }

    async getAllEvents(fromPosition = 0, limit = 1000) {
        const result = await this.db.query(
            `SELECT * FROM events 
             WHERE global_position > $1 
             ORDER BY global_position ASC 
             LIMIT $2`,
            [fromPosition, limit]
        );
        
        return result.rows.map(row => ({
            id: row.id,
            streamId: row.stream_id,
            type: row.event_type,
            data: JSON.parse(row.event_data),
            metadata: JSON.parse(row.metadata),
            version: row.version,
            globalPosition: row.global_position,
            createdAt: row.created_at
        }));
    }

        // Publish to event bus for real-time processing
        if (this.eventBus) {
            await this.eventBus.publish('domain.events', {
                streamId,
                ...event

class UserAggregate {
    constructor(id) {
        this.id = id;
        this.version = 0;
        this.uncommittedEvents = [];
        
        // Current state
        this.email = null;
        this.username = null;
        this.isActive = false;
        this.createdAt = null;
    }

    // Static factory method to create from events
    static fromHistory(id, events) {
        const user = new UserAggregate(id);
        user.replayEvents(events);
        return user;
    }

    // Command methods that generate events
    register(email, username, hashedPassword) {
        if (this.createdAt) {
            throw new Error('User already exists');
        }

        this.addEvent('UserRegistered', {
            userId: this.id,
            email,
            username,
            hashedPassword,
            registeredAt: new Date().toISOString()
        });
    }

    changeEmail(newEmail) {
        if (!this.isActive) {
            throw new Error('Cannot change email for inactive user');
        }

        if (this.email === newEmail) {
            return; // No change needed
        }

        this.addEvent('EmailChanged', {
            userId: this.id,
            oldEmail: this.email,
            newEmail,
            changedAt: new Date().toISOString()
        });
    }

    deactivate(reason) {
        if (!this.isActive) {
            throw new Error('User is already inactive');
        }

        this.addEvent('UserDeactivated', {
            userId: this.id,
            reason,
            deactivatedAt: new Date().toISOString()
        });
    }

    activate() {
        if (this.isActive) {
            throw new Error('User is already active');
        }

        this.addEvent('UserActivated', {
            userId: this.id,
            activatedAt: new Date().toISOString()
        });
    }

    // Event application methods
    applyUserRegistered(event) {
        this.email = event.email;
        this.username = event.username;
        this.isActive = true;
        this.createdAt = new Date(event.registeredAt);
    }

    applyEmailChanged(event) {
        this.email = event.newEmail;
    }

    applyUserDeactivated(event) {
        this.isActive = false;
    }

    applyUserActivated(event) {
        this.isActive = true;
    }

    // Internal methods
    addEvent(eventType, eventData) {
        const event = {
            id: this.generateEventId(),
            type: eventType,
            data: eventData,
            metadata: {
                aggregateId: this.id,
                aggregateType: 'User',
                correlationId: this.generateCorrelationId(),
                causationId: this.generateCausationId()
            }
        };

        this.uncommittedEvents.push(event);
        this.applyEvent(event);
    }

    applyEvent(event) {
        const methodName = `apply${event.type}`;
        if (typeof this[methodName] === 'function') {
            this[methodName](event.data);
            this.version++;
        } else {
            throw new Error(`Unknown event type: ${event.type}`);
        }
    }

    replayEvents(events) {
        for (const event of events) {
            this.applyEvent(event);
        }
        this.uncommittedEvents = [];
    }

    getUncommittedEvents() {
        return [...this.uncommittedEvents];
    }

    markEventsAsCommitted() {
        this.uncommittedEvents = [];
    }

    generateEventId() {
        return require('crypto').randomUUID();
    }

    generateCorrelationId() {
        return require('crypto').randomUUID();
    }

    generateCausationId() {
        return require('crypto').randomUUID();
    }
}

class InMemoryEventBus {
    constructor() {
        this.subscribers = new Map();
        this.eventHistory = [];
        this.maxHistorySize = 1000;
    }

    subscribe(eventType, handler, options = {}) {
        if (!this.subscribers.has(eventType)) {
            this.subscribers.set(eventType, []);
        }

        const subscription = {
            id: Math.random().toString(36).substring(2),
            handler,
            options: {
                once: options.once || false,
                priority: options.priority || 0,
                filter: options.filter || null,
                async: options.async !== false
            }
        };

        this.subscribers.get(eventType).push(subscription);
        
        // Sort by priority (higher numbers first)
        this.subscribers.get(eventType).sort((a, b) => b.options.priority - a.options.priority);

        return {
            unsubscribe: () => this.unsubscribe(eventType, subscription.id)
        };
    }

    unsubscribe(eventType, subscriptionId) {
        if (this.subscribers.has(eventType)) {
            const subscriptions = this.subscribers.get(eventType);
            const index = subscriptions.findIndex(sub => sub.id === subscriptionId);
            if (index > -1) {
                subscriptions.splice(index, 1);
            }
        }
    }

    async publish(eventType, eventData, metadata = {}) {
        const event = {
            id: Math.random().toString(36).substring(2),
            type: eventType,
            data: eventData,
            metadata: {
                timestamp: new Date().toISOString(),
                publisher: metadata.publisher || 'unknown',
                correlationId: metadata.correlationId || Math.random().toString(36).substring(2),
                ...metadata
            }
        };

        // Store in history
        this.eventHistory.push(event);
        if (this.eventHistory.length > this.maxHistorySize) {
            this.eventHistory.shift();
        }

        const subscribers = this.subscribers.get(eventType) || [];
        const errors = [];

        for (const subscription of subscribers) {
            try {
                // Apply filter if present
                if (subscription.options.filter && !subscription.options.filter(event)) {
                    continue;
                }

                if (subscription.options.async) {
                    // Handle asynchronously
                    this.handleAsync(subscription, event);
                } else {
                    // Handle synchronously
                    await subscription.handler(event);
                }

                // Remove subscription if it's a one-time handler
                if (subscription.options.once) {
                    this.unsubscribe(eventType, subscription.id);
                }
            } catch (error) {
                console.error(`Error in event handler for ${eventType}:`, error);
                errors.push({ subscription: subscription.id, error });
            }
        }

        if (errors.length > 0) {
            console.warn(`${errors.length} handlers failed for event ${eventType}`);
        }

        return event;
    }

    async handleAsync(subscription, event) {
        try {
            await subscription.handler(event);
        } catch (error) {
            console.error(`Async event handler error:`, error);
        }
    }

    // Subscribe to multiple event types with pattern matching
    subscribePattern(pattern, handler, options = {}) {
        const subscription = {
            id: Math.random().toString(36).substring(2),
            pattern: new RegExp(pattern),
            handler,
            options
        };

        if (!this.subscribers.has('*pattern*')) {
            this.subscribers.set('*pattern*', []);
        }

        this.subscribers.get('*pattern*').push(subscription);

        return {
            unsubscribe: () => {
                const subscriptions = this.subscribers.get('*pattern*');
                const index = subscriptions.findIndex(sub => sub.id === subscription.id);
                if (index > -1) {
                    subscriptions.splice(index, 1);
                }
            }
        };
    }

    getEventHistory(filter = null) {
        if (filter) {
            return this.eventHistory.filter(filter);
        }
        return [...this.eventHistory];
    }

    clear() {
        this.subscribers.clear();
        this.eventHistory = [];
    }
}

                orderId,
                paymentId,
                error: paymentError,
                failedAt: new Date().toISOString()

                new Date().toISOString()
        };

        if (options.delay && options.delay > 0) {
            // Schedule job for later processing
            await this.redis.zadd(
                `${queueName}:delayed`,
                Date.now() + options.delay,
                JSON.stringify(jobData)
            );
        } else {
            // Add to immediate processing queue
            if (options.priority > 0) {
                // High priority jobs go to front
                await this.redis.lpush(`${queueName}:jobs`, JSON.stringify(jobData));
            } else {
                // Normal priority jobs go to back
                await this.redis.rpush(`${queueName}:jobs`, JSON.stringify(jobData));
            }
        }

        return jobData.id;
    }

    async process(queueName, processor, options = {}) {
        const concurrency = options.concurrency || 1;
        const pollInterval = options.pollInterval || 1000;

        this.processingQueues.set(queueName, {
            processor,
            isActive: true,
            workers: []
        });

        // Start worker processes
        for (let i = 0; i < concurrency; i++) {
            const worker = this.createWorker(queueName, processor, pollInterval);
            this.processingQueues.get(queueName).workers.push(worker);
        }

        // Start delayed job scheduler
        this.startDelayedJobScheduler(queueName);
    }

    createWorker(queueName, processor, pollInterval) {
        const worker = async () => {
            while (!this.isShuttingDown && this.processingQueues.get(queueName)?.isActive) {
                try {
                    const jobData = await this.dequeue(queueName);
                    
                    if (jobData) {
                        await this.processJob(queueName, jobData, processor);
                    } else {
                        // No jobs available, wait before polling again
                        await new Promise(resolve => setTimeout(resolve, pollInterval));
                    }
                } catch (error) {
                    console.error(`Worker error in queue ${queueName}:`, error);
                    await new Promise(resolve => setTimeout(resolve, pollInterval));
                }
            }
        };

        worker();
        return worker;
    }

    async dequeue(queueName) {
        const jobJson = await this.redis.lpop(`${queueName}:jobs`);
        
        if (jobJson) {
            return JSON.parse(jobJson);
        }
        
        return null;
    }

    async processJob(queueName, jobData, processor) {
        const startTime = Date.now();
        
        try {
            // Move job to processing set
            await this.redis.sadd(`${queueName}:processing`, jobData.id);
            
            console.log(`Processing job ${jobData.id} from queue ${queueName}`);
            
            // Process the job
            await processor(jobData.data);
            
            // Job completed successfully
            await this.completeJob(queueName, jobData, startTime);
            
        } catch (error) {
            console.error(`Job ${jobData.id} failed:`, error);
            await this.handleJobFailure(queueName, jobData, error);
        }
    }

    async completeJob(queueName, jobData, startTime) {
        const duration = Date.now() - startTime;
        
        // Remove from processing set
        await this.redis.srem(`${queueName}:processing`, jobData.id);
        
        // Add to completed jobs (with TTL for cleanup)
        const completedData = {
            ...jobData,
            completedAt: new Date().toISOString(),
            duration
        };
        
        await this.redis.setex(
            `${queueName}:completed:${jobData.id}`,
            24 * 60 * 60, // 24 hours TTL
            JSON.stringify(completedData)
        );
        
        console.log(`Job ${jobData.id} completed in ${duration}ms`);
    }

    async handleJobFailure(queueName, jobData, error) {
        jobData.attempts++;
        jobData.lastError = {
            message: error.message,
            stack: error.stack,
            occurredAt: new Date().toISOString()
        };

        // Remove from processing set
        await this.redis.srem(`${queueName}:processing`, jobData.id);

        if (jobData.attempts < jobData.maxAttempts) {
            // Retry with exponential backoff
            const backoffDelay = Math.pow(2, jobData.attempts) * 1000; // 2^attempts seconds
            
            await this.redis.zadd(
                `${queueName}:delayed`,
                Date.now() + backoffDelay,
                JSON.stringify(jobData)
            );
            
            console.log(`Job ${jobData.id} will be retried in ${backoffDelay}ms (attempt ${jobData.attempts}/${jobData.maxAttempts})`);
        } else {
            // Move to failed jobs
            await this.redis.setex(
                `${queueName}:failed:${jobData.id}`,
                7 * 24 * 60 * 60, // 7 days TTL
                JSON.stringify(jobData)
            );
            
            console.error(`Job ${jobData.id} permanently failed after ${jobData.attempts} attempts`);
        }
    }

    async startDelayedJobScheduler(queueName) {
        const scheduler = async () => {
            while (!this.isShuttingDown && this.processingQueues.get(queueName)?.isActive) {
                try {
                    // Get jobs that are ready to be processed
                    const readyJobs = await this.redis.zrangebyscore(
                        `${queueName}:delayed`,
                        0,
                        Date.now(),
                        'WITHSCORES',
                        'LIMIT', 0, 100
                    );

                    for (let i = 0; i < readyJobs.length; i += 2) {
                        const jobJson = readyJobs[i];
                        const score = readyJobs[i + 1];
                        
                        // Move job from delayed to active queue
                        const removed = await this.redis.zrem(`${queueName}:delayed`, jobJson);
                        
                        if (removed) {
                            await this.redis.rpush(`${queueName}:jobs`, jobJson);
                        }
                    }

                    // Wait before next check
                    await new Promise(resolve => setTimeout(resolve, 1000));
                } catch (error) {
                    console.error(`Delayed job scheduler error for queue ${queueName}:`, error);
                    await new Promise(resolve => setTimeout(resolve, 5000));
                }
            }
        };

        scheduler();
    }

    async getQueueStats(queueName) {
        const [
            jobsCount,
            delayedCount,
            processingCount,
            completedCount,
            failedCount
        ] = await Promise.all([
            this.redis.llen(`${queueName}:jobs`),
            this.redis.zcard(`${queueName}:delayed`),
            this.redis.scard(`${queueName}:processing`),
            this.redis.keys(`${queueName}:completed:*`).then(keys => keys.length),
            this.redis.keys(`${queueName}:failed:*`).then(keys => keys.length)
        ]);

        return {
            waiting: jobsCount,
            delayed: delayedCount,
            processing: processingCount,
            completed: completedCount,
            failed: failedCount
        };
    }

    async pauseQueue(queueName) {
        const queueInfo = this.processingQueues.get(queueName);
        if (queueInfo) {
            queueInfo.isActive = false;
        }
    }

    async resumeQueue(queueName) {
        const queueInfo = this.processingQueues.get(queueName);
        if (queueInfo) {
            queueInfo.isActive = true;
        }
    }

    async shutdown() {
        this.isShuttingDown = true;
        
        // Wait for current jobs to complete
        const activeQueues = Array.from(this.processingQueues.keys());
        
        for (const queueName of activeQueues) {
            console.log(`Waiting for queue ${queueName} to finish processing...`);
            
            let processing = 0;
            do {
                processing = await this.redis.scard(`${queueName}:processing`);
                if (processing > 0) {
                    console.log(`${processing} jobs still processing in queue ${queueName}`);
                    await new Promise(resolve => setTimeout(resolve, 1000));
                }
            } while (processing > 0);
        }

        await this.redis.disconnect();
        console.log('Queue system shutdown complete');
    }
}

Benefits of Event-Driven Architecture

  • Loose Coupling: Components don't need to know about each other
  • Scalability: Easy to scale individual components based on event load
  • Real-time Processing: Immediate response to events as they occur
  • Fault Tolerance: System continues functioning if some consumers fail
  • Audit Trail: Events provide natural logging and audit capabilities
  • Extensibility: Easy to add new event consumers without changing producers

Drawbacks and Challenges

  • Eventual Consistency: System state may be temporarily inconsistent
  • Complexity: Distributed event handling can be complex to debug
  • Event Ordering: Ensuring proper event sequence can be challenging
  • Error Handling: Failed event processing requires careful handling
  • Testing Difficulty: Testing distributed event flows is complex
  • Message Broker Dependency: System depends on event infrastructure

When to Use Event-Driven Architecture

  • Real-time Systems: Applications requiring immediate response to changes
  • Microservices: Loose coupling between independent services
  • High Throughput: Systems processing large volumes of events
  • Audit Requirements: Need for complete event history and traceability
  • Complex Workflows: Multi-step processes with conditional logic
  • Integration: Connecting multiple systems with different protocols

CQRS (Command Query Responsibility Segregation)

CQRS separates read and write operations into different models, optimizing each for their specific purpose. This pattern is especially powerful when combined with Event Sourcing.

🔄

Core CQRS Principles

  • Command Model: Handles write operations and business logic
  • Query Model: Optimized for read operations and reporting
  • Separation: Different models for updating vs reading data
  • Eventual Consistency: Read models updated asynchronously

class GetUserListHandler {
    constructor(userViewRepository) {
        this.userViewRepository = userViewRepository;
    }

    async handle(query) {
        const { page, limit, search, filters } = query;
        
        // Build optimized read query
        let whereClause = 'WHERE 1=1';
        const params = [];
        let paramIndex = 1;

        // Search functionality
        if (search) {
            whereClause += ` AND (username ILIKE $${paramIndex} OR email ILIKE $${paramIndex})`;
            params.push(`%${search}%`);
            paramIndex++;
        }

        // Apply filters
        if (filters.isActive !== undefined) {
            whereClause += ` AND is_active = $${paramIndex}`;
            params.push(filters.isActive);
            paramIndex++;
        }

        if (filters.minCompleteness) {
            whereClause += ` AND profile_completeness >= $${paramIndex}`;
            params.push(filters.minCompleteness);
            paramIndex++;
        }

        // Get total count for pagination
        const countQuery = `
            SELECT COUNT(*) as total 
            FROM user_views 
            ${whereClause}
        `;
        
        const countResult = await this.userViewRepository.query(countQuery, params);
        const total = parseInt(countResult.rows[0].total);

        // Get paginated results
        const dataQuery = `
            SELECT 
                id,
                email,
                username,
                display_name,
                is_active,
                created_at,
                last_login_at,
                profile_completeness,
                avatar_url
            FROM user_views 
            ${whereClause}
            ORDER BY ${query.getSortClause()}
            LIMIT $${paramIndex} OFFSET $${paramIndex + 1}
        `;
        
        params.push(limit, query.getOffset());
        
        const result = await this.userViewRepository.query(dataQuery, params);
        
        return {
            users: result.rows,
            pagination: {
                page: query.page,
                limit: query.limit,
                total: total,
                totalPages: Math.ceil(total / limit),
                hasNext: query.page * limit < total,
                hasPrev: query.page > 1
            },
            metadata: {
                executedAt: new Date(),
                queryTime: Date.now() - query.startTime
            }
        };
    }

// Event Sourced User Aggregate
class UserAggregate {
    constructor() {
        this.id = null;
        this.email = null;
        this.username = null;
        this.isActive = true;
        this.version = 0;
        this.uncommittedEvents = [];
    }

    // Command methods that produce events
    static createUser(userData) {
        const aggregate = new UserAggregate();
        const event = {
            type: 'UserCreated',
            aggregateId: userData.id,
            data: {
                email: userData.email,
                username: userData.username,
                hashedPassword: userData.hashedPassword
            },
            version: 1,
            timestamp: new Date()
        };
        
        aggregate.apply(event);
        aggregate.uncommittedEvents.push(event);
        return aggregate;
    }

    updateEmail(newEmail) {
        if (!this.isActive) {
            throw new Error('Cannot update inactive user');
        }
        
        if (this.email === newEmail) {
            return; // No change needed
        }

        const event = {
            type: 'UserEmailUpdated',
            aggregateId: this.id,
            data: {
                oldEmail: this.email,
                newEmail: newEmail
            },
            version: this.version + 1,
            timestamp: new Date()
        };

        this.apply(event);
        this.uncommittedEvents.push(event);
    }

    deactivateUser(reason) {
        if (!this.isActive) {
            throw new Error('User is already inactive');
        }

        const event = {
            type: 'UserDeactivated',
            aggregateId: this.id,
            data: { reason },
            version: this.version + 1,
            timestamp: new Date()
        };

        this.apply(event);
        this.uncommittedEvents.push(event);
    }

    // Apply events to rebuild state
    apply(event) {
        switch (event.type) {
            case 'UserCreated':
                this.id = event.aggregateId;
                this.email = event.data.email;
                this.username = event.data.username;
                this.isActive = true;
                break;
                
            case 'UserEmailUpdated':
                this.email = event.data.newEmail;
                break;
                
            case 'UserDeactivated':
                this.isActive = false;
                break;
                
            default:
                throw new Error(`Unknown event type: ${event.type}`);
        }
        
        this.version = event.version;
    }

    // Rebuild aggregate from event history
    static fromHistory(events) {
        const aggregate = new UserAggregate();
        
        events.forEach(event => {
            aggregate.apply(event);
        });
        
        return aggregate;
    }

    getUncommittedEvents() {
        return [...this.uncommittedEvents];
    }

    markEventsAsCommitted() {
        this.uncommittedEvents = [];
    }
}

            success: true,

            eventsPublished: newEvents.length

        `, [event.timestamp, event.data.reason, event.aggregateId]);
    }

    async save(userAggregate) {
        const client = await this.db.connect();
        
        try {
            await client.query('BEGIN');
            
            // Save aggregate
            await client.query(`
                INSERT INTO users (id, email, username, password_hash, version, created_at, updated_at)
                VALUES ($1, $2, $3, $4, $5, $6, $7)
                ON CONFLICT (id) DO UPDATE SET
                    email = EXCLUDED.email,
                    username = EXCLUDED.username,
                    password_hash = EXCLUDED.password_hash,
                    version = EXCLUDED.version,
                    updated_at = EXCLUDED.updated_at
            `, [

                userAggregate.email,
                userAggregate.username,
                userAggregate.passwordHash,
                userAggregate.version,
                userAggregate.createdAt,
                userAggregate.updatedAt
            ]);

            // Save domain events
            for (const event of userAggregate.getUncommittedEvents()) {
                await client.query(`
                    INSERT INTO domain_events (
                        id, aggregate_id, event_type, event_data, 
                        version, occurred_at, created_at
                    ) VALUES ($1, $2, $3, $4, $5, $6, NOW())
                `, [
                    event.id,
                    event.aggregateId,
                    event.type,
                    JSON.stringify(event.data),
                    event.version,
                    event.timestamp
                ]);
            }

            await client.query('COMMIT');
            
        } catch (error) {
            await client.query('ROLLBACK');
            throw error;
        } finally {
            client.release();
        }
    }

    async findById(id) {
        const result = await this.db.query(
            'SELECT * FROM users WHERE id = $1', 
            [id]
        );
        
        if (result.rows.length === 0) {
            return null;
        }
        
        return this.mapToAggregate(result.rows[0]);
    }

    mapToAggregate(row) {
        // Map database row to domain aggregate
        return UserAggregate.fromData({
            id: row.id,
            email: row.email,
            username: row.username,
            passwordHash: row.password_hash,
            version: row.version,
            createdAt: row.created_at,
            updatedAt: row.updated_at
        });
    }

class UserReadRepository {
    constructor() {
        this.db = readStore;
        this.cache = cacheStore;
    }

    async findById(id) {
        // Try cache first
        const cached = await this.cache.get(`user:${id}`);
        if (cached) {
            return JSON.parse(cached);
        }

        // Query MongoDB read store
        const user = await this.db.collection('users').findOne({ _id: id });
        
        if (user) {
            // Cache for 15 minutes
            await this.cache.setex(`user:${id}`, 900, JSON.stringify(user));
        }
        
        return user;
    }

    async findMany(query, options = {}) {
        const {
            page = 1,
            limit = 10,
            sort = { createdAt: -1 },
            filter = {}
        } = options;

        // Build MongoDB query
        const mongoQuery = this.buildMongoQuery(filter);
        
        // Execute with pagination
        const skip = (page - 1) * limit;
        const [users, total] = await Promise.all([
            this.db.collection('users')
                .find(mongoQuery)
                .sort(sort)
                .skip(skip)
                .limit(limit)
                .toArray(),
            this.db.collection('users').countDocuments(mongoQuery)
        ]);

        return {
            users,
            pagination: {
                page,
                limit,
                total,
                pages: Math.ceil(total / limit)
            }
        };
    }

    async save(userView) {
        // Upsert to MongoDB
        await this.db.collection('users').replaceOne(
            { _id: userView.id },
            userView,
            { upsert: true }
        );

        // Invalidate cache
        await this.cache.del(`user:${userView.id}`);
    }

    buildMongoQuery(filter) {
        const query = {};
        
        if (filter.search) {
            query.$or = [
                { username: { $regex: filter.search, $options: 'i' } },
                { email: { $regex: filter.search, $options: 'i' } }
            ];
        }
        
        if (filter.isActive !== undefined) {
            query.isActive = filter.isActive;
        }
        
        if (filter.createdAfter) {
            query.createdAt = { $gte: new Date(filter.createdAfter) };
        }
        
        return query;
    }
}

Benefits of CQRS

  • Performance Optimization: Read and write models optimized independently
  • Scalability: Scale read and write sides independently
  • Complexity Management: Separates complex business logic from simple queries
  • Technology Flexibility: Different technologies for read vs write sides
  • Business Logic Focus: Commands encapsulate business operations clearly

Drawbacks and Challenges

  • Increased Complexity: More moving parts and infrastructure
  • Eventual Consistency: Read models may be temporarily out of sync
  • Development Overhead: Separate models and synchronization logic
  • Learning Curve: Team must understand CQRS principles
  • Data Consistency: Managing consistency across read/write stores

When to Use CQRS

  • Complex Business Domains: Different read/write patterns and requirements
  • High-Performance Requirements: Need to optimize reads and writes separately
  • Detailed Audit Requirements: Need complete history of all changes
  • Complex Reporting: Sophisticated analytics and reporting needs
  • Event-Driven Systems: Natural fit with event sourcing patterns
