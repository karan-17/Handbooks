Overview & Architecture

LangChain data ingestion provides a standardized framework for converting various data sources into uniform Document objects for downstream LLM processing. The architecture is built around the BaseLoader class, which offers two fundamental methods:

.load()

Returns a complete list of Document objects. Use for smaller datasets where loading everything into memory is acceptable.

.lazy_load()

Yields documents one by one as a generator. Essential for large datasets to prevent memory overflow.

Document Object Structure

Every loaded document follows this standardized structure:

class Document: page_content: str # The actual text content metadata: Dict[str, Any] # Supplementary information (source, page number, etc.)
2025 Update: LangChain v0.3+ includes Pydantic v2 support internally, providing better type validation and performance improvements for Document objects.

File Processing Best Practices

Memory-Efficient Processing

import gc from typing import Generator, List from langchain.schema import Document class MemoryEfficientProcessor: """Memory-conscious document processing for large datasets""" def __init__(self, batch_size: int = 100): self.batch_size = batch_size self.processed_count = 0 def process_directory_lazily(self, directory_path: str) -> Generator[Document, None, None]: """Process files one at a time to avoid memory overflow""" from pathlib import Path pdf_files = list(Path(directory_path).glob("**/*.pdf")) for pdf_file in pdf_files: try: loader = PyPDFLoader(str(pdf_file)) # Use lazy_load to process one page at a time for document in loader.lazy_load(): yield document self.processed_count += 1 # Periodic garbage collection if self.processed_count % self.batch_size == 0: gc.collect() print(f"Processed {self.processed_count} documents, memory cleared") except Exception as e: print(f"Error processing {pdf_file}: {e}") continue # Clear loader reference del loader def batch_process_with_chunking(self, file_paths: List[str], chunk_size: int = 1000): """Process files in chunks with text splitting""" from langchain.text_splitter import RecursiveCharacterTextSplitter text_splitter = RecursiveCharacterTextSplitter( chunk_size=chunk_size, chunk_overlap=200, length_function=len, separators=["\n\n", "\n", ". ", "! ", "? ", " ", ""] ) for file_path in file_paths: loader = self._get_loader_for_file(file_path) try: # Load and immediately split documents = loader.load() chunks = text_splitter.split_documents(documents) # Process chunks in batches for i in range(0, len(chunks), self.batch_size): batch = chunks[i:i + self.batch_size] yield batch # Clear references del documents, chunks gc.collect() except Exception as e: print(f"Error processing {file_path}: {e}") finally: del loader def _get_loader_for_file(self, file_path: str): """Get appropriate loader based on file extension""" from pathlib import Path suffix = Path(file_path).suffix.lower() loaders = { '.pdf': PyPDFLoader, '.docx': Docx2txtLoader, '.txt': TextLoader, '.csv': CSVLoader, '.json': JSONLoader } loader_class = loaders.get(suffix) if not loader_class: raise ValueError(f"Unsupported file type: {suffix}") return loader_class(file_path) # Usage processor = MemoryEfficientProcessor(batch_size=50) # Lazy processing for large directories for document in processor.process_directory_lazily("./large_document_collection/"): # Process each document individually process_document(document) # Batch processing with chunking for batch in processor.batch_process_with_chunking( file_paths=["./large_file1.pdf", "./large_file2.pdf"], chunk_size=1000 ): # Process each batch store_in_vector_db(batch)

Smart Document Chunking

from langchain.text_splitter import ( RecursiveCharacterTextSplitter, MarkdownHeaderTextSplitter, TokenTextSplitter ) from langchain.schema import Document class SmartChunker: """Intelligent document chunking for optimal RAG performance""" def __init__(self): self.splitters = { 'recursive': RecursiveCharacterTextSplitter( chunk_size=1000, chunk_overlap=200, separators=["\n\n", "\n", ". ", "! ", "? ", " ", ""] ), 'markdown': MarkdownHeaderTextSplitter( headers_to_split_on=[ ("#", "Header 1"), ("##", "Header 2"), ("###", "Header 3") ] ), 'token': TokenTextSplitter( encoding_name="cl100k_base", # GPT-3.5/4 tokenizer chunk_size=800, chunk_overlap=100 ) } def chunk_by_document_type(self, documents: List[Document]) -> List[Document]: """Apply different chunking strategies based on document type""" chunked_docs = [] for doc in documents: source = doc.metadata.get('source', '') if source.endswith('.md') or 'markdown' in source.lower(): # Use markdown-aware chunking md_chunks = self.splitters['markdown'].split_text(doc.page_content) for chunk in md_chunks: chunked_docs.append(Document( page_content=chunk.content, metadata={**doc.metadata, **chunk.metadata} )) elif 'api' in source.lower() or 'technical' in source.lower(): # Use token-based chunking for technical content token_chunks = self.splitters['token'].split_documents([doc]) chunked_docs.extend(token_chunks) else: # Use recursive chunking for general content recursive_chunks = self.splitters['recursive'].split_documents([doc]) chunked_docs.extend(recursive_chunks) return chunked_docs def semantic_chunking(self, documents: List[Document]) -> List[Document]: """Advanced semantic chunking using embeddings""" from sentence_transformers import SentenceTransformer import numpy as np from sklearn.metrics.pairwise import cosine_similarity model = SentenceTransformer('all-MiniLM-L6-v2') semantic_chunks = [] for doc in documents: # Split into sentences sentences = doc.page_content.split('. ') if len(sentences) < 3: semantic_chunks.append(doc) continue # Generate embeddings embeddings = model.encode(sentences) # Find semantic boundaries using cosine similarity similarities = [] for i in range(len(embeddings) - 1): sim = cosine_similarity([embeddings[i]], [embeddings[i + 1]])[0][0] similarities.append(sim) # Find split points (low similarity = topic change) threshold = np.percentile(similarities, 25) # Bottom 25% split_points = [0] for i, sim in enumerate(similarities): if sim < threshold: split_points.append(i + 1) split_points.append(len(sentences)) # Create chunks based on semantic boundaries for i in range(len(split_points) - 1): start = split_points[i] end = split_points[i + 1] chunk_text = '. '.join(sentences[start:end]) semantic_chunks.append(Document( page_content=chunk_text, metadata={ **doc.metadata, 'chunk_type': 'semantic', 'chunk_index': i, 'semantic_score': np.mean(similarities[start:end-1]) if start < end-1 else 1.0 } )) return semantic_chunks # Usage chunker = SmartChunker() # Load documents docs = [ PyPDFLoader("./technical_manual.pdf").load()[0], TextLoader("./readme.md").load()[0], WebBaseLoader("https://api-docs.example.com").load()[0] ] # Apply intelligent chunking smart_chunks = chunker.chunk_by_document_type(docs) # Apply semantic chunking for better context preservation semantic_chunks = chunker.semantic_chunking(smart_chunks) print(f"Original documents: {len(docs)}") print(f"Smart chunks: {len(smart_chunks)}") print(f"Semantic chunks: {len(semantic_chunks)}")

Error Handling Patterns

Robust Error Handling Framework

import logging import time from typing import List, Optional, Dict, Any from dataclasses import dataclass from enum import Enum class ErrorSeverity(Enum): LOW = "low" MEDIUM = "medium" HIGH = "high" CRITICAL = "critical" @dataclass class ProcessingError: file_path: str error_type: str error_message: str severity: ErrorSeverity timestamp: float retry_count: int = 0 class RobustDocumentLoader: """Production-ready document loader with comprehensive error handling""" def __init__(self, max_retries: int = 3, retry_delay: float = 1.0, fallback_enabled: bool = True): self.max_retries = max_retries self.retry_delay = retry_delay self.fallback_enabled = fallback_enabled self.errors: List[ProcessingError] = [] self.successful_loads = 0 self.failed_loads = 0 # Configure logging logging.basicConfig( level=logging.INFO, format='%(asctime)s - %(name)s - %(levelname)s - %(message)s', handlers=[ logging.FileHandler('document_processing.log'), logging.StreamHandler() ] ) self.logger = logging.getLogger(__name__) def load_with_retry(self, file_path: str) -> Optional[List[Document]]: """Load document with automatic retry and fallback mechanisms""" for attempt in range(self.max_retries + 1): try: # Determine appropriate loader loader = self._get_loader(file_path) # Add timeout wrapper documents = self._load_with_timeout(loader, timeout=60) self.successful_loads += 1 self.logger.info(f"Successfully loaded {file_path} on attempt {attempt + 1}") return documents except Exception as e: error = ProcessingError( file_path=file_path, error_type=type(e).__name__, error_message=str(e), severity=self._classify_error_severity(e), timestamp=time.time(), retry_count=attempt ) self.errors.append(error) if attempt < self.max_retries: self.logger.warning( f"Attempt {attempt + 1} failed for {file_path}: {e}. " f"Retrying in {self.retry_delay} seconds..." ) time.sleep(self.retry_delay * (2 ** attempt)) # Exponential backoff else: self.logger.error(f"All attempts failed for {file_path}: {e}") # Try fallback processing if self.fallback_enabled: fallback_result = self._fallback_processing(file_path, error) if fallback_result: return fallback_result self.failed_loads += 1 return None def _load_with_timeout(self, loader, timeout: int = 60) -> List[Document]: """Wrap loader with timeout protection""" import signal def timeout_handler(signum, frame): raise TimeoutError(f"Document loading timed out after {timeout} seconds") old_handler = signal.signal(signal.SIGALRM, timeout_handler) signal.alarm(timeout) try: documents = loader.load() return documents finally: signal.alarm(0) signal.signal(signal.SIGALRM, old_handler) def _get_loader(self, file_path: str): """Get appropriate loader with error checking""" from pathlib import Path if not Path(file_path).exists(): raise FileNotFoundError(f"File not found: {file_path}") file_size = Path(file_path).stat().st_size if file_size > 100 * 1024 * 1024: # 100MB self.logger.warning(f"Large file detected: {file_path} ({file_size / 1024 / 1024:.1f}MB)") suffix = Path(file_path).suffix.lower() loaders = { '.pdf': PyPDFLoader, '.docx': Docx2txtLoader, '.txt': lambda x: TextLoader(x, encoding='utf-8'), '.csv': CSVLoader, '.json': JSONLoader, '.md': UnstructuredMarkdownLoader } loader_class = loaders.get(suffix) if not loader_class: raise ValueError(f"Unsupported file type: {suffix}") return loader_class(file_path) def _classify_error_severity(self, error: Exception) -> ErrorSeverity: """Classify error severity for appropriate handling""" error_type = type(error).__name__ critical_errors = ['MemoryError', 'SystemError', 'KeyboardInterrupt'] high_errors = ['PermissionError', 'OSError', 'IOError'] medium_errors = ['TimeoutError', 'UnicodeDecodeError', 'ValueError'] if error_type in critical_errors: return ErrorSeverity.CRITICAL elif error_type in high_errors: return ErrorSeverity.HIGH elif error_type in medium_errors: return ErrorSeverity.MEDIUM else: return ErrorSeverity.LOW def _fallback_processing(self, file_path: str, error: ProcessingError) -> Optional[List[Document]]: """Attempt fallback processing strategies""" self.logger.info(f"Attempting fallback processing for {file_path}") try: # Strategy 1: Try unstructured loader as universal fallback from langchain_unstructured import UnstructuredLoader fallback_loader = UnstructuredLoader( file_path=file_path, partition_via_api=False, mode="single" ) documents = fallback_loader.load() # Mark as fallback processed for doc in documents: doc.metadata['processed_via_fallback'] = True doc.metadata['original_error'] = error.error_message self.logger.info(f"Fallback processing successful for {file_path}") self.successful_loads += 1 return documents except Exception as fallback_error: self.logger.error(f"Fallback processing also failed for {file_path}: {fallback_error}") return None def batch_load_with_monitoring(self, file_paths: List[str]) -> Dict[str, Any]: """Batch load with comprehensive monitoring""" results = { 'successful_documents': [], 'failed_files': [], 'processing_stats': {}, 'error_summary': {} } start_time = time.time() for i, file_path in enumerate(file_paths): self.logger.info(f"Processing {i+1}/{len(file_paths)}: {file_path}") docs = self.load_with_retry(file_path) if docs: results['successful_documents'].extend(docs) else: results['failed_files'].append(file_path) # Progress reporting if (i + 1) % 10 == 0: self.logger.info(f"Progress: {i+1}/{len(file_paths)} files processed") # Compile statistics end_time = time.time() results['processing_stats'] = { 'total_files': len(file_paths), 'successful_loads': self.successful_loads, 'failed_loads': self.failed_loads, 'success_rate': self.successful_loads / len(file_paths) if file_paths else 0, 'processing_time': end_time - start_time, 'documents_per_second': len(results['successful_documents']) / (end_time - start_time) } # Error analysis error_counts = {} for error in self.errors: error_counts[error.error_type] = error_counts.get(error.error_type, 0) + 1 results['error_summary'] = { 'total_errors': len(self.errors), 'error_distribution': error_counts, 'critical_errors': [e for e in self.errors if e.severity == ErrorSeverity.CRITICAL] } return results # Usage example loader = RobustDocumentLoader( max_retries=3, retry_delay=1.0, fallback_enabled=True ) # Process large batch of files file_paths = [ "./documents/report1.pdf", "./documents/data.csv", "./documents/presentation.pptx", "./documents/corrupted_file.pdf", # This will trigger error handling "./documents/large_document.pdf" ] results = loader.batch_load_with_monitoring(file_paths) print(f"Processing completed:") print(f"- Success rate: {results['processing_stats']['success_rate']:.2%}") print(f"- Documents loaded: {len(results['successful_documents'])}") print(f"- Failed files: {len(results['failed_files'])}") print(f"- Processing time: {results['processing_stats']['processing_time']:.2f} seconds") # Handle critical errors if results['error_summary']['critical_errors']: print("CRITICAL ERRORS DETECTED:") for error in results['error_summary']['critical_errors']: print(f"- {error.file_path}: {error.error_message}")

Circuit Breaker Pattern

from enum import Enum import time from typing import Callable, Any class CircuitState(Enum): CLOSED = "closed" # Normal operation OPEN = "open" # Failing, rejecting requests HALF_OPEN = "half_open" # Testing if service recovered class CircuitBreaker: """Implement circuit breaker pattern for external service calls""" def __init__(self, failure_threshold: int = 5, reset_timeout: int = 60, expected_exception: Exception = Exception): self.failure_threshold = failure_threshold self.reset_timeout = reset_timeout self.expected_exception = expected_exception self.failure_count = 0 self.last_failure_time = 0 self.state = CircuitState.CLOSED def call(self, func: Callable, *args, **kwargs) -> Any: """Execute function with circuit breaker protection""" if self.state == CircuitState.OPEN: if time.time() - self.last_failure_time > self.reset_timeout: self.state = CircuitState.HALF_OPEN self.failure_count = 0 else: raise Exception("Circuit breaker is OPEN - service unavailable") try: result = func(*args, **kwargs) self._on_success() return result except self.expected_exception as e: self._on_failure() raise e def _on_success(self): """Handle successful call""" self.failure_count = 0 if self.state == CircuitState.HALF_OPEN: self.state = CircuitState.CLOSED def _on_failure(self): """Handle failed call""" self.failure_count += 1 self.last_failure_time = time.time() if self.failure_count >= self.failure_threshold: self.state = CircuitState.OPEN # Usage with document loaders class ProtectedWebLoader: def __init__(self): self.circuit_breaker = CircuitBreaker( failure_threshold=3, reset_timeout=30, expected_exception=(ConnectionError, TimeoutError) ) def load_url_protected(self, url: str) -> List[Document]: """Load URL with circuit breaker protection""" def _load_url(): loader = WebBaseLoader(url) return loader.load() return self.circuit_breaker.call(_load_url) # Example usage protected_loader = ProtectedWebLoader() urls = [ "https://reliable-site.com/docs", "https://unreliable-site.com/data", # This might fail "https://another-site.com/info" ] for url in urls: try: docs = protected_loader.load_url_protected(url) print(f"Successfully loaded {url}") except Exception as e: print(f"Failed to load {url}: {e}")

import json import pickle from pathlib import Path from typing import Dict, List, Optional, Set from dataclasses import dataclass, asdict import hashlib @dataclass class BatchCheckpoint: batch_id: str processed_files: Set[str] failed_files: Set[str] total_files: int start_time: float last_update: float metadata: Dict[str, Any] class CheckpointedBatchProcessor: """Batch processor with checkpoint/resume capability for long-running jobs""" def __init__(self, batch_size: int = 100, checkpoint_interval: int = 50, checkpoint_dir: str = "./checkpoints"): self.batch_size = batch_size self.checkpoint_interval = checkpoint_interval self.checkpoint_dir = Path(checkpoint_dir) self.checkpoint_dir.mkdir(exist_ok=True) self.current_checkpoint: Optional[BatchCheckpoint] = None def process_batch_with_resume(self, file_paths: List[str], batch_id: Optional[str] = None) -> Dict[str, Any]: """Process batch with automatic checkpointing and resume capability""" if batch_id is None: batch_id = self._generate_batch_id(file_paths) # Also save human-readable JSON version json_file = self.checkpoint_dir / f"{self.current_checkpoint.batch_id}.json" with open(json_file, 'w') as f: # Convert sets to lists for JSON serialization json_data = checkpoint_data.copy() json.dump(json_data, f, indent=2, default=str) except Exception as e: print(f"Error saving checkpoint: {e}") #
