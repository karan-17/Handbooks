1. Memory Fundamentals in AI Systems

Types of Memory in AI Agents

🧠 Short-Term Memory

Purpose: Immediate context and conversation state

Duration: Session-based (minutes to hours)

Storage: In-memory, Redis, or session stores

Use Cases: Chat context, current task state, working memory

📚 Long-Term Memory

Purpose: Persistent knowledge and experiences

Duration: Permanent (days to years)

Storage: Vector databases, knowledge graphs

Use Cases: User preferences, learned facts, historical interactions

🎯 Episodic Memory

Purpose: Specific events and experiences

Duration: Context-dependent

Storage: Time-indexed vector stores

Use Cases: Past conversations, event sequences, temporal context

🗂️ Semantic Memory

Purpose: Factual knowledge and concepts

Duration: Persistent and updatable

Storage: Knowledge graphs, structured databases

Use Cases: Domain expertise, relationships, general knowledge

⚙️ Procedural Memory

Purpose: How-to knowledge and skills

Duration: Persistent with updates

Storage: Rule engines, workflow databases

Use Cases: Process workflows, skill libraries, automation rules

🔄 Working Memory

Purpose: Active processing and manipulation

Duration: Task-based (seconds to minutes)

Storage: Application memory, cache layers

Use Cases: Complex reasoning, multi-step tasks, context switching

            "id": memory_id,
            "values": embedding,
            "metadata": {
                **metadata,
                "content": content,
                "timestamp": datetime.utcnow().isoformat()
            }
        }])
    
    async def search_memories(self, query: str, user_id: str, top_k: int = 5):

        embedding_response = self.openai_client.embeddings.create(
            input=query,
            model="text-embedding-3-small"
        )
        query_embedding = embedding_response.data[0].embedding
        
        # Search Pinecone
        results = self.index.query(
            vector=query_embedding,
            top_k=top_k,
            include_metadata=True,
            filter={"user_id": user_id}
        )
        
        return [{
            "id": match.id,
            "score": match.score,
            "content": match.metadata["content"],
            "metadata": match.metadata

from neo4j import GraphDatabase
from datetime import datetime
import json

class Neo4jMemorySystem:
    def __init__(self, uri: str, user: str, password: str):
        self.driver = GraphDatabase.driver(uri, auth=(user, password))
    
    def close(self):
        self.driver.close()
    
    def create_memory_schema(self):
        """Initialize memory schema with constraints"""
        with self.driver.session() as session:
            # Create constraints
            constraints = [
                "CREATE CONSTRAINT memory_id IF NOT EXISTS FOR (m:Memory) REQUIRE m.id IS UNIQUE",
                "CREATE CONSTRAINT user_id IF NOT EXISTS FOR (u:User) REQUIRE u.id IS UNIQUE",
                "CREATE CONSTRAINT entity_id IF NOT EXISTS FOR (e:Entity) REQUIRE e.id IS UNIQUE"
            ]
            
            for constraint in constraints:
                session.run(constraint)
    
    def store_memory_with_entities(self, memory_id: str, content: str, 
                                 user_id: str, entities: list, relationships: list):
        """Store memory with extracted entities and relationships"""
        with self.driver.session() as session:
            query = """
            // Create or update user
            MERGE (u:User {id: $user_id})
            
            // Create memory node
            CREATE (m:Memory {
                id: $memory_id,
                content: $content,
                timestamp: datetime($timestamp)
            })
            
            // Link memory to user
            CREATE (u)-[:HAS_MEMORY]->(m)
            
            // Process entities
            UNWIND $entities as entity
            MERGE (e:Entity {id: entity.id})
            SET e.name = entity.name, e.type = entity.type
            CREATE (m)-[:MENTIONS]->(e)
            
            // Process relationships
            UNWIND $relationships as rel
            MATCH (e1:Entity {id: rel.from}), (e2:Entity {id: rel.to})
            CREATE (e1)-[r:RELATES_TO {
                type: rel.type,
                confidence: rel.confidence,
                source_memory: $memory_id,
                timestamp: datetime($timestamp)
            }]->(e2)
            """
            
            session.run(query, {
                "memory_id": memory_id,
                "content": content,
                "user_id": user_id,
                "entities": entities,
                "relationships": relationships,
                "timestamp": datetime.utcnow().isoformat()
            })
    
    def temporal_query(self, user_id: str, entity: str, time_range: dict = None):
        """Query memories with temporal context"""
        query = """
        MATCH (u:User {id: $user_id})-[:HAS_MEMORY]->(m:Memory)-[:MENTIONS]->(e:Entity)
        WHERE e.name CONTAINS $entity
        """
        
        params = {"user_id": user_id, "entity": entity}
        
        if time_range:
            query += """
            AND m.timestamp >= datetime($start_time)
            AND m.timestamp <= datetime($end_time)
            """
            params.update({
                "start_time": time_range["start"],
                "end_time": time_range["end"]
            })
        
        query += """
        RETURN m.content as content, m.timestamp as timestamp, 
               collect(e.name) as mentioned_entities
        ORDER BY m.timestamp DESC
        """
        
        with self.driver.session() as session:
            result = session.run(query, params)
            return [record.data() for record in result]
    
    def find_entity_relationships(self, user_id: str, entity: str, max_hops: int = 2):
        """Find relationships around an entity"""
        query = f"""
        MATCH (u:User {{id: $user_id}})-[:HAS_MEMORY]->(m:Memory)-[:MENTIONS]->(start:Entity)
        WHERE start.name CONTAINS $entity
        MATCH path = (start)-[:RELATES_TO*1..{max_hops}]-(connected:Entity)
        RETURN start.name as source_entity, 
               [node in nodes(path) | node.name] as path_entities,
               [rel in relationships(path) | rel.type] as relationship_types
        """
        
        with self.driver.session() as session:
            result = session.run(query, {"user_id": user_id, "entity": entity})
            return [record.data() for record in result]

import boto3
from gremlin_python import statics
from gremlin_python.structure.graph import Graph
from gremlin_python.process.graph_traversal import __
from gremlin_python.driver.driver_remote_connection import DriverRemoteConnection

class NeptuneMemorySystem:
    def __init__(self, endpoint: str):
        self.endpoint = endpoint
        self.graph = Graph()
        self.connection = DriverRemoteConnection(
            f'wss://{endpoint}:8182/gremlin',
            'g'
        )
        self.g = self.graph.traversal().withRemote(self.connection)
    
    def close(self):
        self.connection.close()
    
    def store_conversational_memory(self, conversation_id: str, messages: list, user_id: str):
        """Store conversation as a graph of connected messages"""
        # Create conversation node
        conv = self.g.addV('Conversation').property('id', conversation_id)\
                     .property('user_id', user_id)\
                     .property('timestamp', datetime.utcnow().isoformat()).next()
        
        prev_message = None
        for i, message in enumerate(messages):
            # Create message node
            msg = self.g.addV('Message')\
                       .property('content', message['content'])\
                       .property('role', message['role'])\
                       .property('sequence', i)\
                       .next()
            
            # Connect to conversation
            self.g.addE('CONTAINS').from_(conv).to(msg).next()
            
            # Connect messages in sequence
            if prev_message:
                self.g.addE('FOLLOWS').from_(prev_message).to(msg).next()
            
            prev_message = msg
    
    def semantic_search_with_graph_context(self, query_concepts: list, user_id: str, max_depth: int = 3):
        """Search memories using graph traversal for context"""
        # Find messages containing query concepts
        matching_messages = self.g.V().hasLabel('Message')\
                                   .has('Conversation', 'user_id', user_id)\
                                   .where(__.values('content').is(__.containing(query_concepts[0])))\
                                   .as_('message')
        
        # Expand context through graph relationships
        context = matching_messages.repeat(
            __.both('FOLLOWS', 'CONTAINS').dedup()
        ).times(max_depth).dedup()
        
        return context.valueMap().toList()

class AdvancedMemoryCompression:
    def __init__(self):
        self.quantization_levels = {
            "high_precision": 32,   # Full precision
            "medium_precision": 16, # Half precision 
            "low_precision": 8,     # 8-bit quantization
            "ultra_low": 4         # 4-bit quantization
        }
    
    async def adaptive_compression(self, embeddings: List[np.ndarray], 
                                 workload_metrics: Dict) -> List[bytes]:
        """Adaptive compression based on system load"""
        
        # Determine compression level based on system resources
        memory_pressure = workload_metrics.get("memory_usage", 0.5)
        query_frequency = workload_metrics.get("query_frequency", 10)
        
        if memory_pressure > 0.8:
            compression_level = "ultra_low"
        elif memory_pressure > 0.6:
            compression_level = "low_precision"
        elif query_frequency > 100:  # High query frequency - prioritize speed
            compression_level = "medium_precision"
        else:
            compression_level = "high_precision"
        
        bits = self.quantization_levels[compression_level]
        
        # Compress embeddings
        compressed_embeddings = []
        for embedding in embeddings:
            compressed = await self.quantize_embedding(embedding, bits)
            compressed_embeddings.append(compressed)
        
        return compressed_embeddings
    
    async def quantize_embedding(self, embedding: np.ndarray, bits: int) -> bytes:
        """Quantize embedding to specified bit depth"""
        
        if bits == 32:
            # No compression
            return embedding.tobytes()
        
        elif bits == 16:
            # Half precision
            return embedding.astype(np.float16).tobytes()
        
        elif bits == 8:
            # 8-bit quantization
            min_val, max_val = embedding.min(), embedding.max()
            scale = (max_val - min_val) / 255
            quantized = ((embedding - min_val) / scale).astype(np.uint8)
            
            # Store scale and offset for decompression
            header = np.array([min_val, scale], dtype=np.float32).tobytes()
            return header + quantized.tobytes()
        
        elif bits == 4:
            # 4-bit quantization (more complex implementation)
            min_val, max_val = embedding.min(), embedding.max()
            scale = (max_val - min_val) / 15  # 4-bit = 16 levels (0-15)
            quantized = ((embedding - min_val) / scale).astype(np.uint8)
            
            # Pack two 4-bit values into one byte
            packed = np.zeros(len(quantized) // 2 + len(quantized) % 2, dtype=np.uint8)
            for i in range(0, len(quantized), 2):
                packed[i // 2] = (quantized[i] & 0xF) | ((quantized[i + 1] & 0xF) << 4)
            
            header = np.array([min_val, scale], dtype=np.float32).tobytes()
            return header + packed.tobytes()
    
    async def decompress_embedding(self, compressed_data: bytes, original_shape: Tuple, bits: int) -> np.ndarray:
        """Decompress embedding back to original precision"""
        
        if bits == 32:
            return np.frombuffer(compressed_data, dtype=np.float32).reshape(original_shape)
        
        elif bits == 16:
            return np.frombuffer(compressed_data, dtype=np.float16).astype(np.float32).reshape(original_shape)
        
        elif bits == 8:
            header_size = 2 * 4  # 2 float32 values
            header = np.frombuffer(compressed_data[:header_size], dtype=np.float32)
            min_val, scale = header[0], header[1]
            
            quantized = np.frombuffer(compressed_data[header_size:], dtype=np.uint8)
            return (quantized.astype(np.float32) * scale + min_val).reshape(original_shape)
        
        elif bits == 4:
            header_size = 2 * 4  # 2 float32 values
            header = np.frombuffer(compressed_data[:header_size], dtype=np.float32)
            min_val, scale = header[0], header[1]
            
            packed = np.frombuffer(compressed_data[header_size:], dtype=np.uint8)
            
            # Unpack 4-bit values
            quantized = np.zeros(original_shape[0], dtype=np.uint8)
            for i, byte_val in enumerate(packed):
                if i * 2 < len(quantized):
                    quantized[i * 2] = byte_val & 0xF
                if i * 2 + 1 < len(quantized):
                    quantized[i * 2 + 1] = (byte_val >> 4) & 0xF
            
            return (quantized.astype(np.float32) * scale + min_val).reshape(original_shape)
