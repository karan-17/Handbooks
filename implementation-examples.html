import asyncio
from typing import List, Dict, Any
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.embeddings.openai import OpenAIEmbeddings
from langchain.document_loaders import PyPDFLoader, Docx2txtLoader
import pinecone
from app.core.config import settings
import logging

logger = logging.getLogger(__name__)

class DocumentProcessor:
    def __init__(self):
        self.text_splitter = RecursiveCharacterTextSplitter(
            chunk_size=settings.MAX_CHUNK_SIZE,
            chunk_overlap=settings.CHUNK_OVERLAP,
            separators=["\n\n", "\n", " ", ""]
        )
        self.embeddings = OpenAIEmbeddings(
            openai_api_key=settings.OPENAI_API_KEY,
            model=settings.EMBEDDING_MODEL
        )
        
        # Initialize Pinecone
        pinecone.init(
            api_key=settings.PINECONE_API_KEY,
            environment=settings.PINECONE_ENVIRONMENT
        )
        self.index = pinecone.Index(settings.PINECONE_INDEX_NAME)
    
    async def process_document(self, file_path: str, metadata: Dict[str, Any]) -> str:
        """Process a single document and store in vector database."""
        try:
            # Load document
            content = await self._load_document(file_path)
            
            # Split into chunks
            chunks = self.text_splitter.split_text(content)
            
            # Generate embeddings
            embeddings = await self._generate_embeddings(chunks)
            
            # Store in vector database
            doc_id = await self._store_vectors(chunks, embeddings, metadata)
            
            logger.info(f"Successfully processed document: {file_path}")
            return doc_id
            
        except Exception as e:
            logger.error(f"Error processing document {file_path}: {str(e)}")
            raise
    
    async def _load_document(self, file_path: str) -> str:
        """Load document content based on file type."""
        if file_path.endswith('.pdf'):
            loader = PyPDFLoader(file_path)
        elif file_path.endswith('.docx'):
            loader = Docx2txtLoader(file_path)
        else:
            with open(file_path, 'r', encoding='utf-8') as f:
                return f.read()
        
        docs = loader.load()
        return "\n".join([doc.page_content for doc in docs])
    
    async def _generate_embeddings(self, chunks: List[str]) -> List[List[float]]:
        """Generate embeddings for text chunks."""
        # Batch process embeddings to avoid rate limits
        batch_size = 100
        embeddings = []
        
        for i in range(0, len(chunks), batch_size):
            batch = chunks[i:i + batch_size]
            batch_embeddings = await asyncio.to_thread(
                self.embeddings.embed_documents, batch
            )
            embeddings.extend(batch_embeddings)
            
            # Small delay to respect rate limits
            if i + batch_size < len(chunks):
                await asyncio.sleep(0.1)
        
        return embeddings
    
    async def _store_vectors(self, chunks: List[str], embeddings: List[List[float]], 
                           metadata: Dict[str, Any]) -> str:
        """Store vectors in Pinecone."""
        import uuid
        doc_id = str(uuid.uuid4())
        
        vectors_to_upsert = []
        for i, (chunk, embedding) in enumerate(zip(chunks, embeddings)):
            vector_id = f"{doc_id}_{i}"
            vector_metadata = {
                **metadata,
                "chunk_text": chunk,
                "chunk_index": i,
                "document_id": doc_id
            }
            
            vectors_to_upsert.append({
                "id": vector_id,
                "values": embedding,
                "metadata": vector_metadata
            })
        
        # Batch upsert
        batch_size = 100
        for i in range(0, len(vectors_to_upsert), batch_size):
            batch = vectors_to_upsert[i:i + batch_size]
            await asyncio.to_thread(self.index.upsert, vectors=batch)
        
        return doc_id

import asyncio
from typing import List, Dict, Any, Tuple
import pinecone

from app.core.config import settings

class RetrievalService:
    def __init__(self):
        self.embeddings = OpenAIEmbeddings(
            openai_api_key=settings.OPENAI_API_KEY,
            model=settings.EMBEDDING_MODEL
        )
        
        pinecone.init(
            api_key=settings.PINECONE_API_KEY,
            environment=settings.PINECONE_ENVIRONMENT
        )
        self.index = pinecone.Index(settings.PINECONE_INDEX_NAME)
        
        # Redis for caching
        self.redis_client = redis.from_url(settings.REDIS_URL)
        
    async def retrieve_relevant_docs(self, query: str, 
                                   top_k: int = None,
                                   filters: Dict[str, Any] = None) -> List[Dict[str, Any]]:
        """Retrieve relevant documents for a query."""
        top_k = top_k or settings.TOP_K_RETRIEVAL
        
        # Check cache first
        cache_key = self._get_cache_key(query, top_k, filters)
        cached_result = await self._get_cached_result(cache_key)
        if cached_result:
            return cached_result
        
        # Generate query embedding
        query_embedding = await asyncio.to_thread(
            self.embeddings.embed_query, query
        )
        
        # Search vector database
        search_kwargs = {
            "vector": query_embedding,
            "top_k": top_k,
            "include_metadata": True
        }
        
        if filters:
            search_kwargs["filter"] = filters
        
        search_results = await asyncio.to_thread(
            self.index.query, **search_kwargs
        )
        
        # Filter by similarity threshold
        relevant_docs = []
        for match in search_results.matches:
            if match.score >= settings.SIMILARITY_THRESHOLD:
                relevant_docs.append({
                    "id": match.id,
                    "content": match.metadata.get("chunk_text", ""),
                    "metadata": {k: v for k, v in match.metadata.items() 
                               if k != "chunk_text"},
                    "score": match.score
                })
        
        # Cache result
        await self._cache_result(cache_key, relevant_docs)
        
        return relevant_docs
    
    def _get_cache_key(self, query: str, top_k: int, filters: Dict[str, Any]) -> str:
        """Generate cache key for query."""
        key_data = {
            "query": query,
            "top_k": top_k,
            "filters": filters or {}
        }
        key_string = json.dumps(key_data, sort_keys=True)
        return hashlib.md5(key_string.encode()).hexdigest()
    
    async def _get_cached_result(self, cache_key: str) -> List[Dict[str, Any]]:
        """Get cached search result."""
        try:
            cached_data = await asyncio.to_thread(
                self.redis_client.get, f"search:{cache_key}"
            )
            if cached_data:
                return json.loads(cached_data)
        except Exception:
            pass
        return None
    
    async def _cache_result(self, cache_key: str, result: List[Dict[str, Any]]):
        """Cache search result."""
        try:
            await asyncio.to_thread(
                self.redis_client.setex,
                f"search:{cache_key}",
                3600,  # 1 hour TTL
                json.dumps(result)
            )
        except Exception:
            pass  # Don't fail if caching fails

import asyncio
from typing import List, Dict, Any
import openai
from app.core.config import settings
import logging

logger = logging.getLogger(__name__)

class GenerationService:
    def __init__(self):
        self.client = openai.AsyncOpenAI(api_key=settings.OPENAI_API_KEY)
        
    async def generate_response(self, query: str, 
                              relevant_docs: List[Dict[str, Any]],
                              conversation_history: List[Dict[str, str]] = None) -> Dict[str, Any]:
        """Generate response using retrieved documents."""
        
        # Prepare context
        context = self._prepare_context(relevant_docs)
        
        # Build messages
        messages = self._build_messages(query, context, conversation_history)
        
        try:
            # Generate response
            response = await self.client.chat.completions.create(
                model=settings.OPENAI_MODEL,
                messages=messages,
                temperature=0.1,
                max_tokens=2000,
                stream=False
            )
            
            return {
                "answer": response.choices[0].message.content,
                "token_usage": dict(response.usage),
                "model": settings.OPENAI_MODEL,
                "context_docs": len(relevant_docs)
            }
            
        except Exception as e:
            logger.error(f"Error generating response: {str(e)}")
            raise
    
    def _prepare_context(self, relevant_docs: List[Dict[str, Any]]) -> str:
        """Prepare context from retrieved documents."""
        if not relevant_docs:
            return "No relevant context found."
        
        context_parts = []
        for i, doc in enumerate(relevant_docs, 1):
            source_info = doc.get('metadata', {}).get('filename', 'Unknown source')
            context_parts.append(f"[Document {i} - {source_info}]\n{doc['content']}\n")
        
        return "\n".join(context_parts)
    
    def _build_messages(self, query: str, context: str, 
                       conversation_history: List[Dict[str, str]] = None) -> List[Dict[str, str]]:
        """Build message array for OpenAI API."""
        
        system_prompt = f"""You are a helpful AI assistant that answers questions based on the provided context.
        
INSTRUCTIONS:
1. Use the context below to answer the user's question
2. If the context doesn't contain enough information, say so clearly
3. Cite specific sources when possible
4. Be concise but comprehensive
5. If asked about something not in the context, politely redirect to the available information

CONTEXT:
{context}
"""
        
        messages = [{"role": "system", "content": system_prompt}]
        
        # Add conversation history if provided
        if conversation_history:
            messages.extend(conversation_history[-10:])  # Keep last 10 exchanges
        
        # Add current query
        messages.append({"role": "user", "content": query})
        
        return messages

    async def generate_streaming_response(self, query: str, 
                                        relevant_docs: List[Dict[str, Any]],
                                        conversation_history: List[Dict[str, str]] = None):
        """Generate streaming response for real-time UI updates."""
        
        context = self._prepare_context(relevant_docs)
        messages = self._build_messages(query, context, conversation_history)
        
        try:
            stream = await self.client.chat.completions.create(
                model=settings.OPENAI_MODEL,
                messages=messages,
                temperature=0.1,
                max_tokens=2000,
                stream=True
            )
            
            async for chunk in stream:
                if chunk.choices[0].delta.content is not None:
                    yield chunk.choices[0].delta.content
                    
        except Exception as e:
            logger.error(f"Error in streaming generation: {str(e)}")
            yield f"Error: {str(e)}"

@app.post("/documents/upload")
async def upload_document(
    file: UploadFile = File(...),
    metadata: str = None,
    background_tasks: BackgroundTasks = BackgroundTasks(),
    db = Depends(get_db)
):
    """Upload and process a document."""
    
    if file.content_type not in ["application/pdf", "application/vnd.openxmlformats-officedocument.wordprocessingml.document", "text/plain"]:
        raise HTTPException(status_code=400, detail="Unsupported file type")
    
    # Save uploaded file
    import tempfile
    import os
    
    with tempfile.NamedTemporaryFile(delete=False, suffix=os.path.splitext(file.filename)[1]) as tmp_file:
        content = await file.read()
        tmp_file.write(content)
        tmp_file_path = tmp_file.name
    
    # Parse metadata
    doc_metadata = json.loads(metadata) if metadata else {}
    doc_metadata.update({
        "filename": file.filename,
        "content_type": file.content_type,
        "upload_time": datetime.utcnow().isoformat()
    })
    
    try:
        # Process document asynchronously
        doc_id = await doc_processor.process_document(tmp_file_path, doc_metadata)
        
        # Clean up temp file
        os.unlink(tmp_file_path)
        
        return {
            "document_id": doc_id,
            "status": "processed",
            "filename": file.filename,
            "message": "Document processed successfully"
        }
        
    except Exception as e:
        # Clean up temp file on error
        if os.path.exists(tmp_file_path):
            os.unlink(tmp_file_path)
        raise HTTPException(status_code=500, detail=f"Error processing document: {str(e)}")

import asyncio
from typing import List, Dict, Any, Union
import base64
import openai
from PIL import Image
import io
import fitz  # PyMuPDF
from langchain.text_splitter import RecursiveCharacterTextSplitter
import clip
import torch
import numpy as np

class MultiModalProcessor:
    def __init__(self):
        self.openai_client = openai.AsyncOpenAI()
        self.text_splitter = RecursiveCharacterTextSplitter(
            chunk_size=1000,
            chunk_overlap=200
        )
        
        # Load CLIP model for image embeddings
        self.clip_model, self.clip_preprocess = clip.load("ViT-B/32")
        
    async def process_document(self, file_path: str, metadata: Dict[str, Any]) -> Dict[str, Any]:
        """Process document extracting both text and images."""
        
        if file_path.endswith('.pdf'):
            return await self._process_pdf(file_path, metadata)
        elif file_path.lower().endswith(('.png', '.jpg', '.jpeg')):
            return await self._process_image(file_path, metadata)
        else:
            return await self._process_text(file_path, metadata)
    
    async def _process_pdf(self, file_path: str, metadata: Dict[str, Any]) -> Dict[str, Any]:
        """Extract text and images from PDF."""
        
        doc = fitz.open(file_path)
        text_chunks = []
        images = []
        
        for page_num in range(len(doc)):
            page = doc[page_num]
            
            # Extract text
            text = page.get_text()
            if text.strip():
                chunks = self.text_splitter.split_text(text)
                for i, chunk in enumerate(chunks):
                    text_chunks.append({
                        "content": chunk,
                        "type": "text",
                        "page": page_num + 1,
                        "chunk_index": i,
                        "metadata": {**metadata, "page": page_num + 1}
                    })
            
            # Extract images
            image_list = page.get_images()
            for img_index, img in enumerate(image_list):
                xref = img[0]
                pix = fitz.Pixmap(doc, xref)
                
                if pix.n - pix.alpha < 4:  # GRAY or RGB
                    img_data = pix.tobytes("png")
                    
                    # Generate image description
                    img_description = await self._describe_image(img_data)
                    
                    images.append({
                        "content": img_description,
                        "type": "image",
                        "page": page_num + 1,
                        "image_index": img_index,
                        "image_data": base64.b64encode(img_data).decode(),
                        "metadata": {
                            **metadata, 
                            "page": page_num + 1,
                            "image_index": img_index
                        }
                    })
                
                pix = None
        
        doc.close()
        
        return {
            "text_chunks": text_chunks,
            "images": images,
            "total_chunks": len(text_chunks) + len(images)
        }
    
    async def _describe_image(self, image_data: bytes) -> str:
        """Generate text description of image using GPT-4V."""
        
        base64_image = base64.b64encode(image_data).decode('utf-8')
        
        try:

            "context_summary": ""
        }

    async def add_message(self, session_id: str, role: str, content: str, 
                         metadata: Dict[str, Any] = None) -> bool:
        """Add a message to the conversation."""
        
        session_data = await self.get_session(session_id)
        if not session_data:
            return False
        
        message = {
            "role": role,
            "content": content,
            "timestamp": datetime.utcnow().isoformat(),
            "metadata": metadata or {}
        }
        
        session_data["messages"].append(message)
        
        # Keep only last 20 messages to manage memory
        if len(session_data["messages"]) > 20:
            session_data["messages"] = session_data["messages"][-20:]
        
        # Update context summary if needed
        if len(session_data["messages"]) % 5 == 0:  # Every 5 messages
            session_data["context_summary"] = await self._update_context_summary(session_data)
        
        await asyncio.to_thread(
            self.redis_client.setex,
            f"session:{session_id}",
            self.conversation_ttl,
            json.dumps(session_data)
        )
        
        return True

        """Get conversation session data."""

        )

            return json.loads(session_data)
        return None

    async def evaluate_response_quality(self, query: str, response: str, 
                                      retrieved_docs: List[Dict[str, Any]]) -> Dict[str, float]:
        """Evaluate response quality using multiple metrics."""
        
        metrics = {}
        
        # Relevance evaluation
        metrics['relevance_score'] = await self._evaluate_relevance(query, response, retrieved_docs)
        
        # Faithfulness evaluation (how well response reflects source documents)
        metrics['faithfulness_score'] = await self._evaluate_faithfulness(response, retrieved_docs)
        
        # Completeness evaluation
        metrics['completeness_score'] = await self._evaluate_completeness(query, response)
        
        # Clarity evaluation
        metrics['clarity_score'] = await self._evaluate_clarity(response)
        
        # Overall score
        weights = {'relevance': 0.3, 'faithfulness': 0.3, 'completeness': 0.2, 'clarity': 0.2}
        metrics['overall_score'] = sum(
            metrics[f'{metric}_score'] * weight 
            for metric, weight in weights.items()
        )
        
        return metrics
    
    async def _evaluate_relevance(self, query: str, response: str, 
                                retrieved_docs: List[Dict[str, Any]]) -> float:
        """Evaluate how relevant the response is to the query."""
        
        prompt = f"""
        Evaluate how well this response answers the given query on a scale of 0-1.
        
        Query: {query}
        
        Response: {response}
        
        Rate the relevance (0.0 = not relevant, 1.0 = perfectly relevant):
        """
        
        try:
            evaluation = await self.generation_service.generate_response(
                query=prompt,
                relevant_docs=[],  # No docs needed for this evaluation
            )
            
            # Extract score from evaluation response
            score_match = re.search(r'(\d*\.?\d+)', evaluation['answer'])
            if score_match:
                return min(1.0, max(0.0, float(score_match.group(1))))
            
        except Exception:
            pass
        
        return 0.5  # Default fallback score
    
    async def _evaluate_faithfulness(self, response: str, 
                                   retrieved_docs: List[Dict[str, Any]]) -> float:
        """Evaluate how faithful the response is to the source documents."""
        
        if not retrieved_docs:
            return 0.0
        
        source_content = "\n".join([doc['content'] for doc in retrieved_docs])
        
        prompt = f"""
        Evaluate how well this response is supported by the source documents on a scale of 0-1.
        Check for factual accuracy and consistency.
        
        Source Documents:
        {source_content}
        
        Response: {response}
        
        Rate the faithfulness (0.0 = contradicts sources, 1.0 = fully supported):
        """
        
        try:
            evaluation = await self.generation_service.generate_response(
                query=prompt,
                relevant_docs=[],
            )
            
            score_match = re.search(r'(\d*\.?\d+)', evaluation['answer'])
            if score_match:
                return min(1.0, max(0.0, float(score_match.group(1))))
            
        except Exception:
            pass
        
        return 0.5
    
    async def run_evaluation_suite(self, test_queries: List[Dict[str, Any]]) -> Dict[str, Any]:
        """Run comprehensive evaluation on a set of test queries."""
        
        results = []
        
        for test_case in test_queries:
            query = test_case['query']
            expected_answer = test_case.get('expected_answer')
            
            # Get RAG response
            rag_response = await self.generation_service.generate_response(
                query=query,
                relevant_docs=test_case.get('relevant_docs', [])
            )
            
            # Evaluate response
            quality_scores = await self.evaluate_response_quality(
                query=query,
                response=rag_response['answer'],
                retrieved_docs=test_case.get('relevant_docs', [])
            )
            
            result = {
                'query': query,
                'response': rag_response['answer'],
                'expected': expected_answer,
                'scores': quality_scores,
                'latency_ms': test_case.get('latency_ms', 0)
            }
            
            results.append(result)
        
        # Calculate aggregate metrics
        avg_scores = {}
        for metric in ['relevance_score', 'faithfulness_score', 'completeness_score', 'clarity_score', 'overall_score']:
            scores = [r['scores'].get(metric, 0) for r in results]
            avg_scores[f'avg_{metric}'] = sum(scores) / len(scores) if scores else 0
        
        return {
            'individual_results': results,
            'aggregate_metrics': avg_scores,
            'total_tests': len(results),
            'evaluation_timestamp': datetime.utcnow().isoformat()
        }

class CostTracker:
    def __init__(self):
        self.daily_costs = {}
        
    async def track_openai_cost(self, usage_data: Dict[str, Any]):
        """Track OpenAI API costs."""
        
        today = datetime.utcnow().strftime('%Y-%m-%d')
        
        if today not in self.daily_costs:
            self.daily_costs[today] = {
                'embedding_tokens': 0,
                'completion_tokens': 0,
                'prompt_tokens': 0,
                'total_cost': 0.0
            }
        
        # OpenAI pricing (approximate)
        embedding_cost_per_1k = 0.0001
        completion_cost_per_1k = 0.002
        
        if 'embedding_tokens' in usage_data:
            tokens = usage_data['embedding_tokens']
            cost = (tokens / 1000) * embedding_cost_per_1k
            self.daily_costs[today]['embedding_tokens'] += tokens
            self.daily_costs[today]['total_cost'] += cost
        
        if 'completion_tokens' in usage_data:
            tokens = usage_data['completion_tokens']
            cost = (tokens / 1000) * completion_cost_per_1k
            self.daily_costs[today]['completion_tokens'] += tokens
            self.daily_costs[today]['total_cost'] += cost
    
    async def get_cost_breakdown(self, days: int = 30) -> Dict[str, Any]:
        """Get cost breakdown for specified period."""
        
        end_date = datetime.utcnow()
        start_date = end_date - timedelta(days=days)
        
        total_cost = 0.0
        daily_breakdown = []
        
        current_date = start_date
        while current_date <= end_date:
            date_str = current_date.strftime('%Y-%m-%d')
            day_cost = self.daily_costs.get(date_str, {}).get('total_cost', 0.0)
            
            daily_breakdown.append({
                'date': date_str,
                'cost_usd': day_cost
            })
            
            total_cost += day_cost
            current_date += timedelta(days=1)
        
        average_daily_cost = total_cost / days if days > 0 else 0
        
        return {
            'period_days': days,
            'total_cost_usd': total_cost,
            'average_daily_cost_usd': average_daily_cost,
            'projected_monthly_cost_usd': average_daily_cost * 30,
            'daily_breakdown': daily_breakdown
        }
